1//! An implementation of a diff functionality for the [`ABinaryTree`] struct.
2//!
3//! # About
4//!
5//! This module provides the [`AbDiff`] and [`StagedAbDiff`] structs that allow
6//! performing changes to an [`ABinaryTree`] instance without immediately applying
7//! the them. Instead, the changes can be applied to the diff and the results
8//! examined before merging the given diff back into the tree (or not).
9//!
10//! # Don't Panic!
11//!
12//! Functions in this module should never panic. However, if there is a bug in
13//! the implementation, a function will return an unrecoverable
14//! [`LibraryError`](ABinaryTreeDiffError::LibraryError). This means that some
15//! functions that are not expected to fail and throw an error, will still
16//! return a [`Result`] since they may throw a
17//! [`LibraryError`](ABinaryTreeDiffError::LibraryError).
18

36/// The [`StagedAbDiff`] can be created from an [`AbDiff`] instance. It's sole
37/// purpose is to be subsequently merged into an existing [`ABinaryTree`]
38/// instance. The difference between [`StagedAbDiff`] and an [`AbDiff`] is that a
39/// [`StagedAbDiff`] is immutable and does not contain a reference to the original
40/// tree. Since it only contains the actual diff without reference to the
41/// original content, it can't provide the same information as the [`AbDiff`] it
42/// was created from. However, the lack of the internal reference means that its
43/// lifetime is not tied to that of the original tree.
44#[derive(Debug, Serialize, Deserialize)]
45#[cfg_attr(any(test, feature = "test-utils"), derive(Clone, PartialEq))]
46pub(crate) struct StagedAbDiff<L: Clone + Debug + Default, P: Clone + Debug + Default> {
47    leaf_diff: BTreeMap<LeafNodeIndex, L>,
48    parent_diff: BTreeMap<ParentNodeIndex, P>,
49    size: TreeSize,
50}
51
52impl<'a, L: Clone + Debug + Default, P: Clone + Debug + Default> From<AbDiff<'a, L, P>>
53    for StagedAbDiff<L, P>
54{
55    fn from(diff: AbDiff<'a, L, P>) -> Self {
56        StagedAbDiff {
57            leaf_diff: diff.leaf_diff,
58            parent_diff: diff.parent_diff,
59            size: diff.size,
60        }
61    }
62}

64impl<L: Clone + Debug + Default, P: Clone + Debug + Default> StagedAbDiff<L, P> {
65    /// Return the leaf and parent diffs as a tuple.
66    pub(super) fn into_diffs(self) -> (BTreeMap<LeafNodeIndex, L>, BTreeMap<ParentNodeIndex, P>) {
67        (self.leaf_diff, self.parent_diff)
68    }
69
70    /// Return the projected size of the tree after a merge with the diff.
71    pub(super) fn tree_size(&self) -> TreeSize {
72        self.size
73    }
74}

76/// The [`AbDiff`] represents a set of differences (i.e. a "Diff") for an
77/// [`ABinaryTree`]. It can be created from an [`ABinaryTree`] instance and then
78/// accessed mutably or immutably. Any changes are saved by the [`AbDiff`] applied
79/// to the original [`ABinaryTree`] instance by converting it to a [`StagedAbDiff`]
80/// and subsequently merging it.
81pub(crate) struct AbDiff<'a, L: Clone + Debug + Default, P: Clone + Debug + Default> {
82    original_tree: &'a ABinaryTree<L, P>,
83    leaf_diff: BTreeMap<LeafNodeIndex, L>,
84    parent_diff: BTreeMap<ParentNodeIndex, P>,
85    size: TreeSize,
86    default_leaf: L,
87    default_parent: P,
88}
89
90impl<'a, L: Clone + Debug + Default, P: Clone + Debug + Default> From<&'a ABinaryTree<L, P>>
91    for AbDiff<'a, L, P>
92{
93    fn from(tree: &'a ABinaryTree<L, P>) -> AbDiff<'a, L, P> {
94        AbDiff {
95            original_tree: tree,
96            leaf_diff: BTreeMap::new(),
97            parent_diff: BTreeMap::new(),
98            size: tree.tree_size(),
99            default_leaf: L::default(),
100            default_parent: P::default(),
101        }
102    }
103}

105impl<L: Clone + Debug + Default, P: Clone + Debug + Default> AbDiff<'_, L, P> {
106    // Functions handling interactions with leaves.
107    ///////////////////////////////////////////////
108
109    /// Grow the tree by adding a new subtree to the right of the tree.
110    ///
111    /// Returns an error if the the size of the diff increases beyond
112    /// [`MAX_TREE_SIZE`].
113    pub(crate) fn grow_tree(&mut self) -> Result<(), ABinaryTreeDiffError> {
114        // Prevent the tree from becoming too large.
115        if self.size().u32() > MAX_TREE_SIZE / 2 {
116            return Err(ABinaryTreeDiffError::TreeTooLarge);
117        }
118        self.size.inc();
119        Ok(())
120    }
121
122    /// Shrink the tree by removing the rightmost subtree.
123    ///
124    /// Returns an error if the diff only has one leaf left.
125    pub(crate) fn shrink_tree(&mut self) -> Result<(), ABinaryTreeDiffError> {
126        // First make sure that the tree isn't getting too small.
127        if self.size().u32() <= MIN_TREE_SIZE {
128            return Err(ABinaryTreeDiffError::TreeTooSmall);
129        }
130        self.size.dec();
131        // We also need to remove nodes from the diff that are now outside the
132        // tree.
133        self.leaf_diff
134            .retain(|&index, _| index.u32() < self.size.leaf_count());
135
136        self.parent_diff
137            .retain(|&index, _| index.u32() < self.size.parent_count());
138        Ok(())
139    }
140
141    /// Replace the content of the leaf node at the given leaf index with new
142    /// content.
143    pub(crate) fn replace_leaf(&mut self, leaf_index: LeafNodeIndex, new_leaf: L) {
144        debug_assert!(leaf_index.u32() < self.leaf_count());
145        self.leaf_diff.insert(leaf_index, new_leaf);
146    }
147
148    /// Replace the content of the parent node at the given leaf index with new
149    /// content.
150    pub(crate) fn replace_parent(&mut self, parent_index: ParentNodeIndex, node: P) {
151        debug_assert!(parent_index.u32() < self.parent_count());
152        self.parent_diff.insert(parent_index, node);
153    }
154
155    /// Returns an iterator over a tuple of the leaf index and a reference to a
156    /// leaf, sorted according to their position in the tree from left to right.
157    pub(crate) fn leaves(&self) -> impl Iterator<Item = (LeafNodeIndex, &L)> {
158        let original_leaves = self.original_tree.leaves().peekable();
159        let diff_leaves = self
160            .leaf_diff
161            .iter()
162            .map(|(index, leaf)| (*index, leaf))
163            .peekable();
164
165        // Combine the original leaves with the leaves from the diff. Since both
166        // iterators are sorted, we can just iterate over them and don't need
167        // additional sorting. If one of the iterators is exhausted, we just add
168        // the remaining leaves from the other iterator. We also make sure that
169        // we don't add leaves from the original leaves that are also in the
170        // diff.
171
172        // Harmonize the iterator types
173        let a_iter = Box::new(diff_leaves) as Box<dyn Iterator<Item = (LeafNodeIndex, &L)>>;
174        let b_iter = Box::new(original_leaves) as Box<dyn Iterator<Item = (LeafNodeIndex, &L)>>;
175
176        // We only compare indices, not the actual leaves
177        let cmp = |&(x, _): &(LeafNodeIndex, &L)| x;
178
179        sorted_iter(a_iter, b_iter, cmp, self.leaf_count() as usize)
180    }
181
182    pub(crate) fn parents(&self) -> impl Iterator<Item = (ParentNodeIndex, &P)> {
183        let original_parents = self.original_tree.parents().peekable();
184        let diff_parents = self
185            .parent_diff
186            .iter()
187            .map(|(index, parent)| (*index, parent))
188            .peekable();
189
190        // Combine the original parents with the parents from the diff. Since
191        // both iterators are sorted, we can just iterate over them and don't
192        // need additional sorting. If one of the iterators is exhausted, we
193        // just add the remaining parents from the other iterator. We also make
194        // sure that we don't add parents from the original parents that are
195        // also in the diff.
196
197        // Harmonize the iterator types
198        let a_iter = Box::new(diff_parents) as Box<dyn Iterator<Item = (ParentNodeIndex, &P)>>;
199        let b_iter = Box::new(original_parents) as Box<dyn Iterator<Item = (ParentNodeIndex, &P)>>;
200
201        // We only compare indices, not the actual parents
202        let cmp = |&(x, _): &(ParentNodeIndex, &P)| x;
203
204        sorted_iter(a_iter, b_iter, cmp, self.parent_count() as usize)
205    }

210    /// Returns a vector of [`ParentNodeIndex`] instances, each one referencing a
211    /// node in the direct path of the given [`LeafNodeIndex`], ordered from the
212    /// parent of the corresponding leaf to the root of the tree.
213    pub(crate) fn direct_path(&self, leaf_index: LeafNodeIndex) -> Vec<ParentNodeIndex> {
214        direct_path(leaf_index, self.size())
215    }
216
217    /// Sets all nodes in the direct path to a copy of the given node.
218    pub(crate) fn set_direct_path_to_node(&mut self, leaf_index: LeafNodeIndex, node: &P) {
219        let direct_path = self.direct_path(leaf_index);
220        for node_index in &direct_path {
221            self.replace_parent(*node_index, node.clone());
222        }
223    }

225    /// Returns the copath of a leaf node.
226    pub(crate) fn copath(&self, leaf_index: LeafNodeIndex) -> Vec<TreeNodeIndex> {
227        copath(leaf_index, self.size())
228    }
229
230    /// Returns the lowest common ancestor of two leaf nodes.
231    pub(crate) fn lowest_common_ancestor(
232        &self,
233        leaf_index_1: LeafNodeIndex,
234        leaf_index_2: LeafNodeIndex,
235    ) -> ParentNodeIndex {
236        debug_assert!(leaf_index_1 != leaf_index_2);
237        debug_assert!(leaf_index_1.u32() < self.leaf_count());
238        debug_assert!(leaf_index_2.u32() < self.leaf_count());
239        lowest_common_ancestor(leaf_index_1, leaf_index_2)
240    }
241
242    // Functions related to the shared subtree of two given leaves
243    //////////////////////////////////////////////////////////////
244
245    /// Returns [`TreeNodeIndex`] to the copath node of the `leaf_index_1` that is
246    /// in the direct path of `leaf_index_2`.
247    pub(crate) fn subtree_root_copath_node(
248        &self,
249        leaf_index_1: LeafNodeIndex,
250        leaf_index_2: LeafNodeIndex,
251    ) -> TreeNodeIndex {
252        debug_assert!(leaf_index_1 != leaf_index_2);
253        debug_assert!(leaf_index_1.u32() < self.leaf_count());
254        debug_assert!(leaf_index_2.u32() < self.leaf_count());
255
256        // We want to return the position of the lowest common ancestor in the
257        // direct path of `leaf_index_1` (i.e. the sender_leaf_index).
258        let subtree_root_node_index = lowest_common_ancestor(leaf_index_1, leaf_index_2);
259
260        // Figure out which child is the relevant copath node.
261        if leaf_index_2 < leaf_index_1 {
262            left(subtree_root_node_index)
263        } else {
264            right(subtree_root_node_index)
265        }
266    }

312    /// Returns a reference to the leaf node at index `leaf_index`.
313    pub(crate) fn leaf(&self, leaf_index: LeafNodeIndex) -> &L {
314        // Check if it's in the diff.
315        if let Some(node) = self.leaf_diff.get(&leaf_index) {
316            node
317        // If it's not in the diff, it could be that it's outside of the
318        // diff, in which case we want to return a blank here, because the
319        // diff might have been trimmed in the mean time.
320        } else if leaf_index.u32() >= self.leaf_count() {
321            &self.default_leaf
322            // If it isn't in the diff, it must be in the tree.
323        } else {
324            self.original_tree.leaf_by_index(leaf_index)
325        }
326    }
327
328    /// Returns a reference to the parent node at index `parent_index`.
329    pub(crate) fn parent(&self, parent_index: ParentNodeIndex) -> &P {
330        // Check if it's in the diff.
331        if let Some(node) = self.parent_diff.get(&parent_index) {
332            return node;
333        }
334        // If it isn't in the diff, it must be in the tree.
335        self.original_tree.parent_by_index(parent_index)
336    }
337
338    /// Returns a mutable reference to the parent node in the diff at index
339    /// `parent_index`. If the diff doesn't have a node at that index, it clones
340    /// the node to the diff and returns a mutable reference to that node.
341    pub(crate) fn parent_mut(&mut self, parent_index: ParentNodeIndex) -> &mut P {
342        debug_assert!(parent_index.u32() < self.parent_count());
343        // We then check if the node is already in the diff. (Not using `if let
344        // ...` here, because the borrow checker doesn't like that).
345        if self.parent_diff.contains_key(&parent_index) {
346            return self
347                .parent_diff
348                .get_mut(&parent_index)
349                // We just checked that this index exists, so this must be Some.
350                .unwrap_or(&mut self.default_parent);
351            // If not, we take a copy from the original tree and put it in the
352            // diff before returning a mutable reference to it.
353        }
354        let tree_node = self.original_tree.parent_by_index(parent_index);
355        self.replace_parent(parent_index, tree_node.clone());
356        self.parent_diff
357            .get_mut(&parent_index)
358            // We just inserted this into the diff, so this should be Some.
359            .unwrap_or(&mut self.default_parent)
360    }

362    // Index checking
363
364    #[cfg(test)]
365    pub(crate) fn deref_vec(
366        &self,
367        parent_index_vec: Vec<ParentNodeIndex>,
368    ) -> Result<Vec<&P>, ABinaryTreeDiffError> {
369        let mut parent_vec = Vec::new();
370        for parent_index in parent_index_vec {
371            let node = self.parent(parent_index);
372            parent_vec.push(node);
373        }
374        Ok(parent_vec)
375    }
