1//! An implementation of a diff functionality for the [`ABinaryTree`] struct.
2//!
3//! # About
4//!
5//! This module provides the [`AbDiff`] and [`StagedAbDiff`] structs that allow
6//! performing changes to an [`ABinaryTree`] instance without immediately applying
7//! the them. Instead, the changes can be applied to the diff and the results
8//! examined before merging the given diff back into the tree (or not).
9//!
10//! # Don't Panic!
11//!
12//! Functions in this module should never panic. However, if there is a bug in
13//! the implementation, a function will return an unrecoverable
14//! [`LibraryError`](ABinaryTreeDiffError::LibraryError). This means that some
15//! functions that are not expected to fail and throw an error, will still
16//! return a [`Result`] since they may throw a
17//! [`LibraryError`](ABinaryTreeDiffError::LibraryError).
18

36/// The [`StagedAbDiff`] can be created from an [`AbDiff`] instance. It's sole
37/// purpose is to be subsequently merged into an existing [`ABinaryTree`]
38/// instance. The difference between [`StagedAbDiff`] and an [`AbDiff`] is that a
39/// [`StagedAbDiff`] is immutable and does not contain a reference to the original
40/// tree. Since it only contains the actual diff without reference to the
41/// original content, it can't provide the same information as the [`AbDiff`] it
42/// was created from. However, the lack of the internal reference means that its
43/// lifetime is not tied to that of the original tree.
44#[derive(Debug, Serialize, Deserialize)]
45#[cfg_attr(any(test, feature = "test-utils"), derive(Clone, PartialEq))]
46pub(crate) struct StagedAbDiff<L: Clone + Debug + Default, P: Clone + Debug + Default> {
47    leaf_diff: BTreeMap<LeafNodeIndex, L>,
48    parent_diff: BTreeMap<ParentNodeIndex, P>,
49    size: TreeSize,
50}
51
52impl<'a, L: Clone + Debug + Default, P: Clone + Debug + Default> From<AbDiff<'a, L, P>>
53    for StagedAbDiff<L, P>
54{
55    fn from(diff: AbDiff<'a, L, P>) -> Self {
56        StagedAbDiff {
57            leaf_diff: diff.leaf_diff,
58            parent_diff: diff.parent_diff,
59            size: diff.size,
60        }
61    }
62}

102/// The [`AbDiff`] represents a set of differences (i.e. a "Diff") for an
103/// [`ABinaryTree`]. It can be created from an [`ABinaryTree`] instance and then
104/// accessed mutably or immutably. Any changes are saved by the [`AbDiff`] applied
105/// to the original [`ABinaryTree`] instance by converting it to a [`StagedAbDiff`]
106/// and subsequently merging it.
107pub(crate) struct AbDiff<'a, L: Clone + Debug + Default, P: Clone + Debug + Default> {
108    original_tree: &'a ABinaryTree<L, P>,
109    leaf_diff: BTreeMap<LeafNodeIndex, L>,
110    parent_diff: BTreeMap<ParentNodeIndex, P>,
111    size: TreeSize,
112    default_leaf: L,
113    default_parent: P,
114}
115
116impl<'a, L: Clone + Debug + Default, P: Clone + Debug + Default> From<&'a ABinaryTree<L, P>>
117    for AbDiff<'a, L, P>
118{
119    fn from(tree: &'a ABinaryTree<L, P>) -> AbDiff<'a, L, P> {
120        AbDiff {
121            original_tree: tree,
122            leaf_diff: BTreeMap::new(),
123            parent_diff: BTreeMap::new(),
124            size: tree.tree_size(),
125            default_leaf: L::default(),
126            default_parent: P::default(),
127        }
128    }
129}

131impl<L: Clone + Debug + Default, P: Clone + Debug + Default> AbDiff<'_, L, P> {

135    /// Grow the tree by adding a new subtree to the right of the tree.
136    ///
137    /// Returns an error if the the size of the diff increases beyond
138    /// [`MAX_TREE_SIZE`].
139    pub(crate) fn grow_tree(&mut self) -> Result<(), ABinaryTreeDiffError> {
140        // Prevent the tree from becoming too large.
141        if self.size().u32() > MAX_TREE_SIZE / 2 {
142            return Err(ABinaryTreeDiffError::TreeTooLarge);
143        }
144        self.size.inc();
145        Ok(())
146    }
147
148    /// Shrink the tree by removing the rightmost subtree.
149    ///
150    /// Returns an error if the diff only has one leaf left.
151    pub(crate) fn shrink_tree(&mut self) -> Result<(), ABinaryTreeDiffError> {
152        // First make sure that the tree isn't getting too small.
153        if self.size().u32() <= MIN_TREE_SIZE {
154            return Err(ABinaryTreeDiffError::TreeTooSmall);
155        }
156        self.size.dec();
157        // We also need to remove nodes from the diff that are now outside the
158        // tree.
159        self.leaf_diff
160            .retain(|&index, _| index.u32() < self.size.leaf_count());
161
162        self.parent_diff
163            .retain(|&index, _| index.u32() < self.size.parent_count());
164        Ok(())
165    }
166
167    /// Replace the content of the leaf node at the given leaf index with new
168    /// content.
169    pub(crate) fn replace_leaf(&mut self, leaf_index: LeafNodeIndex, new_leaf: L) {
170        debug_assert!(leaf_index.u32() < self.leaf_count());
171        self.leaf_diff.insert(leaf_index, new_leaf);
172    }
173
174    /// Replace the content of the parent node at the given leaf index with new
175    /// content.
176    pub(crate) fn replace_parent(&mut self, parent_index: ParentNodeIndex, node: P) {
177        debug_assert!(parent_index.u32() < self.parent_count());
178        self.parent_diff.insert(parent_index, node);
179    }

181    /// Returns an iterator over a tuple of the parent index and a reference to a
182    /// node, sorted according to their position in the tree from left to right.
183    fn raw_parents<'a>(
184        original_tree: &'a ABinaryTree<L, P>,
185        parent_diff: &'a BTreeMap<ParentNodeIndex, P>,
186        parent_count: usize,
187    ) -> impl Iterator<Item = (ParentNodeIndex, &'a P)> {
188        let original_parents = original_tree.parents().peekable();
189        let diff_parents = parent_diff
190            .iter()
191            .map(|(index, parent)| (*index, parent))
192            .peekable();
193
194        // Combine the original parents with the parents from the diff. Since
195        // both iterators are sorted, we can just iterate over them and don't
196        // need additional sorting. If one of the iterators is exhausted, we
197        // just add the remaining parents from the other iterator. We also make
198        // sure that we don't add parents from the original parents that are
199        // also in the diff.
200
201        // Harmonize the iterator types
202        let a_iter = Box::new(diff_parents) as Box<dyn Iterator<Item = (ParentNodeIndex, &P)>>;
203        let b_iter = Box::new(original_parents) as Box<dyn Iterator<Item = (ParentNodeIndex, &P)>>;
204
205        // We only compare indices, not the actual parents
206        let cmp = |&(x, _): &(ParentNodeIndex, &P)| x;
207
208        sorted_iter(a_iter, b_iter, cmp, parent_count)
209    }
210
211    /// Returns an iterator over a tuple of the leaf index and a reference to a
212    /// leaf, sorted according to their position in the tree from left to right.
213    fn raw_leaves<'a>(
214        original_tree: &'a ABinaryTree<L, P>,
215        leaf_diff: &'a BTreeMap<LeafNodeIndex, L>,
216        leaf_count: usize,
217    ) -> impl Iterator<Item = (LeafNodeIndex, &'a L)> {
218        let original_leaves = original_tree.leaves().peekable();
219        let diff_leaves = leaf_diff
220            .iter()
221            .map(|(index, leaf)| (*index, leaf))
222            .peekable();
223
224        // Combine the original leaves with the leaves from the diff. Since both
225        // iterators are sorted, we can just iterate over them and don't need
226        // additional sorting. If one of the iterators is exhausted, we just add
227        // the remaining leaves from the other iterator. We also make sure that
228        // we don't add leaves from the original leaves that are also in the
229        // diff.
230
231        // Harmonize the iterator types
232        let a_iter = Box::new(diff_leaves) as Box<dyn Iterator<Item = (LeafNodeIndex, &L)>>;
233        let b_iter = Box::new(original_leaves) as Box<dyn Iterator<Item = (LeafNodeIndex, &L)>>;
234
235        // We only compare indices, not the actual leaves
236        let cmp = |&(x, _): &(LeafNodeIndex, &L)| x;
237
238        sorted_iter(a_iter, b_iter, cmp, leaf_count)
239    }

284    /// Returns the lowest common ancestor of two leaf nodes.
285    pub(crate) fn lowest_common_ancestor(
286        &self,
287        leaf_index_1: LeafNodeIndex,
288        leaf_index_2: LeafNodeIndex,
289    ) -> ParentNodeIndex {
290        debug_assert!(leaf_index_1 != leaf_index_2);
291        debug_assert!(leaf_index_1.u32() < self.leaf_count());
292        debug_assert!(leaf_index_2.u32() < self.leaf_count());
293        lowest_common_ancestor(leaf_index_1, leaf_index_2)
294    }

299    /// Returns [`TreeNodeIndex`] to the copath node of the `leaf_index_1` that is
300    /// in the direct path of `leaf_index_2`.
301    pub(crate) fn subtree_root_copath_node(
302        &self,
303        leaf_index_1: LeafNodeIndex,
304        leaf_index_2: LeafNodeIndex,
305    ) -> TreeNodeIndex {
306        debug_assert!(leaf_index_1 != leaf_index_2);
307        debug_assert!(leaf_index_1.u32() < self.leaf_count());
308        debug_assert!(leaf_index_2.u32() < self.leaf_count());
309
310        // We want to return the position of the lowest common ancestor in the
311        // direct path of `leaf_index_1` (i.e. the sender_leaf_index).
312        let subtree_root_node_index = lowest_common_ancestor(leaf_index_1, leaf_index_2);
313
314        // Figure out which child is the relevant copath node.
315        if leaf_index_2 < leaf_index_1 {
316            left(subtree_root_node_index)
317        } else {
318            right(subtree_root_node_index)
319        }
320    }

366    /// Returns a reference to the leaf node at index `leaf_index`.
367    pub(crate) fn leaf(&self, leaf_index: LeafNodeIndex) -> &L {
368        // Check if it's in the diff.
369        if let Some(node) = self.leaf_diff.get(&leaf_index) {
370            node
371        // If it's not in the diff, it could be that it's outside of the
372        // diff, in which case we want to return a blank here, because the
373        // diff might have been trimmed in the mean time.
374        } else if leaf_index.u32() >= self.leaf_count() {
375            &self.default_leaf
376            // If it isn't in the diff, it must be in the tree.
377        } else {
378            self.original_tree.leaf_by_index(leaf_index)
379        }
380    }
381
382    /// Returns a reference to the parent node at index `parent_index`.
383    pub(crate) fn parent(&self, parent_index: ParentNodeIndex) -> &P {
384        // Check if it's in the diff.
385        if let Some(node) = self.parent_diff.get(&parent_index) {
386            return node;
387        }
388        // If it isn't in the diff, it must be in the tree.
389        self.original_tree.parent_by_index(parent_index)
390    }
391
392    /// Returns a mutable reference to the parent node in the diff at index
393    /// `parent_index`. If the diff doesn't have a node at that index, it clones
394    /// the node to the diff and returns a mutable reference to that node.
395    pub(crate) fn parent_mut(&mut self, parent_index: ParentNodeIndex) -> &mut P {
396        debug_assert!(parent_index.u32() < self.parent_count());
397        // We then check if the node is already in the diff. (Not using `if let
398        // ...` here, because the borrow checker doesn't like that).
399        if self.parent_diff.contains_key(&parent_index) {
400            return self
401                .parent_diff
402                .get_mut(&parent_index)
403                // We just checked that this index exists, so this must be Some.
404                .unwrap_or(&mut self.default_parent);
405            // If not, we take a copy from the original tree and put it in the
406            // diff before returning a mutable reference to it.
407        }
408        let tree_node = self.original_tree.parent_by_index(parent_index);
409        self.replace_parent(parent_index, tree_node.clone());
410        self.parent_diff
411            .get_mut(&parent_index)
412            // We just inserted this into the diff, so this should be Some.
413            .unwrap_or(&mut self.default_parent)
414    }

416    // Index checking

418    #[cfg(test)]
419    pub(crate) fn deref_vec(
420        &self,
421        parent_index_vec: Vec<ParentNodeIndex>,
422    ) -> Result<Vec<&P>, ABinaryTreeDiffError> {
423        let mut parent_vec = Vec::new();
424        for parent_index in parent_index_vec {
425            let node = self.parent(parent_index);
426            parent_vec.push(node);
427        }
428        Ok(parent_vec)
429    }

432/// Binary Tree Diff error
433#[derive(Error, Debug, PartialEq, Clone)]
434pub(crate) enum ABinaryTreeDiffError {
435    /// See [`LibraryError`] for more details.
436    #[error(transparent)]
437    LibraryError(#[from] LibraryError),
438    /// Maximum tree size reached.
439    #[error("Maximum tree size reached.")]
440    TreeTooLarge,
441    /// Minimum tree size reached.
442    #[error("Minimum tree size reached.")]
443    TreeTooSmall,
444    /// See [`ABinaryTreeError`] for more details.
445    #[error(transparent)]
446    ABinaryTreeError(#[from] ABinaryTreeError),
447}
