3//! # About
4//!
5//! This module contains an implementation of a binary tree based on an array
6//! representation. The main [`ABinaryTree`] struct is generally immutable, but
7//! allows the creation of an [`AbDiff`] struct, where changes can be made before

49/// A representation of a full, left-balanced binary tree that uses a simple
50/// vector to store nodes. Each tree has to consist of at least one node.
51pub(crate) struct ABinaryTree<L: Clone + Debug + Default, P: Clone + Debug + Default> {
52    leaf_nodes: Vec<L>,
53    parent_nodes: Vec<P>,
54    default_leaf: L,
55    default_parent: P,
56}
57
58impl<L: Clone + Debug + Default, P: Clone + Debug + Default> ABinaryTree<L, P> {
59    /// Create a tree from the given vector of nodes. The vector of nodes can't
60    /// be empty and has to yield a full, left-balanced binary tree. The nodes
61    /// in the tree are ordered in the array-representation. This function
62    /// throws a [`ABinaryTreeError::InvalidNumberOfNodes`] error if the number
63    /// of nodes does not allow the creation of a full, left-balanced binary
64    /// tree and an [`ABinaryTreeError::OutOfRange`] error if the number of
65    /// given nodes exceeds the range of [`TreeNodeIndex`].
66    pub(crate) fn new(nodes: Vec<TreeNode<L, P>>) -> Result<Self, ABinaryTreeError> {
67        // No more than 2^30 nodes
68        if nodes.len() > MAX_TREE_SIZE as usize {
69            return Err(ABinaryTreeError::OutOfRange);
70        }
71        if nodes.len() % 2 != 1 {
72            return Err(ABinaryTreeError::InvalidNumberOfNodes);
73        }
74        let mut leaf_nodes = Vec::new();
75        let mut parent_nodes = Vec::new();
76
77        // Split the nodes intow two vectors, one for the leaf nodes and one for
78        // the parent nodes.
79        for (i, node) in nodes.into_iter().enumerate() {
80            match node {
81                TreeNode::Leaf(l) => {
82                    if i % 2 == 0 {
83                        leaf_nodes.push(*l)
84                    } else {
85                        return Err(ABinaryTreeError::WrongNodeType);
86                    }
87                }
88                TreeNode::Parent(p) => {
89                    if i % 2 == 1 {
90                        parent_nodes.push(*p)
91                    } else {
92                        return Err(ABinaryTreeError::WrongNodeType);
93                    }
94                }
95            }
96        }
97
98        Ok(ABinaryTree {
99            leaf_nodes,
100            parent_nodes,
101            default_leaf: L::default(),
102            default_parent: P::default(),
103        })
104    }
105
106    /// Obtain a reference to the data contained in the leaf node at index
107    /// `leaf_index`, where the indexing corresponds to the array representation
108    /// of the underlying binary tree. Returns the default value if the node
109    /// cannot be found.
110    pub(in crate::binary_tree) fn leaf_by_index(&self, leaf_index: LeafNodeIndex) -> &L {
111        self.leaf_nodes
112            .get(leaf_index.usize())
113            .unwrap_or(&self.default_leaf)
114    }
115
116    /// Obtain a reference to the data contained in the parent node at index
117    /// `parent_index`, where the indexing corresponds to the array
118    /// representation of the underlying binary tree. Returns the default value
119    /// if the node cannot be found.
120    pub(crate) fn parent_by_index(&self, parent_index: ParentNodeIndex) -> &P {
121        self.parent_nodes
122            .get(parent_index.usize())
123            .unwrap_or(&self.default_parent)
124    }
125
126    /// Return the number of nodes in the tree.
127    pub(crate) fn tree_size(&self) -> TreeSize {
128        // We can cast the size to a u32, because the maximum size of a
129        // tree is 2^30.
130        TreeSize::new((self.leaf_nodes.len() + self.parent_nodes.len()) as u32)
131    }
132
133    /// Return the number of leaf nodes in the tree.
134    pub(crate) fn leaf_count(&self) -> u32 {
135        // This works, because the tree always has at least one leaf.
136        self.leaf_nodes.len() as u32
137    }
138
139    /// Return the number of parent nodes in the tree.
140    pub(crate) fn parent_count(&self) -> u32 {
141        // This works, because the tree always has at least one leaf.
142        self.parent_nodes.len() as u32
143    }
144
145    /// Returns an iterator over a tuple of the leaf index and a reference to a
146    /// leaf, sorted according to their position in the tree from left to right.
147    pub(crate) fn leaves(&self) -> impl Iterator<Item = (LeafNodeIndex, &L)> {
148        self.leaf_nodes
149            .iter()
150            .enumerate()
151            .map(|(index, leave)| (LeafNodeIndex::new(index as u32), leave))
152    }
153
154    /// Returns an iterator over a tuple of the parent index and a reference to
155    /// a parent, sorted according to their position in the tree from left to
156    /// right.
157    pub(crate) fn parents(&self) -> impl Iterator<Item = (ParentNodeIndex, &P)> {
158        self.parent_nodes
159            .iter()
160            .enumerate()
161            .map(|(index, leave)| (ParentNodeIndex::new(index as u32), leave))
162    }
163
164    /// Creates and returns an empty [`AbDiff`].
165    pub(crate) fn empty_diff(&self) -> AbDiff<'_, L, P> {
166        self.into()
167    }
168
169    /// Merges the changes applied to the [`StagedAbDiff`] into the tree.
170    /// Depending on the changes made to the diff, this can either increase or
171    /// decrease the size of the tree, although not beyond the minimum size of
172    /// leaf or the maximum size of `u32::MAX`.
173    pub(crate) fn merge_diff(&mut self, diff: StagedAbDiff<L, P>) {
174        let tree_size = diff.tree_size();
175
176        let (leaf_diff, parent_diff) = diff.into_diffs();
177
178        // Resize the tree to the new size.
179        self.leaf_nodes
180            .resize_with(tree_size.leaf_count() as usize, Default::default);
181        self.parent_nodes
182            .resize_with(tree_size.parent_count() as usize, Default::default);
183
184        // Merge leaves
185        // Iterate over the BTreeMap in order of indices.
186        for (leaf_index, diff_leaf) in leaf_diff.into_iter() {
187            // Assert that the node index is within the range of the tree.
188            debug_assert!(leaf_index.u32() < self.leaf_count());
189
190            match self.leaf_nodes.get_mut(leaf_index.usize()) {
191                Some(n) => *n = diff_leaf,
192                None => {
193                    // Panic in debug mode
194                    debug_assert!(false);
195                }
196            }
197        }
198
199        // Merge parents
200        // Iterate over the BTreeMap in order of indices.
201        for (parent_index, diff_parent) in parent_diff.into_iter() {
202            // Assert that the node index is within the range of the tree.
203            debug_assert!(parent_index.u32() < self.parent_count());
204
205            match self.parent_nodes.get_mut(parent_index.usize()) {
206                Some(n) => *n = diff_parent,
207                None => {
208                    // Panic in debug mode
209                    debug_assert!(false);
210                }
211            }
212        }
213    }
214
215    /// Return a reference to the leaf at the given `LeafNodeIndex`, or the default
216    /// value if the leaf is not found.
217    pub(crate) fn leaf(&self, leaf_index: LeafNodeIndex) -> &L {
218        self.leaf_nodes
219            .get(leaf_index.usize())
220            .unwrap_or(&self.default_leaf)
221    }
222
223    /// Returns a vector of [`ParentNodeIndex`]es, where the first reference is to
224    /// the root of the shared subtree of the two given leaf indices followed by
225    /// references to the nodes in the direct path of said subtree root.
226    pub(crate) fn subtree_path(
227        &self,
228        leaf_index_1: LeafNodeIndex,
229        leaf_index_2: LeafNodeIndex,
230    ) -> Vec<ParentNodeIndex> {
231        common_direct_path(leaf_index_1, leaf_index_2, self.tree_size())
232    }
233
234    pub(crate) fn parent(&self, parent_index: ParentNodeIndex) -> &P {
235        self.parent_nodes
236            .get(parent_index.usize())
237            .unwrap_or(&self.default_parent)
238    }
239}

241/// Binary Tree error
242#[derive(Error, Debug, PartialEq, Clone)]
243pub(crate) enum ABinaryTreeError {
244    /// Adding nodes exceeds the maximum possible size of the tree.
245    #[error("Adding nodes exceeds the maximum possible size of the tree.")]
246    OutOfRange,
247    /// Not enough nodes to remove.
248    #[error("Not enough nodes to remove.")]
249    InvalidNumberOfNodes,
250    /// Wrong node type.
251    #[error("Wrong node type.")]
252    WrongNodeType,
253}
