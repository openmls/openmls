47/// Compare two byte slices in a way that's hopefully not optimised out by the
48/// compiler.
49#[inline(never)]
50fn equal_ct(a: &[u8], b: &[u8]) -> bool {
51    let mut diff = 0u8;
52    for (l, r) in a.iter().zip(b.iter()) {
53        diff |= l ^ r;
54    }
55    diff == 0
