13/// Re-use guard size.
14pub(crate) const REUSE_GUARD_BYTES: usize = 4;
15
16#[derive(Debug, Clone, Copy, TlsSerialize, TlsDeserialize, TlsDeserializeBytes, TlsSize)]
17#[cfg_attr(test, derive(PartialEq, Eq))]
18pub struct ReuseGuard {
19    pub(in crate::ciphersuite) value: [u8; REUSE_GUARD_BYTES],
20}
21
22impl ReuseGuard {
23    /// Returns the raw 4 bytes of the guard.
24    #[cfg(feature = "virtual-clients-draft")]
25    pub(crate) fn bytes(&self) -> [u8; REUSE_GUARD_BYTES] {
26        self.value
27    }
28
29    /// Samples a fresh reuse guard uniformly at random.
30    pub(crate) fn try_from_random(rng: &impl OpenMlsRand) -> Result<Self, CryptoError> {
31        Ok(Self {
32            value: rng
33                .random_array()
34                .map_err(|_| CryptoError::InsufficientRandomness)?,
35        })
36    }
37
38    /// Build a `reuse_guard` for a virtual-clients sender as specified by
39    /// the mls-virtual-clients draft (Reuse Guard section). Picks a
40    /// random `x` uniformly from the largest `N_e`-divisible prefix of
41    /// `[0, 2^32)` with `x mod N_e == leaf_index_e`, then encrypts `x`
42    /// under FF1-AES128 with `prp_key = ExpandWithLabel(reuse_guard_secret,
43    /// "reuse guard", key_schedule_nonce, 16)`.
44    ///
45    /// `emulation_ciphersuite` is the emulation group's ciphersuite.
46    /// `ratchet_nonce` is the pre-XOR per-message nonce produced by the
47    /// higher-level group's key schedule.
48    #[cfg(feature = "virtual-clients-draft")]
49    pub(crate) fn for_emulator_sender(
50        crypto: &impl OpenMlsCrypto,
51        rand: &impl OpenMlsRand,
52        reuse_guard_secret: &ReuseGuardSecret,
53        emulation_ciphersuite: openmls_traits::types::Ciphersuite,
54        ratchet_nonce: &AeadNonce,
55        emulation_leaf_index: LeafNodeIndex,
56        emulation_group_size: TreeSize,
57    ) -> Result<Self, ReuseGuardDerivationError> {
58        let n_e = u64::from(emulation_group_size.leaf_count());
59        let leaf_index_e = u64::from(emulation_leaf_index.u32());
60        if n_e == 0 {
61            return Err(ReuseGuardDerivationError::Library(LibraryError::custom(
62                "emulation_group_size is zero",
63            )));
64        }
65        if leaf_index_e >= n_e {
66            return Err(ReuseGuardDerivationError::Library(LibraryError::custom(
67                "emulation_leaf_index is out of range for the emulation group",
68            )));
69        }
70
71        // Largest multiple of N_e that fits in 2^32. Sampling uniformly
72        // from [0, full) and then computing `(r / n_e) * n_e +
73        // leaf_index_e` keeps the distribution uniform over the support
74        // {leaf_index_e, leaf_index_e + n_e, ..., full - n_e + leaf_index_e}.
75        let full: u64 = (1u64 << 32) / n_e * n_e;
76        debug_assert!(full > 0);
77        let r = loop {
78            let candidate = u32::from_be_bytes(
79                rand.random_array::<4>()
80                    .map_err(|_| LibraryError::custom("rand failure"))?,
81            );
82            let candidate_u64 = u64::from(candidate);
83            if candidate_u64 < full {
84                break candidate_u64;
85            }
86        };
87        let x_u64 = r / n_e * n_e + leaf_index_e;
88        debug_assert!(x_u64 < (1u64 << 32));
89        debug_assert_eq!(x_u64 % n_e, leaf_index_e);
90        let x = x_u64 as u32;
91
92        let prp_key = reuse_guard_secret.derive_prp_key(
93            crypto,
94            emulation_ciphersuite,
95            ratchet_nonce.raw_bytes(),
96        )?;
97        let permuted = crypto.ff1_aes128_encrypt(&prp_key, x).map_err(|e| {
98            log::error!("vc: FF1 encryption of reuse_guard failed: {e:?}");
99            ReuseGuardDerivationError::Library(LibraryError::custom("FF1 encrypt failed"))
100        })?;
101        Ok(Self {
102            value: permuted.to_be_bytes(),
103        })
104    }
105}

107/// Errors returned by [`ReuseGuard::for_emulator_sender`].
108#[cfg(feature = "virtual-clients-draft")]
109#[derive(thiserror::Error, Debug, PartialEq, Clone)]
110pub(crate) enum ReuseGuardDerivationError {
111    #[error(transparent)]
112    VirtualClients(#[from] VirtualClientsError),
113    #[error(transparent)]
114    Library(#[from] LibraryError),
115}

118mod vc_tests {
119    use super::*;
120    use crate::ciphersuite::Secret;
121    use openmls_rust_crypto::OpenMlsRustCrypto;
122    use openmls_traits::types::Ciphersuite;
123    use openmls_traits::OpenMlsProvider as _;
124
125    fn provider() -> OpenMlsRustCrypto {
126        OpenMlsRustCrypto::default()
127    }
128
129    fn dummy_reuse_guard_secret(
130        _p: &OpenMlsRustCrypto,
131        ciphersuite: Ciphersuite,
132    ) -> ReuseGuardSecret {
133        let bytes = vec![0xa5u8; ciphersuite.hash_length()];
134        ReuseGuardSecret::from_secret_for_tests(Secret::from_slice(&bytes))
135    }
136
137    /// `leaf_index_e >= N_e` is a malformed `EmulationEpochState`. The
138    /// sender path catches it before drawing any randomness or running
139    /// FF1.
140    #[test]
141    fn for_emulator_sender_rejects_leaf_index_out_of_range() {
142        let p = provider();
143        let ciphersuite = Ciphersuite::MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519;
144        let secret = dummy_reuse_guard_secret(&p, ciphersuite);
145        let nonce = crate::ciphersuite::aead::AeadNonce::random(p.rand());
146        let n_e = TreeSize::from_leaf_count(1);
147        debug_assert_eq!(n_e.leaf_count(), 2);
148        let err = ReuseGuard::for_emulator_sender(
149            p.crypto(),
150            p.rand(),
151            &secret,
152            ciphersuite,
153            &nonce,
154            LeafNodeIndex::new(5),
155            n_e,
156        )
157        .expect_err("malformed leaf_index must not produce a guard");
158        assert!(
159            matches!(err, ReuseGuardDerivationError::Library(_)),
160            "expected LibraryError, got {err:?}"
161        );
162    }
