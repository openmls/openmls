7/// A struct to contain secrets. This is to provide better visibility into where
8/// and how secrets are used and to avoid passing secrets in their raw
9/// representation.
10///
11/// Note: This has a hand-written `Debug` implementation.
12///       Please update as well when changing this struct.
13#[derive(Clone, Serialize, Deserialize, Eq)]
14pub(crate) struct Secret {
15    pub(in crate::ciphersuite) value: SecretVLBytes,
16}
17
18impl Debug for Secret {
19    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
20        let mut ds = f.debug_struct("Secret");
21
22        #[cfg(feature = "crypto-debug")]
23        return ds.field("value", &self.value).finish();
24        #[cfg(not(feature = "crypto-debug"))]
25        ds.field("value", &"***").finish()
26    }
27}

37impl PartialEq for Secret {
38    // Constant time comparison.
39    fn eq(&self, other: &Secret) -> bool {
40        // These values can be considered public and checked before the actual
41        // comparison.
42        if self.value.as_slice().len() != other.value.as_slice().len() {
43            log::error!("Incompatible secrets");
44            log::trace!("  {}", self.value.as_slice().len());
45            log::trace!("  {}", other.value.as_slice().len());
46            return false;
47        }
48        equal_ct(self.value.as_slice(), other.value.as_slice())
49    }
50}

52impl Secret {
53    /// Randomly sample a fresh `Secret`.
54    /// This default random initialiser uses the default Secret length of `hash_length`.
55    /// The function can return a [`CryptoError`] if there is insufficient randomness.
56    pub(crate) fn random(
57        ciphersuite: Ciphersuite,
58        rand: &impl OpenMlsRand,
59    ) -> Result<Self, CryptoError> {
60        Ok(Secret {
61            value: rand
62                .random_vec(ciphersuite.hash_length())
63                .map_err(|_| CryptoError::InsufficientRandomness)?
64                .into(),
65        })
66    }
67
68    /// Create an all zero secret.
69    pub(crate) fn zero(ciphersuite: Ciphersuite) -> Self {
70        Self {
71            value: vec![0u8; ciphersuite.hash_length()].into(),
72        }
73    }
74
75    /// Create a new secret from a byte vector.
76    pub(crate) fn from_slice(bytes: &[u8]) -> Self {
77        Secret {
78            value: bytes.into(),
79        }
80    }

82    /// HKDF extract where `self` is `salt`.
83    pub(crate) fn hkdf_extract<'a>(
84        &self,
85        crypto: &impl OpenMlsCrypto,
86        ciphersuite: Ciphersuite,
87        ikm_option: impl Into<Option<&'a Secret>>,
88    ) -> Result<Self, CryptoError> {
89        log::trace!("HKDF extract with");
90        log_crypto!(trace, "  salt: {:x?}", self.value);
91        let zero_secret = Self::zero(ciphersuite);
92        let ikm = ikm_option.into().unwrap_or(&zero_secret);
93        log_crypto!(trace, "  ikm:  {:x?}", ikm.value);
94
95        Ok(Self {
96            value: crypto.hkdf_extract(
97                ciphersuite.hash_algorithm(),
98                self.value.as_slice(),
99                ikm.value.as_slice(),
100            )?,
101        })
102    }
103
104    /// HKDF expand where `self` is `prk`.
105    pub(crate) fn hkdf_expand(
106        &self,
107        crypto: &impl OpenMlsCrypto,
108        ciphersuite: Ciphersuite,
109        info: &[u8],
110        okm_len: usize,
111    ) -> Result<Self, CryptoError> {
112        let key = crypto
113            .hkdf_expand(
114                ciphersuite.hash_algorithm(),
115                self.value.as_slice(),
116                info,
117                okm_len,
118            )
119            .map_err(|_| CryptoError::CryptoLibraryError)?;
120        if key.as_slice().is_empty() {
121            return Err(CryptoError::InvalidLength);
122        }
123        Ok(Self { value: key })
124    }

126    /// Expand a `Secret` to a new `Secret` of length `length` including a
127    /// `label` and a `context`.
128    pub(crate) fn kdf_expand_label(
129        &self,
130        crypto: &impl OpenMlsCrypto,
131        ciphersuite: Ciphersuite,
132        label: &str,
133        context: &[u8],
134        length: usize,
135    ) -> Result<Secret, CryptoError> {
136        let full_label = format!("MLS 1.0 {}", label);
137        log::trace!(
138            "KDF expand with label \"{}\" and {:?} with context {:x?}",
139            &full_label,
140            ciphersuite,
141            context
142        );
143        let info = KdfLabel::serialized_label(context, full_label, length)?;
144        log::trace!("  serialized info: {:x?}", info);
145        log_crypto!(trace, "  secret: {:x?}", self.value);
146        self.hkdf_expand(crypto, ciphersuite, &info, length)
147    }
148
149    /// Derive a new `Secret` from the this one by expanding it with the given
150    /// `label` and an empty `context`.
151    pub(crate) fn derive_secret(
152        &self,
153        crypto: &impl OpenMlsCrypto,
154        ciphersuite: Ciphersuite,
155        label: &str,
156    ) -> Result<Secret, CryptoError> {
157        log_crypto!(
158            trace,
159            "derive secret from {:x?} with label {}",
160            self.value,
161            label
162        );
163        self.kdf_expand_label(crypto, ciphersuite, label, &[], ciphersuite.hash_length())
164    }
165
166    /// Returns the inner bytes of a secret
167    pub(crate) fn as_slice(&self) -> &[u8] {
168        self.value.as_slice()
169    }
170}
