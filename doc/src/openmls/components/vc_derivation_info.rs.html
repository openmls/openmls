69#[derive(Error, Debug, PartialEq, Clone)]
70pub enum VirtualClientsError {
71    /// The derivation-info bytes failed to deserialize.
72    #[error("Failed to deserialize derivation info.")]
73    DerivationInfoMalformed,
74    /// AEAD decryption of the encrypted derivation info failed (wrong key,
75    /// tampered ciphertext, or mismatched AAD).
76    #[error("Failed to decrypt derivation info.")]
77    DerivationInfoDecryptionFailed,
78    /// No virtual-clients operation secret tree was registered for this
79    /// epoch.
80    #[error("No virtual-clients operation secret tree for this epoch.")]

83    /// epoch, or it has been deleted.

86    /// Loading or storing virtual-clients state via the storage provider
87    /// failed.
88    #[error("Virtual-clients storage error")]
89    StorageError,
90    /// The leaf encryption key in the path does not match the key derived
91    /// from the path secret.
92    #[error("Leaf encryption key from path does not match the derived key.")]
93    EncryptionKeyMismatch,
94    /// A cryptographic operation failed during virtual-clients processing.
95    #[error("Cryptographic operation failed.")]
96    CryptoError(#[from] CryptoError),
97    /// Hash function produced output of unexpected length.
98    #[error(
99        "Hash function produced output of length {actual_length}, expected {expected_length}."
100    )]
101    HashOutputLengthMismatch {
102        /// The number of bytes in the hash output.

104        /// The required number of bytes in the hash output.
105        expected_length: usize,
106    },
107    /// TLS encoding/decoding of a virtual-clients structure failed. Covers
108    /// both serialization on the sender side and deserialization of the
109    /// decrypted `DerivationInfoTbe` on the receiver side.

112    /// The leaf carrying (or about to carry) a VC derivation-info entry
113    /// does not declare `AppDataDictionary` in its capabilities.
114    #[error("Leaf does not declare AppDataDictionary support in its capabilities.")]
115    AppDataDictionaryNotSupported,
116    /// The leaf's `AppDataDictionary` extension is missing the
117    /// `AppComponents` entry, or that entry does not list
118    /// [`VC_COMPONENT_ID`].
119    #[error("Leaf's AppComponents entry does not list the virtual-clients component id.")]
120    VcComponentNotListed,
121    /// The requested leaf index lies outside the operation secret tree.
122    #[error("Leaf index is outside the operation secret tree.")]
123    IndexOutOfBounds,
124    /// The operation secret for the requested generation was already derived
125    /// and deleted for forward secrecy.
126    #[error("The operation secret for this generation was already consumed.")]
127    OperationGenerationConsumed,
128    /// The requested operation generation lies too far beyond the current

131    #[error("The requested operation generation is too far beyond the ratchet head.")]
132    OperationGenerationTooDistant,
133    /// An operation ratchet has reached the maximum generation.

135    OperationRatchetTooLong,
136    /// An unrecoverable error has occurred due to a bug in the
137    /// implementation.
138    #[error("An unrecoverable error has occurred due to a bug in the implementation.")]
139    LibraryError,
140    /// The `KeyPackageUpload` lists the same `key_package_index` more than
141    /// once. Each batch index must appear at most once.
142    #[error("KeyPackageUpload contains a duplicate key_package_index: {0}.")]
143    DuplicateKeyPackageIndex(u32),

145    /// Each KeyPackage reference must appear at most once.

397/// The virtual-clients derivation info carried in the leaf node's
398/// `app_data_dictionary` extension under [`VC_COMPONENT_ID`]
399/// (mls-virtual-clients draft):
400///
401/// ```text
402/// struct {
403///   opaque epoch_id<V>;
404///   opaque ciphertext<V>;
405/// } DerivationInfo
406/// ```
407///
408/// `ciphertext` is the AEAD-wrapped [`DerivationInfoTbe`], encrypted in the
409/// emulation group's ciphersuite with key and nonce derived from the
410/// per-epoch [`EpochEncryptionKey`] and the carrying leaf's serialized
411/// `encryption_key`, with `epoch_id` as AAD.
412#[derive(Debug, TlsSize, TlsSerialize, TlsDeserializeBytes)]
413pub(crate) struct DerivationInfo {
414    epoch_id: EpochId,
415    ciphertext: VLBytes,
416}
417
418impl DerivationInfo {
419    /// Encrypt `tbe` under the per-epoch AEAD key, binding it to the leaf
420    /// that carries the resulting derivation info via the leaf's serialized
421    /// `encryption_key` (the key/nonce derivation context) and to
422    /// `epoch_id` (the AAD).
423    pub(crate) fn encrypt(
424        crypto: &impl OpenMlsCrypto,
425        ciphersuite: Ciphersuite,
426        key: &EpochEncryptionKey,
427        epoch_id: EpochId,
428        leaf_encryption_key: &[u8],
429        tbe: &DerivationInfoTbe,
430    ) -> Result<Self, VirtualClientsError> {
431        let (aead_key, aead_nonce) =
432            key.derive_key_nonce(crypto, ciphersuite, leaf_encryption_key)?;
433        let payload = tbe.tls_serialize_detached()?;
434        let ciphertext = crypto.aead_encrypt(
435            ciphersuite.aead_algorithm(),
436            aead_key.as_slice(),
437            payload.as_slice(),
438            aead_nonce.as_slice(),
439            epoch_id.0.as_slice(),
440        )?;
441        Ok(Self {
442            epoch_id,
443            ciphertext: ciphertext.into(),
444        })
445    }
446
447    pub(crate) fn epoch_id(&self) -> &EpochId {
448        &self.epoch_id
449    }
450
451    /// Decrypt the wrapped [`DerivationInfoTbe`]. `leaf_encryption_key` is
452    /// the serialized `encryption_key` of the leaf node that carries this
453    /// derivation info.
454    pub(crate) fn decrypt(
455        &self,
456        crypto: &impl OpenMlsCrypto,
457        ciphersuite: Ciphersuite,
458        key: &EpochEncryptionKey,
459        leaf_encryption_key: &[u8],
460        operation_type: VirtualClientOperationType,
461    ) -> Result<DerivationInfoTbe, VirtualClientsError> {
462        let (aead_key, aead_nonce) =
463            key.derive_key_nonce(crypto, ciphersuite, leaf_encryption_key)?;
464        let plaintext = crypto
465            .aead_decrypt(
466                ciphersuite.aead_algorithm(),
467                aead_key.as_slice(),
468                self.ciphertext.as_slice(),
469                aead_nonce.as_slice(),
470                self.epoch_id.0.as_slice(),
471            )
472            .map_err(|e| {
473                log::error!("vc: aead decrypt derivation info failed: {e:?}");
474                VirtualClientsError::DerivationInfoDecryptionFailed
475            })?;
476        DerivationInfoTbe::deserialize_for_operation(&plaintext, operation_type)
477    }
478}

544    pub leaf_index: LeafNodeIndex,

546    pub generation: u32,
547    /// One entry per uploaded KeyPackage.
548    pub key_package_info: Vec<KeyPackageInfo>,

551/// Per-`KeyPackageRef` material a sibling retains when it processes a
552/// [`KeyPackageUpload`]. It captures what the Welcome path needs to later
553/// rederive the KeyPackage's init and leaf-encryption keys without touching

555/// epoch, leaf index, generation, and batch index used to validate the leaf
556/// found in the ratchet tree.
557///

559/// independent of the operation tree's bounded out-of-order tolerance: a batch

563#[derive(Debug, Serialize, Deserialize)]
564pub struct RetainedKeyPackageMaterial {

566    pub epoch_id: EpochId,
567    /// Uploader's emulation-group leaf index, identifying the operation
568    /// ratchet the batch generation was allocated from.
569    pub leaf_index: LeafNodeIndex,
570    /// Operation-ratchet generation consumed for the whole batch.
571    pub generation: u32,
572    /// Position of this KeyPackage within the batch.
573    pub key_package_index: u32,
574    /// Per-KeyPackage seed secret from which the init and leaf-encryption keys
575    /// are derived at Welcome time.
576    pub key_package_seed_secret: KeyPackageSeedSecret,
577}

582/// `key_package_index` repeats, and
583/// [`VirtualClientsError::DuplicateKeyPackageRef`] if any `key_package_ref`
584/// repeats. A duplicate index would map two KeyPackages onto the same
585/// per-index seed, and a duplicate reference would have the second upload
586/// entry overwrite the first's retained material, so both are rejected before
587/// any state is loaded or any operation generation is consumed.
588fn validate_key_package_infos(infos: &[KeyPackageInfo]) -> Result<(), VirtualClientsError> {
589    let mut seen_indices = std::collections::BTreeSet::new();
590    let mut seen_refs = std::collections::BTreeSet::new();
591    for info in infos {
592        if !seen_indices.insert(info.key_package_index) {
593            return Err(VirtualClientsError::DuplicateKeyPackageIndex(
594                info.key_package_index,
595            ));
596        }
597        if !seen_refs.insert(&info.key_package_ref) {
598            return Err(VirtualClientsError::DuplicateKeyPackageRef);
599        }
600    }
601    Ok(())
602}

639/// Process a [`KeyPackageUpload`] received from a sibling virtual client.
640///
641/// Derives the batch's single `key_package` operation secret once from the
642/// uploader's leaf ratchet at `(epoch_id, leaf_index, generation)`, then
643/// stores the advanced operation tree and one [`RetainedKeyPackageMaterial`]
644/// per [`KeyPackageInfo`] (keyed by the info's [`KeyPackageRef`]) in a single
645/// atomic batch write.
646///
647/// The operation secret and the per-index seeds are derived under the
648/// emulation ciphersuite (the operation tree's ciphersuite). The init and
649/// leaf-encryption keys are later derived from each seed under the KeyPackage's
650/// own ciphersuite at Welcome time. The operation secret is dropped once all
651/// seeds are derived. The batch generation is consumed in the tree exactly
652/// once.
653pub fn process_vc_key_package_upload<Provider: OpenMlsProvider>(
654    provider: &Provider,
655    upload: &KeyPackageUpload,
656) -> Result<(), VirtualClientsError> {
657    use crate::components::vc_operation_tree::OperationSecretTree;
658    use openmls_traits::storage::StorageProvider as _;
659
660    validate_key_package_infos(&upload.key_package_info)?;
661
662    let storage = provider.storage();
663    let crypto = provider.crypto();
664
665    let state: EmulationEpochState = storage
666        .vc_emulation_epoch_state(&upload.epoch_id)
667        .map_err(|e| {
668            log::error!("vc: load emulation epoch state in process upload failed: {e:?}");
669            VirtualClientsError::StorageError
670        })?
671        .ok_or(VirtualClientsError::MissingEmulationEpochState)?;
672    let mut operation_tree: OperationSecretTree = storage
673        .vc_operation_tree(&upload.epoch_id)
674        .map_err(|e| {
675            log::error!("vc: load operation tree in process upload failed: {e:?}");
676            VirtualClientsError::StorageError
677        })?
678        .ok_or(VirtualClientsError::MissingOperationTree)?;
679    let emulation_ciphersuite = state.emulation_ciphersuite;
680
681    // The KeyPackage operation context is empty, matching `build_vc_batch`.
682    let operation_secret = operation_tree.derive_operation_secret(
683        crypto,
684        emulation_ciphersuite,
685        &upload.epoch_id,
686        upload.leaf_index,
687        VirtualClientOperationType::KeyPackage,
688        upload.generation,
689        b"",
690    )?;
691
692    let mut materials = Vec::with_capacity(upload.key_package_info.len());
693    for info in &upload.key_package_info {
694        let key_package_seed_secret = operation_secret.derive_key_package_seed_secret(
695            crypto,
696            emulation_ciphersuite,
697            info.key_package_index,
698        )?;
699        let material = RetainedKeyPackageMaterial {
700            epoch_id: upload.epoch_id.clone(),
701            leaf_index: upload.leaf_index,
702            generation: upload.generation,
703            key_package_index: info.key_package_index,
704            key_package_seed_secret,
705        };
706        materials.push((info.key_package_ref.clone(), material));
707    }
708
709    storage
710        .write_retained_key_package_material_batch(&upload.epoch_id, &operation_tree, &materials)
711        .map_err(|e| {
712            log::error!("vc: persist batch key package material in process upload failed: {e:?}");
713            VirtualClientsError::StorageError
714        })?;
715    Ok(())
716}

749/// Per-higher-level-group record of which emulation-group epoch produced the
750/// virtual-client LeafNode that was active at each recent epoch of that
751/// group.
752///
753/// Reuse guards must be resolved with the emulation epoch that was bound at
754/// the higher-level epoch a message was sent in, not the latest one: a
755/// delayed PrivateMessage from a past higher-level epoch has to be
756/// deprotected with the state that was active then. Entries are written at
757/// commit merge and retained for as many past epochs as the group's message
758/// secrets store keeps, since a binding is only useful while the matching
759/// message secrets still exist.
760#[derive(Debug, Default, Clone, PartialEq, Serialize, Deserialize)]
761pub struct VcEmulationBindings {
762    // In order of insertion, oldest at the front.
763    bindings: std::collections::VecDeque<(crate::group::GroupEpoch, EpochId)>,
764}
765
766impl VcEmulationBindings {
767    /// Look up the emulation epoch bound at the given higher-level epoch.
768    pub fn get(&self, epoch: crate::group::GroupEpoch) -> Option<&EpochId> {
769        for (bound_epoch, epoch_id) in &self.bindings {
770            if *bound_epoch == epoch {
771                return Some(epoch_id);
772            }
773        }
774        None
775    }
776
777    /// Record `epoch_id` as the binding for `epoch`, keeping at most
778    /// `max_entries` entries by dropping the oldest ones.
779    pub(crate) fn insert(
780        &mut self,
781        epoch: crate::group::GroupEpoch,
782        epoch_id: EpochId,
783        max_entries: usize,
784    ) {
785        self.bindings
786            .retain(|(bound_epoch, _)| *bound_epoch != epoch);
787        self.bindings.push_back((epoch, epoch_id));
788        while self.bindings.len() > max_entries {
789            self.bindings.pop_front();
790        }
791    }
792}

1116/// What virtual-clients operation a per-operation secret is being derived
1117/// for (mls-virtual-clients draft `VirtualClientOperationType`). Mixed into
1118/// the `OperationContext` of every operation-secret derivation so that
1119/// secrets derived for different operations cannot collide even if the other
1120/// fields happen to match.
1121///
1122/// The operation type does not travel on the wire. Receivers infer it from
1123/// the carrying LeafNode's `leaf_node_source`: `key_package` maps to
1124/// [`KeyPackage`](Self::KeyPackage), `update` and `commit` map to
1125/// [`LeafNode`](Self::LeafNode).
1126///
1127/// Only `LeafNode` is wired into a sender path today (see `apply_vc_emulation`
1128/// in the commit builder). `KeyPackage` and `Application` are reserved
1129/// variants that a follow-up PR will emit, once the KeyPackage and
1130/// application-message operation paths exist.
1131#[derive(Debug, Clone, Copy, PartialEq, Eq, TlsSize, TlsSerialize, TlsDeserializeBytes)]
1132#[repr(u8)]
1133pub enum VirtualClientOperationType {
1134    /// Derivation of KeyPackage material for the virtual client.
1135    KeyPackage = 1,
1136    /// Derivation of LeafNode material for the virtual client (e.g. the
1137    /// leaf carried by a commit).
1138    LeafNode = 2,
1139    /// Derivation of application-message material for the virtual client.
1140    Application = 3,
1141}

1143/// AEAD plaintext attached to the leaf via the VC component
1144/// (mls-virtual-clients draft):
1145///
1146/// ```text
1147/// struct {
1148///   uint32 leaf_index;
1149///   uint32 generation;
1150///   select (LeafNode.leaf_node_source) {
1151///     case key_package:  uint32 key_package_index;
1152///     case update:
1153///     case commit:       struct{};
1154///   };
1155/// } DerivationInfoTBE
1156/// ```
1157///
1158/// `leaf_index` is the *emulation*-group leaf index of the sending virtual
1159/// client, *not* the leaf index in the group that carries this commit.
1160/// `generation` is the operation-ratchet generation the sender consumed for
1161/// this operation. `key_package_index`, present only for the `KeyPackage`
1162/// variant, is the KeyPackage's position within its `key_package` operation
1163/// batch.
1164#[derive(Debug, PartialEq, Eq)]
1165pub(crate) enum DerivationInfoTbe {
1166    /// Carried by `update` and `commit` leaves. No `key_package_index`.
1167    LeafNode {
1168        leaf_index: LeafNodeIndex,
1169        generation: u32,
1170    },
1171    /// Carried by `key_package` leaves. Adds the position within the batch.
1172    KeyPackage {
1173        leaf_index: LeafNodeIndex,
1174        generation: u32,
1175        key_package_index: u32,
1176    },
1177}
1178
1179impl DerivationInfoTbe {
1180    /// The emulation-group leaf index of the sending virtual client.
1181    pub(crate) fn leaf_index(&self) -> LeafNodeIndex {
1182        match self {
1183            Self::LeafNode { leaf_index, .. } | Self::KeyPackage { leaf_index, .. } => *leaf_index,
1184        }
1185    }
1186
1187    /// The operation-ratchet generation the sender consumed.
1188    pub(crate) fn generation(&self) -> u32 {
1189        match self {
1190            Self::LeafNode { generation, .. } | Self::KeyPackage { generation, .. } => *generation,
1191        }
1192    }
1193
1194    /// Serialize the variant's fields in order, with no variant tag, matching
1195    /// the `DerivationInfoTBE` select. The TLS derive macros cannot express a
1196    /// tagless select, so this codec is written by hand.
1197    fn tls_serialize_detached(&self) -> Result<Vec<u8>, tls_codec::Error> {
1198        match self {
1199            Self::LeafNode {
1200                leaf_index,
1201                generation,
1202            } => {
1203                let mut out = Vec::with_capacity(
1204                    leaf_index.tls_serialized_len() + generation.tls_serialized_len(),
1205                );
1206                leaf_index.tls_serialize(&mut out)?;
1207                generation.tls_serialize(&mut out)?;
1208                Ok(out)
1209            }
1210            Self::KeyPackage {
1211                leaf_index,
1212                generation,
1213                key_package_index,
1214            } => {
1215                let mut out = Vec::with_capacity(
1216                    leaf_index.tls_serialized_len()
1217                        + generation.tls_serialized_len()
1218                        + key_package_index.tls_serialized_len(),
1219                );
1220                leaf_index.tls_serialize(&mut out)?;
1221                generation.tls_serialize(&mut out)?;
1222                key_package_index.tls_serialize(&mut out)?;
1223                Ok(out)
1224            }
1225        }
1226    }
1227
1228    /// Deserialize the tagless select for the given operation type. The
1229    /// operation type stands in for the carrying leaf's `leaf_node_source`:
1230    /// [`KeyPackage`](VirtualClientOperationType::KeyPackage) parses the
1231    /// `KeyPackage` variant, [`LeafNode`](VirtualClientOperationType::LeafNode)
1232    /// the `LeafNode` variant. The plaintext must be consumed exactly.
1233    fn deserialize_for_operation(
1234        bytes: &[u8],
1235        operation_type: VirtualClientOperationType,
1236    ) -> Result<Self, VirtualClientsError> {
1237        let (leaf_index, rest) = LeafNodeIndex::tls_deserialize_bytes(bytes)?;
1238        let (generation, rest) = u32::tls_deserialize_bytes(rest)?;
1239        let tbe = match operation_type {
1240            VirtualClientOperationType::KeyPackage => {
1241                let (key_package_index, rest) = u32::tls_deserialize_bytes(rest)?;
1242                if !rest.is_empty() {
1243                    return Err(VirtualClientsError::DerivationInfoMalformed);
1244                }
1245                Self::KeyPackage {
1246                    leaf_index,
1247                    generation,
1248                    key_package_index,
1249                }
1250            }
1251            VirtualClientOperationType::LeafNode => {
1252                if !rest.is_empty() {
1253                    return Err(VirtualClientsError::DerivationInfoMalformed);
1254                }
1255                Self::LeafNode {
1256                    leaf_index,
1257                    generation,
1258                }
1259            }
1260            VirtualClientOperationType::Application => {
1261                return Err(VirtualClientsError::DerivationInfoMalformed);
1262            }
1263        };
1264        Ok(tbe)
1265    }
1266}

1268/// Verify that the effective leaf about to carry a VC derivation-info entry
1269/// declares `AppDataDictionary` and lists [`VC_COMPONENT_ID`] in its
1270/// `AppComponents` entry, and return the resolved `AppDataDictionary`.
1271///
1272/// `caller_capabilities` and `caller_extensions` are the leaf parameters the
1273/// caller supplied for this operation. `current_leaf` is the leaf being
1274/// replaced, or `None` when there is none (a fresh KeyPackage, or an external
1275/// commit). The caller's `AppDataDictionary` is merged over the current
1276/// leaf's, with the caller winning on duplicate component ids, so injecting
1277/// the VC derivation-info preserves the `AppComponents` entry across
1278/// operations.
1279pub(crate) fn resolve_vc_leaf_dictionary(
1280    caller_capabilities: Option<&crate::treesync::node::leaf_node::Capabilities>,
1281    caller_extensions: Option<
1282        &crate::extensions::Extensions<crate::treesync::node::leaf_node::LeafNode>,
1283    >,
1284    current_leaf: Option<&crate::treesync::node::leaf_node::LeafNode>,
1285) -> Result<crate::extensions::AppDataDictionary, VirtualClientsError> {
1286    use crate::{
1287        component::{ComponentId, ComponentType},
1288        extensions::ExtensionType,
1289    };
1290    use tls_codec::DeserializeBytes as _;
1291
1292    let supports_app_data_dictionary = match caller_capabilities {
1293        Some(c) => c.extensions().contains(&ExtensionType::AppDataDictionary),
1294        None => current_leaf
1295            .map(|leaf| {
1296                leaf.capabilities()
1297                    .extensions()
1298                    .contains(&ExtensionType::AppDataDictionary)
1299            })
1300            .unwrap_or(false),
1301    };
1302    if !supports_app_data_dictionary {
1303        return Err(VirtualClientsError::AppDataDictionaryNotSupported);
1304    }
1305
1306    let mut resolved_dictionary = current_leaf
1307        .and_then(|leaf| leaf.extensions().app_data_dictionary())
1308        .map(|ext| ext.dictionary().clone())
1309        .unwrap_or_default();
1310    if let Some(caller_dict) = caller_extensions.and_then(|exts| exts.app_data_dictionary()) {
1311        for entry in caller_dict.dictionary().entries() {
1312            resolved_dictionary.insert(entry.id(), entry.data().to_vec());
1313        }
1314    }
1315
1316    let app_components_bytes = resolved_dictionary
1317        .get(&ComponentId::from(ComponentType::AppComponents))
1318        .map(<[u8]>::to_vec);
1319    let Some(app_components_bytes) = app_components_bytes else {
1320        return Err(VirtualClientsError::VcComponentNotListed);
1321    };
1322
1323    // The AppComponents body is `ComponentID supported_components<V>`, i.e.
1324    // a TLS-encoded variable-length vector of u16.
1325    let supported_components = Vec::<u16>::tls_deserialize_exact_bytes(&app_components_bytes)
1326        .map_err(|e| {
1327            log::error!("vc: AppComponents body failed to deserialize: {e:?}");
1328            VirtualClientsError::VcComponentNotListed
1329        })?;
1330    if !supported_components.contains(&VC_COMPONENT_ID) {
1331        return Err(VirtualClientsError::VcComponentNotListed);
1332    }
1333
1334    Ok(resolved_dictionary)
1335}

1337/// Merge a virtual-clients derivation-info blob into `resolved_dictionary`
1338/// under [`VC_COMPONENT_ID`] and build the resulting leaf-node extensions.
1339///
1340/// Every other component id in `resolved_dictionary` (notably `AppComponents`)
1341/// is preserved, as is every non-`AppDataDictionary` extension the caller
1342/// supplied in `caller_extensions`. The rebuilt dictionary replaces any
1343/// `AppDataDictionary` entry already in that list.
1344pub(crate) fn merge_vc_derivation_info(
1345    caller_extensions: Option<
1346        &crate::extensions::Extensions<crate::treesync::node::leaf_node::LeafNode>,
1347    >,
1348    mut resolved_dictionary: crate::extensions::AppDataDictionary,
1349    derivation_info_bytes: Vec<u8>,
1350) -> Result<
1351    crate::extensions::Extensions<crate::treesync::node::leaf_node::LeafNode>,
1352    crate::error::LibraryError,
1353> {
1354    use crate::extensions::{AppDataDictionaryExtension, Extension, Extensions};
1355
1356    resolved_dictionary.insert(VC_COMPONENT_ID, derivation_info_bytes);
1357    let vc_extension =
1358        Extension::AppDataDictionary(AppDataDictionaryExtension::new(resolved_dictionary));
1359
1360    let other_extensions = caller_extensions
1361        .map(|exts| {
1362            exts.iter()
1363                .filter(|ext| !matches!(ext, Extension::AppDataDictionary(_)))
1364                .cloned()
1365                .collect::<Vec<_>>()
1366        })
1367        .unwrap_or_default();
1368    let new_extensions: Vec<Extension> = other_extensions
1369        .into_iter()
1370        .chain(std::iter::once(vc_extension))
1371        .collect();
1372    Extensions::from_vec(new_extensions)
1373        .map_err(|_| crate::error::LibraryError::custom("Failed to build VC leaf-node extensions"))
1374}

1884    /// The generation ID is deterministic for fixed inputs, changes when any
1885    /// `PrivateMessageContext` field changes, and two siblings that share the
1886    /// same emulator epoch secret derive the same value (so a DS can compare
1887    /// them for equality across siblings).
1888    #[test]
1889    fn generation_id_is_deterministic_and_context_sensitive() {
1890        let provider = OpenMlsRustCrypto::default();
1891        let secret_bytes = provider
1892            .rand()
1893            .random_vec(CIPHERSUITE.hash_length())
1894            .expect("randomness");
1895        let state = state_from_secret_bytes(&provider, &secret_bytes, LeafNodeIndex::new(0));
1896
1897        let group_id = GroupId::from_slice(b"higher-level-group");
1898        let epoch = GroupEpoch::from(7);
1899        let derive = |group_id: &GroupId, epoch, generation, ratchet_type| {
1900            state
1901                .derive_generation_id(provider.crypto(), group_id, epoch, generation, ratchet_type)
1902                .expect("derive generation id")
1903        };
1904
1905        let base = derive(&group_id, epoch, 3, RatchetType::Application);
1906        // The generation ID is `Kdf.Nh` bytes long.
1907        assert_eq!(base.as_slice().len(), CIPHERSUITE.hash_length());
1908        // Deterministic for fixed inputs.
1909        assert_eq!(base, derive(&group_id, epoch, 3, RatchetType::Application));
1910        // Sensitive to the generation, the epoch, the group id, and the
1911        // ratchet type.
1912        assert_ne!(base, derive(&group_id, epoch, 4, RatchetType::Application));
1913        assert_ne!(
1914            base,
1915            derive(&group_id, GroupEpoch::from(8), 3, RatchetType::Application)
1916        );
1917        assert_ne!(
1918            base,
1919            derive(
1920                &GroupId::from_slice(b"other-group"),
1921                epoch,
1922                3,
1923                RatchetType::Application
1924            )
1925        );
1926        assert_ne!(base, derive(&group_id, epoch, 3, RatchetType::Handshake));
1927
1928        // A sibling sharing the same emulator epoch secret derives the same
1929        // generation ID, even from a different leaf index: the leaf index is
1930        // not part of the PrivateMessageContext.
1931        let sibling = state_from_secret_bytes(&provider, &secret_bytes, LeafNodeIndex::new(5));
1932        let sibling_id = sibling
1933            .derive_generation_id(
1934                provider.crypto(),
1935                &group_id,
1936                epoch,
1937                3,
1938                RatchetType::Application,
1939            )
1940            .expect("sibling derive generation id");
1941        assert_eq!(base, sibling_id);
1942    }
