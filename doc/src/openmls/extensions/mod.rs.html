1//! # Extensions
2//!
3//! In MLS, extensions appear in the following places:
4//!
5//! - In [`KeyPackages`](`crate::key_packages`), to describe client capabilities
6//!   and aspects of their participation in the group.
7//!
8//! - In `GroupInfo`, to inform new members of the group's parameters and to
9//!   provide any additional information required to join the group.
10//!
11//! - In the `GroupContext` object, to ensure that all members of the group have
12//!   a consistent view of the parameters in use.
13//!
14//! Note that `GroupInfo` and `GroupContext` are not exposed via OpenMLS' public
15//! API.
16//!
17//! OpenMLS supports the following extensions:
18//!
19//! - [`ApplicationIdExtension`] (KeyPackage extension)
20//! - [`RatchetTreeExtension`] (GroupInfo extension)
21//! - [`RequiredCapabilitiesExtension`] (GroupContext extension)
22//! - [`ExternalPubExtension`] (GroupInfo extension)
23

66/// MLS Extension Types
67///
68/// Copied from draft-ietf-mls-protocol-16:
69///
70/// | Value            | Name                     | Message(s) | Recommended | Reference |
71/// |:-----------------|:-------------------------|:-----------|:------------|:----------|
72/// | 0x0000           | RESERVED                 | N/A        | N/A         | RFC XXXX  |
73/// | 0x0001           | application_id           | LN         | Y           | RFC XXXX  |
74/// | 0x0002           | ratchet_tree             | GI         | Y           | RFC XXXX  |
75/// | 0x0003           | required_capabilities    | GC         | Y           | RFC XXXX  |
76/// | 0x0004           | external_pub             | GI         | Y           | RFC XXXX  |
77/// | 0x0005           | external_senders         | GC         | Y           | RFC XXXX  |
78/// | 0xff00  - 0xffff | Reserved for Private Use | N/A        | N/A         | RFC XXXX  |
79///
80/// Note: OpenMLS does not provide a `Reserved` variant in [ExtensionType].
81#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, Serialize, Deserialize, Ord, PartialOrd)]
82pub enum ExtensionType {
83    /// The application id extension allows applications to add an explicit,
84    /// application-defined identifier to a KeyPackage.
85    ApplicationId,
86
87    /// The ratchet tree extensions provides the whole public state of the
88    /// ratchet tree.
89    RatchetTree,
90
91    /// The required capabilities extension defines the configuration of a group
92    /// that imposes certain requirements on clients in the group.
93    RequiredCapabilities,
94
95    /// To join a group via an External Commit, a new member needs a GroupInfo
96    /// with an ExternalPub extension present in its extensions field.
97    ExternalPub,
98
99    /// Group context extension that contains the credentials and signature keys
100    /// of senders that are permitted to send external proposals to the group.
101    ExternalSenders,
102
103    /// KeyPackage extension that marks a KeyPackage for use in a last resort
104    /// scenario.
105    LastResort,
106
107    #[cfg(feature = "extensions-draft-08")]
108    /// AppDataDictionary extension
109    AppDataDictionary,
110
111    /// A GREASE extension type for ensuring extensibility.
112    Grease(u16),
113
114    /// A currently unknown extension type.
115    Unknown(u16),
116}
117
118impl ExtensionType {
119    /// Returns true for all extension types that are considered "default" by the spec.
120    pub(crate) fn is_default(self) -> bool {
121        match self {
122            ExtensionType::ApplicationId
123            | ExtensionType::RatchetTree
124            | ExtensionType::RequiredCapabilities
125            | ExtensionType::ExternalPub
126            | ExtensionType::ExternalSenders => true,
127            ExtensionType::LastResort | ExtensionType::Grease(_) | ExtensionType::Unknown(_) => {
128                false
129            }
130            #[cfg(feature = "extensions-draft-08")]
131            ExtensionType::AppDataDictionary => false,
132        }
133    }
134
135    /// Returns whether an extension type is valid when used in leaf nodes.
136    /// Returns None if validity can not be determined.
137    /// This is the case for unknown extensions.
138    //  https://validation.openmls.tech/#valn1601
139    pub(crate) fn is_valid_in_leaf_node(self) -> Option<bool> {
140        match self {
141            ExtensionType::Grease(_)
142            | ExtensionType::LastResort
143            | ExtensionType::RatchetTree
144            | ExtensionType::RequiredCapabilities
145            | ExtensionType::ExternalPub
146            | ExtensionType::ExternalSenders => Some(false),
147            ExtensionType::ApplicationId => Some(true),
148            ExtensionType::Unknown(_) => None,
149            #[cfg(feature = "extensions-draft-08")]
150            ExtensionType::AppDataDictionary => Some(true),
151        }
152    }
153    pub(crate) fn is_valid_in_group_info(self) -> Option<bool> {
154        match self {
155            ExtensionType::Grease(_)
156            | ExtensionType::LastResort
157            | ExtensionType::RequiredCapabilities
158            | ExtensionType::ExternalSenders
159            | ExtensionType::ApplicationId => Some(false),
160            ExtensionType::RatchetTree | ExtensionType::ExternalPub => Some(true),
161            ExtensionType::Unknown(_) => None,
162            #[cfg(feature = "extensions-draft-08")]
163            ExtensionType::AppDataDictionary => Some(true),
164        }
165    }
166
167    /// Returns true if this is a GREASE extension type.
168    ///
169    /// GREASE values are used to ensure implementations properly handle unknown
170    /// extension types. See [RFC 9420 Section 13.5](https://www.rfc-editor.org/rfc/rfc9420.html#section-13.5).
171    pub fn is_grease(&self) -> bool {
172        matches!(self, ExtensionType::Grease(_))
173    }
174}

248/// # Extension
249///
250/// An extension is one of the [`Extension`] enum values.
251/// The enum provides a set of common functionality for all extensions.
252///
253/// See the individual extensions for more details on each extension.
254///
255/// ```c
256/// // draft-ietf-mls-protocol-16
257/// struct {
258///     ExtensionType extension_type;
259///     opaque extension_data<V>;
260/// } Extension;
261/// ```
262#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
263pub enum Extension {
264    /// An [`ApplicationIdExtension`]
265    ApplicationId(ApplicationIdExtension),
266
267    /// A [`RatchetTreeExtension`]
268    RatchetTree(RatchetTreeExtension),
269
270    /// A [`RequiredCapabilitiesExtension`]
271    RequiredCapabilities(RequiredCapabilitiesExtension),
272
273    /// An [`ExternalPubExtension`]
274    ExternalPub(ExternalPubExtension),
275
276    /// An [`ExternalSendersExtension`]
277    ExternalSenders(ExternalSendersExtension),
278
279    /// An [`AppDataDictionaryExtension`]
280    #[cfg(feature = "extensions-draft-08")]
281    AppDataDictionary(AppDataDictionaryExtension),
282
283    /// A [`LastResortExtension`]
284    LastResort(LastResortExtension),
285
286    /// A currently unknown extension.
287    Unknown(u16, UnknownExtension),
288}
289
290/// A unknown/unparsed extension represented by raw bytes.
291#[derive(PartialEq, Eq, Clone, Debug, Serialize, Deserialize)]
292pub struct UnknownExtension(pub Vec<u8>);

294/// A list of extensions with unique extension types.
295#[derive(Default, Debug, Clone, PartialEq, Eq, Serialize, Deserialize, TlsSize)]
296pub struct Extensions {
297    unique: Vec<Extension>,
298}
299
300impl TlsSerializeTrait for Extensions {
301    fn tls_serialize<W: Write>(&self, writer: &mut W) -> Result<usize, Error> {
302        self.unique.tls_serialize(writer)
303    }
304}
305
306impl TlsDeserializeTrait for Extensions {
307    fn tls_deserialize<R: Read>(bytes: &mut R) -> Result<Self, Error>
308    where
309        Self: Sized,
310    {
311        let candidate: Vec<Extension> = Vec::tls_deserialize(bytes)?;
312        Extensions::try_from(candidate)
313            .map_err(|_| Error::DecodingError("Found duplicate extensions".into()))
314    }
315}
316
317impl DeserializeBytes for Extensions {
318    fn tls_deserialize_bytes(bytes: &[u8]) -> Result<(Self, &[u8]), Error>
319    where
320        Self: Sized,
321    {
322        let mut bytes_ref = bytes;
323        let extensions = Extensions::tls_deserialize(&mut bytes_ref)?;
324        let remainder = &bytes[extensions.tls_serialized_len()..];
325        Ok((extensions, remainder))
326    }
327}

329impl Extensions {
330    /// Create an empty extension list.
331    pub fn empty() -> Self {
332        Self { unique: vec![] }
333    }
334
335    /// Create an extension list with a single extension.
336    pub fn single(extension: Extension) -> Self {
337        Self {
338            unique: vec![extension],
339        }
340    }
341
342    /// Create an extension list with multiple extensions.
343    ///
344    /// This function will fail when the list of extensions contains duplicate
345    /// extension types.
346    pub fn from_vec(extensions: Vec<Extension>) -> Result<Self, InvalidExtensionError> {
347        extensions.try_into()
348    }
349
350    /// Returns an iterator over the extension list.
351    pub fn iter(&self) -> impl Iterator<Item = &Extension> {
352        self.unique.iter()
353    }
354
355    /// Add an extension to the extension list.
356    ///
357    /// Returns an error when there already is an extension with the same
358    /// extension type.
359    pub fn add(&mut self, extension: Extension) -> Result<(), InvalidExtensionError> {
360        if self.contains(extension.extension_type()) {
361            return Err(InvalidExtensionError::Duplicate);
362        }
363
364        self.unique.push(extension);
365
366        Ok(())
367    }
368
369    /// Add an extension to the extension list (or replace an existing one.)
370    ///
371    /// Returns the replaced extension (if any).
372    pub fn add_or_replace(&mut self, extension: Extension) -> Option<Extension> {
373        let replaced = self.remove(extension.extension_type());
374        self.unique.push(extension);
375        replaced
376    }
377
378    /// Remove an extension from the extension list.
379    ///
380    /// Returns the removed extension or `None` when there is no extension with
381    /// the given extension type.
382    pub fn remove(&mut self, extension_type: ExtensionType) -> Option<Extension> {
383        if let Some(pos) = self
384            .unique
385            .iter()
386            .position(|ext| ext.extension_type() == extension_type)
387        {
388            Some(self.unique.remove(pos))
389        } else {
390            None
391        }
392    }
393
394    /// Returns `true` iff the extension list contains an extension with the
395    /// given extension type.
396    pub fn contains(&self, extension_type: ExtensionType) -> bool {
397        self.unique
398            .iter()
399            .any(|ext| ext.extension_type() == extension_type)
400    }
401
402    // validate that all extensions can be added to a leaf node.
403    // https://validation.openmls.tech/#valn1601
404    pub(crate) fn validate_extension_types_for_leaf_node(
405        &self,
406    ) -> Result<(), InvalidExtensionError> {
407        for extension_type in self.unique.iter().map(Extension::extension_type) {
408            // also allow unknown extensions, which return `None` here
409            if extension_type.is_valid_in_leaf_node() == Some(false) {
410                return Err(InvalidExtensionError::IllegalInLeafNodes);
411            }
412        }
413        Ok(())
414    }
415}

417impl TryFrom<Vec<Extension>> for Extensions {
418    type Error = InvalidExtensionError;
419
420    fn try_from(candidate: Vec<Extension>) -> Result<Self, Self::Error> {
421        let mut unique: Vec<Extension> = Vec::new();
422
423        for extension in candidate.into_iter() {
424            if unique
425                .iter()
426                .any(|ext| ext.extension_type() == extension.extension_type())
427            {
428                return Err(InvalidExtensionError::Duplicate);
429            } else {
430                unique.push(extension);
431            }
432        }
433
434        Ok(Self { unique })
435    }
436}

688    fn ensure_ordering() {
689        // Create some extensions with different extension types and test
690        // that all permutations keep their order after being (de)serialized.
