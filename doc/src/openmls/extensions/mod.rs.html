1//! # Extensions
2//!
3//! In MLS, extensions appear in the following places:
4//!
5//! - In [`KeyPackages`](`crate::key_packages`), to describe client capabilities
6//!   and aspects of their participation in the group.
7//!
8//! - In `GroupInfo`, to inform new members of the group's parameters and to
9//!   provide any additional information required to join the group.
10//!
11//! - In the `GroupContext` object, to ensure that all members of the group have
12//!   a consistent view of the parameters in use.
13//!
14//! Note that `GroupInfo` and `GroupContext` are not exposed via OpenMLS' public
15//! API.
16//!
17//! OpenMLS supports the following extensions:
18//!
19//! - [`ApplicationIdExtension`] (KeyPackage extension)
20//! - [`RatchetTreeExtension`] (GroupInfo extension)
21//! - [`RequiredCapabilitiesExtension`] (GroupContext extension)
22//! - [`ExternalPubExtension`] (GroupInfo extension)
23

65/// MLS Extension Types
66///
67/// Copied from draft-ietf-mls-protocol-16:
68///
69/// | Value            | Name                     | Message(s) | Recommended | Reference |
70/// |:-----------------|:-------------------------|:-----------|:------------|:----------|
71/// | 0x0000           | RESERVED                 | N/A        | N/A         | RFC XXXX  |
72/// | 0x0001           | application_id           | LN         | Y           | RFC XXXX  |
73/// | 0x0002           | ratchet_tree             | GI         | Y           | RFC XXXX  |
74/// | 0x0003           | required_capabilities    | GC         | Y           | RFC XXXX  |
75/// | 0x0004           | external_pub             | GI         | Y           | RFC XXXX  |
76/// | 0x0005           | external_senders         | GC         | Y           | RFC XXXX  |
77/// | 0xff00  - 0xffff | Reserved for Private Use | N/A        | N/A         | RFC XXXX  |
78///
79/// Note: OpenMLS does not provide a `Reserved` variant in [ExtensionType].
80#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, Serialize, Deserialize, Ord, PartialOrd)]
81pub enum ExtensionType {
82    /// The application id extension allows applications to add an explicit,
83    /// application-defined identifier to a KeyPackage.
84    ApplicationId,
85
86    /// The ratchet tree extensions provides the whole public state of the
87    /// ratchet tree.
88    RatchetTree,
89
90    /// The required capabilities extension defines the configuration of a group
91    /// that imposes certain requirements on clients in the group.
92    RequiredCapabilities,
93
94    /// To join a group via an External Commit, a new member needs a GroupInfo
95    /// with an ExternalPub extension present in its extensions field.
96    ExternalPub,
97
98    /// Group context extension that contains the credentials and signature keys
99    /// of senders that are permitted to send external proposals to the group.
100    ExternalSenders,
101
102    /// KeyPackage extension that marks a KeyPackage for use in a last resort
103    /// scenario.
104    LastResort,
105
106    /// A currently unknown extension type.
107    Unknown(u16),
108}
109
110impl ExtensionType {
111    /// Returns true for all extension types that are considered "default" by the spec.
112    pub(crate) fn is_default(self) -> bool {
113        match self {
114            ExtensionType::ApplicationId
115            | ExtensionType::RatchetTree
116            | ExtensionType::RequiredCapabilities
117            | ExtensionType::ExternalPub
118            | ExtensionType::ExternalSenders => true,
119            ExtensionType::LastResort | ExtensionType::Unknown(_) => false,
120        }
121    }
122
123    /// Returns whether an extension type is valid when used in leaf nodes.
124    /// Returns None if validity can not be determined.
125    /// This is the case for unknown extensions.
126    //  https://validation.openmls.tech/#valn1601
127    pub(crate) fn is_valid_in_leaf_node(self) -> Option<bool> {
128        match self {
129            ExtensionType::LastResort
130            | ExtensionType::RatchetTree
131            | ExtensionType::RequiredCapabilities
132            | ExtensionType::ExternalPub
133            | ExtensionType::ExternalSenders => Some(false),
134            ExtensionType::ApplicationId => Some(true),
135            ExtensionType::Unknown(_) => None,
136        }
137    }
138}

146impl TlsDeserializeTrait for ExtensionType {
147    fn tls_deserialize<R: Read>(bytes: &mut R) -> Result<Self, Error>
148    where
149        Self: Sized,
150    {
151        let mut extension_type = [0u8; 2];

164        let extension_type = ExtensionType::tls_deserialize(&mut bytes_ref)?;

206/// # Extension
207///
208/// An extension is one of the [`Extension`] enum values.
209/// The enum provides a set of common functionality for all extensions.
210///
211/// See the individual extensions for more details on each extension.
212///
213/// ```c
214/// // draft-ietf-mls-protocol-16
215/// struct {
216///     ExtensionType extension_type;
217///     opaque extension_data<V>;
218/// } Extension;
219/// ```
220#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
221pub enum Extension {
222    /// An [`ApplicationIdExtension`]
223    ApplicationId(ApplicationIdExtension),
224
225    /// A [`RatchetTreeExtension`]
226    RatchetTree(RatchetTreeExtension),
227
228    /// A [`RequiredCapabilitiesExtension`]
229    RequiredCapabilities(RequiredCapabilitiesExtension),
230
231    /// An [`ExternalPubExtension`]
232    ExternalPub(ExternalPubExtension),
233
234    /// An [`ExternalSendersExtension`]
235    ExternalSenders(ExternalSendersExtension),
236
237    /// A [`LastResortExtension`]
238    LastResort(LastResortExtension),
239
240    /// A currently unknown extension.
241    Unknown(u16, UnknownExtension),
242}
243
244/// A unknown/unparsed extension represented by raw bytes.
245#[derive(PartialEq, Eq, Clone, Debug, Serialize, Deserialize)]
246pub struct UnknownExtension(pub Vec<u8>);
247
248/// A list of extensions with unique extension types.
249#[derive(Default, Debug, Clone, PartialEq, Eq, Serialize, Deserialize, TlsSize)]
250pub struct Extensions {
251    unique: Vec<Extension>,
252}
253
254impl TlsSerializeTrait for Extensions {
255    fn tls_serialize<W: Write>(&self, writer: &mut W) -> Result<usize, Error> {
256        self.unique.tls_serialize(writer)
257    }
258}
259
260impl TlsDeserializeTrait for Extensions {
261    fn tls_deserialize<R: Read>(bytes: &mut R) -> Result<Self, Error>
262    where
263        Self: Sized,
264    {
265        let candidate: Vec<Extension> = Vec::tls_deserialize(bytes)?;
266        Extensions::try_from(candidate)
267            .map_err(|_| Error::DecodingError("Found duplicate extensions".into()))
268    }
269}
270
271impl DeserializeBytes for Extensions {
272    fn tls_deserialize_bytes(bytes: &[u8]) -> Result<(Self, &[u8]), Error>
273    where
274        Self: Sized,
275    {
276        let mut bytes_ref = bytes;
277        let extensions = Extensions::tls_deserialize(&mut bytes_ref)?;
278        let remainder = &bytes[extensions.tls_serialized_len()..];
279        Ok((extensions, remainder))
280    }
281}

283impl Extensions {
284    /// Create an empty extension list.
285    pub fn empty() -> Self {
286        Self { unique: vec![] }
287    }
288
289    /// Create an extension list with a single extension.
290    pub fn single(extension: Extension) -> Self {
291        Self {
292            unique: vec![extension],
293        }
294    }
295
296    /// Create an extension list with multiple extensions.
297    ///
298    /// This function will fail when the list of extensions contains duplicate
299    /// extension types.
300    pub fn from_vec(extensions: Vec<Extension>) -> Result<Self, InvalidExtensionError> {
301        extensions.try_into()
302    }
303
304    /// Returns an iterator over the extension list.
305    pub fn iter(&self) -> impl Iterator<Item = &Extension> {
306        self.unique.iter()
307    }
308
309    /// Add an extension to the extension list.
310    ///
311    /// Returns an error when there already is an extension with the same
312    /// extension type.
313    pub fn add(&mut self, extension: Extension) -> Result<(), InvalidExtensionError> {
314        if self.contains(extension.extension_type()) {
315            return Err(InvalidExtensionError::Duplicate);
316        }
317
318        self.unique.push(extension);
319
320        Ok(())
321    }
322
323    /// Add an extension to the extension list (or replace an existing one.)
324    ///
325    /// Returns the replaced extension (if any).
326    pub fn add_or_replace(&mut self, extension: Extension) -> Option<Extension> {
327        let replaced = self.remove(extension.extension_type());
328        self.unique.push(extension);
329        replaced
330    }
331
332    /// Remove an extension from the extension list.
333    ///
334    /// Returns the removed extension or `None` when there is no extension with
335    /// the given extension type.
336    pub fn remove(&mut self, extension_type: ExtensionType) -> Option<Extension> {
337        if let Some(pos) = self
338            .unique
339            .iter()
340            .position(|ext| ext.extension_type() == extension_type)
341        {
342            Some(self.unique.remove(pos))
343        } else {
344            None
345        }
346    }
347
348    /// Returns `true` iff the extension list contains an extension with the
349    /// given extension type.
350    pub fn contains(&self, extension_type: ExtensionType) -> bool {
351        self.unique
352            .iter()
353            .any(|ext| ext.extension_type() == extension_type)
354    }
355
356    // validate that all extensions can be added to a leaf node.
357    // https://validation.openmls.tech/#valn1601
358    pub(crate) fn validate_extension_types_for_leaf_node(
359        &self,
360    ) -> Result<(), InvalidExtensionError> {
361        for extension_type in self.unique.iter().map(Extension::extension_type) {
362            // also allow unknown extensions, which return `None` here
363            if extension_type.is_valid_in_leaf_node() == Some(false) {
364                return Err(InvalidExtensionError::IllegalInLeafNodes);
365            }
366        }
367        Ok(())
368    }
369}
370
371impl TryFrom<Vec<Extension>> for Extensions {
372    type Error = InvalidExtensionError;
373
374    fn try_from(candidate: Vec<Extension>) -> Result<Self, Self::Error> {
375        let mut unique: Vec<Extension> = Vec::new();
376
377        for extension in candidate.into_iter() {
378            if unique
379                .iter()
380                .any(|ext| ext.extension_type() == extension.extension_type())
381            {
382                return Err(InvalidExtensionError::Duplicate);
383            } else {
384                unique.push(extension);
385            }
386        }
387
388        Ok(Self { unique })
389    }
390}

561    #[test]
562    fn add_try_from() {
563        // Create some extensions with different extension types and test that
564        // duplicates are rejected. The extension content does not matter in this test.
565        let ext_x = Extension::ApplicationId(ApplicationIdExtension::new(b"Test"));
566        let ext_y = Extension::RequiredCapabilities(RequiredCapabilitiesExtension::default());
567
568        let tests = [
569            (vec![], true),
570            (vec![ext_x.clone()], true),
571            (vec![ext_x.clone(), ext_x.clone()], false),
572            (vec![ext_x.clone(), ext_x.clone(), ext_x.clone()], false),
573            (vec![ext_y.clone()], true),
574            (vec![ext_y.clone(), ext_y.clone()], false),
575            (vec![ext_y.clone(), ext_y.clone(), ext_y.clone()], false),
576            (vec![ext_x.clone(), ext_y.clone()], true),
577            (vec![ext_y.clone(), ext_x.clone()], true),
578            (vec![ext_x.clone(), ext_x.clone(), ext_y.clone()], false),
579            (vec![ext_y.clone(), ext_y.clone(), ext_x.clone()], false),
580            (vec![ext_x.clone(), ext_y.clone(), ext_y.clone()], false),
581            (vec![ext_y.clone(), ext_x.clone(), ext_x.clone()], false),
582            (vec![ext_x.clone(), ext_y.clone(), ext_x.clone()], false),
583            (vec![ext_y.clone(), ext_x, ext_y], false),
584        ];
585
586        for (test, should_work) in tests.into_iter() {
587            // Test `add`.
588            {
589                let mut extensions = Extensions::default();
590
591                let mut works = true;
592                for ext in test.iter() {
593                    match extensions.add(ext.clone()) {
594                        Ok(_) => {}
595                        Err(InvalidExtensionError::Duplicate) => {
596                            works = false;
597                        }
598                        _ => panic!("This should have never happened."),
599                    }
600                }
601
602                println!("{:?}, {:?}", test.clone(), should_work);
603                assert_eq!(works, should_work);
604            }
605
606            // Test `try_from`.
607            if should_work {
608                assert!(Extensions::try_from(test).is_ok());
609            } else {
610                assert!(Extensions::try_from(test).is_err());
611            }
612        }
613    }

615    #[test]
616    fn ensure_ordering() {
617        // Create some extensions with different extension types and test
618        // that all permutations keep their order after being (de)serialized.
619        // The extension content does not matter in this test.
620        let ext_x = Extension::ApplicationId(ApplicationIdExtension::new(b"Test"));
621        let ext_y = Extension::ExternalPub(ExternalPubExtension::new(HpkePublicKey::new(vec![])));
622        let ext_z = Extension::RequiredCapabilities(RequiredCapabilitiesExtension::default());
623
624        for candidate in [ext_x, ext_y, ext_z]
625            .into_iter()
626            .permutations(3)
627            .collect::<Vec<_>>()
628        {
629            let candidate: Extensions = Extensions::try_from(candidate).unwrap();
630            let bytes = candidate.tls_serialize_detached().unwrap();
631            let got = Extensions::tls_deserialize(&mut bytes.as_slice()).unwrap();
632            assert_eq!(candidate, got);
633        }
634    }

636    #[test]
637    fn that_unknown_extensions_are_de_serialized_correctly() {
638        let extension_types = [0x0000u16, 0x0A0A, 0x7A7A, 0xF100, 0xFFFF];
639        let extension_datas = [vec![], vec![0], vec![1, 2, 3]];
640
641        for extension_type in extension_types.into_iter() {
642            for extension_data in extension_datas.iter() {
643                // Construct an unknown extension manually.
644                let test = {
645                    let mut buf = extension_type.to_be_bytes().to_vec();
646                    buf.append(
647                        &mut VLBytes::new(extension_data.clone())
648                            .tls_serialize_detached()
649                            .unwrap(),
650                    );
651                    buf
652                };
653
654                // Test deserialization.
655                let got = Extension::tls_deserialize_exact(&test).unwrap();
656
657                match got {
658                    Extension::Unknown(got_extension_type, ref got_extension_data) => {
659                        assert_eq!(extension_type, got_extension_type);
660                        assert_eq!(extension_data, &got_extension_data.0);
661                    }
662                    other => panic!("Expected `Extension::Unknown`, got {other:?}"),
663                }
664
665                // Test serialization.
666                let got_serialized = got.tls_serialize_detached().unwrap();
667                assert_eq!(test, got_serialized);
668            }
669        }
