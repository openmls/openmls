26/// 7.1 Content Authentication
27///
28/// ```c
29/// // draft-ietf-mls-protocol-17
30///
31/// struct {
32///    /* SignWithLabel(., "FramedContentTBS", FramedContentTBS) */
33///    opaque signature<V>;
34///    select (FramedContent.content_type) {
35///        case commit:
36///            /*
37///              MAC(confirmation_key,
38///                  GroupContext.confirmed_transcript_hash)
39///            */
40///            MAC confirmation_tag;
41///        case application:
42///        case proposal:
43///            struct{};
44///    };
45///} FramedContentAuthData;
46/// ```
47#[derive(PartialEq, Debug, Clone, Serialize, Deserialize)]
48pub(crate) struct FramedContentAuthData {
49    pub(super) signature: Signature,
50    pub(super) confirmation_tag: Option<ConfirmationTag>,
51}
52
53impl FramedContentAuthData {
54    pub(crate) fn deserialize<R: Read>(
55        bytes: &mut R,
56        content_type: ContentType,
57    ) -> Result<Self, tls_codec::Error> {
58        let signature = Signature::tls_deserialize(bytes)?;
59        let confirmation_tag = if matches!(content_type, ContentType::Commit) {
60            Some(ConfirmationTag::tls_deserialize(bytes)?)
61        } else {
62            None
63        };
64        Ok(Self {
65            signature,
66            confirmation_tag,
67        })
68    }
69}

71/// 6 Message Framing
72///
73/// ```c
74/// // draft-ietf-mls-protocol-17
75///
76/// struct {
77///     WireFormat wire_format;
78///     FramedContent content;
79///     FramedContentAuthData auth;
80/// } AuthenticatedContent;
81/// ```
82#[derive(PartialEq, Debug, Clone, TlsSerialize, TlsSize)]
83pub(crate) struct AuthenticatedContent {
84    pub(super) wire_format: WireFormat,
85    pub(super) content: FramedContent,
86    pub(super) auth: FramedContentAuthData,
87}

89impl AuthenticatedContent {
90    /// Convenience function for creating a [`VerifiableAuthenticatedContent`].
91    #[inline]
92    fn new_and_sign(
93        framing_parameters: FramingParameters,
94        sender: Sender,
95        body: FramedContentBody,
96        context: &GroupContext,
97        signer: &impl Signer,
98    ) -> Result<Self, LibraryError> {
99        let mut content_tbs = FramedContentTbs::new(
100            framing_parameters.wire_format(),
101            context.group_id().clone(),
102            context.epoch(),
103            sender.clone(),
104            framing_parameters.aad().into(),
105            body,
106        );
107
108        if matches!(sender, Sender::NewMemberCommit | Sender::Member(_)) {
109            let serialized_context = context
110                .tls_serialize_detached()
111                .map_err(LibraryError::missing_bound_check)?;
112            content_tbs = content_tbs.with_context(serialized_context);
113        }
114
115        content_tbs
116            .sign(signer)
117            .map_err(|_| LibraryError::custom("Signing failed"))
118    }
119
120    /// This constructor builds an `AuthenticatedContent` containing an application
121    /// message. The sender type is always `SenderType::Member`.
122    pub(crate) fn new_application(
123        sender_leaf_index: LeafNodeIndex,
124        authenticated_data: &[u8],
125        application_message: &[u8],
126        context: &GroupContext,
127        signer: &impl Signer,
128    ) -> Result<Self, LibraryError> {
129        let framing_parameters =
130            FramingParameters::new(authenticated_data, WireFormat::PrivateMessage);
131        Self::new_and_sign(
132            framing_parameters,
133            Sender::Member(sender_leaf_index),
134            FramedContentBody::application(application_message),
135            context,
136            signer,
137        )
138    }
139
140    /// This constructor builds an `PublicMessage` containing a Proposal.
141    /// The sender type is always `SenderType::Member`.
142    pub(crate) fn member_proposal(
143        framing_parameters: FramingParameters,
144        sender_leaf_index: LeafNodeIndex,
145        proposal: Proposal,
146        context: &GroupContext,
147        signer: &impl Signer,
148    ) -> Result<Self, LibraryError> {
149        Self::new_and_sign(
150            framing_parameters,
151            Sender::Member(sender_leaf_index),
152            FramedContentBody::Proposal(proposal),
153            context,
154            signer,
155        )
156    }
157
158    /// This constructor builds an `PublicMessage` containing an External Proposal.
159    pub(crate) fn new_join_proposal(
160        proposal: Proposal,
161        group_id: GroupId,
162        epoch: GroupEpoch,
163        signer: &impl Signer,
164    ) -> Result<Self, LibraryError> {
165        let body = FramedContentBody::Proposal(proposal);
166
167        let content_tbs = FramedContentTbs::new(
168            WireFormat::PublicMessage,
169            group_id,
170            epoch,
171            Sender::NewMemberProposal,
172            vec![].into(),
173            body,
174        );
175
176        content_tbs
177            .sign(signer)
178            .map_err(|_| LibraryError::custom("Signing failed"))
179    }
180
181    /// This constructor builds an `PublicMessage` containing an External Proposal.
182    pub(crate) fn new_external_proposal(
183        proposal: Proposal,
184        group_id: GroupId,
185        epoch: GroupEpoch,
186        signer: &impl Signer,
187        sender_index: SenderExtensionIndex,
188    ) -> Result<Self, LibraryError> {
189        let body = FramedContentBody::Proposal(proposal);
190
191        let content_tbs = FramedContentTbs::new(
192            WireFormat::PublicMessage,
193            group_id,
194            epoch,
195            Sender::External(sender_index),
196            vec![].into(),
197            body,
198        );
199
200        content_tbs
201            .sign(signer)
202            .map_err(|_| LibraryError::custom("Signing failed"))
203    }
204
205    /// This constructor builds an `PublicMessage` containing a Commit. If the
206    /// given `CommitType` is `Member`, the `SenderType` is `Member` as well. If
207    /// it's an `External` commit, the `SenderType` is `NewMemberCommit`. If it is an
208    /// `External` commit, the context is not signed along with the rest of the
209    /// commit.
210    pub(crate) fn commit(
211        framing_parameters: FramingParameters,
212        sender: Sender,
213        commit: Commit,
214        context: &GroupContext,
215        signer: &impl Signer,
216    ) -> Result<Self, LibraryError> {
217        Self::new_and_sign(
218            framing_parameters,
219            sender,
220            FramedContentBody::commit(commit),
221            context,
222            signer,
223        )
224    }

295impl SignedStruct<FramedContentTbs> for AuthenticatedContent {
296    fn from_payload(tbs: FramedContentTbs, signature: Signature) -> Self {
297        let auth = FramedContentAuthData {
298            signature,
299            // Tags must always be added after the signature
300            confirmation_tag: None,
301        };
302        Self {
303            wire_format: tbs.wire_format,
304            content: tbs.content,
305            auth,
306        }
307    }
308}

310impl Size for FramedContentAuthData {
311    #[inline]
312    fn tls_serialized_len(&self) -> usize {
313        self.signature.tls_serialized_len()
314            + if let Some(confirmation_tag) = &self.confirmation_tag {
315                confirmation_tag.tls_serialized_len()
316            } else {
317                0
318            }
319    }
320}
321
322impl TlsSerializeTrait for FramedContentAuthData {
323    fn tls_serialize<W: Write>(&self, writer: &mut W) -> Result<usize, tls_codec::Error> {
324        let mut written = self.signature.tls_serialize(writer)?;
325        written += if let Some(confirmation_tag) = &self.confirmation_tag {
326            confirmation_tag.tls_serialize(writer)?
327        } else {
328            0
329        };
330        Ok(written)
331    }
