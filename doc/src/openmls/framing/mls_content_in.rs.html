28/// ```c
29/// // draft-ietf-mls-protocol-17
30/// struct {
31///     opaque group_id<V>;
32///     uint64 epoch;
33///     Sender sender;
34///     opaque authenticated_data<V>;
35///
36///     // ... continued in [FramedContentBody] ...
37/// } FramedContent;
38/// ```
39#[derive(
40    Debug,
41    PartialEq,
42    Clone,
43    Serialize,
44    Deserialize,
45    TlsSerialize,
46    TlsDeserialize,
47    TlsDeserializeBytes,
48    TlsSize,
49)]
50pub(crate) struct FramedContentIn {
51    pub(super) group_id: GroupId,
52    pub(super) epoch: GroupEpoch,
53    pub(super) sender: Sender,
54    pub(super) authenticated_data: VLBytes,
55    pub(super) body: FramedContentBodyIn,
56}
57
58impl FramedContentIn {
59    /// Returns a [`FramedContent`] after successful validation.
60    pub(crate) fn validate(
61        self,
62        ciphersuite: Ciphersuite,
63        crypto: &impl OpenMlsCrypto,
64        sender_context: Option<SenderContext>,
65        protocol_version: ProtocolVersion,
66    ) -> Result<FramedContent, ValidationError> {
67        Ok(FramedContent {
68            group_id: self.group_id,
69            epoch: self.epoch,
70            sender: self.sender,
71            authenticated_data: self.authenticated_data,
72            body: self
73                .body
74                .validate(ciphersuite, crypto, sender_context, protocol_version)?,
75        })
76    }
77}

85/// ```c
86/// // draft-ietf-mls-protocol-17
87/// struct {
88///     // ... continued from [FramedContent] ...
89///
90///     ContentType content_type;
91///     select (FramedContent.content_type) {
92///         case application:
93///           opaque application_data<V>;
94///         case proposal:
95///           Proposal proposal;
96///         case commit:
97///           Commit commit;
98///     }
99/// } FramedContent;
100/// ```
101#[derive(
102    Debug,
103    PartialEq,
104    Clone,
105    Serialize,
106    Deserialize,
107    TlsSerialize,
108    TlsDeserialize,
109    TlsDeserializeBytes,
110    TlsSize,
111)]
112#[repr(u8)]
113pub(crate) enum FramedContentBodyIn {
114    #[tls_codec(discriminant = 1)]
115    Application(VLBytes),
116    #[tls_codec(discriminant = 2)]
117    Proposal(ProposalIn),
118    #[tls_codec(discriminant = 3)]
119    Commit(CommitIn),
120}
121
122impl FramedContentBodyIn {
123    /// Returns the [`ContentType`].
124    pub(crate) fn content_type(&self) -> ContentType {
125        match self {
126            FramedContentBodyIn::Application(_) => ContentType::Application,
127            FramedContentBodyIn::Proposal(_) => ContentType::Proposal,
128            FramedContentBodyIn::Commit(_) => ContentType::Commit,
129        }
130    }
131
132    pub(super) fn deserialize_without_type<R: Read>(
133        bytes: &mut R,
134        content_type: ContentType,
135    ) -> Result<Self, tls_codec::Error> {
136        Ok(match content_type {
137            ContentType::Application => {
138                FramedContentBodyIn::Application(VLBytes::tls_deserialize(bytes)?)
139            }
140            ContentType::Proposal => {
141                FramedContentBodyIn::Proposal(ProposalIn::tls_deserialize(bytes)?)
142            }
143            ContentType::Commit => FramedContentBodyIn::Commit(CommitIn::tls_deserialize(bytes)?),
144        })
145    }
146
147    /// Returns a [`FramedContentBody`] after successful validation.
148    pub(crate) fn validate(
149        self,
150        ciphersuite: Ciphersuite,
151        crypto: &impl OpenMlsCrypto,
152        sender_context: Option<SenderContext>,
153        protocol_version: ProtocolVersion,
154    ) -> Result<FramedContentBody, ValidationError> {
155        Ok(match self {
156            FramedContentBodyIn::Application(bytes) => FramedContentBody::Application(bytes),
157            FramedContentBodyIn::Proposal(proposal_in) => FramedContentBody::Proposal(
158                proposal_in.validate(crypto, ciphersuite, sender_context, protocol_version)?,
159            ),
160            FramedContentBodyIn::Commit(commit_in) => {
161                let sender_context = sender_context
162                    .ok_or(LibraryError::custom("Forgot the commit sender context"))?;
163                FramedContentBody::Commit(commit_in.validate(
164                    ciphersuite,
165                    crypto,
166                    sender_context,
167                    protocol_version,
168                )?)
169            }
170        })
171    }
172}

175pub(crate) struct FramedContentTbsIn {
176    pub(super) version: ProtocolVersion,
177    pub(super) wire_format: WireFormat,
178    pub(super) content: FramedContentIn,
179    pub(super) serialized_context: Option<Vec<u8>>,
180}

182impl Signable for FramedContentTbsIn {
183    type SignedOutput = AuthenticatedContentIn;
184
185    fn unsigned_payload(&self) -> Result<Vec<u8>, tls_codec::Error> {
186        self.tls_serialize_detached()
187    }
188
189    fn label(&self) -> &str {
190        "FramedContentTBS"
191    }
192}
193
194impl Size for FramedContentTbsIn {
195    #[inline]
196    fn tls_serialized_len(&self) -> usize {
197        self.version.tls_serialized_len()
198            + self.wire_format.tls_serialized_len()
199            + self.content.tls_serialized_len()
200            + match &self.serialized_context {
201                Some(context)
202                    if matches!(
203                        self.content.sender,
204                        Sender::Member(_) | Sender::NewMemberCommit
205                    ) =>
206                {
207                    context.len()
208                }
209                _ => 0,
210            }
211    }
212}

224    }
