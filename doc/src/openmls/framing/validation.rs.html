52/// Intermediate message that can be constructed either from a public message or from private message.
53/// If it it constructed from a ciphertext message, the ciphertext message is decrypted first.
54/// This function implements the following checks:
55///  - ValSem005
56///  - ValSem007
57///  - ValSem009
58#[derive(Debug)]
59pub(crate) struct DecryptedMessage {
60    verifiable_content: VerifiableAuthenticatedContentIn,
61    /// Recovered sender emulation-group leaf index for an application
62    /// message from a sibling emulator client.
63    #[cfg(feature = "virtual-clients-draft")]
64    emulator_sender_leaf_index: Option<LeafNodeIndex>,
65}
66
67impl DecryptedMessage {
68    /// Constructs a [DecryptedMessage] from a [VerifiableAuthenticatedContent].
69    pub(crate) fn from_inbound_public_message<'a>(
70        public_message: PublicMessageIn,
71        message_secrets_option: impl Into<Option<&'a MessageSecrets>>,
72        serialized_context: Vec<u8>,
73        crypto: &impl OpenMlsCrypto,
74        ciphersuite: Ciphersuite,
75    ) -> Result<Self, ValidationError> {
76        if public_message.sender().is_member() {
77            // ValSem007 Membership tag presence
78            if public_message.membership_tag().is_none() {
79                return Err(ValidationError::MissingMembershipTag);
80            }
81
82            if let Some(message_secrets) = message_secrets_option.into() {
83                // Verify the membership tag. This needs to be done explicitly for PublicMessage messages,
84                // it is implicit for PrivateMessage messages (because the encryption can only be known by members).
85                // ValSem008
86                // https://validation.openmls.tech/#valn1302
87                public_message.verify_membership(
88                    crypto,
89                    ciphersuite,
90                    message_secrets.membership_key(),
91                    message_secrets.serialized_context(),
92                )?;
93            }
94        }
95
96        let verifiable_content = public_message.into_verifiable_content(serialized_context);
97
98        // Public messages don't carry a reuse_guard, so no emulator
99        // sender leaf index to recover.
100        Self::from_verifiable_content(
101            verifiable_content,
102            #[cfg(feature = "virtual-clients-draft")]
103            None,
104        )
105    }

107    /// Constructs a [DecryptedMessage] from a [PrivateMessage] by attempting to decrypt it
108    /// to a [VerifiableAuthenticatedContent] first.
109    pub(crate) fn from_inbound_ciphertext(
110        ciphertext: PrivateMessageIn,
111        crypto: &impl OpenMlsCrypto,
112        group: &mut MlsGroup,
113        sender_ratchet_configuration: &SenderRatchetConfiguration,
114        #[cfg(feature = "virtual-clients-draft")] emulator_ctx: Option<
115            &crate::framing::private_message::EmulatorReuseGuardCtx<'_>,
116        >,
117    ) -> Result<Self, ValidationError> {
118        // This will be refactored with #265.
119        let ciphersuite = group.ciphersuite();
120        // TODO: #819 The old leaves should not be needed any more.
121        //       Revisit when the transition is further along.
122        let (message_secrets, _old_leaves) = group
123            .message_secrets_and_leaves(ciphertext.epoch())
124            .map_err(MessageDecryptionError::SecretTreeError)?;
125        let sender_data = ciphertext.sender_data(message_secrets, crypto, ciphersuite)?;
126        // Check if we are the sender. With the `virtual-clients` feature,
127        // decrypting own messages is allowed, so we skip this check
128        #[cfg(not(feature = "virtual-clients-draft"))]
129        if sender_data.leaf_index == group.own_leaf_index() {
130            return Err(ValidationError::CannotDecryptOwnMessage);
131        }
132        #[cfg(feature = "virtual-clients-draft")]
133        let effective_emulator_ctx = match emulator_ctx {
134            Some(ctx) if sender_data.leaf_index == group.own_leaf_index() => Some(ctx),
135            _ => None,
136        };
137        let message_secrets = group
138            .message_secrets_for_epoch_mut(ciphertext.epoch())
139            .map_err(|_| MessageDecryptionError::AeadError)?;
140        let decrypted = ciphertext.to_verifiable_content(
141            ciphersuite,
142            crypto,
143            message_secrets,
144            sender_data.leaf_index,
145            sender_ratchet_configuration,
146            sender_data,
147            #[cfg(feature = "virtual-clients-draft")]
148            effective_emulator_ctx,
149        )?;
150        Self::from_verifiable_content(
151            decrypted.verifiable,
152            #[cfg(feature = "virtual-clients-draft")]
153            decrypted.emulator_sender_leaf_index,
154        )
155    }

157    // Internal constructor function. Does the following checks:
158    // - Confirmation tag must be present for Commit messages
159    // - Membership tag must be present for member messages, if the original incoming message was not an PrivateMessage
160    // - Ensures application messages were originally PrivateMessage messages
161    fn from_verifiable_content(
162        verifiable_content: VerifiableAuthenticatedContentIn,
163        #[cfg(feature = "virtual-clients-draft")] emulator_sender_leaf_index: Option<LeafNodeIndex>,
164    ) -> Result<Self, ValidationError> {
165        // ValSem009
166        if verifiable_content.content_type() == ContentType::Commit
167            && verifiable_content.confirmation_tag().is_none()
168        {
169            return Err(ValidationError::MissingConfirmationTag);
170        }
171        // ValSem005
172        if verifiable_content.content_type() == ContentType::Application {
173            if verifiable_content.wire_format() != WireFormat::PrivateMessage {
174                return Err(ValidationError::UnencryptedApplicationMessage);
175            } else if !verifiable_content.sender().is_member() {
176                // This should not happen because the sender of an PrivateMessage should always be a member
177                return Err(LibraryError::custom("Expected sender to be member.").into());
178            }
179        }
180        Ok(DecryptedMessage {
181            verifiable_content,
182            #[cfg(feature = "virtual-clients-draft")]
183            emulator_sender_leaf_index,
184        })
185    }

195    /// Gets the correct credential from the message depending on the sender type.
196    ///
197    /// The closure argument is used to look up the credential and signature key. If the epoch of
198    /// the message is the same as that of the group, look it up in the tree; else, look in up in
199    /// the past trees of the message secret store.
200    ///
201    /// Checks the following semantic validation:
202    ///  - ValSem112
203    ///  - ValSem245
204    ///  - Prepares ValSem246 by setting the right credential. The remainder
205    ///    of ValSem246 is validated as part of ValSem010.
206    ///  - [valn1301](https://validation.openmls.tech/#valn1301)
207    ///
208    /// Returns the [`Credential`] and the leaf's [`SignaturePublicKey`].
209    pub(crate) fn credential(
210        &self,
211        look_up_credential_with_key: impl Fn(LeafNodeIndex) -> Option<CredentialWithKey>,
212        external_senders: Option<&ExternalSendersExtension>,
213    ) -> Result<CredentialWithKey, ValidationError> {
214        let sender = self.sender();
215        match sender {
216            Sender::Member(leaf_index) => {
217                // https://validation.openmls.tech/#valn1306
218                look_up_credential_with_key(*leaf_index).ok_or(ValidationError::UnknownMember)
219            }
220            Sender::External(index) => {
221                let sender = external_senders
222                    .ok_or(ValidationError::NoExternalSendersExtension)?
223                    .get(index.index())
224                    .ok_or(ValidationError::UnauthorizedExternalSender)?;
225                Ok(CredentialWithKey {
226                    credential: sender.credential().clone(),
227                    signature_key: sender.signature_key().clone(),
228                })
229            }
230            Sender::NewMemberCommit | Sender::NewMemberProposal => {
231                // Fetch the credential from the message itself.
232                // https://validation.openmls.tech/#valn0407
233                self.verifiable_content.new_member_credential()
234            }
235        }
236    }

269/// Partially checked and potentially decrypted message (if it was originally encrypted).
270/// Use this to inspect the [`Credential`] of the message sender
271/// and the optional `aad` if the original message was encrypted.
272/// The [`OpenMlsSignaturePublicKey`] is used to verify the signature of the
273/// message.
274#[derive(Debug, Clone)]
275pub struct UnverifiedMessage {
276    verifiable_content: VerifiableAuthenticatedContentIn,
277    credential: Credential,
278    sender_pk: OpenMlsSignaturePublicKey,
279    sender_context: Option<SenderContext>,
280    /// See [`DecryptedMessage::emulator_sender_leaf_index`].
281    #[cfg(feature = "virtual-clients-draft")]
282    emulator_sender_leaf_index: Option<LeafNodeIndex>,
283}

285impl UnverifiedMessage {
286    /// Construct an [UnverifiedMessage] from a [DecryptedMessage] and an optional [Credential].
287    pub(crate) fn from_decrypted_message(
288        decrypted_message: DecryptedMessage,
289        credential: Credential,
290        sender_pk: OpenMlsSignaturePublicKey,
291        sender_context: Option<SenderContext>,
292    ) -> Self {
293        #[cfg(feature = "virtual-clients-draft")]
294        let emulator_sender_leaf_index = decrypted_message.emulator_sender_leaf_index;
295        UnverifiedMessage {
296            verifiable_content: decrypted_message.verifiable_content,
297            credential,
298            sender_pk,
299            sender_context,
300            #[cfg(feature = "virtual-clients-draft")]
301            emulator_sender_leaf_index,
302        }
303    }
304
305    /// Verify the [`UnverifiedMessage`].
306    pub(crate) fn verify(
307        self,
308        ciphersuite: Ciphersuite,
309        crypto: &impl OpenMlsCrypto,
310        protocol_version: ProtocolVersion,
311    ) -> Result<VerifiedMessage, ValidationError> {
312        let content: AuthenticatedContentIn = self
313            .verifiable_content
314            .verify(crypto, &self.sender_pk)
315            .map_err(|_| ValidationError::InvalidSignature)?;
316        // https://validation.openmls.tech/#valn1302
317        // https://validation.openmls.tech/#valn1304
318        let content =
319            content.validate(ciphersuite, crypto, self.sender_context, protocol_version)?;
320        Ok(VerifiedMessage {
321            content,
322            credential: self.credential,
323            #[cfg(feature = "virtual-clients-draft")]
324            emulator_sender_leaf_index: self.emulator_sender_leaf_index,
325        })
326    }

384    /// Parse the Safe AAD prefix at the start of `authenticated_data` and
385    /// attach it to this message. Callers should invoke this only when the receiving
386    /// group's GroupContext requires Safe AAD framing. Otherwise, `safe_aad`
387    /// stays `None` and `authenticated_data` is the caller-supplied bytes
388    /// untouched.
389    #[cfg(feature = "extensions-draft")]
390    pub(crate) fn try_attach_safe_aad(&mut self) -> Result<(), crate::framing::SafeAadError> {
391        let (safe_aad, prefix_len) =
392            crate::framing::safe_aad::parse_authenticated_data_prefix(&self.authenticated_data)?;
393        self.safe_aad = Some(safe_aad);
394        self.safe_aad_prefix_len = prefix_len;
395        Ok(())
396    }

402        self.safe_aad.as_ref()

462    /// Safely export a value if the content of the processed message is a
463    /// [`StagedCommit`].
464    #[cfg(feature = "extensions-draft")]
465    pub fn safe_export_secret<Crypto: OpenMlsCrypto>(
466        &mut self,
467        crypto: &Crypto,
468        component_id: ComponentId,
469    ) -> Result<Vec<u8>, ProcessedMessageSafeExportSecretError> {
470        if let ProcessedMessageContent::StagedCommitMessage(ref mut staged_commit) =
471            &mut self.content
472        {
473            let secret = staged_commit.safe_export_secret(crypto, component_id)?;
474            Ok(secret)
475        } else {
476            Err(ProcessedMessageSafeExportSecretError::NotACommit)
477        }
478    }

481/// Content of a processed message.
482///
483/// See the content variants' documentation for more information.
484/// [`StagedCommit`] and [`QueuedProposal`] can be inspected for authorization purposes.
485#[derive(Debug)]
486pub enum ProcessedMessageContent {
487    /// An application message.
488    ///
489    /// The [`ApplicationMessage`] contains a vector of bytes that can be used right-away.
490    ApplicationMessage(ApplicationMessage),
491    /// A standalone proposal.
492    ///
493    /// The [`QueuedProposal`] can be inspected for authorization purposes by the application.
494    /// If the proposal is deemed to be allowed, it should be added to the group's proposal
495    /// queue using [`MlsGroup::store_pending_proposal()`](crate::group::mls_group::MlsGroup::store_pending_proposal()).
496    ProposalMessage(Box<QueuedProposal>),
497    /// An [external join proposal](crate::prelude::JoinProposal) sent by a
498    /// [NewMemberProposal](crate::prelude::Sender::NewMemberProposal) sender which is outside the group.
499    ///
500    /// Since this originates from a party outside the group, the [`QueuedProposal`] SHOULD be
501    /// inspected for authorization purposes by the application. If the proposal is deemed to be
502    /// allowed, it should be added to the group's proposal queue using
503    /// [`MlsGroup::store_pending_proposal()`](crate::group::mls_group::MlsGroup::store_pending_proposal()).
504    ExternalJoinProposalMessage(Box<QueuedProposal>),
505    /// A Commit message.
506    ///
507    /// The [`StagedCommit`] can be inspected for authorization purposes by the application.
508    /// If the type of the commit and the proposals it covers are deemed to be allowed,
509    /// the commit should be merged into the group's state using
510    /// [`MlsGroup::merge_staged_commit()`](crate::group::mls_group::MlsGroup::merge_staged_commit()).
511    StagedCommitMessage(Box<StagedCommit>),
512    /// A Commit authored by this client that it got fanned out by the delivery
513    /// service, matching the group's pending commit.
514    ///
515    /// This is returned instead of
516    /// [`StagedCommitMessage`](Self::StagedCommitMessage) when the processed
517    /// Commit was created by this client and matches the group's pending commit.
518    /// Since this client already holds the corresponding pending commit, the
519    /// incoming Commit is not staged. To apply it, merge the pending commit
520    /// using
521    /// [`MlsGroup::merge_pending_commit()`](crate::group::mls_group::MlsGroup::merge_pending_commit()).
522    /// An own Commit that does not match the pending commit is instead returned
523    /// as a [`StagedCommitMessage`](Self::StagedCommitMessage) (if it has no
524    /// UpdatePath) or rejected (if it has an UpdatePath we cannot decrypt).
525    ///
526    /// The match against the pending commit is established by comparing the
527    /// confirmation tag of the incoming Commit against the one stored with the
528    /// pending commit. The message signature has already been verified, which
529    /// authenticates the Commit as ours, and a matching confirmation tag binds
530    /// the confirmed transcript hash of the new epoch. We do not otherwise
531    /// compare the contents of the incoming Commit against the pending commit,
532    /// and the incoming Commit's state is never adopted.
533    ///
534    /// This is only produced for Commits framed as
535    /// [`PublicMessage`](crate::framing::MlsMessageBodyIn::PublicMessage). A
536    /// Commit framed as a
537    /// [`PrivateMessage`](crate::framing::MlsMessageBodyIn::PrivateMessage)
538    /// cannot be decrypted by its own author and is instead rejected during
539    /// decryption.
540    OwnPendingCommit,
541    /// A Commit message covering AppDataUpdate proposals.
542    ///
543    /// The proposals carry diffs in an application-defined format, so the
544    /// commit cannot be staged before the application has interpreted them and
545    /// computed the resulting dictionary entries. Inspect the proposals via
546    /// [`UnresolvedAppDataCommit::app_data_update_proposals()`], compute the
547    /// updates with the help of
548    /// [`MlsGroup::app_data_dictionary_updater()`](crate::group::mls_group::MlsGroup::app_data_dictionary_updater)
549    /// and resume staging via
550    /// [`MlsGroup::stage_app_data_commit()`](crate::group::mls_group::MlsGroup::stage_app_data_commit).
551    #[cfg(feature = "extensions-draft")]
552    UnresolvedAppDataCommit(Box<UnresolvedAppDataCommit>),
553}
