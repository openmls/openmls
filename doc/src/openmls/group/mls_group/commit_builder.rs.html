54/// Per-commit virtual-clients state allocated by
55/// [`CommitBuilder::vc_emulation`] and consumed by `build`.
56///
57/// `vc_emulation` advances the own `LeafNode` operation ratchet by one
58/// generation and immediately persists the advanced tree, before the commit
59/// message exists. A builder that is discarded after the setter therefore
60/// burns a generation, as does a commit the DS rejects. That is harmless:
61/// sibling ratchets skip over a burned generation, retaining the skipped
62/// generation secrets inside their copy of the operation secret tree.
63#[cfg(feature = "virtual-clients-draft")]
64#[derive(Debug)]
65struct VcLoaded {
66    epoch_id: EpochId,
67    emulation_leaf_index: LeafNodeIndex,
68    epoch_encryption_key: EpochEncryptionKey,
69    emulation_ciphersuite: openmls_traits::types::Ciphersuite,
70    generation: u32,
71    operation_secret: OperationSecret,
72    /// The resolved `AppDataDictionary` produced by the leaf-configuration
73    /// pre-check in `vc_emulation`, carried to `build` so the VC
74    /// derivation-info injection preserves every other entry.
75    resolved_dictionary: AppDataDictionary,
76}

158/// The [`CommitBuilder`] is used to easily and dynamically build commit messages.
159/// It operates in a series of stages:
160///
161/// The [`Initial`] stage is used to populate the builder with proposals and other data using
162/// method calls on the builder that let the builder stay in the same stage.
163///
164/// The next stage is [`LoadedPsks`], and it signifies the stage after the builder loaded the the
165/// pre-shared keys for the PreSharedKey proposals in this commit.
166///
167/// Then comes the [`Complete`] stage, which denotes that all data has been validated. From this
168/// stage, the commit can be staged in the group, and the outgoing messages returned.
169///
170/// For example, to create a commit to a new Add proposal with a KeyPackage `key_package_to_add`
171/// that does not commit to the proposals in the proposal store, one could build the commit as
172/// follows:
173///
174/// ```rust,ignore
175/// let message_bundle: CommitMessageBundle = mls_group
176///   .commit_builder()
177///   .consume_proposal_store(false)
178///   .add_proposal(key_package_to_add)
179///   .load_psks(provider.storage())?
180///   .build(provider.rand(), provider.crypto(), signer, app_policy_proposals)?
181///   .stage_commit(provider)?;
182///
183/// let commit = message_bundle.commit();
184/// let welcome = message_bundle.welcome().expect("expected a welcome since there was an add");
185/// let group_info = message_bundle.welcome().expect("expected a group info since there was an add");
186/// ```
187///
188/// In this example `signer` is a reference to a [`Signer`] and `app_policy_proposals` is the
189/// application-defined policy for which proposals to accept, implemented by an
190/// `FnMut(&QueuedProposal) -> bool`.
191///
192/// See the [book] for another example.
193///
194/// [book]: https://book.openmls.tech/user_manual/add_members.html
195#[derive(Debug)]
196pub struct CommitBuilder<'a, T, G: BorrowMut<MlsGroup> = &'a mut MlsGroup> {
197    /// A mutable reference to the MlsGroup. This means that we hold an exclusive lock on the group
198    /// for the lifetime of this builder.
199    group: G,
200
201    /// The current stage
202    stage: T,
203
204    /// Virtual-clients material allocated by [`Self::vc_emulation`] and
205    /// consumed by `build`. Lives on the builder rather than on a stage
206    /// struct so the stage transitions can carry it through unchanged.
207    #[cfg(feature = "virtual-clients-draft")]
208    vc_loaded: Option<VcLoaded>,
209
210    pd: PhantomData<&'a ()>,
211}

213impl<'a, T, G: BorrowMut<MlsGroup>> CommitBuilder<'a, T, G> {
214    pub(crate) fn replace_stage<NextStage>(
215        self,
216        next_stage: NextStage,
217    ) -> (T, CommitBuilder<'a, NextStage, G>) {
218        self.map_stage(|prev_stage| (prev_stage, next_stage))
219    }
220
221    pub(crate) fn into_stage<NextStage>(
222        self,
223        next_stage: NextStage,
224    ) -> CommitBuilder<'a, NextStage, G> {
225        self.replace_stage(next_stage).1
226    }
227
228    fn take_stage(self) -> (T, CommitBuilder<'a, (), G>) {
229        self.replace_stage(())
230    }
231
232    fn map_stage<NextStage, Aux, F: FnOnce(T) -> (Aux, NextStage)>(
233        self,
234        f: F,
235    ) -> (Aux, CommitBuilder<'a, NextStage, G>) {
236        let Self {
237            group,
238            stage,
239            #[cfg(feature = "virtual-clients-draft")]
240            vc_loaded,
241            pd: PhantomData,
242        } = self;
243
244        let (aux, stage) = f(stage);
245
246        (
247            aux,
248            CommitBuilder {
249                group,
250                stage,
251                #[cfg(feature = "virtual-clients-draft")]
252                vc_loaded,
253                pd: PhantomData,
254            },
255        )
256    }
257
258    #[cfg(feature = "fork-resolution")]
259    pub(crate) fn stage(&self) -> &T {
260        &self.stage
261    }
262}

358    /// Opt this commit into the virtual-clients-draft sender flow.
359    ///
360    /// The application supplies the [`EpochId`] of an already-registered
361    /// emulation epoch (see
362    /// [`MlsGroup::register_vc_emulation_epoch`]). This method loads the
363    /// per-epoch operation secret tree and AEAD key from the storage
364    /// provider, validates the leaf configuration (see the preconditions
365    /// below), then advances the own `LeafNode` operation ratchet by one
366    /// generation and immediately persists the advanced tree. `build` then:
367    ///
368    /// - derives the path secret and the new leaf's encryption keypair
369    ///   from the allocated `OperationSecret`, so a sibling virtual
370    ///   client can rederive them on the receiver side, and
371    /// - embeds an encrypted `DerivationInfo` blob under
372    ///   [`VC_COMPONENT_ID`](crate::components::vc_derivation_info::VC_COMPONENT_ID)
373    ///   in the new leaf's `app_data_dictionary` extension.
374    ///
375    /// Because the ratchet advance is persisted here, a builder that is
376    /// discarded after this call burns a generation. The same happens when
377    /// the DS rejects the commit. That is harmless because sibling ratchets
378    /// skip over a burned generation, retaining the skipped generation
379    /// secrets inside their copy of the operation secret tree.
380    ///
381    /// The leaf configuration is validated against the
382    /// `leaf_node_parameters` set on the builder so far, so call this after
383    /// configuring the self-update leaf. The application must ensure the new
384    /// leaf:
385    ///
386    /// - lists [`ExtensionType::AppDataDictionary`](crate::extensions::ExtensionType::AppDataDictionary)
387    ///   in its `Capabilities.extensions`, and
388    /// - signals support for
389    ///   [`VC_COMPONENT_ID`](crate::components::vc_derivation_info::VC_COMPONENT_ID).
390    ///
391    /// If those preconditions are not met this method fails with
392    /// `VirtualClientsError::AppDataDictionaryNotSupported` or
393    /// `VirtualClientsError::VcComponentNotListed` (wrapped in
394    /// [`CreateCommitError::VirtualClientsError`]) before allocating a
395    /// generation, so no operation secret is burned in that case.
396    ///
397    /// Fails with `VirtualClientsError::MissingEmulationEpochState` or
398    /// `VirtualClientsError::MissingOperationTree` if the epoch was never
399    /// registered. Neither the state nor the tree is instantiated on the
400    /// fly, since that could diverge from a sibling virtual client's
401    /// already-advanced ratchets.
402    ///
403    /// Implies that a self-update takes place: the commit will always have
404    /// a path even if no other proposals are queued.
405    ///
406    /// [`MlsGroup::register_vc_emulation_epoch`]: crate::group::MlsGroup::register_vc_emulation_epoch
407    #[cfg(feature = "virtual-clients-draft")]
408    pub fn vc_emulation<Crypto: OpenMlsCrypto, Storage: StorageProvider>(
409        mut self,
410        crypto: &Crypto,
411        storage: &Storage,
412        epoch_id: EpochId,
413    ) -> Result<Self, CreateCommitError> {
414        let state: EmulationEpochState = storage
415            .vc_emulation_epoch_state(&epoch_id)
416            .map_err(|e| {
417                log::error!("vc: load emulation epoch state in vc_emulation failed: {e:?}");
418                CreateCommitError::VirtualClientsError(VirtualClientsError::StorageError)
419            })?
420            .ok_or(VirtualClientsError::MissingEmulationEpochState)?;
421        let mut operation_tree: OperationSecretTree = storage
422            .vc_operation_tree(&epoch_id)
423            .map_err(|e| {
424                log::error!("vc: load operation tree in vc_emulation failed: {e:?}");
425                CreateCommitError::VirtualClientsError(VirtualClientsError::StorageError)
426            })?
427            .ok_or(VirtualClientsError::MissingOperationTree)?;
428        let (emulation_leaf_index, epoch_encryption_key, emulation_ciphersuite) =
429            state.into_parts();
430
431        // Validate the leaf configuration before allocating a generation, so
432        // a deterministic precondition failure (the new leaf not declaring
433        // `AppDataDictionary` or not listing `VC_COMPONENT_ID`) does not burn
434        // an operation secret. Returns the resolved `AppDataDictionary`, which
435        // `build` reuses so the injection preserves the AppComponents entry
436        // across commits.
437        let own_leaf_index = self.group.borrow().own_leaf_index();
438        let is_external_commit = self.stage.external_commit_info.is_some();
439        let resolved_dictionary = check_vc_leaf_configuration(
440            &self.stage.leaf_node_parameters,
441            self.group.borrow(),
442            own_leaf_index,
443            is_external_commit,
444        )?;
445
446        // Update-path leaf-node derivations are the only operation type
447        // wired up so far. KeyPackage / Application will get their own
448        // allocation entry points when emitted. The operation context for
449        // LeafNode operations is the higher-level group's id.
450        let (generation, operation_secret) = operation_tree.next_operation_secret(
451            crypto,
452            emulation_ciphersuite,
453            &epoch_id,
454            emulation_leaf_index,
455            VirtualClientOperationType::LeafNode,
456            self.group.borrow().group_id().as_slice(),
457        )?;
458        // Persist the advanced tree right away, so the allocation can never
459        // be observed on the wire before it is persisted.
460        storage
461            .write_vc_operation_tree(&epoch_id, &operation_tree)
462            .map_err(|e| {
463                log::error!("vc: persist advanced operation tree failed: {e:?}");
464                CreateCommitError::VirtualClientsError(VirtualClientsError::StorageError)
465            })?;
466
467        self.vc_loaded = Some(VcLoaded {
468            epoch_id,
469            emulation_leaf_index,
470            epoch_encryption_key,
471            emulation_ciphersuite,
472            generation,
473            operation_secret,
474            resolved_dictionary,
475        });
476        Ok(self)
477    }

479    /// Loads the PSKs for the PskProposals marked for inclusion and moves on to the next phase.
480    pub fn load_psks<Storage: StorageProvider>(
481        self,
482        storage: &'a Storage,
483    ) -> Result<CommitBuilder<'a, LoadedPsks, G>, CreateCommitError> {
484        let psk_ids: Vec<_> = self
485            .stage
486            .own_proposals
487            .iter()
488            .chain(
489                self.group
490                    .borrow()
491                    .proposal_store()
492                    .proposals()
493                    .map(|queued_proposal| queued_proposal.proposal()),
494            )
495            .filter_map(|proposal| match proposal {
496                Proposal::PreSharedKey(psk_proposal) => Some(psk_proposal.clone().into_psk_id()),
497                _ => None,
498            })
499            .collect();
500
501        // Load the PSKs and make the PskIds owned.
502        let psks = load_psks(storage, &self.group.borrow().resumption_psk_store, &psk_ids)?
503            .into_iter()
504            .map(|(psk_id_ref, key)| (psk_id_ref.clone(), key))
505            .collect();
506
507        // Initialize GroupInfoConfig
508        let use_ratchet_tree_extension = self
509            .group
510            .borrow()
511            .configuration()
512            .use_ratchet_tree_extension;
513
514        let group_info_config = GroupInfoConfig {
515            use_ratchet_tree_extension,
516            create_group_info: use_ratchet_tree_extension,
517            other_extensions: vec![],
518        };
519
520        Ok(self
521            .map_stage(|stage| {
522                (
523                    (),
524                    LoadedPsks {
525                        own_proposals: stage.own_proposals,
526                        psks,
527                        force_self_update: stage.force_self_update,
528                        leaf_node_parameters: stage.leaf_node_parameters,
529                        consume_proposal_store: stage.consume_proposal_store,
530                        group_info_config,
531                        external_commit_info: stage.external_commit_info,
532                        #[cfg(feature = "extensions-draft")]
533                        app_data_dictionary_updates: None,
534                    },
535                )
536            })
537            .1)
538    }

559    /// Add the provided [`Extension`]s to the [`GroupInfo`].
560    ///
561    ///  Returns an error if a  [`RatchetTreeExtension`] or [`ExternalPubExtension`] is added
562    ///  directly here.
563    pub fn create_group_info_with_extensions(
564        mut self,
565        extensions: impl IntoIterator<Item = Extension>,
566    ) -> Result<Self, InvalidExtensionError> {
567        self.stage.group_info_config.create_group_info = true;
568        self.stage.group_info_config.other_extensions = extensions
569            .into_iter()
570            .map(|extension| {
571                if extension.as_ratchet_tree_extension().is_ok()
572                    || extension.as_external_pub_extension().is_ok()
573                {
574                    Err(InvalidExtensionError::CannotAddDirectlyToGroupInfo)
575                } else {
576                    Ok(extension)
577                }
578            })
579            .collect::<Result<Vec<_>, _>>()?;
580
581        Ok(self)
582    }
583    /// Validates the inputs and builds the commit. The last argument `f` is a function that lets
584    /// the caller filter the proposals that are considered for inclusion. This provides a way for
585    /// the application to enforce custom policies in the creation of commits.
586    pub fn build<S: Signer>(
587        self,
588        rand: &impl OpenMlsRand,
589        crypto: &impl OpenMlsCrypto,
590        signer: &S,
591        f: impl FnMut(&QueuedProposal) -> bool,
592    ) -> Result<CommitBuilder<'a, Complete, G>, CreateCommitError> {
593        self.build_internal(rand, crypto, signer, None::<NewSignerBundle<'_, S>>, f)
594    }

596    /// Just like `build`, this function validates the inputs and builds the
597    /// commit. The last argument `f` is a function that lets the caller filter
598    /// the proposals that are considered for inclusion. This provides a way for
599    /// the application to enforce custom policies in the creation of commits.
600    ///
601    /// In contrast to `build`, this function can be used to create commits that
602    /// rotate the own leaf node's signature key.
603    pub fn build_with_new_signer<S: Signer>(
604        self,
605        rand: &impl OpenMlsRand,
606        crypto: &impl OpenMlsCrypto,
607        old_signer: &impl Signer,
608        new_signer: NewSignerBundle<'_, S>,
609        f: impl FnMut(&QueuedProposal) -> bool,
610    ) -> Result<CommitBuilder<'a, Complete, G>, CreateCommitError> {
611        self.build_internal(rand, crypto, old_signer, Some(new_signer), f)
612    }

614    fn build_internal<S: Signer>(
615        self,
616        rand: &impl OpenMlsRand,
617        crypto: &impl OpenMlsCrypto,
618        old_signer: &impl Signer,
619        new_signer: Option<NewSignerBundle<'_, S>>,
620        f: impl FnMut(&QueuedProposal) -> bool,
621    ) -> Result<CommitBuilder<'a, Complete, G>, CreateCommitError> {
622        #[cfg_attr(not(feature = "virtual-clients-draft"), allow(unused_mut))]
623        let (mut cur_stage, mut builder) = self.take_stage();
624
625        // retrieve the config
626        let GroupInfoConfig {
627            create_group_info,
628            use_ratchet_tree_extension,
629            other_extensions,
630        } = cur_stage.group_info_config;
631
632        let group = builder.group.borrow();
633        let ciphersuite = group.ciphersuite();
634        let own_leaf_index = group.own_leaf_index();
635        let (sender, is_external_commit) = match cur_stage.external_commit_info {
636            None => (Sender::build_member(own_leaf_index), false),
637            Some(_) => (Sender::NewMemberCommit, true),
638        };
639        let psks = cur_stage.psks;
640
641        // put the pending and uniform proposals into a uniform shape,
642        // i.e. produce queued proposals from the own proposals
643        let own_proposals: Vec<_> = cur_stage
644            .own_proposals
645            .into_iter()
646            .map(|proposal| {
647                QueuedProposal::from_proposal_and_sender(ciphersuite, crypto, proposal, &sender)
648            })
649            .collect::<Result<_, _>>()?;
650
651        // prepare an iterator for the proposals in the group's proposal store, but only if the
652        // flag is set.
653        let group_proposal_store_queue = group
654            .pending_proposals()
655            .filter(|_| cur_stage.consume_proposal_store)
656            .cloned();
657
658        // prepare the iterator for the proposal validation and seletion function. That function
659        // assumes that "earlier in the list" means "older", so since our own proposals are
660        // newest, we have to put them last.
661        let proposal_queue = group_proposal_store_queue.chain(own_proposals).filter(f);
662
663        let (proposal_queue, contains_own_updates) =
664            ProposalQueue::filter_proposals(proposal_queue, group.own_leaf_index).map_err(|e| {
665                match e {
666                    ProposalQueueError::LibraryError(e) => e.into(),
667                    ProposalQueueError::ProposalNotFound => CreateCommitError::MissingProposal,
668                    ProposalQueueError::UpdateFromExternalSender
669                    | ProposalQueueError::SelfRemoveFromNonMember => {
670                        CreateCommitError::WrongProposalSenderType
671                    }
672                }
673            })?;
674
675        // Validate the proposals by doing the following checks:
676
677        // ValSem113: All Proposals: The proposal type must be supported by all
678        // members of the group
679        group
680            .public_group
681            .validate_proposal_type_support(&proposal_queue)?;
682        // ValSem101
683        // ValSem102
684        // ValSem103
685        // ValSem104
686        group
687            .public_group
688            .validate_key_uniqueness(&proposal_queue, None)?;
689        // ValSem105
690        group.public_group.validate_add_proposals(&proposal_queue)?;
691        // ValSem106
692        // ValSem109
693        group.public_group.validate_capabilities(&proposal_queue)?;
694        // ValSem107
695        // ValSem108
696        group
697            .public_group
698            .validate_remove_proposals(&proposal_queue)?;
699        group
700            .public_group
701            .validate_pre_shared_key_proposals(&proposal_queue)?;
702        // Validate update proposals for member commits
703        // ValSem110
704        // ValSem111
705        // ValSem112
706        group
707            .public_group
708            .validate_update_proposals(&proposal_queue, own_leaf_index)?;
709
710        // ValSem208
711        // ValSem209
712        group
713            .public_group
714            .validate_group_context_extensions_proposal(&proposal_queue)?;
715
716        #[cfg(feature = "extensions-draft")]
717        group
718            .public_group
719            .validate_app_data_update_proposals_and_group_context(&proposal_queue)?;
720
721        if is_external_commit {
722            group
723                .public_group
724                .validate_external_commit(&proposal_queue)?;
725        }
726
727        let proposal_reference_list = proposal_queue.commit_list();
728
729        // Make a copy of the public group to apply proposals safely
730        let mut diff = group.public_group.empty_diff();
731
732        // Apply proposals to tree
733        #[cfg(feature = "extensions-draft")]
734        let apply_proposals_values = diff.apply_proposals_with_app_data_updates(
735            &proposal_queue,
736            own_leaf_index,
737            cur_stage.app_data_dictionary_updates,
738        )?;
739        #[cfg(not(feature = "extensions-draft"))]
740        let apply_proposals_values = diff.apply_proposals(&proposal_queue, own_leaf_index)?;
741        if apply_proposals_values.self_removed && !is_external_commit {
742            return Err(CreateCommitError::CannotRemoveSelf);
743        }
744
745        // Virtual-clients sender hook: when the caller opted into VC for
746        // this commit, validate that the effective leaf is configured to
747        // accept the derivation-info entry (capabilities + AppComponents),
748        // then derive the path-secret + leaf-keypair override from the
749        // operation secret allocated in `vc_emulation` and embed the
750        // `DerivationInfo` blob in the leaf's `app_data_dictionary`
751        // extension.
752        #[cfg(feature = "virtual-clients-draft")]
753        let vc_loaded = builder.vc_loaded.take();
754        #[cfg(feature = "virtual-clients-draft")]
755        let own_update_override = if let Some(loaded) = vc_loaded.as_ref() {
756            // The leaf-configuration pre-check already ran in `vc_emulation`,
757            // before the generation was allocated. Reuse the resolved
758            // `AppDataDictionary` it produced so the inject step preserves
759            // every other entry, including the AppComponents entry that
760            // survives across multiple VC commits.
761            Some(apply_vc_emulation(
762                loaded,
763                &mut cur_stage.leaf_node_parameters,
764                loaded.resolved_dictionary.clone(),
765                crypto,
766                ciphersuite,
767            )?)
768        } else {
769            None
770        };
771        #[cfg(not(feature = "virtual-clients-draft"))]
772        let own_update_override: Option<crate::treesync::diff::OwnUpdatePathOverride> = None;
773
774        let path_computation_result =
775            // If path is needed, compute path values
776            if apply_proposals_values.path_required
777                || contains_own_updates
778                || cur_stage.force_self_update
779                || !cur_stage.leaf_node_parameters.is_empty()
780            {
781                let commit_type = match &cur_stage.external_commit_info {
782                    Some(ExternalCommitInfo { credential , ..}) => {
783                        CommitType::External(credential.clone())
784                    }
785                    None => CommitType::Member,
786                };
787                // Process the path. This includes updating the provisional
788                // group context by updating the epoch and computing the new
789                // tree hash.
790                if let Some(new_signer) = new_signer {
791                    if let Some(credential_with_key) =
792                        cur_stage.leaf_node_parameters.credential_with_key()
793                    {
794                        if credential_with_key != &new_signer.credential_with_key {
795                            return Err(CreateCommitError::InvalidLeafNodeParameters);
796                        }
797                    }
798                    cur_stage.leaf_node_parameters.set_credential_with_key(
799                        new_signer.credential_with_key,
800                    );
801
802                    diff.compute_path(
803                        rand,
804                        crypto,
805                        own_leaf_index,
806                        apply_proposals_values.exclusion_list(),
807                        &commit_type,
808                        &cur_stage.leaf_node_parameters,
809                        new_signer.signer,
810                        apply_proposals_values.extensions.clone(),
811                        own_update_override,
812                    )?
813                } else {
814                    diff.compute_path(
815                        rand,
816                        crypto,
817                        own_leaf_index,
818                        apply_proposals_values.exclusion_list(),
819                        &commit_type,
820                        &cur_stage.leaf_node_parameters,
821                        old_signer,
822                        apply_proposals_values.extensions.clone(),
823                        own_update_override,
824                    )?
825                }
826            } else {
827                // If path is not needed, update the group context and return
828                // empty path processing results
829                diff.update_group_context(crypto, apply_proposals_values.extensions.clone())?;
830                PathComputationResult::default()
831            };
832
833        let update_path_leaf_node = path_computation_result
834            .encrypted_path
835            .as_ref()
836            .map(|path| path.leaf_node().clone());
837
838        // Validate that the update path leaf node's capabilities
839        if let Some(ref leaf_node) = update_path_leaf_node {
840            // Check that all extension types in the group context that are valid in leaf nodes
841            // are supported by the leaf node
842            //
843            // This is currently not required by the RFC, likely by mistake:
844            // https://mailarchive.ietf.org/arch/msg/mls/k18P4FP7dfS2cBmP0kL6Uh50-ok/
845            if !diff
846                .group_context()
847                .extensions()
848                .iter()
849                .map(Extension::extension_type)
850                .all(|ext_type| leaf_node.supports_extension(&ext_type))
851            {
852                return Err(CreateCommitError::LeafNodeValidation(
853                    LeafNodeValidationError::UnsupportedExtensions,
854                ));
855            }
856
857            // Check that the leaf node supports everything listed in the required capabilities.
858            // https://validation.openmls.tech/#valn0103
859            if let Some(required_capabilities) =
860                diff.group_context().extensions().required_capabilities()
861            {
862                leaf_node
863                    .capabilities()
864                    .supports_required_capabilities(required_capabilities)?
865            }
866        }
867
868        // Create commit message
869        let commit = Commit {
870            proposals: proposal_reference_list,
871            path: path_computation_result.encrypted_path,
872        };
873
874        let (outgoing_aad, wire_format): (Vec<u8>, WireFormat) =
875            match &cur_stage.external_commit_info {
876                None => (
877                    group.outgoing_authenticated_data()?,
878                    group.outgoing_wire_format(),
879                ),
880                Some(ExternalCommitInfo { aad, .. }) => {
881                    // The spec requires the SafeAAD prefix even with zero items
882                    // when the target GroupContext has `safe_aad` present, so a
883                    // bare `aad` would be rejected by SafeAAD-aware receivers.
884                    #[cfg(feature = "extensions-draft")]
885                    let aad_bytes = if group.context().safe_aad_required() {
886                        crate::framing::safe_aad::assemble_authenticated_data(
887                            &crate::framing::SafeAad::empty(),
888                            aad,
889                        )
890                        .map_err(|_| LibraryError::custom("SafeAad serialization failed"))?
891                    } else {
892                        aad.clone()
893                    };
894                    #[cfg(not(feature = "extensions-draft"))]
895                    let aad_bytes = aad.clone();
896                    (aad_bytes, WireFormat::PublicMessage)
897                }
898            };
899        let framing_parameters = FramingParameters::new(&outgoing_aad, wire_format);
900
901        // Build AuthenticatedContent
902        let mut authenticated_content = AuthenticatedContent::commit(
903            framing_parameters,
904            sender,
905            commit,
906            group.public_group.group_context(),
907            old_signer,
908        )?;
909
910        // Update the confirmed transcript hash using the commit we just created.
911        diff.update_confirmed_transcript_hash(crypto, &authenticated_content)?;
912
913        let serialized_provisional_group_context = diff
914            .group_context()
915            .tls_serialize_detached()
916            .map_err(LibraryError::missing_bound_check)?;
917
918        let joiner_secret = JoinerSecret::new(
919            crypto,
920            ciphersuite,
921            path_computation_result.commit_secret,
922            group.group_epoch_secrets().init_secret(),
923            &serialized_provisional_group_context,
924        )
925        .map_err(LibraryError::unexpected_crypto_error)?;
926
927        // Prepare the PskSecret
928        let psk_secret = { PskSecret::new(crypto, ciphersuite, psks)? };
929
930        // Create key schedule
931        let mut key_schedule = KeySchedule::init(ciphersuite, crypto, &joiner_secret, psk_secret)?;
932
933        let serialized_provisional_group_context = diff
934            .group_context()
935            .tls_serialize_detached()
936            .map_err(LibraryError::missing_bound_check)?;
937
938        let welcome_secret = key_schedule
939            .welcome(crypto, ciphersuite)
940            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
941        key_schedule
942            .add_context(crypto, &serialized_provisional_group_context)
943            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
944        let EpochSecretsResult {
945            epoch_secrets: provisional_epoch_secrets,
946            #[cfg(feature = "extensions-draft")]
947            application_exporter,
948        } = key_schedule
949            .epoch_secrets(crypto, ciphersuite)
950            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
951
952        // Calculate the confirmation tag
953        let confirmation_tag = provisional_epoch_secrets
954            .confirmation_key()
955            .tag(
956                crypto,
957                ciphersuite,
958                diff.group_context().confirmed_transcript_hash(),
959            )
960            .map_err(LibraryError::unexpected_crypto_error)?;
961
962        // Set the confirmation tag
963        authenticated_content.set_confirmation_tag(confirmation_tag.clone());
964
965        diff.update_interim_transcript_hash(ciphersuite, crypto, confirmation_tag.clone())?;
966
967        // If there are invitations, we need to build a welcome
968        let needs_welcome = !apply_proposals_values.invitation_list.is_empty();
969
970        // We need a GroupInfo if we need to build a Welcome, or if
971        // `create_group_info` is set to `true`. If not overridden, `create_group_info`
972        // is set to the `use_ratchet_tree` flag in the group configuration.
973        let needs_group_info = needs_welcome || create_group_info;
974
975        let (welcome_option, group_info) = if !needs_group_info {
976            (None, None)
977        } else {
978            // Create the ratchet tree extension if necessary
979            let mut extensions_list = vec![];
980            if use_ratchet_tree_extension {
981                extensions_list.push(Extension::RatchetTree(RatchetTreeExtension::new(
982                    diff.export_ratchet_tree(),
983                )));
984            };
985            // Append rest of extensions
986            extensions_list.extend(other_extensions);
987
988            let mut extensions = Extensions::from_vec(extensions_list)?;
989
990            let welcome_option = needs_welcome
991                .then(|| -> Result<_, CreateCommitError> {
992                    let group_info_tbs = {
993                        GroupInfoTBS::new(
994                            diff.group_context().clone(),
995                            extensions.clone(),
996                            confirmation_tag.clone(),
997                            own_leaf_index,
998                        )?
999                    };
1000                    // Sign to-be-signed group info.
1001                    let group_info = group_info_tbs.sign(old_signer)?;
1002
1003                    // Encrypt GroupInfo object
1004                    let (welcome_key, welcome_nonce) = welcome_secret
1005                        .derive_welcome_key_nonce(crypto, ciphersuite)
1006                        .map_err(LibraryError::unexpected_crypto_error)?;
1007                    let encrypted_group_info = welcome_key
1008                        .aead_seal(
1009                            crypto,
1010                            group_info
1011                                .tls_serialize_detached()
1012                                .map_err(LibraryError::missing_bound_check)?
1013                                .as_slice(),
1014                            &[],
1015                            &welcome_nonce,
1016                        )
1017                        .map_err(LibraryError::unexpected_crypto_error)?;
1018
1019                    // Create group secrets for later use, so we can afterwards consume the
1020                    // `joiner_secret`.
1021                    let encrypted_secrets = diff.encrypt_group_secrets(
1022                        &joiner_secret,
1023                        apply_proposals_values.invitation_list,
1024                        path_computation_result.plain_path.as_deref(),
1025                        &apply_proposals_values.presharedkeys,
1026                        &encrypted_group_info,
1027                        crypto,
1028                        own_leaf_index,
1029                    )?;
1030
1031                    // Create welcome message
1032                    let welcome =
1033                        Welcome::new(ciphersuite, encrypted_secrets, encrypted_group_info);
1034                    Ok(welcome)
1035                })
1036                .transpose()?;
1037
1038            // Create the GroupInfo for export if needed. In contrast to the Welcome, this
1039            // group info contains the external public key extension.
1040            let exported_group_info = create_group_info
1041                .then(|| -> Result<_, CreateCommitError> {
1042                    let external_pub = provisional_epoch_secrets
1043                        .external_secret()
1044                        .derive_external_keypair(crypto, ciphersuite)
1045                        .map_err(LibraryError::unexpected_crypto_error)?
1046                        .public;
1047
1048                    let external_pub_extension =
1049                        Extension::ExternalPub(ExternalPubExtension::new(external_pub.into()));
1050                    extensions.add(external_pub_extension)?;
1051                    let group_info_tbs = {
1052                        GroupInfoTBS::new(
1053                            diff.group_context().clone(),
1054                            extensions,
1055                            confirmation_tag.clone(),
1056                            own_leaf_index,
1057                        )?
1058                    };
1059                    // Sign to-be-signed group info.
1060                    Ok(group_info_tbs.sign(old_signer)?)
1061                })
1062                .transpose()?;
1063
1064            (welcome_option, exported_group_info)
1065        };
1066
1067        let (provisional_group_epoch_secrets, provisional_message_secrets) =
1068            provisional_epoch_secrets.split_secrets(
1069                serialized_provisional_group_context,
1070                diff.tree_size(),
1071                own_leaf_index,
1072            );
1073
1074        #[cfg(feature = "extensions-draft")]
1075        let application_export_tree = ApplicationExportTree::new(application_exporter);
1076        let staged_commit_state = MemberStagedCommitState::new(
1077            provisional_group_epoch_secrets,
1078            provisional_message_secrets,
1079            diff.into_staged_diff(crypto, ciphersuite)?,
1080            path_computation_result.new_keypairs,
1081            // The committer is not allowed to include their own update
1082            // proposal, so there is no extra keypair to store here.
1083            None,
1084            update_path_leaf_node,
1085            #[cfg(feature = "extensions-draft")]
1086            application_export_tree,
1087            // The committer's `own_leaf_index` is already set to the new
1088            // leaf (in `build_group` for external commits, or unchanged for
1089            // regular commits), so `merge_commit` has nothing to overwrite.
1090            #[cfg(feature = "virtual-clients-draft")]
1091            None,
1092        );
1093        let staged_commit = StagedCommit::new(
1094            proposal_queue,
1095            StagedCommitState::GroupMember(Box::new(staged_commit_state)),
1096            #[cfg(feature = "virtual-clients-draft")]
1097            vc_loaded.as_ref().map(|loaded| loaded.epoch_id.clone()),
1098        );
1099
1100        Ok(builder.into_stage(Complete {
1101            result: CreateCommitResult {
1102                commit: authenticated_content,
1103                welcome_option,
1104                staged_commit,
1105                group_info: group_info.filter(|_| create_group_info),
1106            },
1107            original_wire_format_policy: cur_stage
1108                .external_commit_info
1109                .as_ref()
1110                .map(|info| info.wire_format_policy),
1111        }))
1112    }
1113
1114    /// Creates a new [`AppDataUpdates`] based on the current state of the
1115    /// [`AppDataDictionary`] of the group.
1116    ///
1117    /// [`AppDataDictionary`]: crate::extensions::AppDataDictionary
1118    #[cfg(feature = "extensions-draft")]
1119    pub fn app_data_dictionary_updater(&self) -> AppDataDictionaryUpdater<'_> {
1120        AppDataDictionaryUpdater::new(self.group.borrow().context().app_data_dict())
1121    }
1122
1123    /// Sets the [`AppDataUpdates`] that contain the changes made by the AppDataUpdate proposals
1124    #[cfg(feature = "extensions-draft")]
1125    pub fn with_app_data_dictionary_updates(
1126        &mut self,
1127        app_data_dictionary_updates: Option<AppDataUpdates>,
1128    ) {
1129        self.stage.app_data_dictionary_updates = app_data_dictionary_updates;
1130    }
1131
1132    /// Returns an iterator over all AppDataUpdate proposals in the proposal store of the group
1133    #[cfg(feature = "extensions-draft")]
1134    pub fn app_data_update_proposals(&self) -> impl Iterator<Item = &AppDataUpdateProposal> {
1135        let proposal_store_proposals = self
1136            .group
1137            .borrow()
1138            .proposal_store()
1139            .proposals()
1140            .map(|queued_proposal| queued_proposal.proposal());
1141
1142        // The proposals in the proposal store come earlier than the own_proposals.
1143        let all_proposals = proposal_store_proposals.chain(self.stage.own_proposals.iter());
1144
1145        // Filter for AppDataUpdate proposals
1146        let mut app_data_update_proposals: Vec<&AppDataUpdateProposal> = all_proposals
1147            .filter_map(|proposal| match proposal {
1148                Proposal::AppDataUpdate(proposal) => Some(proposal.as_ref()),
1149                _ => None,
1150            })
1151            .collect();
1152
1153        app_data_update_proposals.sort_by_key(|prop| prop.component_id());
1154        app_data_update_proposals.into_iter()
1155    }

1165    /// Stages the commit and returns the protocol messages.
1166    pub fn stage_commit<Provider: OpenMlsProvider>(
1167        self,
1168        provider: &Provider,
1169    ) -> Result<CommitMessageBundle, CommitBuilderStageError<Provider::StorageError>> {
1170        let Self {
1171            group,
1172            stage:
1173                Complete {
1174                    result: create_commit_result,
1175                    original_wire_format_policy: _,
1176                },
1177            ..
1178        } = self;
1179
1180        // Set the current group state to [`MlsGroupState::PendingCommit`],
1181        // storing the current [`StagedCommit`] from the commit results
1182        group.group_state = MlsGroupState::PendingCommit(Box::new(PendingCommitState::Member(
1183            create_commit_result.staged_commit,
1184        )));
1185
1186        provider
1187            .storage()
1188            .write_group_state(group.group_id(), &group.group_state)
1189            .map_err(CommitBuilderStageError::KeyStoreError)?;
1190
1191        group.reset_aad();
1192
1193        // Convert PublicMessage messages to MLSMessage and encrypt them if required by the
1194        // configuration.
1195        //
1196        // Note that this performs writes to the storage, so we should do that here, rather than
1197        // when working with the result.
1198        let mls_message = group.content_to_mls_message(create_commit_result.commit, provider)?;
1199
1200        Ok(CommitMessageBundle {
1201            version: group.version(),
1202            commit: mls_message,
1203            welcome: create_commit_result.welcome_option,
1204            group_info: create_commit_result.group_info,
1205        })
1206    }

1209/// Build the path-secret + leaf-keypair override from the
1210/// [`OperationSecret`] allocated in [`CommitBuilder::vc_emulation`] and
1211/// inject the corresponding `DerivationInfo` blob into
1212/// `leaf_node_parameters`'s `app_data_dictionary` extension.
1213///
1214/// The `DerivationInfoTbe` wrapping stays in the emulation epoch's
1215/// ciphersuite, while the operation secret is expanded under the
1216/// higher-level group ciphersuite to produce MLS path material for this
1217/// group. The generation was consumed and the advanced tree persisted when
1218/// `vc_emulation` was called, so this helper neither allocates nor
1219/// persists anything.
1220#[cfg(feature = "virtual-clients-draft")]
1221fn apply_vc_emulation(
1222    loaded: &VcLoaded,
1223    leaf_node_parameters: &mut LeafNodeParameters,
1224    resolved_dictionary: AppDataDictionary,
1225    crypto: &impl OpenMlsCrypto,
1226    group_ciphersuite: openmls_traits::types::Ciphersuite,
1227) -> Result<crate::treesync::diff::OwnUpdatePathOverride, CreateCommitError> {
1228    let emulation_ciphersuite = loaded.emulation_ciphersuite;
1229
1230    let path_secret = loaded
1231        .operation_secret
1232        .derive_path_generation_secret(crypto, group_ciphersuite)?
1233        .into();
1234    let leaf_encryption_keypair = loaded
1235        .operation_secret
1236        .derive_encryption_key_secret(crypto, group_ciphersuite)?
1237        .generate_encryption_key_pair(crypto, group_ciphersuite)?;
1238
1239    // Wrap the TBE under the per-epoch AEAD key, bound to the new leaf via
1240    // its serialized encryption key as derivation context.
1241    let leaf_encryption_key = leaf_encryption_keypair
1242        .public_key()
1243        .tls_serialize_detached()
1244        .map_err(VirtualClientsError::from)?;
1245    // leaf_node operations are not batched, so the TBE carries no
1246    // key_package_index: the select resolves to the empty `update`/`commit`
1247    // case and the field is absent on the wire.
1248    let tbe = DerivationInfoTbe::LeafNode {
1249        leaf_index: loaded.emulation_leaf_index,
1250        generation: loaded.generation,
1251    };
1252    let derivation_info = DerivationInfo::encrypt(
1253        crypto,
1254        emulation_ciphersuite,
1255        &loaded.epoch_encryption_key,
1256        loaded.epoch_id.clone(),
1257        &leaf_encryption_key,
1258        &tbe,
1259    )?;
1260    let derivation_info_bytes = derivation_info
1261        .tls_serialize_detached()
1262        .map_err(VirtualClientsError::from)?;
1263
1264    inject_vc_derivation_info(
1265        leaf_node_parameters,
1266        resolved_dictionary,
1267        derivation_info_bytes,
1268    )?;
1269
1270    Ok(crate::treesync::diff::OwnUpdatePathOverride {
1271        path_secret,
1272        leaf_encryption_keypair,
1273    })
1274}
1275
1276/// Verify that the effective leaf for this commit (= the merged view of
1277/// `leaf_node_parameters` over the existing leaf, or `leaf_node_parameters`
1278/// alone for external commits) declares `AppDataDictionary` and lists
1279/// [`VC_COMPONENT_ID`] in its `AppComponents` entry. Without both, the
1280/// receiver cannot reliably surface the derivation-info entry to the
1281/// virtual-clients consumer, so we reject the commit at build time.
1282///
1283/// Returns the resolved `AppDataDictionary` (caller's override merged
1284/// over the existing leaf's, with the caller winning on duplicate keys)
1285/// so subsequent injection of the VC derivation-info preserves the
1286/// AppComponents entry across commits.
1287#[cfg(feature = "virtual-clients-draft")]
1288fn check_vc_leaf_configuration(
1289    leaf_node_parameters: &LeafNodeParameters,
1290    group: &MlsGroup,
1291    own_leaf_index: LeafNodeIndex,
1292    is_external_commit: bool,
1293) -> Result<AppDataDictionary, CreateCommitError> {
1294    let current_leaf = if is_external_commit {
1295        None
1296    } else {
1297        Some(group.public_group().leaf(own_leaf_index).ok_or_else(|| {
1298            LibraryError::custom("Couldn't find own leaf for VC capability check")
1299        })?)
1300    };
1301
1302    crate::components::vc_derivation_info::resolve_vc_leaf_dictionary(
1303        leaf_node_parameters.capabilities(),
1304        leaf_node_parameters.extensions(),
1305        current_leaf,
1306    )
1307    .map_err(CreateCommitError::VirtualClientsError)
1308}
1309
1310/// Merge a virtual-clients derivation info blob into
1311/// `leaf_node_parameters.app_data_dictionary[VC_COMPONENT_ID]`,
1312/// preserving every other component id from `resolved_dictionary` and
1313/// every non-`AppDataDictionary` leaf-node extension the caller put in.
1314#[cfg(feature = "virtual-clients-draft")]
1315fn inject_vc_derivation_info(
1316    leaf_node_parameters: &mut LeafNodeParameters,
1317    resolved_dictionary: AppDataDictionary,
1318    derivation_info_bytes: Vec<u8>,
1319) -> Result<(), CreateCommitError> {
1320    let extensions = crate::components::vc_derivation_info::merge_vc_derivation_info(
1321        leaf_node_parameters.extensions(),
1322        resolved_dictionary,
1323        derivation_info_bytes,
1324    )?;
1325    leaf_node_parameters.set_extensions(extensions);
1326    Ok(())
1327}

1329/// Contains the messages that are produced by committing. The messages can be accessed individually
1330/// using getters or through the [`IntoIterator`] interface.
1331#[derive(Debug, Clone)]
1332pub struct CommitMessageBundle {
1333    version: ProtocolVersion,
1334    commit: MlsMessageOut,
1335    welcome: Option<Welcome>,
1336    group_info: Option<GroupInfo>,
1337}
1338
1339/// The result of a commit with an add proposal. This includes
1340/// - The Commit as an [`MlsMessageOut`]
1341/// - The [`Welcome`] as an [`MlsMessageOut`]
1342/// - Optionally a [`GroupInfo`] as an [`MlsMessageOut`]
1343pub struct WelcomeCommitMessages {
1344    /// The Commit as an [`MlsMessageOut`].
1345    pub commit: MlsMessageOut,
1346
1347    /// The [`Welcome`] as an [`MlsMessageOut`].
1348    pub welcome: MlsMessageOut,
1349
1350    /// Optionally a [`GroupInfo`] as an [`MlsMessageOut`].
1351    pub group_info: Option<MlsMessageOut>,
1352}
1353
1354impl TryFrom<CommitMessageBundle> for WelcomeCommitMessages {
1355    type Error = LibraryError;
1356
1357    fn try_from(value: CommitMessageBundle) -> Result<Self, Self::Error> {
1358        let (commit, welcome_opt, group_info) = value.into_messages();
1359        Ok(Self {
1360            commit,
1361            welcome: welcome_opt.ok_or(LibraryError::custom(
1362                "WelcomeCommitMessages must only be used with commits that produce a welcome.",
1363            ))?,
1364            group_info,
1365        })
1366    }
1367}

1386impl CommitMessageBundle {
1387    // borrowed getters
1388
1389    /// Gets the Commit messsage. For owned version, see [`Self::into_commit`].
1390    pub fn commit(&self) -> &MlsMessageOut {
1391        &self.commit
1392    }
1393
1394    /// Gets the Welcome messsage. Only [`Some`] if new clients have been added in the commit.
1395    /// For owned version, see [`Self::into_welcome`].
1396    pub fn welcome(&self) -> Option<&Welcome> {
1397        self.welcome.as_ref()
1398    }
1399
1400    /// Gets the Welcome messsage. Only [`Some`] if new clients have been added in the commit.
1401    /// Performs a copy of the Welcome. For owned version, see [`Self::into_welcome_msg`].
1402    pub fn to_welcome_msg(&self) -> Option<MlsMessageOut> {
1403        self.welcome
1404            .as_ref()
1405            .map(|welcome| MlsMessageOut::from_welcome(welcome.clone(), self.version))
1406    }
1407
1408    /// Gets the GroupInfo message. Only [`Some`] if new clients have been added or the group
1409    /// configuration has `use_ratchet_tree_extension` set.
1410    /// For owned version, see [`Self::into_group_info`].
1411    pub fn group_info(&self) -> Option<&GroupInfo> {
1412        self.group_info.as_ref()
1413    }
1414
1415    /// Gets all three messages, some of which optional. For owned version, see
1416    /// [`Self::into_contents`].
1417    pub fn contents(&self) -> (&MlsMessageOut, Option<&Welcome>, Option<&GroupInfo>) {
1418        (
1419            &self.commit,
1420            self.welcome.as_ref(),
1421            self.group_info.as_ref(),
1422        )
1423    }
1424
1425    // owned getters
1426    /// Gets the Commit messsage. This method consumes the [`CommitMessageBundle`]. For a borrowed
1427    /// version see [`Self::commit`].
1428    pub fn into_commit(self) -> MlsMessageOut {
1429        self.commit
1430    }
1431
1432    /// Gets the Welcome messsage. Only [`Some`] if new clients have been added in the commit.
1433    /// This method consumes the [`CommitMessageBundle`]. For a borrowed version see
1434    /// [`Self::welcome`].
1435    pub fn into_welcome(self) -> Option<Welcome> {
1436        self.welcome
1437    }
1438
1439    /// Gets the Welcome messsage. Only [`Some`] if new clients have been added in the commit.
1440    /// For a borrowed version, see [`Self::to_welcome_msg`].
1441    pub fn into_welcome_msg(self) -> Option<MlsMessageOut> {
1442        self.welcome
1443            .map(|welcome| MlsMessageOut::from_welcome(welcome, self.version))
1444    }
1445
1446    /// Gets the GroupInfo message. Only [`Some`] if new clients have been added or the group
1447    /// configuration has `use_ratchet_tree_extension` set.
1448    /// This method consumes the [`CommitMessageBundle`]. For a borrowed version see
1449    /// [`Self::group_info`].
1450    pub fn into_group_info(self) -> Option<GroupInfo> {
1451        self.group_info
1452    }
1453
1454    /// Gets the GroupInfo messsage. Only [`Some`] if new clients have been added in the commit.
1455    pub fn into_group_info_msg(self) -> Option<MlsMessageOut> {
1456        self.group_info.map(|group_info| group_info.into())
1457    }
1458
1459    /// Gets all three messages, some of which optional. This method consumes the
1460    /// [`CommitMessageBundle`]. For a borrowed version see [`Self::contents`].
1461    pub fn into_contents(self) -> (MlsMessageOut, Option<Welcome>, Option<GroupInfo>) {
1462        (self.commit, self.welcome, self.group_info)
1463    }
1464
1465    /// Gets all three messages, some of which optional, as [`MlsMessageOut`].
1466    /// This method consumes the [`CommitMessageBundle`].
1467    pub fn into_messages(self) -> (MlsMessageOut, Option<MlsMessageOut>, Option<MlsMessageOut>) {
1468        (
1469            self.commit,
1470            self.welcome
1471                .map(|welcome| MlsMessageOut::from_welcome(welcome, self.version)),
1472            self.group_info.map(|group_info| group_info.into()),
1473        )
1474    }
1475}

1477impl IntoIterator for CommitMessageBundle {
1478    type Item = MlsMessageOut;
1479
1480    type IntoIter = core::iter::Chain<
1481        core::iter::Chain<
1482            core::option::IntoIter<MlsMessageOut>,
1483            core::option::IntoIter<MlsMessageOut>,
1484        >,
1485        core::option::IntoIter<MlsMessageOut>,
1486    >;
1487
1488    fn into_iter(self) -> Self::IntoIter {
1489        let welcome = self.to_welcome_msg();
1490        let group_info = self.group_info.map(|group_info| group_info.into());
1491
1492        Some(self.commit)
1493            .into_iter()
1494            .chain(welcome)
1495            .chain(group_info)
1496    }
1497}
