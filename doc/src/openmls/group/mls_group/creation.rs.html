75    /// Join an existing group through an External Commit.
76    /// The resulting [`MlsGroup`] instance starts off with a pending
77    /// commit (the external commit, which adds this client to the group).
78    /// Merging this commit is necessary for this [`MlsGroup`] instance to
79    /// function properly, as, for example, this client is not yet part of the
80    /// tree. As a result, it is not possible to clear the pending commit. If
81    /// the external commit was rejected due to an epoch change, the
82    /// [`MlsGroup`] instance has to be discarded and a new one has to be
83    /// created using this function based on the latest `ratchet_tree` and
84    /// group info. For more information on the external init process,
85    /// please see Section 11.2.1 in the MLS specification.
86    ///
87    /// Note: If there is a group member in the group with the same identity as
88    /// us, this will create a remove proposal.
89    #[allow(clippy::too_many_arguments)]
90    #[deprecated(
91        since = "0.7.1",
92        note = "Use the `MlsGroup::external_commit_builder` instead."
93    )]
94    pub fn join_by_external_commit<Provider: OpenMlsProvider>(
95        provider: &Provider,
96        signer: &impl Signer,
97        ratchet_tree: Option<RatchetTreeIn>,
98        verifiable_group_info: VerifiableGroupInfo,
99        mls_group_config: &MlsGroupJoinConfig,
100        capabilities: Option<Capabilities>,
101        extensions: Option<Extensions<LeafNode>>,
102        aad: &[u8],
103        credential_with_key: CredentialWithKey,
104    ) -> Result<(Self, MlsMessageOut, Option<GroupInfo>), ExternalCommitError<Provider::StorageError>>
105    {
106        let leaf_node_parameters = LeafNodeParameters::builder()
107            .with_capabilities(capabilities.unwrap_or_default())
108            .with_extensions(extensions.unwrap_or_default())
109            .build();
110
111        let mut external_commit_builder = ExternalCommitBuilder::new()
112            .with_aad(aad.to_vec())
113            .with_config(mls_group_config.clone());
114
115        if let Some(ratchet_tree) = ratchet_tree {
116            external_commit_builder = external_commit_builder.with_ratchet_tree(ratchet_tree)
117        }
118
119        let (mls_group, commit_message_bundle) = external_commit_builder
120            .build_group(provider, verifiable_group_info, credential_with_key)?
121            .leaf_node_parameters(leaf_node_parameters)
122            .load_psks(provider.storage())
123            .map_err(|e| {
124                log::error!("Error loading PSKs for external commit: {e:?}");
125                LibraryError::custom("Error loading PSKs for external commit")
126            })?
127            .build(provider.rand(), provider.crypto(), signer, |_| true)?
128            .finalize(provider)?;
129
130        let (commit, _, group_info) = commit_message_bundle.into_contents();
131
132        Ok((mls_group, commit, group_info))
133    }

137    /// Creates a new processed [`Welcome`] message , which can be
138    /// inspected before creating a [`StagedWelcome`].
139    ///
140    /// This does not require a ratchet tree yet.
141    ///
142    /// [`Welcome`]: crate::messages::Welcome
143    pub fn new_from_welcome<Provider: OpenMlsProvider>(
144        provider: &Provider,
145        mls_group_config: &MlsGroupJoinConfig,
146        welcome: Welcome,
147    ) -> Result<Self, WelcomeError<Provider::StorageError>> {
148        let (resumption_psk_store, key_material) =
149            keys_for_welcome(mls_group_config, &welcome, provider)?;
150
151        let ciphersuite = welcome.ciphersuite();
152        let Some(egs) =
153            welcome.find_encrypted_group_secret(key_material.key_package_ref(provider.crypto())?)
154        else {
155            return Err(WelcomeError::JoinerSecretNotFound);
156        };
157
158        // This check seems to be superfluous from the perspective of the RFC, but still doesn't
159        // seem like a bad idea. There is no local KeyPackage to compare against on the
160        // virtual-client path, where the derived material is implicitly the welcome's ciphersuite.
161        if let Some(key_package_bundle) = key_material.key_package_bundle() {
162            if welcome.ciphersuite() != key_package_bundle.key_package().ciphersuite() {
163                let e = WelcomeError::CiphersuiteMismatch;
164                log::debug!("new_from_welcome {e:?}");
165                return Err(e);
166            }
167        }
168
169        let group_secrets = GroupSecrets::try_from_ciphertext(
170            key_material.init_private_key(),
171            egs.encrypted_group_secrets(),
172            welcome.encrypted_group_info(),
173            ciphersuite,
174            provider.crypto(),
175        )?;
176
177        // Validate PSKs
178        PreSharedKeyId::validate_in_welcome(&group_secrets.psks, ciphersuite)?;
179
180        let psk_secret = {
181            let psks = load_psks(
182                provider.storage(),
183                &resumption_psk_store,
184                &group_secrets.psks,
185            )?;
186
187            PskSecret::new(provider.crypto(), ciphersuite, psks)?
188        };
189
190        // prepare the key schedule
191        let mut key_schedule = KeySchedule::init(
192            ciphersuite,
193            provider.crypto(),
194            &group_secrets.joiner_secret,
195            psk_secret,
196        )?;
197
198        // derive the keys for decrypting the group info
199        let (welcome_key, welcome_nonce) = key_schedule
200            .welcome(provider.crypto(), ciphersuite)
201            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?
202            .derive_welcome_key_nonce(provider.crypto(), ciphersuite)
203            .map_err(LibraryError::unexpected_crypto_error)?;
204
205        let verifiable_group_info = VerifiableGroupInfo::try_from_ciphertext(
206            &welcome_key,
207            &welcome_nonce,
208            welcome.encrypted_group_info(),
209            &[],
210            provider.crypto(),
211        )?;
212
213        let serialized_group_context = verifiable_group_info
214            .group_context()
215            .tls_serialize_detached()
216            .map_err(LibraryError::missing_bound_check)?;
217
218        // TODO #751: Implement PSK
219        key_schedule.add_context(provider.crypto(), &serialized_group_context)?;
220
221        let epoch_secrets = key_schedule.epoch_secrets(provider.crypto(), ciphersuite)?;
222
223        // On the bundle path, check the required capabilities and the
224        // ciphersuite against the local KeyPackage. On the virtual-client path
225        // there is no local KeyPackage: these are checked in staging against
226        // the own tree leaf instead.
227        if let Some(key_package_bundle) = key_material.key_package_bundle() {
228            if let Some(required_capabilities) =
229                verifiable_group_info.extensions().required_capabilities()
230            {
231                // Also check that our key package actually supports the extensions.
232                // As per the spec, the sender must have checked this. But you never know.
233                key_package_bundle
234                    .key_package()
235                    .leaf_node()
236                    .capabilities()
237                    .supports_required_capabilities(required_capabilities)?;
238            }
239
240            // https://validation.openmls.tech/#valn1404
241            // Verify that the cipher_suite in the GroupInfo matches the cipher_suite in the
242            // KeyPackage.
243            if verifiable_group_info.ciphersuite() != key_package_bundle.key_package().ciphersuite()
244            {
245                let e = WelcomeError::CiphersuiteMismatch;
246                log::debug!("new_from_welcome {e:?}");
247                return Err(e);
248            }
249        }
250
251        Ok(Self {
252            mls_group_config: mls_group_config.clone(),
253            ciphersuite,
254            group_secrets,
255            epoch_secrets,
256            verifiable_group_info,
257            resumption_psk_store,
258            key_material,
259        })
260    }

291    /// Consume the `ProcessedWelcome` and combine it with the ratchet tree into
292    /// a `StagedWelcome`.
293    pub(crate) fn into_staged_welcome_inner<Provider: OpenMlsProvider>(
294        mut self,
295        provider: &Provider,
296        ratchet_tree: Option<RatchetTreeIn>,
297        validate_lifetimes: LeafNodeLifetimePolicy,
298        replace_old_group: bool,
299    ) -> Result<StagedWelcome, WelcomeError<Provider::StorageError>> {
300        // Check if we need to replace an old group
301        if !replace_old_group
302            && MlsGroup::load(provider.storage(), self.verifiable_group_info.group_id())
303                .map_err(WelcomeError::StorageError)?
304                .is_some()
305        {
306            return Err(WelcomeError::GroupAlreadyExists);
307        }
308
309        // Build the ratchet tree and group
310
311        // Set nodes either from the extension or from the `nodes_option`.
312        // If we got a ratchet tree extension in the welcome, we enable it for
313        // this group. Note that this is not strictly necessary. But there's
314        // currently no other mechanism to enable the extension.
315        let ratchet_tree = match self.verifiable_group_info.extensions().ratchet_tree() {
316            Some(extension) => extension.ratchet_tree().clone(),
317            None => match ratchet_tree {
318                Some(ratchet_tree) => ratchet_tree,
319                None => return Err(WelcomeError::MissingRatchetTree),
320            },
321        };
322
323        // Since there is currently only the external pub extension, there is no
324        // group info extension of interest here.
325        let (public_group, _group_info_extensions) = PublicGroup::from_ratchet_tree(
326            provider.crypto(),
327            ratchet_tree,
328            self.verifiable_group_info.clone(),
329            ProposalStore::new(),
330            validate_lifetimes,
331        )?;
332
333        // Find our own leaf in the tree. On the bundle path this is the leaf
334        // whose signature key matches the local KeyPackage. On the
335        // virtual-client path there is no local signature key, so the leaf is
336        // located by its derived encryption key and validated against the
337        // emulation epoch's derivation info.
338        let own_leaf_index = match &self.key_material {
339            WelcomeKeyMaterial::KeyPackage(key_package_bundle) => {
340                // Check that the leaf node of the added key package supports all extensions in
341                // the group context.
342                // https://validation.openmls.tech/#valn1415
343                let added_leaf_supports_all_group_context_extensions = public_group
344                    .group_context()
345                    .extensions()
346                    .iter()
347                    .all(|extension| {
348                        key_package_bundle
349                            .key_package
350                            .leaf_node()
351                            .supports_extension(&extension.extension_type())
352                    });
353                if !added_leaf_supports_all_group_context_extensions {
354                    return Err(WelcomeError::UnsupportedExtensions);
355                }
356
357                public_group
358                    .members()
359                    .find_map(|m| {
360                        if m.signature_key
361                            == key_package_bundle
362                                .key_package()
363                                .leaf_node()
364                                .signature_key()
365                                .as_slice()
366                        {
367                            Some(m.index)
368                        } else {
369                            None
370                        }
371                    })
372                    .ok_or(WelcomeError::PublicTreeError(
373                        PublicTreeError::MalformedTree,
374                    ))?
375            }
376            #[cfg(feature = "virtual-clients-draft")]
377            WelcomeKeyMaterial::VirtualClient(material) => {
378                find_and_validate_vc_own_leaf(provider, &public_group, material)?
379            }
380        };
381
382        struct KeyScheduleResult {
383            group_epoch_secrets: GroupEpochSecrets,
384            message_secrets: MessageSecrets,
385            #[cfg(feature = "extensions-draft")]
386            application_exporter: ApplicationExportSecret,
387        }
388        let KeyScheduleResult {
389            group_epoch_secrets,
390            message_secrets,
391            #[cfg(feature = "extensions-draft")]
392                application_exporter: application_export_secret,
393        } = {
394            let serialized_group_context = public_group
395                .group_context()
396                .tls_serialize_detached()
397                .map_err(LibraryError::missing_bound_check)?;
398
399            let EpochSecretsResult {
400                epoch_secrets,
401                #[cfg(feature = "extensions-draft")]
402                application_exporter,
403            } = self.epoch_secrets;
404
405            let (group_epoch_secrets, message_secrets) = epoch_secrets.split_secrets(
406                serialized_group_context,
407                public_group.tree_size(),
408                own_leaf_index,
409            );
410
411            KeyScheduleResult {
412                group_epoch_secrets,
413                message_secrets,
414                #[cfg(feature = "extensions-draft")]
415                application_exporter,
416            }
417        };
418
419        let confirmation_tag = message_secrets
420            .confirmation_key()
421            .tag(
422                provider.crypto(),
423                self.ciphersuite,
424                public_group.group_context().confirmed_transcript_hash(),
425            )
426            .map_err(LibraryError::unexpected_crypto_error)?;
427
428        // Verify confirmation tag
429        // https://validation.openmls.tech/#valn1411
430        if &confirmation_tag != public_group.confirmation_tag() {
431            log::error!("Confirmation tag mismatch");
432            log_crypto!(trace, "  Got:      {:x?}", confirmation_tag);
433            log_crypto!(trace, "  Expected: {:x?}", public_group.confirmation_tag());
434
435            // in some tests we need to be able to proceed despite the tag being wrong,
436            // e.g. to test whether a later validation check is performed correctly.
437            if !crate::skip_validation::is_disabled::confirmation_tag() {
438                return Err(WelcomeError::ConfirmationTagMismatch);
439            }
440        }
441
442        let message_secrets_store = MessageSecretsStore::new_with_secret(
443            &PastEpochDeletionPolicy::MaxEpochs(0),
444            message_secrets,
445        );
446
447        // Extract and store the resumption PSK for the current epoch.
448        let resumption_psk = group_epoch_secrets.resumption_psk();
449        self.resumption_psk_store
450            .add(public_group.group_context().epoch(), resumption_psk.clone());
451
452        let welcome_sender_index = self.verifiable_group_info.signer();
453        let path_keypairs = if let Some(path_secret) = self.group_secrets.path_secret {
454            let (path_keypairs, _commit_secret) = public_group
455                .derive_path_secrets(
456                    provider.crypto(),
457                    self.ciphersuite,
458                    path_secret,
459                    welcome_sender_index,
460                    own_leaf_index,
461                )
462                .map_err(|e| match e {
463                    DerivePathError::LibraryError(e) => e.into(),
464                    DerivePathError::PublicKeyMismatch => {
465                        WelcomeError::PublicTreeError(PublicTreeError::PublicKeyMismatch)
466                    }
467                })?;
468            Some(path_keypairs)
469        } else {
470            None
471        };
472
473        let staged_welcome = StagedWelcome {
474            mls_group_config: self.mls_group_config,
475            public_group,
476            group_epoch_secrets,
477            own_leaf_index,
478            message_secrets_store,
479            #[cfg(feature = "extensions-draft")]
480            application_export_secret,
481            resumption_psk_store: self.resumption_psk_store,
482            verifiable_group_info: self.verifiable_group_info,
483            key_material: self.key_material,
484            path_keypairs,
485        };
486
487        Ok(staged_welcome)
488    }

490    /// Exports a secret from the epoch of the group that is joined
491    /// using this [`ProcessedWelcome`].
492    /// Returns [`ExportSecretError::KeyLengthTooLong`] if the requested
493    /// key length is too long.
494    pub fn export_secret<CryptoProvider: OpenMlsCrypto>(
495        &self,
496        crypto: &CryptoProvider,
497        label: &str,
498        context: &[u8],
499        key_length: usize,
500    ) -> Result<Vec<u8>, ExportSecretError> {
501        if key_length > u16::MAX as usize {
502            log::error!("Got a key that is larger than u16::MAX");
503            return Err(ExportSecretError::KeyLengthTooLong);
504        }
505
506        Ok(self
507            .epoch_secrets
508            .epoch_secrets
509            .exporter_secret()
510            .derive_exported_secret(self.ciphersuite, crypto, label, context, key_length)
511            .map_err(LibraryError::unexpected_crypto_error)?)
512    }

587    /// Consumes the [`StagedWelcome`] and returns the respective [`MlsGroup`].
588    pub fn into_group<Provider: OpenMlsProvider>(
589        self,
590        provider: &Provider,
591    ) -> Result<MlsGroup, WelcomeError<Provider::StorageError>> {
592        // If we got a path secret, derive the path (which also checks if the
593        // public keys match) and store the derived keys in the key store.
594        let group_keypairs = if let Some(path_keypairs) = self.path_keypairs {
595            let mut keypairs = vec![self.key_material.encryption_key_pair()];
596            keypairs.extend_from_slice(&path_keypairs);
597            keypairs
598        } else {
599            vec![self.key_material.encryption_key_pair()]
600        };
601
602        #[cfg(feature = "extensions-draft")]
603        let application_export_tree = ApplicationExportTree::new(self.application_export_secret);
604
605        let past_epoch_deletion_policy = self.mls_group_config.past_epoch_deletion_policy().clone();
606
607        let mut mls_group = MlsGroup {
608            mls_group_config: self.mls_group_config,
609            own_leaf_nodes: vec![],
610            aad: vec![],
611            #[cfg(feature = "extensions-draft")]
612            safe_aad: crate::framing::SafeAad::empty(),
613            group_state: MlsGroupState::Operational,
614            public_group: self.public_group,
615            group_epoch_secrets: self.group_epoch_secrets,
616            own_leaf_index: self.own_leaf_index,
617            message_secrets_store: self.message_secrets_store,
618            resumption_psk_store: self.resumption_psk_store,
619            #[cfg(feature = "extensions-draft")]
620            application_export_tree: Some(application_export_tree),
621        };
622
623        mls_group
624            .store_epoch_keypairs(provider.storage(), group_keypairs.as_slice())
625            .map_err(WelcomeError::StorageError)?;
626        // resize the store
627        mls_group.resize_message_secrets_store(&past_epoch_deletion_policy);
628
629        mls_group
630            .store(provider.storage())
631            .map_err(WelcomeError::StorageError)?;
632
633        Ok(mls_group)
634    }

637    /// using this [`StagedWelcome`].
638    /// Returns [`ExportSecretError::KeyLengthTooLong`] if the requested
639    /// key length is too long.
640    pub fn export_secret<CryptoProvider: OpenMlsCrypto>(
641        &self,
642        crypto: &CryptoProvider,
643        label: &str,
644        context: &[u8],
645        key_length: usize,
646    ) -> Result<Vec<u8>, ExportSecretError> {
647        if key_length > u16::MAX as usize {
648            log::error!("Got a key that is larger than u16::MAX");
649            return Err(ExportSecretError::KeyLengthTooLong);
650        }
651
652        Ok(self
653            .group_epoch_secrets
654            .exporter_secret()
655            .derive_exported_secret(
656                self.group_context().ciphersuite(),
657                crypto,
658                label,
659                context,
660                key_length,
661            )
662            .map_err(LibraryError::unexpected_crypto_error)?)
663    }
664}

666fn keys_for_welcome<Provider: OpenMlsProvider>(
667    mls_group_config: &MlsGroupJoinConfig,
668    welcome: &Welcome,
669    provider: &Provider,
670) -> Result<
671    (ResumptionPskStore, WelcomeKeyMaterial),
672    WelcomeError<<Provider as OpenMlsProvider>::StorageError>,
673> {
674    let resumption_psk_store = ResumptionPskStore::new(mls_group_config.number_of_resumption_psks);
675
676    for egs in welcome.secrets() {
677        let hash_ref = egs.new_member();
678        if let Some(key_package_bundle) = provider
679            .storage()
680            .key_package(&hash_ref)
681            .map_err(WelcomeError::StorageError)?
682        {
683            let key_package_bundle: KeyPackageBundle = key_package_bundle;
684            if !key_package_bundle.key_package().last_resort() {
685                provider
686                    .storage()
687                    .delete_key_package(
688                        &key_package_bundle.key_package.hash_ref(provider.crypto())?,
689                    )
690                    .map_err(WelcomeError::StorageError)?;
691            } else {
692                log::debug!("Key package has last resort extension, not deleting");
693            }
694            return Ok((
695                resumption_psk_store,
696                WelcomeKeyMaterial::KeyPackage(Box::new(key_package_bundle)),
697            ));
698        }
699
700        #[cfg(feature = "virtual-clients-draft")]
701        if let Some(material) = vc_welcome_material(provider, welcome.ciphersuite(), &hash_ref)? {
702            return Ok((
703                resumption_psk_store,
704                WelcomeKeyMaterial::VirtualClient(material),
705            ));
706        }
707    }
708
709    Err(WelcomeError::NoMatchingKeyPackage)
710}

712/// Try to derive virtual-client welcome material for `hash_ref`. Returns
713/// `None` if no [`RetainedKeyPackageMaterial`] is stored for the ref, so the
714/// caller can keep scanning the welcome's encrypted group secrets.
715///
716/// On a match this reconstructs the per-KeyPackage seed pinned in the retained
717/// material at upload-processing time, deletes the consumed material, and
718/// derives the init and leaf-encryption keypairs from the seed under the
719/// Welcome's ciphersuite. It does not touch the operation tree: the batch
720/// generation was already consumed once when the upload was processed, and the
721/// seed is enough to reproduce the keys.
722///
723/// [`RetainedKeyPackageMaterial`]: crate::components::vc_derivation_info::RetainedKeyPackageMaterial
724#[cfg(feature = "virtual-clients-draft")]
725fn vc_welcome_material<Provider: OpenMlsProvider>(
726    provider: &Provider,
727    ciphersuite: Ciphersuite,
728    hash_ref: &crate::ciphersuite::hash_ref::KeyPackageRef,
729) -> Result<
730    Option<crate::components::vc_derivation_info::VcWelcomeMaterial>,
731    WelcomeError<<Provider as OpenMlsProvider>::StorageError>,
732> {
733    use crate::components::vc_derivation_info::{
734        RetainedKeyPackageMaterial, VcWelcomeMaterial, VirtualClientsError,
735    };
736
737    let storage = provider.storage();
738    let Some(material) = storage
739        .retained_key_package_material::<_, RetainedKeyPackageMaterial>(hash_ref)
740        .map_err(WelcomeError::StorageError)?
741    else {
742        return Ok(None);
743    };
744
745    let crypto = provider.crypto();
746    // The material is consumed once, mirroring the non-last-resort
747    // `delete_key_package` on the bundle path.
748    storage
749        .delete_retained_key_package_material(hash_ref)
750        .map_err(|e| {
751            log::error!("vc: delete retained key package material in welcome failed: {e:?}");
752            VirtualClientsError::StorageError
753        })?;
754
755    // The seed was derived under the emulation ciphersuite at
756    // upload-processing time. The init and leaf-encryption keys are derived
757    // from it under the KeyPackage's own (the welcome's) ciphersuite.
758    let init_key_pair = material
759        .key_package_seed_secret
760        .derive_init_key_secret(crypto, ciphersuite)?
761        .generate_init_key_pair(crypto, ciphersuite)?;
762    let encryption_keypair = material
763        .key_package_seed_secret
764        .derive_encryption_key_secret(crypto, ciphersuite)?
765        .generate_encryption_key_pair(crypto, ciphersuite)?;
766
767    Ok(Some(VcWelcomeMaterial {
768        key_package_ref: hash_ref.clone(),
769        epoch_id: material.epoch_id,
770        leaf_index: material.leaf_index,
771        generation: material.generation,
772        key_package_index: material.key_package_index,
773        init_private_key: init_key_pair.private,
774        encryption_keypair,
775    }))
776}

778/// Find and validate the virtual client's own leaf in the ratchet tree.
779///
780/// A virtual client has no signature key of its own, so the leaf is located by
781/// the derivation info the VC sender embedded under [`VC_COMPONENT_ID`]: the
782/// leaf whose cleartext `epoch_id` equals the material's and whose
783/// `encryption_key` equals the key derived in the first welcome stage. That
784/// leaf's encrypted [`DerivationInfoTbe`] is then decrypted with the emulation
785/// epoch state and its `leaf_index`, `generation`, and `key_package_index`
786/// must equal the material's.
787///
788/// [`VC_COMPONENT_ID`]: crate::components::vc_derivation_info::VC_COMPONENT_ID
789/// [`DerivationInfoTbe`]: crate::components::vc_derivation_info::DerivationInfoTbe
790#[cfg(feature = "virtual-clients-draft")]
791fn find_and_validate_vc_own_leaf<Provider: OpenMlsProvider>(
792    provider: &Provider,
793    public_group: &PublicGroup,
794    material: &crate::components::vc_derivation_info::VcWelcomeMaterial,
795) -> Result<LeafNodeIndex, WelcomeError<<Provider as OpenMlsProvider>::StorageError>> {
796    use tls_codec::{DeserializeBytes as _, Serialize as _};
797
798    use crate::components::vc_derivation_info::{
799        DerivationInfo, DerivationInfoTbe, EmulationEpochState, VirtualClientOperationType,
800        VirtualClientsError, VC_COMPONENT_ID,
801    };
802
803    let crypto = provider.crypto();
804    let derived_encryption_key = material.encryption_keypair.public_key().as_slice().to_vec();
805
806    let own_index = public_group
807        .members()
808        .find(|m| m.encryption_key == derived_encryption_key)
809        .map(|m| m.index)
810        .ok_or(WelcomeError::PublicTreeError(
811            PublicTreeError::MalformedTree,
812        ))?;
813
814    let own_leaf = public_group
815        .leaf(own_index)
816        .ok_or(WelcomeError::PublicTreeError(
817            PublicTreeError::MalformedTree,
818        ))?;
819
820    let derivation_info_bytes = own_leaf
821        .extensions()
822        .app_data_dictionary()
823        .and_then(|dict| dict.dictionary().get(&VC_COMPONENT_ID))
824        .ok_or(VirtualClientsError::VcComponentNotListed)?;
825    let derivation_info = DerivationInfo::tls_deserialize_exact_bytes(derivation_info_bytes)
826        .map_err(|e| {
827            log::error!("vc: welcome leaf derivation info deserialize failed: {e:?}");
828            VirtualClientsError::DerivationInfoMalformed
829        })?;
830    if derivation_info.epoch_id() != &material.epoch_id {
831        log::error!("vc: welcome leaf epoch id does not match the retained material");
832        return Err(VirtualClientsError::DerivationInfoMalformed.into());
833    }
834
835    let state: EmulationEpochState = provider
836        .storage()
837        .vc_emulation_epoch_state(&material.epoch_id)
838        .map_err(|e| {
839            log::error!("vc: load emulation epoch state in welcome staging failed: {e:?}");
840            VirtualClientsError::StorageError
841        })?
842        .ok_or(VirtualClientsError::MissingEmulationEpochState)?;
843    let (_state_leaf_index, epoch_encryption_key, emulation_ciphersuite) = state.into_parts();
844
845    let leaf_encryption_key = own_leaf
846        .encryption_key()
847        .tls_serialize_detached()
848        .map_err(VirtualClientsError::from)?;
849    // A KeyPackage leaf carries the `KeyPackage` variant of the tagless
850    // select, so it decodes with `key_package_index` present.
851    let tbe = derivation_info.decrypt(
852        crypto,
853        emulation_ciphersuite,
854        &epoch_encryption_key,
855        &leaf_encryption_key,
856        VirtualClientOperationType::KeyPackage,
857    )?;
858
859    let DerivationInfoTbe::KeyPackage {
860        leaf_index,
861        generation,
862        key_package_index,
863    } = tbe
864    else {
865        log::error!("vc: welcome leaf derivation info is not a key-package variant");
866        return Err(VirtualClientsError::DerivationInfoMalformed.into());
867    };
868    if leaf_index != material.leaf_index
869        || generation != material.generation
870        || key_package_index != material.key_package_index
871    {
872        log::error!("vc: welcome leaf derivation info does not match the retained material");
873        return Err(VirtualClientsError::DerivationInfoMalformed.into());
874    }
875
876    Ok(own_index)
877}

879#[cfg(feature = "virtual-clients-draft")]
880impl MlsGroup {
881    /// Bootstrap a virtual client's sibling emulator client into a higher-level
882    /// group by processing another sibling's external commit, when this client
883    /// is not yet a member of that group.
884    ///
885    /// The first emulator client joined the higher-level group via an external
886    /// commit. A second emulator client (this one), sharing the same emulation
887    /// epoch (`epoch_id`), reconstructs the resulting group state from that
888    /// commit. It cannot decapsulate the external init secret from the previous
889    /// epoch's `external_secret` (it never held it), so it uses the external
890    /// init secret the committing sibling carried in the commit's
891    /// `DerivationInfoTBE` (mls-virtual-clients draft), and recreates
892    /// the commit path from the shared operation secret tree.
893    ///
894    /// `verifiable_group_info` and `ratchet_tree` describe the higher-level
895    /// group at the epoch *before* the external commit (the ratchet tree may
896    /// instead travel in the GroupInfo's `ratchet_tree` extension).
897    /// `external_commit` is the sibling's external commit. On success the
898    /// returned group sits at the epoch the commit installs, with this client
899    /// on the shared virtual-client leaf.
900    pub fn vc_join_via_sibling_external_commit<Provider: OpenMlsProvider>(
901        provider: &Provider,
902        join_config: &MlsGroupJoinConfig,
903        verifiable_group_info: VerifiableGroupInfo,
904        ratchet_tree: Option<RatchetTreeIn>,
905        external_commit: impl Into<crate::framing::ProtocolMessage>,
906        epoch_id: crate::components::vc_derivation_info::EpochId,
907    ) -> Result<MlsGroup, crate::group::errors::VcExternalCommitJoinError<Provider::StorageError>>
908    {
909        use crate::{
910            framing::Sender,
911            group::config::PastEpochDeletionPolicy,
912            group::errors::{ProcessMessageError, VcExternalCommitJoinError as Error},
913            group::public_group::PublicGroup,
914            prelude::mls_content::FramedContentBody,
915            schedule::{EpochSecrets, InitSecret},
916        };
917
918        // Resolve the prior-epoch ratchet tree (from the GroupInfo extension or
919        // the argument) and rebuild the prior-epoch public group.
920        let ratchet_tree = match verifiable_group_info.extensions().ratchet_tree() {
921            Some(extension) => extension.ratchet_tree().clone(),
922            None => ratchet_tree.ok_or(Error::MissingRatchetTree)?,
923        };
924        let (public_group, _group_info) = PublicGroup::from_ratchet_tree(
925            provider.crypto(),
926            ratchet_tree,
927            verifiable_group_info,
928            ProposalStore::new(),
929            LeafNodeLifetimePolicy::default(),
930        )?;
931
932        // Assemble a transient group at the prior epoch. Its epoch secrets are
933        // never used cryptographically here: the external commit carries the
934        // external init secret and the operation secret tree supplies the path,
935        // so a random init-secret stub is sufficient. The own leaf index is the
936        // leftmost free index, where the committing sibling installs the shared
937        // virtual-client leaf.
938        let ciphersuite = public_group.ciphersuite();
939        let serialized_group_context = public_group
940            .group_context()
941            .tls_serialize_detached()
942            .map_err(LibraryError::missing_bound_check)?;
943        let own_leaf_index = public_group.leftmost_free_index(std::iter::empty())?;
944        let init_secret = InitSecret::random(ciphersuite, provider.rand())
945            .map_err(LibraryError::unexpected_crypto_error)?;
946        let epoch_secrets =
947            EpochSecrets::with_init_secret(provider.crypto(), ciphersuite, init_secret)
948                .map_err(LibraryError::unexpected_crypto_error)?;
949        let (group_epoch_secrets, message_secrets) = epoch_secrets.split_secrets(
950            serialized_group_context,
951            public_group.tree_size(),
952            LeafNodeIndex::new(0u32),
953        );
954        // Do not retain the synthetic prior-epoch secrets used only to stage
955        // this external commit.
956        let message_secrets_store = MessageSecretsStore::new_with_secret(
957            &PastEpochDeletionPolicy::MaxEpochs(0),
958            message_secrets,
959        );
960        let mut group = MlsGroup {
961            mls_group_config: join_config.clone(),
962            own_leaf_nodes: vec![],
963            aad: vec![],
964            #[cfg(feature = "extensions-draft")]
965            safe_aad: crate::framing::SafeAad::empty(),
966            group_state: MlsGroupState::Operational,
967            public_group,
968            group_epoch_secrets,
969            own_leaf_index,
970            message_secrets_store,
971            resumption_psk_store: ResumptionPskStore::new(join_config.number_of_resumption_psks),
972            #[cfg(feature = "extensions-draft")]
973            application_export_tree: None,
974        };
975
976        // Parse and verify the external commit against the prior-epoch group.
977        let unverified = group.unprotect_message(provider, external_commit)?;
978        let verified = unverified
979            .verify(group.ciphersuite(), provider.crypto(), group.version())
980            .map_err(ProcessMessageError::from)?;
981        if !matches!(verified.content.sender(), Sender::NewMemberCommit) {
982            return Err(Error::NotAnExternalCommit);
983        }
984        let content = verified.content;
985        let FramedContentBody::Commit(commit) = content.content() else {
986            return Err(Error::NotAnExternalCommit);
987        };
988
989        // Recover the sibling-VC commit material (operation secret + emulation
990        // epoch id + carried external init secret) from the leaf's derivation
991        // info, then stage and merge the commit through the sibling-VC path.
992        let material = group
993            .load_vc_commit_material(provider, commit)?
994            .ok_or(Error::MissingDerivationInfo)?;
995        if material.epoch_id != epoch_id {
996            return Err(Error::EpochIdMismatch);
997        }
998        let staged = group.stage_commit(&content, vec![], vec![], provider, Some(material))?;
999        group.merge_staged_commit(provider, staged)?;
1000        group.resize_message_secrets_store(join_config.past_epoch_deletion_policy());
1001        group
1002            .store(provider.storage())
1003            .map_err(Error::StorageError)?;
1004        Ok(group)
1005    }
1006
1007    /// Bootstrap a virtual client's sibling emulator client into a higher-level
1008    /// group the virtual client created, when this client is not yet a member.
1009    ///
1010    /// The creator emulator client built the group with
1011    /// [`MlsGroupBuilder::vc_emulation`], its `key_package`-sourced leaf key
1012    /// material derived from a `key_package` operation secret. Sharing the same
1013    /// emulation epoch (`epoch_id`), this client reconstructs the epoch-0 state:
1014    /// it verifies the GroupInfo and single-leaf ratchet tree (which may instead
1015    /// travel in the GroupInfo's `ratchet_tree` extension), rederives the creator
1016    /// leaf's key material from the shared operation secret tree, and derives the
1017    /// same epoch-0 `epoch_secret` from the creator's KeyPackage seed. No secret
1018    /// travels on the wire. On success the returned group sits at epoch 0 with
1019    /// this client on the shared virtual-client leaf (index 0).
1020    ///
1021    /// [`MlsGroupBuilder::vc_emulation`]: crate::group::MlsGroupBuilder::vc_emulation
1022    pub fn vc_join_at_creation<Provider: OpenMlsProvider>(
1023        provider: &Provider,
1024        join_config: &MlsGroupJoinConfig,
1025        verifiable_group_info: VerifiableGroupInfo,
1026        ratchet_tree: Option<RatchetTreeIn>,
1027        epoch_id: crate::components::vc_derivation_info::EpochId,
1028    ) -> Result<MlsGroup, crate::group::errors::VcGroupCreationJoinError<Provider::StorageError>>
1029    {
1030        use tls_codec::{DeserializeBytes as _, Serialize as _};
1031
1032        use crate::{
1033            components::vc_derivation_info::{
1034                load_vc_epoch_state_and_tree, DerivationInfo, DerivationInfoTbe,
1035                VirtualClientOperationType, VirtualClientsError, VC_COMPONENT_ID,
1036            },
1037            group::errors::VcGroupCreationJoinError as Error,
1038            group::public_group::PublicGroup,
1039            schedule::EpochSecrets,
1040            treesync::node::leaf_node::LeafNodeSource,
1041        };
1042
1043        // Resolve the ratchet tree (from the GroupInfo extension or the
1044        // argument) and verify the GroupInfo and tree.
1045        let ratchet_tree = match verifiable_group_info.extensions().ratchet_tree() {
1046            Some(extension) => extension.ratchet_tree().clone(),
1047            None => ratchet_tree.ok_or(Error::MissingRatchetTree)?,
1048        };
1049        let (public_group, group_info) = PublicGroup::from_ratchet_tree(
1050            provider.crypto(),
1051            ratchet_tree,
1052            verifiable_group_info,
1053            ProposalStore::new(),
1054            LeafNodeLifetimePolicy::default(),
1055        )?;
1056        let ciphersuite = public_group.ciphersuite();
1057
1058        // The created group consists of exactly the creator's leaf at index 0.
1059        if public_group.members().count() != 1 {
1060            return Err(Error::NotASingleLeafTree);
1061        }
1062        let creator_index = LeafNodeIndex::new(0);
1063        let creator_leaf = public_group
1064            .leaf(creator_index)
1065            .ok_or(Error::NotASingleLeafTree)?;
1066
1067        // A virtual client's group-creation leaf is key_package-sourced.
1068        let LeafNodeSource::KeyPackage(_) = creator_leaf.leaf_node_source() else {
1069            return Err(Error::CreatorLeafNotKeyPackageSourced);
1070        };
1071
1072        // Read the creator leaf's derivation info and check the emulation epoch.
1073        let derivation_info_bytes = creator_leaf
1074            .extensions()
1075            .app_data_dictionary()
1076            .and_then(|dict| dict.dictionary().get(&VC_COMPONENT_ID))
1077            .ok_or(Error::MissingDerivationInfo)?;
1078        let derivation_info = DerivationInfo::tls_deserialize_exact_bytes(derivation_info_bytes)
1079            .map_err(|_| VirtualClientsError::DerivationInfoMalformed)?;
1080        if derivation_info.epoch_id() != &epoch_id {
1081            return Err(Error::EpochIdMismatch);
1082        }
1083
1084        // Load the emulation epoch state and operation tree.
1085        let (state, mut operation_tree) = load_vc_epoch_state_and_tree(provider, &epoch_id)?;
1086        let (_leaf_index, epoch_encryption_key, emulation_ciphersuite) = state.into_parts();
1087
1088        // Decrypt the derivation info (KeyPackage case) and read the batch index
1089        // the creator's seed was derived under.
1090        let leaf_encryption_key = creator_leaf
1091            .encryption_key()
1092            .tls_serialize_detached()
1093            .map_err(VirtualClientsError::from)?;
1094        let tbe = derivation_info.decrypt(
1095            provider.crypto(),
1096            emulation_ciphersuite,
1097            &epoch_encryption_key,
1098            &leaf_encryption_key,
1099            VirtualClientOperationType::KeyPackage,
1100        )?;
1101        let DerivationInfoTbe::KeyPackage {
1102            leaf_index,
1103            generation,
1104            key_package_index,
1105        } = tbe
1106        else {
1107            // Decrypting with the KeyPackage operation type always yields the
1108            // KeyPackage variant.
1109            return Err(LibraryError::custom("unexpected derivation info variant").into());
1110        };
1111
1112        // Rederive the creator's KeyPackage seed positionally from the shared
1113        // operation tree, then the creator leaf's encryption keypair from that
1114        // seed. The `key_package` operation context is empty. The advanced tree
1115        // is persisted only after the confirmation tag verifies, so a corrupted
1116        // GroupInfo naming the genuine creator leaf does not burn this sibling's
1117        // generation.
1118        let operation_secret = operation_tree.derive_operation_secret(
1119            provider.crypto(),
1120            emulation_ciphersuite,
1121            &epoch_id,
1122            leaf_index,
1123            VirtualClientOperationType::KeyPackage,
1124            generation,
1125            b"",
1126        )?;
1127        let key_package_seed = operation_secret.derive_key_package_seed_secret(
1128            provider.crypto(),
1129            emulation_ciphersuite,
1130            key_package_index,
1131        )?;
1132        let leaf_keypair = key_package_seed
1133            .derive_encryption_key_secret(provider.crypto(), ciphersuite)?
1134            .generate_encryption_key_pair(provider.crypto(), ciphersuite)?;
1135        if leaf_keypair.public_key() != creator_leaf.encryption_key() {
1136            return Err(Error::LeafKeyMismatch);
1137        }
1138
1139        // Derive epoch-0 secrets from the same KeyPackage seed and verify the
1140        // GroupInfo's confirmation tag against them. A mismatch means the
1141        // reconstruction did not reproduce the creator's epoch secrets.
1142        let serialized_group_context = public_group
1143            .group_context()
1144            .tls_serialize_detached()
1145            .map_err(LibraryError::missing_bound_check)?;
1146        let epoch_secret =
1147            key_package_seed.derive_group_creation_secret(provider.crypto(), ciphersuite)?;
1148        let epoch_secrets =
1149            EpochSecrets::from_epoch_secret(provider.crypto(), ciphersuite, epoch_secret)
1150                .map_err(LibraryError::unexpected_crypto_error)?;
1151        let (group_epoch_secrets, message_secrets) = epoch_secrets.split_secrets(
1152            serialized_group_context,
1153            public_group.tree_size(),
1154            creator_index,
1155        );
1156        let expected_confirmation_tag = message_secrets
1157            .confirmation_key()
1158            .tag(
1159                provider.crypto(),
1160                ciphersuite,
1161                public_group.group_context().confirmed_transcript_hash(),
1162            )
1163            .map_err(LibraryError::unexpected_crypto_error)?;
1164        if &expected_confirmation_tag != group_info.confirmation_tag() {
1165            return Err(Error::ConfirmationTagMismatch);
1166        }
1167
1168        // The reconstruction reproduced the creator's secrets, so this is a
1169        // genuine sibling-created group: persist the advanced operation tree.
1170        provider
1171            .storage()
1172            .write_vc_operation_tree(&epoch_id, &operation_tree)
1173            .map_err(Error::StorageError)?;
1174
1175        let message_secrets_store = MessageSecretsStore::new_with_secret(
1176            join_config.past_epoch_deletion_policy(),
1177            message_secrets,
1178        );
1179        let mut resumption_psk_store =
1180            ResumptionPskStore::new(join_config.number_of_resumption_psks);
1181        resumption_psk_store.add(
1182            public_group.group_context().epoch(),
1183            group_epoch_secrets.resumption_psk().clone(),
1184        );
1185
1186        // Bind epoch 0 of the created group to the emulation epoch so later VC
1187        // operations in this group resolve the right emulation state. Written
1188        // before the group itself, so an error between the writes cannot leave a
1189        // loadable group without a binding (a bound group is required for the
1190        // reuse-guard MUST).
1191        let mut bindings: crate::components::vc_derivation_info::VcEmulationBindings = provider
1192            .storage()
1193            .vc_emulation_bindings(public_group.group_id())
1194            .map_err(Error::StorageError)?
1195            .unwrap_or_default();
1196        let max_entries = message_secrets_store.max_epochs.saturating_add(1);
1197        bindings.insert(public_group.group_context().epoch(), epoch_id, max_entries);
1198        provider
1199            .storage()
1200            .write_vc_emulation_bindings(public_group.group_id(), &bindings)
1201            .map_err(Error::StorageError)?;
1202
1203        let mls_group = MlsGroup {
1204            mls_group_config: join_config.clone(),
1205            own_leaf_nodes: vec![],
1206            aad: vec![],
1207            #[cfg(feature = "extensions-draft")]
1208            safe_aad: crate::framing::SafeAad::empty(),
1209            group_state: MlsGroupState::Operational,
1210            public_group,
1211            group_epoch_secrets,
1212            own_leaf_index: creator_index,
1213            message_secrets_store,
1214            resumption_psk_store,
1215            #[cfg(feature = "extensions-draft")]
1216            application_export_tree: None,
1217        };
1218        mls_group
1219            .store(provider.storage())
1220            .map_err(Error::StorageError)?;
1221        mls_group
1222            .store_epoch_keypairs(provider.storage(), &[leaf_keypair])
1223            .map_err(Error::StorageError)?;
1224
1225        Ok(mls_group)
1226    }
1227}

1229/// Verify or skip the validation of leaf node lifetimes in the ratchet tree
1230/// when joining a group.
1231#[derive(Default, Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize)]
1232pub enum LeafNodeLifetimePolicy {
1233    /// Verify the lifetime of leaf nodes in the ratchet tree.
1234    ///
1235    /// **NOTE:** Only leaf nodes that have never been updated have a lifetime.
1236    #[default]
1237    Verify,
1238
1239    /// Skip the verification of the lifeimte in leaf nodes in the ratchet tree.
1240    Skip,
1241}

1243/// Builder for joining a group.
1244///
1245/// Create this with [`StagedWelcome::build_from_welcome`].
1246pub struct JoinBuilder<'a, Provider: OpenMlsProvider> {
1247    provider: &'a Provider,
1248    processed_welcome: ProcessedWelcome,
1249    ratchet_tree: Option<RatchetTreeIn>,
1250    validate_lifetimes: LeafNodeLifetimePolicy,
1251    replace_old_group: bool,
1252}
1253
1254impl<'a, Provider: OpenMlsProvider> JoinBuilder<'a, Provider> {
1255    /// Create a new builder for the [`JoinBuilder`].
1256    pub fn new(provider: &'a Provider, processed_welcome: ProcessedWelcome) -> Self {
1257        Self {
1258            provider,
1259            processed_welcome,
1260            ratchet_tree: None,
1261            replace_old_group: false,
1262            validate_lifetimes: LeafNodeLifetimePolicy::Verify,
1263        }
1264    }
1265
1266    /// The ratchet tree to use for the new group.
1267    pub fn with_ratchet_tree(mut self, ratchet_tree: RatchetTreeIn) -> Self {
1268        self.ratchet_tree = Some(ratchet_tree);
1269        self
1270    }
1271
1272    /// Instruct the builder to replace any existing group with the same ID.
1273    pub fn replace_old_group(mut self) -> Self {
1274        self.replace_old_group = true;
1275        self
1276    }
1277
1278    /// Skip the validation of lifetimes in leaf nodes in the ratchet tree.
1279    /// Note that only the leaf nodes are checked that were never updated.
1280    ///
1281    /// By default they are validated.
1282    pub fn skip_lifetime_validation(mut self) -> Self {
1283        self.validate_lifetimes = LeafNodeLifetimePolicy::Skip;
1284        self
1285    }
1286
1287    /// Get a reference to the [`ProcessedWelcome`].
1288    ///
1289    /// Use this to inspect the [`Welcome`] message before validation.
1290    pub fn processed_welcome(&self) -> &ProcessedWelcome {
1291        &self.processed_welcome
1292    }
1293
1294    /// Build the [`StagedWelcome`].
1295    pub fn build(self) -> Result<StagedWelcome, WelcomeError<Provider::StorageError>> {
1296        self.processed_welcome.into_staged_welcome_inner(
1297            self.provider,
1298            self.ratchet_tree,
1299            self.validate_lifetimes,
1300            self.replace_old_group,
1301        )
1302    }
