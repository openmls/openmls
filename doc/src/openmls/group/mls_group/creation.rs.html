77    /// Join an existing group through an External Commit.
78    /// The resulting [`MlsGroup`] instance starts off with a pending
79    /// commit (the external commit, which adds this client to the group).
80    /// Merging this commit is necessary for this [`MlsGroup`] instance to
81    /// function properly, as, for example, this client is not yet part of the
82    /// tree. As a result, it is not possible to clear the pending commit. If
83    /// the external commit was rejected due to an epoch change, the
84    /// [`MlsGroup`] instance has to be discarded and a new one has to be
85    /// created using this function based on the latest `ratchet_tree` and
86    /// group info. For more information on the external init process,
87    /// please see Section 11.2.1 in the MLS specification.
88    ///
89    /// Note: If there is a group member in the group with the same identity as
90    /// us, this will create a remove proposal.
91    #[allow(clippy::too_many_arguments)]
92    #[deprecated(
93        since = "0.7.1",
94        note = "Use the `MlsGroup::external_commit_builder` instead."
95    )]
96    pub fn join_by_external_commit<Provider: OpenMlsProvider>(
97        provider: &Provider,
98        signer: &impl Signer,
99        ratchet_tree: Option<RatchetTreeIn>,
100        verifiable_group_info: VerifiableGroupInfo,
101        mls_group_config: &MlsGroupJoinConfig,
102        capabilities: Option<Capabilities>,
103        extensions: Option<Extensions>,
104        aad: &[u8],
105        credential_with_key: CredentialWithKey,
106    ) -> Result<(Self, MlsMessageOut, Option<GroupInfo>), ExternalCommitError<Provider::StorageError>>
107    {
108        let leaf_node_parameters = LeafNodeParameters::builder()
109            .with_capabilities(capabilities.unwrap_or_default())
110            .with_extensions(extensions.unwrap_or_default())
111            .build();
112
113        let mut external_commit_builder = ExternalCommitBuilder::new()
114            .with_aad(aad.to_vec())
115            .with_config(mls_group_config.clone());
116
117        if let Some(ratchet_tree) = ratchet_tree {
118            external_commit_builder = external_commit_builder.with_ratchet_tree(ratchet_tree)
119        }
120
121        let (mls_group, commit_message_bundle) = external_commit_builder
122            .build_group(provider, verifiable_group_info, credential_with_key)?
123            .leaf_node_parameters(leaf_node_parameters)
124            .load_psks(provider.storage())
125            .map_err(|e| {
126                log::error!("Error loading PSKs for external commit: {e:?}");
127                LibraryError::custom("Error loading PSKs for external commit")
128            })?
129            .build(provider.rand(), provider.crypto(), signer, |_| true)?
130            .finalize(provider)?;
131
132        let (commit, _, group_info) = commit_message_bundle.into_contents();
133
134        Ok((mls_group, commit, group_info))
135    }

139    /// Creates a new processed [`Welcome`] message , which can be
140    /// inspected before creating a [`StagedWelcome`].
141    ///
142    /// This does not require a ratchet tree yet.
143    ///
144    /// [`Welcome`]: crate::messages::Welcome
145    pub fn new_from_welcome<Provider: OpenMlsProvider>(
146        provider: &Provider,
147        mls_group_config: &MlsGroupJoinConfig,
148        welcome: Welcome,
149    ) -> Result<Self, WelcomeError<Provider::StorageError>> {
150        let (resumption_psk_store, key_package_bundle) =
151            keys_for_welcome(mls_group_config, &welcome, provider)?;
152
153        let ciphersuite = welcome.ciphersuite();
154        let Some(egs) = welcome.find_encrypted_group_secret(
155            key_package_bundle
156                .key_package()
157                .hash_ref(provider.crypto())?,
158        ) else {
159            return Err(WelcomeError::JoinerSecretNotFound);
160        };
161
162        // This check seems to be superfluous from the perspective of the RFC, but still doesn't
163        // seem like a bad idea.
164        if welcome.ciphersuite() != key_package_bundle.key_package().ciphersuite() {
165            let e = WelcomeError::CiphersuiteMismatch;
166            log::debug!("new_from_welcome {e:?}");
167            return Err(e);
168        }
169
170        let group_secrets = GroupSecrets::try_from_ciphertext(
171            key_package_bundle.init_private_key(),
172            egs.encrypted_group_secrets(),
173            welcome.encrypted_group_info(),
174            ciphersuite,
175            provider.crypto(),
176        )?;
177        let psk_secret = {
178            let psks = load_psks(
179                provider.storage(),
180                &resumption_psk_store,
181                &group_secrets.psks,
182            )?;
183
184            PskSecret::new(provider.crypto(), ciphersuite, psks)?
185        };
186        let key_schedule = KeySchedule::init(
187            ciphersuite,
188            provider.crypto(),
189            &group_secrets.joiner_secret,
190            psk_secret,
191        )?;
192        let (welcome_key, welcome_nonce) = key_schedule
193            .welcome(provider.crypto(), ciphersuite)
194            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?
195            .derive_welcome_key_nonce(provider.crypto(), ciphersuite)
196            .map_err(LibraryError::unexpected_crypto_error)?;
197        let verifiable_group_info = VerifiableGroupInfo::try_from_ciphertext(
198            &welcome_key,
199            &welcome_nonce,
200            welcome.encrypted_group_info(),
201            &[],
202            provider.crypto(),
203        )?;
204        if let Some(required_capabilities) =
205            verifiable_group_info.extensions().required_capabilities()
206        {
207            // Also check that our key package actually supports the extensions.
208            // As per the spec, the sender must have checked this. But you never know.
209            key_package_bundle
210                .key_package()
211                .leaf_node()
212                .capabilities()
213                .supports_required_capabilities(required_capabilities)?;
214        }
215
216        // https://validation.openmls.tech/#valn1404
217        // Verify that the cipher_suite in the GroupInfo matches the cipher_suite in the
218        // KeyPackage.
219        if verifiable_group_info.ciphersuite() != key_package_bundle.key_package().ciphersuite() {
220            let e = WelcomeError::CiphersuiteMismatch;
221            log::debug!("new_from_welcome {e:?}");
222            return Err(e);
223        }
224
225        Ok(Self {
226            mls_group_config: mls_group_config.clone(),
227            ciphersuite,
228            group_secrets,
229            key_schedule,
230            verifiable_group_info,
231            resumption_psk_store,
232            key_package_bundle,
233        })
234    }

260    /// Consume the `ProcessedWelcome` and combine it with the ratchet tree into
261    /// a `StagedWelcome`.
262    pub(crate) fn into_staged_welcome_inner<Provider: OpenMlsProvider>(
263        mut self,
264        provider: &Provider,
265        ratchet_tree: Option<RatchetTreeIn>,
266        validate_lifetimes: LeafNodeLifetimePolicy,
267    ) -> Result<StagedWelcome, WelcomeError<Provider::StorageError>> {
268        // Build the ratchet tree and group
269
270        // Set nodes either from the extension or from the `nodes_option`.
271        // If we got a ratchet tree extension in the welcome, we enable it for
272        // this group. Note that this is not strictly necessary. But there's
273        // currently no other mechanism to enable the extension.
274        let ratchet_tree = match self.verifiable_group_info.extensions().ratchet_tree() {
275            Some(extension) => extension.ratchet_tree().clone(),
276            None => match ratchet_tree {
277                Some(ratchet_tree) => ratchet_tree,
278                None => return Err(WelcomeError::MissingRatchetTree),
279            },
280        };
281
282        // Since there is currently only the external pub extension, there is no
283        // group info extension of interest here.
284        let (public_group, _group_info_extensions) = PublicGroup::from_ratchet_tree(
285            provider.crypto(),
286            ratchet_tree,
287            self.verifiable_group_info.clone(),
288            ProposalStore::new(),
289            validate_lifetimes,
290        )?;
291
292        // Find our own leaf in the tree.
293        let own_leaf_index = public_group
294            .members()
295            .find_map(|m| {
296                if m.signature_key
297                    == self
298                        .key_package_bundle
299                        .key_package()
300                        .leaf_node()
301                        .signature_key()
302                        .as_slice()
303                {
304                    Some(m.index)
305                } else {
306                    None
307                }
308            })
309            .ok_or(WelcomeError::PublicTreeError(
310                PublicTreeError::MalformedTree,
311            ))?;
312
313        struct KeyScheduleResult {
314            group_epoch_secrets: GroupEpochSecrets,
315            message_secrets: MessageSecrets,
316            #[cfg(feature = "extensions-draft-08")]
317            application_exporter: ApplicationExportSecret,
318        }
319        let KeyScheduleResult {
320            group_epoch_secrets,
321            message_secrets,
322            #[cfg(feature = "extensions-draft-08")]
323                application_exporter: application_export_secret,
324        } = {
325            let serialized_group_context = public_group
326                .group_context()
327                .tls_serialize_detached()
328                .map_err(LibraryError::missing_bound_check)?;
329
330            // TODO #751: Implement PSK
331            self.key_schedule
332                .add_context(provider.crypto(), &serialized_group_context)
333                .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
334
335            let EpochSecretsResult {
336                epoch_secrets,
337                #[cfg(feature = "extensions-draft-08")]
338                application_exporter,
339            } = self
340                .key_schedule
341                .epoch_secrets(provider.crypto(), self.ciphersuite)
342                .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
343
344            let (group_epoch_secrets, message_secrets) = epoch_secrets.split_secrets(
345                serialized_group_context,
346                public_group.tree_size(),
347                own_leaf_index,
348            );
349
350            KeyScheduleResult {
351                group_epoch_secrets,
352                message_secrets,
353                #[cfg(feature = "extensions-draft-08")]
354                application_exporter,
355            }
356        };
357
358        let confirmation_tag = message_secrets
359            .confirmation_key()
360            .tag(
361                provider.crypto(),
362                self.ciphersuite,
363                public_group.group_context().confirmed_transcript_hash(),
364            )
365            .map_err(LibraryError::unexpected_crypto_error)?;
366
367        // Verify confirmation tag
368        // https://validation.openmls.tech/#valn1410
369        if &confirmation_tag != public_group.confirmation_tag() {
370            log::error!("Confirmation tag mismatch");
371            log_crypto!(trace, "  Got:      {:x?}", confirmation_tag);
372            log_crypto!(trace, "  Expected: {:x?}", public_group.confirmation_tag());
373            debug_assert!(false, "Confirmation tag mismatch");
374
375            // in some tests we need to be able to proceed despite the tag being wrong,
376            // e.g. to test whether a later validation check is performed correctly.
377            if !crate::skip_validation::is_disabled::confirmation_tag() {
378                return Err(WelcomeError::ConfirmationTagMismatch);
379            }
380        }
381
382        let message_secrets_store = MessageSecretsStore::new_with_secret(0, message_secrets);
383
384        // Extract and store the resumption PSK for the current epoch.
385        let resumption_psk = group_epoch_secrets.resumption_psk();
386        self.resumption_psk_store
387            .add(public_group.group_context().epoch(), resumption_psk.clone());
388
389        let welcome_sender_index = self.verifiable_group_info.signer();
390        let path_keypairs = if let Some(path_secret) = self.group_secrets.path_secret {
391            let (path_keypairs, _commit_secret) = public_group
392                .derive_path_secrets(
393                    provider.crypto(),
394                    self.ciphersuite,
395                    path_secret,
396                    welcome_sender_index,
397                    own_leaf_index,
398                )
399                .map_err(|e| match e {
400                    DerivePathError::LibraryError(e) => e.into(),
401                    DerivePathError::PublicKeyMismatch => {
402                        WelcomeError::PublicTreeError(PublicTreeError::PublicKeyMismatch)
403                    }
404                })?;
405            Some(path_keypairs)
406        } else {
407            None
408        };
409
410        let staged_welcome = StagedWelcome {
411            mls_group_config: self.mls_group_config,
412            public_group,
413            group_epoch_secrets,
414            own_leaf_index,
415            message_secrets_store,
416            #[cfg(feature = "extensions-draft-08")]
417            application_export_secret,
418            resumption_psk_store: self.resumption_psk_store,
419            verifiable_group_info: self.verifiable_group_info,
420            key_package_bundle: self.key_package_bundle,
421            path_keypairs,
422        };
423
424        Ok(staged_welcome)
425    }

500    /// Consumes the [`StagedWelcome`] and returns the respective [`MlsGroup`].
501    pub fn into_group<Provider: OpenMlsProvider>(
502        self,
503        provider: &Provider,
504    ) -> Result<MlsGroup, WelcomeError<Provider::StorageError>> {
505        // If we got a path secret, derive the path (which also checks if the
506        // public keys match) and store the derived keys in the key store.
507        let group_keypairs = if let Some(path_keypairs) = self.path_keypairs {
508            let mut keypairs = vec![self.key_package_bundle.encryption_key_pair()];
509            keypairs.extend_from_slice(&path_keypairs);
510            keypairs
511        } else {
512            vec![self.key_package_bundle.encryption_key_pair()]
513        };
514
515        #[cfg(feature = "extensions-draft-08")]
516        let application_export_tree = ApplicationExportTree::new(self.application_export_secret);
517
518        let mut mls_group = MlsGroup {
519            mls_group_config: self.mls_group_config,
520            own_leaf_nodes: vec![],
521            aad: vec![],
522            group_state: MlsGroupState::Operational,
523            public_group: self.public_group,
524            group_epoch_secrets: self.group_epoch_secrets,
525            own_leaf_index: self.own_leaf_index,
526            message_secrets_store: self.message_secrets_store,
527            resumption_psk_store: self.resumption_psk_store,
528            #[cfg(feature = "extensions-draft-08")]
529            application_export_tree: Some(application_export_tree),
530        };
531
532        mls_group
533            .store_epoch_keypairs(provider.storage(), group_keypairs.as_slice())
534            .map_err(WelcomeError::StorageError)?;
535        mls_group.set_max_past_epochs(mls_group.mls_group_config.max_past_epochs);
536
537        mls_group
538            .store(provider.storage())
539            .map_err(WelcomeError::StorageError)?;
540
541        Ok(mls_group)
542    }

545fn keys_for_welcome<Provider: OpenMlsProvider>(
546    mls_group_config: &MlsGroupJoinConfig,
547    welcome: &Welcome,
548    provider: &Provider,
549) -> Result<
550    (ResumptionPskStore, KeyPackageBundle),
551    WelcomeError<<Provider as OpenMlsProvider>::StorageError>,
552> {
553    let resumption_psk_store = ResumptionPskStore::new(mls_group_config.number_of_resumption_psks);
554    let key_package_bundle: KeyPackageBundle = welcome
555        .secrets()
556        .iter()
557        .find_map(|egs| {
558            let hash_ref = egs.new_member();
559
560            provider
561                .storage()
562                .key_package(&hash_ref)
563                .map_err(WelcomeError::StorageError)
564                .transpose()
565        })
566        .ok_or(WelcomeError::NoMatchingKeyPackage)??;
567    if !key_package_bundle.key_package().last_resort() {
568        provider
569            .storage()
570            .delete_key_package(&key_package_bundle.key_package.hash_ref(provider.crypto())?)
571            .map_err(WelcomeError::StorageError)?;
572    } else {
573        log::debug!("Key package has last resort extension, not deleting");
574    }
575    Ok((resumption_psk_store, key_package_bundle))
576}

578/// Verify or skip the validation of leaf node lifetimes in the ratchet tree
579/// when joining a group.
580#[derive(Default, Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize)]
581pub enum LeafNodeLifetimePolicy {
582    /// Verify the lifetime of leaf nodes in the ratchet tree.
583    ///
584    /// **NOTE:** Only leaf nodes that have never been updated have a lifetime.
585    #[default]
586    Verify,
587
588    /// Skip the verification of the lifeimte in leaf nodes in the ratchet tree.
589    Skip,
590}

602impl<'a, Provider: OpenMlsProvider> JoinBuilder<'a, Provider> {
603    /// Create a new builder for the [`JoinBuilder`].
604    pub(crate) fn new(provider: &'a Provider, processed_welcome: ProcessedWelcome) -> Self {

606            provider,
607            processed_welcome,
608            ratchet_tree: None,
609            validate_lifetimes: LeafNodeLifetimePolicy::Verify,

615        self.ratchet_tree = Some(ratchet_tree);
616        self
617    }

619    /// Skip the validation of lifetimes in leaf nodes in the ratchet tree.
620    /// Note that only the leaf nodes are checked that were never updated.
621    ///
622    /// By default they are validated.
623    pub fn skip_lifetime_validation(mut self) -> Self {
624        self.validate_lifetimes = LeafNodeLifetimePolicy::Skip;
625        self

628    /// Get a reference to the [`ProcessedWelcome`].
629    ///
630    /// Use this to inspect the [`Welcome`] message before validation.
631    pub fn processed_welcome(&self) -> &ProcessedWelcome {
632        &self.processed_welcome
633    }
634
635    /// Build the [`StagedWelcome`].
636    pub fn build(self) -> Result<StagedWelcome, WelcomeError<Provider::StorageError>> {
637        self.processed_welcome.into_staged_welcome_inner(
