155/// [`MlsGroupState`] determines the state of an [`MlsGroup`]. The different
156/// states and their transitions are as follows:
157///
158/// * [`MlsGroupState::Operational`]: This is the main state of the group, which
159///   allows access to all of its functionality, (except merging pending commits,
160///   see the [`MlsGroupState::PendingCommit`] for more information) and it's the
161///   state the group starts in (except when created via
162///   [`MlsGroup::external_commit_builder()`], see the functions documentation for
163///   more information). From this `Operational`, the group state can either
164///   transition to [`MlsGroupState::Inactive`], when it processes a commit that
165///   removes this client from the group, or to [`MlsGroupState::PendingCommit`],
166///   when this client creates a commit.
167///
168/// * [`MlsGroupState::Inactive`]: A group can enter this state from any other
169///   state when it processes a commit that removes this client from the group.
170///   This is a terminal state that the group can not exit from. If the clients
171///   wants to re-join the group, it can either be added by a group member or it
172///   can join via external commit.
173///
174/// * [`MlsGroupState::PendingCommit`]: This state is split into two possible
175///   sub-states, one for each Commit type:
176///   [`PendingCommitState::Member`] and [`PendingCommitState::External`]:
177///
178///   * If the client creates a commit for this group, the `PendingCommit` state
179///     is entered with [`PendingCommitState::Member`] and with the [`StagedCommit`] as
180///     additional state variable. In this state, it can perform the same
181///     operations as in the [`MlsGroupState::Operational`], except that it cannot
182///     create proposals or commits. However, it can merge or clear the stored
183///     [`StagedCommit`], where both actions result in a transition to the
184///     [`MlsGroupState::Operational`]. Additionally, if a commit from another
185///     group member is processed, the own pending commit is also cleared and
186///     either the `Inactive` state is entered (if this client was removed from
187///     the group as part of the processed commit), or the `Operational` state is
188///     entered.
189///
190///   * A group can enter the [`PendingCommitState::External`] sub-state only as
191///     the initial state when the group is created via
192///     [`MlsGroup::external_commit_builder()`]. In contrast to the
193///     [`PendingCommitState::Member`] `PendingCommit` state, the only possible
194///     functionality that can be used is the [`MlsGroup::merge_pending_commit()`]
195///     function, which merges the pending external commit and transitions the
196///     state to [`MlsGroupState::PendingCommit`]. For more information on the
197///     external commit process, see [`MlsGroup::external_commit_builder()`] or
198///     Section 11.2.1 of the MLS specification.
199#[derive(Debug, Serialize, Deserialize)]
200#[cfg_attr(any(test, feature = "test-utils"), derive(Clone, PartialEq))]
201pub enum MlsGroupState {
202    /// There is currently a pending Commit that hasn't been merged yet.
203    PendingCommit(Box<PendingCommitState>),
204    /// The group state is in an opertaional state, where new messages and Commits can be created.
205    Operational,
206    /// The group is inactive because the member has been removed.
207    Inactive,
208}

210/// A `MlsGroup` represents an MLS group with a high-level API. The API exposes
211/// high level functions to manage a group by adding/removing members, get the
212/// current member list, etc.
213///
214/// The API is modeled such that it can serve as a direct interface to the
215/// Delivery Service. Functions that modify the public state of the group will
216/// return a `Vec<MLSMessageOut>` that can be sent to the Delivery Service
217/// directly. Conversely, incoming messages from the Delivery Service can be fed
218/// into [process_message()](`MlsGroup::process_message()`).
219///
220/// An `MlsGroup` has an internal queue of pending proposals that builds up as
221/// new messages are processed. When creating proposals, those messages are not
222/// automatically appended to this queue, instead they have to be processed
223/// again through [process_message()](`MlsGroup::process_message()`). This
224/// allows the Delivery Service to reject them (e.g. if they reference the wrong
225/// epoch).
226///
227/// If incoming messages or applied operations are semantically or syntactically
228/// incorrect, an error event will be returned with a corresponding error
229/// message and the state of the group will remain unchanged.
230///
231/// An `MlsGroup` has an internal state variable determining if it is active or
232/// inactive, as well as if it has a pending commit. See [`MlsGroupState`] for
233/// more information.
234#[derive(Debug)]
235#[cfg_attr(feature = "test-utils", derive(Clone, PartialEq))]
236pub struct MlsGroup {
237    /// The group configuration. See [`MlsGroupJoinConfig`] for more information.
238    mls_group_config: MlsGroupJoinConfig,
239    /// The public state of the group.
240    public_group: PublicGroup,
241    /// Epoch-specific secrets of the group.
242    group_epoch_secrets: GroupEpochSecrets,
243    /// The own leaf index in the ratchet tree.
244    own_leaf_index: LeafNodeIndex,
245    /// A [`MessageSecretsStore`] that stores message secrets.
246    /// By default this store has the length of 1, i.e. only the [`MessageSecrets`]
247    /// of the current epoch is kept.
248    /// If more secrets from past epochs should be kept in order to be
249    /// able to decrypt application messages from previous epochs, the size of
250    /// the store must be increased through [`max_past_epochs()`].
251    message_secrets_store: MessageSecretsStore,
252    // Resumption psk store. This is where the resumption psks are kept in a rollover list.
253    resumption_psk_store: ResumptionPskStore,
254    // Own [`LeafNode`]s that were created for update proposals and that
255    // are needed in case an update proposal is committed by another group
256    // member. The vector is emptied after every epoch change.
257    own_leaf_nodes: Vec<LeafNode>,
258    // Additional authenticated data (AAD) for the next outgoing message. This
259    // is ephemeral and will be reset by every API call that successfully
260    // returns an [`MlsMessageOut`].
261    aad: Vec<u8>,
262    // Safe AAD items to attach to the next outgoing message. Ephemeral, reset
263    // alongside `aad`. Only consulted when the group's GroupContext requires
264    // Safe AAD framing.
265    #[cfg(feature = "extensions-draft")]
266    safe_aad: SafeAad,
267    // A variable that indicates the state of the group. See [`MlsGroupState`]
268    // for more information.
269    group_state: MlsGroupState,
270    /// The state of the Application Exporter. See the MLS Extensions Draft 08
271    /// for more information. This is `None` if an old OpenMLS group state was
272    /// loaded and has not yet merged a commit.
273    #[cfg(feature = "extensions-draft")]
274    application_export_tree: Option<ApplicationExportTree>,
275}

390    /// Returns a reference to the [`StagedCommit`] of the most recently created

403    /// Sets the `group_state` to [`MlsGroupState::Operational`], thus clearing
404    /// any potentially pending commits.
405    ///
406    /// Note that this has no effect if the group was created through an external commit and
407    /// the resulting external commit has not been merged yet. For more
408    /// information, see [`MlsGroup::external_commit_builder()`].
409    ///
410    /// Use with caution! This function should only be used if it is clear that
411    /// the pending commit will not be used in the group. In particular, if a
412    /// pending commit is later accepted by the group, this client will lack the
413    /// key material to encrypt or decrypt group messages.
414    pub fn clear_pending_commit<Storage: StorageProvider>(
415        &mut self,
416        storage: &Storage,
417    ) -> Result<(), Storage::Error> {
418        match self.group_state {
419            MlsGroupState::PendingCommit(ref pending_commit_state) => {
420                if let PendingCommitState::Member(_) = **pending_commit_state {
421                    self.group_state = MlsGroupState::Operational;
422                    storage.write_group_state(self.group_id(), &self.group_state)
423                } else {
424                    Ok(())
425                }
426            }
427            MlsGroupState::Operational | MlsGroupState::Inactive => Ok(()),
428        }
429    }

431    /// Clear the pending proposals, if the proposal store is not empty.
432    ///
433    /// Warning: Once the pending proposals are cleared it will be impossible to process
434    /// a Commit message that references those proposals. Only use this
435    /// function as a last resort, e.g. when a call to
436    /// `MlsGroup::commit_to_pending_proposals` fails.
437    pub fn clear_pending_proposals<Storage: StorageProvider>(
438        &mut self,
439        storage: &Storage,
440    ) -> Result<(), Storage::Error> {
441        // If the proposal store is not empty...
442        if !self.proposal_store().is_empty() {
443            // Empty the proposal store
444            self.proposal_store_mut().empty();
445
446            // Clear proposals in storage
447            storage.clear_proposal_queue::<GroupId, ProposalRef>(self.group_id())?;
448        }
449
450        Ok(())
451    }

468    /// Loads the state of the group with given id from persisted state.
469    pub fn load<Storage: crate::storage::StorageProvider>(
470        storage: &Storage,
471        group_id: &GroupId,
472    ) -> Result<Option<MlsGroup>, Storage::Error> {
473        let public_group = PublicGroup::load(storage, group_id)?;
474        let group_epoch_secrets = storage.group_epoch_secrets(group_id)?;
475        let own_leaf_index = storage.own_leaf_index(group_id)?;
476        let message_secrets_store = storage.message_secrets(group_id)?;
477        let resumption_psk_store = storage.resumption_psk_store(group_id)?;
478        let mls_group_config = storage.mls_group_join_config(group_id)?;
479        let own_leaf_nodes = storage.own_leaf_nodes(group_id)?;
480        let group_state = storage.group_state(group_id)?;
481        #[cfg(feature = "extensions-draft")]
482        let application_export_tree = storage.application_export_tree(group_id)?;
483
484        let build = || -> Option<Self> {
485            Some(Self {
486                public_group: public_group?,
487                group_epoch_secrets: group_epoch_secrets?,
488                own_leaf_index: own_leaf_index?,
489                message_secrets_store: message_secrets_store?,
490                resumption_psk_store: resumption_psk_store?,
491                mls_group_config: mls_group_config?,
492                own_leaf_nodes,
493                aad: vec![],
494                #[cfg(feature = "extensions-draft")]
495                safe_aad: SafeAad::empty(),
496                group_state: group_state?,
497                #[cfg(feature = "extensions-draft")]
498                application_export_tree,
499            })
500        };
501
502        Ok(build())
503    }

505    /// Remove the persisted state of this group from storage. Note that
506    /// signature key material is not managed by OpenMLS and has to be removed
507    /// from the storage provider separately (if desired).
508    pub fn delete<Storage: crate::storage::StorageProvider>(
509        &mut self,
510        storage: &Storage,
511    ) -> Result<(), Storage::Error> {
512        PublicGroup::delete(storage, self.group_id())?;
513        storage.delete_own_leaf_index(self.group_id())?;
514        storage.delete_group_epoch_secrets(self.group_id())?;
515        storage.delete_message_secrets(self.group_id())?;
516        storage.delete_all_resumption_psk_secrets(self.group_id())?;
517        storage.delete_group_config(self.group_id())?;
518        storage.delete_own_leaf_nodes(self.group_id())?;
519        storage.delete_group_state(self.group_id())?;
520        storage.clear_proposal_queue::<GroupId, ProposalRef>(self.group_id())?;
521
522        #[cfg(feature = "extensions-draft")]
523        storage.delete_application_export_tree::<_, ApplicationExportTree>(self.group_id())?;
524
525        // Drop this group's emulation-epoch bindings. `EmulationEpochState`
526        // and the operation secret tree are keyed on the emulation epoch
527        // and may still be referenced by other higher-level groups, so
528        // they're not deleted here.
529        #[cfg(feature = "virtual-clients-draft")]
530        storage.delete_vc_emulation_bindings(self.group_id())?;
531
532        self.proposal_store_mut().empty();
533        storage.delete_encryption_epoch_key_pairs(
534            self.group_id(),
535            &self.epoch(),
536            self.own_leaf_index().u32(),
537        )?;
538
539        Ok(())
540    }

595    /// Set the past epoch secret deletion policy for the group.

668    /// Create a new group context extension proposal
669    pub(crate) fn create_group_context_ext_proposal<Provider: OpenMlsProvider>(
670        &self,
671        framing_parameters: FramingParameters,
672        extensions: Extensions<GroupContext>,
673        signer: &impl Signer,
674    ) -> Result<AuthenticatedContent, CreateGroupContextExtProposalError<Provider::StorageError>>
675    {
676        // Ensure that the group supports all the extensions that are wanted.
677        let required_extension = extensions
678            .iter()
679            .find(|extension| extension.extension_type() == ExtensionType::RequiredCapabilities);
680        if let Some(required_extension) = required_extension {
681            let required_capabilities = required_extension.as_required_capabilities_extension()?;
682            // Ensure we support all the capabilities.
683            self.own_leaf_node()
684                .ok_or_else(|| LibraryError::custom("Tree has no own leaf."))?
685                .capabilities()
686                .supports_required_capabilities(required_capabilities)?;
687
688            // Ensure that all other leaf nodes support all the required
689            // extensions as well.
690            self.public_group()
691                .check_extension_support(required_capabilities.extension_types())?;
692        }
693        let proposal = GroupContextExtensionProposal::new(extensions);
694        let proposal = Proposal::GroupContextExtensions(Box::new(proposal));
695        AuthenticatedContent::member_proposal(
696            framing_parameters,
697            self.own_leaf_index(),
698            proposal,
699            self.context(),
700            signer,
701        )
702        .map_err(|e| e.into())
703    }

705    /// Load the [`EmulationEpochState`] this group is bound to at `epoch`, if
706    /// any. Returns `None` when the group has no virtual-clients binding for
707    /// that epoch. The binding is resolved at the epoch a message was sent in,
708    /// so a delayed message from a past epoch deprotects with the state that
709    /// was bound then, not the latest one.
710    ///
711    /// [`EmulationEpochState`]: crate::components::vc_derivation_info::EmulationEpochState
712    #[cfg(feature = "virtual-clients-draft")]
713    pub(crate) fn vc_emulation_state_at_epoch<Storage: StorageProvider>(
714        &self,
715        storage: &Storage,
716        epoch: GroupEpoch,
717    ) -> Result<
718        Option<crate::components::vc_derivation_info::EmulationEpochState>,
719        VcEmulationStateError<Storage::Error>,
720    > {
721        let bindings: Option<crate::components::vc_derivation_info::VcEmulationBindings> = storage
722            .vc_emulation_bindings(self.group_id())
723            .map_err(VcEmulationStateError::Storage)?;
724        let Some(epoch_id) = bindings.and_then(|bindings| bindings.get(epoch).cloned()) else {
725            return Ok(None);
726        };
727        let state = storage
728            .vc_emulation_epoch_state(&epoch_id)
729            .map_err(VcEmulationStateError::Storage)?
730            .ok_or_else(|| {
731                log::error!("vc: group is bound to emulation epoch, but state is missing");
732                VcEmulationStateError::MissingEmulationEpochState
733            })?;
734        Ok(Some(state))
735    }

737    // Encrypt an AuthenticatedContent into an PrivateMessage
738    pub(crate) fn encrypt<Provider: OpenMlsProvider>(
739        &mut self,
740        public_message: AuthenticatedContent,
741        provider: &Provider,
742    ) -> Result<EncryptionOutput, MessageEncryptionError<Provider::StorageError>> {
743        let padding_size = self.configuration().padding_size();
744
745        // If this group is bound to an emulation epoch at its current epoch,
746        // load the state so the framing layer can derive a deterministic
747        // reuse guard.
748        #[cfg(feature = "virtual-clients-draft")]
749        let emulation_state = self
750            .vc_emulation_state_at_epoch(provider.storage(), self.epoch())
751            .map_err(|e| match e {
752                VcEmulationStateError::Storage(e) => MessageEncryptionError::StorageError(e),
753                VcEmulationStateError::MissingEmulationEpochState => {
754                    MessageEncryptionError::VirtualClientsError(
755                        crate::components::vc_derivation_info::VirtualClientsError::MissingEmulationEpochState,
756                    )
757                }
758            })?;
759        #[cfg(feature = "virtual-clients-draft")]
760        let emulator_ctx: Option<crate::framing::EmulatorReuseGuardCtx<'_>> = emulation_state
761            .as_ref()
762            .map(|state| state.reuse_guard_inputs());
763
764        let msg = PrivateMessage::try_from_authenticated_content(
765            provider.crypto(),
766            provider.rand(),
767            &public_message,
768            self.ciphersuite(),
769            self.message_secrets_store.message_secrets_mut(),
770            padding_size,
771            #[cfg(feature = "virtual-clients-draft")]
772            emulator_ctx.as_ref(),
773        )?;
774
775        // When the group is bound to an emulation epoch, derive the generation
776        // ID the application hands to the DS to detect generation collisions
777        // between siblings. Only application messages carry one for now,
778        // matching the deferral of handshake VC framing in higher-level groups.
779        #[cfg(feature = "virtual-clients-draft")]
780        let msg = {
781            let mut msg = msg;
782            if let Some(state) = &emulation_state {
783                if public_message.content().content_type() == ContentType::Application {
784                    let generation_id = state
785                        .derive_generation_id(
786                            provider.crypto(),
787                            self.group_id(),
788                            self.epoch(),
789                            msg.generation,
790                            crate::components::vc_derivation_info::RatchetType::Application,
791                        )
792                        .map_err(MessageEncryptionError::VirtualClientsError)?;
793                    msg.generation_id = Some(generation_id);
794                }
795            }
796            msg
797        };
798
799        provider
800            .storage()
801            .write_message_secrets(self.group_id(), &self.message_secrets_store)
802            .map_err(MessageEncryptionError::StorageError)?;
803
804        Ok(msg)
805    }

984    /// Converts PublicMessage to MlsMessage. Depending on whether handshake
985    /// message should be encrypted, PublicMessage messages are encrypted to
986    /// PrivateMessage first.
987    fn content_to_mls_message(
988        &mut self,
989        mls_auth_content: AuthenticatedContent,
990        provider: &impl OpenMlsProvider,
991    ) -> Result<MlsMessageOut, LibraryError> {
992        let msg = match self.configuration().wire_format_policy().outgoing() {
993            OutgoingWireFormatPolicy::AlwaysPlaintext => {
994                let mut plaintext: PublicMessage = mls_auth_content.into();
995                // Set the membership tag only if the sender type is `Member`.
996                if plaintext.sender().is_member() {
997                    plaintext.set_membership_tag(
998                        provider.crypto(),
999                        self.ciphersuite(),
1000                        self.message_secrets().membership_key(),
1001                        self.message_secrets().serialized_context(),
1002                    )?;
1003                }
1004                plaintext.into()
1005            }
1006            OutgoingWireFormatPolicy::AlwaysCiphertext => {
1007                // Decrypting own handshake messages is not supported yet with
1008                // the `virtual-clients` feature, so the generation is unused.
1009                let EncryptionOutput {
1010                    private_message, ..
1011                } = self
1012                    .encrypt(mls_auth_content, provider)
1013                    // We can be sure the encryption will work because the plaintext was created by us
1014                    .map_err(|_| LibraryError::custom("Malformed plaintext"))?;
1015                MlsMessageOut::from_private_message(private_message, self.version())
1016            }
1017        };
1018        Ok(msg)
1019    }

1181    /// By default this store has the length of 1, i.e. only the [`MessageSecrets`]
1182    /// of the current epoch is kept.
1183    /// If more secrets from past epochs should be kept in order to be
1184    /// able to decrypt application messages from previous epochs, the size of
1185    /// the store must be increased through [`max_past_epochs()`].
1186    message_secrets_store: MessageSecretsStore,
1187
1188    /// A secret that is not stored as part of the [`MlsGroup`] after the group is created.
1189    /// It can be used by the application to derive forward secure secrets.
1190    #[cfg(feature = "extensions-draft")]
1191    application_export_secret: ApplicationExportSecret,

1226/// The key material a client uses to process a [`Welcome`] message.
1227///
1228/// A regular member holds a local [`KeyPackageBundle`]. A sibling emulator
1229/// joining a higher-level group as a virtual client has no local bundle: it
1230/// derives the init and leaf-encryption keys from the operation secret tree of
1231/// the emulation epoch the KeyPackage belongs to.
1232///
1233/// [`Welcome`]: crate::messages::Welcome
1234#[derive(Debug)]
1235pub(crate) enum WelcomeKeyMaterial {
1236    /// A locally stored [`KeyPackageBundle`]. Boxed to keep the enum small,
1237    /// since the virtual-client variant is much smaller.
1238    KeyPackage(Box<KeyPackageBundle>),
1239    /// Virtual-client material derived from an emulation epoch's operation
1240    /// secret tree.
1241    #[cfg(feature = "virtual-clients-draft")]
1242    VirtualClient(crate::components::vc_derivation_info::VcWelcomeMaterial),
1243}
1244
1245impl WelcomeKeyMaterial {
1246    /// The [`KeyPackageRef`] addressed by the welcome's encrypted group
1247    /// secrets. The bundle computes it from its KeyPackage, the virtual-client
1248    /// material carries the ref it was matched on.
1249    ///
1250    /// [`KeyPackageRef`]: crate::ciphersuite::hash_ref::KeyPackageRef
1251    fn key_package_ref(
1252        &self,
1253        crypto: &impl OpenMlsCrypto,
1254    ) -> Result<crate::ciphersuite::hash_ref::KeyPackageRef, LibraryError> {
1255        match self {
1256            WelcomeKeyMaterial::KeyPackage(bundle) => bundle.key_package().hash_ref(crypto),
1257            #[cfg(feature = "virtual-clients-draft")]
1258            WelcomeKeyMaterial::VirtualClient(material) => Ok(material.key_package_ref.clone()),
1259        }
1260    }
1261
1262    /// The init private key used to decrypt the encrypted group secrets.
1263    fn init_private_key(&self) -> &crate::ciphersuite::HpkePrivateKey {
1264        match self {
1265            WelcomeKeyMaterial::KeyPackage(bundle) => bundle.init_private_key(),
1266            #[cfg(feature = "virtual-clients-draft")]
1267            WelcomeKeyMaterial::VirtualClient(material) => &material.init_private_key,
1268        }
1269    }
1270
1271    /// The local [`KeyPackageBundle`] on the regular path, or `None` on the
1272    /// virtual-client path. Checks that only apply when there is a local
1273    /// KeyPackage to compare against branch on this.
1274    fn key_package_bundle(&self) -> Option<&KeyPackageBundle> {
1275        match self {
1276            WelcomeKeyMaterial::KeyPackage(bundle) => Some(bundle),
1277            #[cfg(feature = "virtual-clients-draft")]
1278            WelcomeKeyMaterial::VirtualClient(_) => None,
1279        }
1280    }
1281
1282    /// The joiner's leaf encryption keypair.
1283    fn encryption_key_pair(&self) -> EncryptionKeyPair {
1284        match self {
1285            WelcomeKeyMaterial::KeyPackage(bundle) => bundle.encryption_key_pair(),
1286            #[cfg(feature = "virtual-clients-draft")]
1287            WelcomeKeyMaterial::VirtualClient(material) => material.encryption_keypair.clone(),
1288        }
1289    }
1290}
