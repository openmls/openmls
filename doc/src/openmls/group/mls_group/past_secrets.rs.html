7// Internal helper struct
8#[derive(Serialize, Deserialize)]
9#[cfg_attr(any(test, feature = "test-utils"), derive(Clone, PartialEq))]
10#[cfg_attr(feature = "crypto-debug", derive(Debug))]
11struct EpochTree {
12    epoch: u64,
13    message_secrets: MessageSecrets,
14    leaves: Vec<Member>,
15}
16
17/// Can store message secrets for up to `max_epochs`. The trees are added with [`self::add()`] and can be queried
18/// with [`Self::get_epoch()`].
19#[derive(Serialize, Deserialize)]
20#[cfg_attr(any(test, feature = "test-utils"), derive(Clone, PartialEq))]
21#[cfg_attr(feature = "crypto-debug", derive(Debug))]
22pub(crate) struct MessageSecretsStore {
23    // Maximum size of the `past_epoch_trees` list.
24    pub(crate) max_epochs: usize,
25    // Past message secrets.
26    past_epoch_trees: VecDeque<EpochTree>,
27    // The message secrets of the current epoch.
28    message_secrets: MessageSecrets,
29}
30
31#[cfg(not(feature = "crypto-debug"))]
32impl core::fmt::Debug for MessageSecretsStore {
33    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
34        f.debug_struct("MessageSecretsStore")
35            .field("max_epochs", &"***")
36            .field("past_epoch_trees", &"***")
37            .field("message_secrets", &"***")
38            .finish()
39    }
40}
41
42impl MessageSecretsStore {
43    /// Create a new store that can hold up to `max_past_epochs` message secrets.
44    /// If `max_past_epochs` is 0, only the current epoch is being stored.
45    pub(crate) fn new_with_secret(max_epochs: usize, message_secrets: MessageSecrets) -> Self {
46        Self {
47            max_epochs,
48            past_epoch_trees: VecDeque::new(),
49            message_secrets,
50        }
51    }
52
53    /// Resize the store.
54    pub(crate) fn resize(&mut self, max_past_epochs: usize) {
55        let old_size = self.max_epochs;
56        self.max_epochs = max_past_epochs;
57        if old_size > max_past_epochs {
58            let num_epochs_out = old_size - max_past_epochs;
59            self.past_epoch_trees.rotate_left(num_epochs_out);
60            self.past_epoch_trees.truncate(max_past_epochs);
61        }
62    }
63
64    /// Add a secret tree for a given epoch `group_epoch`.
65    /// Note that this does not take the epoch into account and pops out the
66    /// oldest element.
67    pub(crate) fn add(
68        &mut self,
69        group_epoch: impl Into<GroupEpoch>,
70        message_secrets: MessageSecrets,
71        leaves: Vec<Member>,
72    ) {
73        // Don't store the tree if it's not intended
74        if self.max_epochs == 0 {
75            return;
76        }
77        if self.past_epoch_trees.len() >= self.max_epochs {
78            self.past_epoch_trees.rotate_left(1);
79            self.past_epoch_trees.truncate(self.max_epochs - 1);
80        }
81        self.past_epoch_trees.push_back(EpochTree {
82            epoch: group_epoch.into().as_u64(),
83            message_secrets,
84            leaves,
85        });
86        debug_assert!(
87            self.max_epochs >= self.past_epoch_trees.len(),
88            "Only {} past secrets must be stored but we found {}",
89            self.max_epochs,
90            self.past_epoch_trees.len()
91        );
92    }
93
94    /// Get a mutable reference to a secret tree for a given epoch `group_epoch`.
95    /// If no message secrets are found for that epoch, `None` is returned.
96    pub(crate) fn secrets_for_epoch_mut(
97        &mut self,
98        group_epoch: impl Into<GroupEpoch>,
99    ) -> Option<&mut MessageSecrets> {
100        let epoch = group_epoch.into().as_u64();
101        for epoch_tree in self.past_epoch_trees.iter_mut() {
102            if epoch_tree.epoch == epoch {
103                return Some(&mut epoch_tree.message_secrets);
104            }
105        }
106        None
107    }
108
109    /// Get a reference to a secret tree for a given epoch `group_epoch`.
110    /// If no message secrets are found for that epoch, `None` is returned.
111    pub(crate) fn secrets_for_epoch(
112        &self,
113        group_epoch: impl Into<GroupEpoch>,
114    ) -> Option<&MessageSecrets> {
115        let epoch = group_epoch.into().as_u64();
116        for epoch_tree in self.past_epoch_trees.iter() {
117            if epoch_tree.epoch == epoch {
118                return Some(&epoch_tree.message_secrets);
119            }
120        }
121        None
122    }
123
124    /// Get a mutable reference to a secret tree for a given epoch `group_epoch`.
125    /// Return a mutable reference to the [`MessageSecrets`] and a slice to the
126    /// [`Member`]s of the epoch.
127    pub(crate) fn secrets_and_leaves_for_epoch_mut(
128        &mut self,
129        group_epoch: impl Into<GroupEpoch>,
130    ) -> Option<(&mut MessageSecrets, &[Member])> {
131        let epoch = group_epoch.into().as_u64();
132        for epoch_tree in self.past_epoch_trees.iter_mut() {
133            if epoch_tree.epoch == epoch {
134                return Some((&mut epoch_tree.message_secrets, &epoch_tree.leaves));
135            }
136        }
137        None
138    }
139
140    /// Return a slice with the [`Member`]s of the `group_epoch`.
141    pub(crate) fn leaves_for_epoch(&self, group_epoch: impl Into<GroupEpoch>) -> &[Member] {
142        let epoch = group_epoch.into().as_u64();
143        for epoch_tree in self.past_epoch_trees.iter() {
144            if epoch_tree.epoch == epoch {
145                return &epoch_tree.leaves;
146            }
147        }
148        &[]
149    }
150
151    /// Check if the provided epoch contains a leaf index.
152    pub(crate) fn epoch_has_leaf(
153        &self,
154        group_epoch: GroupEpoch,
155        leaf_index: LeafNodeIndex,
156    ) -> bool {
157        self.past_epoch_trees.iter().any(|t| {
158            t.epoch == group_epoch.0
159                && t.leaves
160                    .iter()
161                    .any(|Member { index, .. }| *index == leaf_index)
162        })
163    }
164
165    /// Get a mutable reference to the message secrets of the current epoch.
166    pub(crate) fn message_secrets_mut(&mut self) -> &mut MessageSecrets {
167        &mut self.message_secrets
168    }
169
170    /// Get a reference to the message secrets of the current epoch.
171    pub(crate) fn message_secrets(&self) -> &MessageSecrets {
172        &self.message_secrets
173    }
