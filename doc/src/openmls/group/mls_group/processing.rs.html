113    /// Parses incoming messages from the DS. Checks for syntactic errors and
114    /// makes some semantic checks as well. If the input is an encrypted
115    /// message, it will be decrypted. This processing function does syntactic
116    /// and semantic validation of the message. It returns a [ProcessedMessage]
117    /// enum.
118    ///
119    /// # Errors:
120    /// Returns an [`ProcessMessageError`] when the validation checks fail
121    /// with the exact reason of the failure.
122    pub fn process_message<Provider: OpenMlsProvider>(
123        &mut self,
124        provider: &Provider,
125        message: impl Into<ProtocolMessage>,
126    ) -> Result<ProcessedMessage, ProcessMessageError<Provider::StorageError>> {
127        let unverified_message = self.unprotect_message(provider, message)?;
128
129        // Check if the commit contains AppDataUpdate proposals - if so, the caller
130        // must use process_unverified_message_with_app_data_updates instead
131        #[cfg(feature = "extensions-draft-08")]
132        if let Some(proposals) = unverified_message.committed_proposals() {
133            for proposal_or_ref in proposals {
134                if let ProposalOrRefIn::Proposal(proposal) = proposal_or_ref {
135                    if matches!(proposal.as_ref(), ProposalIn::AppDataUpdate(_)) {
136                        return Err(ProcessMessageError::FoundAppDataUpdateProposal);
137                    }
138                }
139            }
140        }
141        self.process_unverified_message(provider, unverified_message)
142    }

150    /// Parses and deprotects incoming messages from the DS. Checks for syntactic errors, but only
151    /// performs limited semantic checks.
152    pub fn unprotect_message<Provider: OpenMlsProvider>(
153        &mut self,
154        provider: &Provider,
155        message: impl Into<ProtocolMessage>,
156    ) -> Result<UnverifiedMessage, ProcessMessageError<Provider::StorageError>> {
157        // Make sure we are still a member of the group
158        if !self.is_active() {
159            return Err(ProcessMessageError::GroupStateError(
160                MlsGroupStateError::UseAfterEviction,
161            ));
162        }
163        let message = message.into();
164
165        // Check that handshake messages are compatible with the incoming wire format policy
166        if !message.is_external()
167            && message.is_handshake_message()
168            && !self
169                .configuration()
170                .wire_format_policy()
171                .incoming()
172                .is_compatible_with(message.wire_format())
173        {
174            return Err(ProcessMessageError::IncompatibleWireFormat);
175        }
176
177        // Parse the message
178        let sender_ratchet_configuration = *self.configuration().sender_ratchet_configuration();
179
180        // Check if this message will modify the secret tree when decrypting a
181        // private message
182        let will_modify_secret_tree = matches!(message, ProtocolMessage::PrivateMessage(_));
183
184        // Checks the following semantic validation:
185        //  - ValSem002
186        //  - ValSem003
187        //  - ValSem006
188        //  - ValSem007 MembershipTag presence
189        let decrypted_message =
190            self.decrypt_message(provider.crypto(), message, &sender_ratchet_configuration)?;
191
192        // Persist the secret tree if it was modified to ensure forward secrecy
193        if will_modify_secret_tree {
194            provider
195                .storage()
196                .write_message_secrets(self.group_id(), &self.message_secrets_store)
197                .map_err(ProcessMessageError::StorageError)?;
198        }
199
200        let unverified_message = self
201            .public_group
202            .parse_message(decrypted_message, &self.message_secrets_store)
203            .map_err(ProcessMessageError::from)?;
204
205        Ok(unverified_message)
206    }

226    /// Creates a Commit message that covers the pending proposals that are
227    /// currently stored in the group's [ProposalStore]. The Commit message is
228    /// created even if there are no valid pending proposals.
229    ///
230    /// Returns an error if there is a pending commit. Otherwise it returns a
231    /// tuple of `Commit, Option<Welcome>, Option<GroupInfo>`, where `Commit`
232    /// and [`Welcome`] are MlsMessages of the type [`MlsMessageOut`].
233    ///
234    /// [`Welcome`]: crate::messages::Welcome
235    // FIXME: #1217
236    #[allow(clippy::type_complexity)]
237    pub fn commit_to_pending_proposals<Provider: OpenMlsProvider>(
238        &mut self,
239        provider: &Provider,
240        signer: &impl Signer,
241    ) -> Result<
242        (MlsMessageOut, Option<MlsMessageOut>, Option<GroupInfo>),
243        CommitToPendingProposalsError<Provider::StorageError>,
244    > {
245        self.is_operational()?;
246
247        // Build and stage the commit using the commit builder
248        // TODO #751
249        let (commit, welcome, group_info) = self
250            .commit_builder()
251            // This forces committing to the proposals in the proposal store:
252            .consume_proposal_store(true)
253            .load_psks(provider.storage())?
254            .build(provider.rand(), provider.crypto(), signer, |_| true)?
255            .stage_commit(provider)?
256            .into_contents();
257
258        Ok((
259            commit,
260            // Turn the [`Welcome`] to an [`MlsMessageOut`], if there is one
261            welcome.map(|welcome| MlsMessageOut::from_welcome(welcome, self.version())),

266    /// Merge a [StagedCommit] into the group after inspection. As this advances
267    /// the epoch of the group, it also clears any pending commits.
268    pub fn merge_staged_commit<Provider: OpenMlsProvider>(
269        &mut self,
270        provider: &Provider,
271        staged_commit: StagedCommit,
272    ) -> Result<(), MergeCommitError<Provider::StorageError>> {
273        // Check if we were removed from the group
274        if staged_commit.self_removed() {
275            self.group_state = MlsGroupState::Inactive;
276        }
277        provider
278            .storage()
279            .write_group_state(self.group_id(), &self.group_state)
280            .map_err(MergeCommitError::StorageError)?;
281
282        // Merge staged commit
283        self.merge_commit(provider, staged_commit)?;
284
285        // Extract and store the resumption psk for the current epoch
286        let resumption_psk = self.group_epoch_secrets().resumption_psk();
287        self.resumption_psk_store
288            .add(self.context().epoch(), resumption_psk.clone());
289        provider
290            .storage()
291            .write_resumption_psk_store(self.group_id(), &self.resumption_psk_store)
292            .map_err(MergeCommitError::StorageError)?;
293
294        // Delete own KeyPackageBundles
295        self.own_leaf_nodes.clear();
296        provider
297            .storage()
298            .delete_own_leaf_nodes(self.group_id())
299            .map_err(MergeCommitError::StorageError)?;
300
301        // Delete a potential pending commit
302        self.clear_pending_commit(provider.storage())
303            .map_err(MergeCommitError::StorageError)?;
304
305        Ok(())
306    }

308    /// Merges the pending [`StagedCommit`] if there is one, and
309    /// clears the field by setting it to `None`.
310    pub fn merge_pending_commit<Provider: OpenMlsProvider>(
311        &mut self,
312        provider: &Provider,
313    ) -> Result<(), MergePendingCommitError<Provider::StorageError>> {
314        match &self.group_state {
315            MlsGroupState::PendingCommit(_) => {
316                let old_state = mem::replace(&mut self.group_state, MlsGroupState::Operational);
317                if let MlsGroupState::PendingCommit(pending_commit_state) = old_state {
318                    self.merge_staged_commit(provider, (*pending_commit_state).into())?;
319                }
320                Ok(())
321            }
322            MlsGroupState::Inactive => Err(MlsGroupStateError::UseAfterEviction)?,
323            MlsGroupState::Operational => Ok(()),
324        }
325    }

327    /// Resolve a commit's virtual-clients derivation info to the per-commit
328    /// `OperationSecret` the receiver needs in order to recreate the path of an
329    /// own (sibling-virtual-client) commit.
330    ///
331    /// Only call this when the commit's sender is our own leaf. Other senders'
332    /// commits are processed via the normal HPKE path and never need to look up
333    /// VC secrets.
334    ///
335    /// Returns `Ok(None)` when the commit carries no virtual-clients
336    /// derivation-info entry on its update-path leaf (path-less commits, or
337    /// commits without an `app_data_dictionary`). Otherwise:
338    ///   - looks up the per-epoch `VcPprf` and `EpochEncryptionKey` the
339    ///     application registered via `register_vc_emulation_epoch`,
340    ///   - decrypts the wrapped `EpochInfoTbe` with the AEAD key,
341    ///   - hashes the decrypted `EpochInfoTbe` to produce the PPRF input,
342    ///   - evaluates the PPRF (puncturing it) and persists the punctured state
343    ///     back to storage,
344    ///   - returns the resulting `OperationSecret`.
345    ///
346    /// Once the precondition holds, missing PPRF / AEAD-key registration or
347    /// AEAD decryption failure is reported as an error: an own-leaf commit
348    /// carrying the VC component must be derivable.
349    #[cfg(feature = "virtual-clients-draft")]
350    fn load_vc_commit_material<Provider: OpenMlsProvider>(
351        &self,
352        provider: &Provider,
353        commit: &Commit,
354    ) -> Result<Option<crate::components::vc_derivation_info::OperationSecret>, StageCommitError>
355    {
356        use tls_codec::DeserializeBytes;
357
358        use crate::components::vc_derivation_info::{
359            pprf_input, DerivationInfo, EmulationEpochState, OperationSecret, VcPprf,
360            VirtualClientOperationType, VirtualClientsError, VC_COMPONENT_ID,
361        };
362
363        let Some(path) = commit.path.as_ref() else {
364            return Ok(None);
365        };
366        let Some(app_data_dict) = path.leaf_node().extensions().app_data_dictionary() else {
367            return Ok(None);
368        };
369        let Some(derivation_info_bytes) = app_data_dict.dictionary().get(&VC_COMPONENT_ID) else {
370            return Ok(None);
371        };
372        let derivation_info = DerivationInfo::tls_deserialize_exact_bytes(derivation_info_bytes)
373            .map_err(|e| {
374                log::error!("vc: derivation info deserialize failed: {e:?}");
375                VirtualClientsError::DerivationInfoMalformed
376            })?;
377
378        let epoch_id = derivation_info.epoch_id();
379        let storage = provider.storage();
380        let mut pprf: VcPprf = storage
381            .vc_pprf(epoch_id)
382            .map_err(|e| {
383                log::error!("vc: load pprf failed: {e:?}");
384                VirtualClientsError::StorageError
385            })?
386            .ok_or(VirtualClientsError::MissingPprf)?;
387        let state: EmulationEpochState = storage
388            .vc_emulation_epoch_state(epoch_id)
389            .map_err(|e| {
390                log::error!("vc: load emulation epoch state failed: {e:?}");
391                VirtualClientsError::StorageError
392            })?
393            .ok_or(VirtualClientsError::MissingEpochEncryptionKey)?;
394        // Receiver only needs the AEAD key; the leaf index travels on the
395        // wire inside `EpochInfoTbe` (sender's view), so it doesn't have
396        // to come from storage on this side.
397        let (_state_leaf_index, epoch_encryption_key) = state.into_parts();
398
399        let ciphersuite = self.ciphersuite();
400        let crypto = provider.crypto();
401        let epoch_info = derivation_info.decrypt(crypto, ciphersuite, &epoch_encryption_key)?;
402
403        // We only know how to consume `LeafNode` operation type for an
404        // update-path commit — anything else here is a sender bug or a
405        // cross-context replay. Treat it as malformed rather than silently
406        // deriving a secret in the wrong domain.
407        if !matches!(
408            epoch_info.operation_type,
409            VirtualClientOperationType::LeafNode
410        ) {
411            log::error!(
412                "vc: unexpected operation_type on update-path leaf: {:?}",
413                epoch_info.operation_type
414            );
415            return Err(VirtualClientsError::DerivationInfoMalformed.into());
416        }
417
418        let pprf_input_bytes = pprf_input(crypto, ciphersuite, &epoch_info)?;
419        let operation_secret: OperationSecret = pprf
420            .evaluate(crypto, ciphersuite, &pprf_input_bytes)
421            .map_err(VirtualClientsError::from)?
422            .into();
423
424        // Persist puncture immediately. Reprocessing the same commit will
425        // fail with `PprfError::PuncturedInput`, which is exactly what we
426        // want for forward secrecy.
427        storage.write_vc_pprf(epoch_id, &pprf).map_err(|e| {
428            log::error!("vc: persist punctured pprf failed: {e:?}");
429            VirtualClientsError::StorageError
430        })?;
431
432        Ok(Some(operation_secret))
433    }

435    /// Helper function to read decryption keypairs.
436    pub(super) fn read_decryption_keypairs(
437        &self,
438        provider: &impl OpenMlsProvider,
439        own_leaf_nodes: &[LeafNode],
440    ) -> Result<(Vec<EncryptionKeyPair>, Vec<EncryptionKeyPair>), StageCommitError> {
441        // All keys from the previous epoch are potential decryption keypairs.
442        let old_epoch_keypairs = self.read_epoch_keypairs(provider.storage()).map_err(|e| {
443            log::error!("Error reading epoch keypairs: {e:?}");
444            StageCommitError::MissingDecryptionKey
445        })?;
446
447        // If we are processing an update proposal that originally came from
448        // us, the keypair corresponding to the leaf in the update is also a
449        // potential decryption keypair.
450        let leaf_node_keypairs = own_leaf_nodes
451            .iter()
452            .map(|leaf_node| {
453                EncryptionKeyPair::read(provider, leaf_node.encryption_key())
454                    .ok_or(StageCommitError::MissingDecryptionKey)
455            })
456            .collect::<Result<Vec<EncryptionKeyPair>, StageCommitError>>()?;
457
458        Ok((old_epoch_keypairs, leaf_node_keypairs))
459    }

461    /// This function processes a message that may contain AppDataUpdate proposals.
462    /// Process these first an create an AppDataUpdates, then pass that into this function.
463    #[cfg(feature = "extensions-draft-08")]
464    pub fn process_unverified_message_with_app_data_updates<Provider: OpenMlsProvider>(
465        &self,
466        provider: &Provider,
467        unverified_message: UnverifiedMessage,
468        app_data_dict_updates: Option<AppDataUpdates>,
469    ) -> Result<ProcessedMessage, ProcessMessageError<Provider::StorageError>> {
470        // Checks the following semantic validation:
471        //  - ValSem010
472        //  - ValSem246 (as part of ValSem010)
473        //  - https://validation.openmls.tech/#valn1302
474        //  - https://validation.openmls.tech/#valn1304
475        let (content, credential) =
476            unverified_message.verify(self.ciphersuite(), provider.crypto(), self.version())?;
477
478        #[cfg_attr(not(feature = "extensions-draft-08"), allow(unused_mut))]
479        let mut processed = match content.sender() {
480            Sender::Member(_) | Sender::NewMemberProposal | Sender::NewMemberCommit => self
481                .process_internal_authenticated_content_with_app_data_updates(
482                    provider,
483                    content,
484                    credential,
485                    app_data_dict_updates,
486                )?,
487            Sender::External(_) => {
488                self.process_external_authenticated_content(provider, content, credential)?
489            }
490        };
491        #[cfg(feature = "extensions-draft-08")]
492        if self.context().safe_aad_required() {
493            processed
494                .try_attach_safe_aad()
495                .map_err(|_| ProcessMessageError::MalformedSafeAad)?;
496        }
497        Ok(processed)
498    }

558    /// This processing function does most of the semantic verifications.
559    /// It returns a [ProcessedMessage] enum.
560    /// Checks the following semantic validation:
561    ///  - ValSem008
562    ///  - ValSem010
563    ///  - ValSem101
564    ///  - ValSem102
565    ///  - ValSem104
566    ///  - ValSem106
567    ///  - ValSem107
568    ///  - ValSem108
569    ///  - ValSem110
570    ///  - ValSem111
571    ///  - ValSem112
572    ///  - ValSem113: All Proposals: The proposal type must be supported by all
573    ///    members of the group
574    ///  - ValSem200
575    ///  - ValSem201
576    ///  - ValSem202: Path must be the right length
577    ///  - ValSem203: Path secrets must decrypt correctly
578    ///  - ValSem204: Public keys from Path must be verified and match the
579    ///    private keys from the direct path
580    ///  - ValSem205
581    #[cfg(feature = "extensions-draft-08")]
582    fn process_internal_authenticated_content_with_app_data_updates<Provider: OpenMlsProvider>(
583        &self,
584        provider: &Provider,
585        content: AuthenticatedContent,
586        credential: Credential,
587        app_data_dict_updates: Option<AppDataUpdates>,
588    ) -> Result<ProcessedMessage, ProcessMessageError<Provider::StorageError>> {
589        let sender = content.sender().clone();
590        let authenticated_data = content.authenticated_data().to_owned();
591        let epoch = content.epoch();
592
593        let content = match content.content() {
594            FramedContentBody::Application(application_message) => {
595                ProcessedMessageContent::ApplicationMessage(ApplicationMessage::new(
596                    application_message.as_slice().to_owned(),
597                ))
598            }
599            FramedContentBody::Proposal(_) => {
600                let proposal = Box::new(QueuedProposal::from_authenticated_content_by_ref(
601                    self.ciphersuite(),
602                    provider.crypto(),
603                    content,
604                )?);
605
606                if matches!(sender, Sender::NewMemberProposal) {
607                    ProcessedMessageContent::ExternalJoinProposalMessage(proposal)
608                } else {
609                    ProcessedMessageContent::ProposalMessage(proposal)
610                }
611            }
612            FramedContentBody::Commit(commit) => {
613                // Since this is a commit, we need to load the private key material we need for decryption.
614                let (old_epoch_keypairs, leaf_node_keypairs) =
615                    self.read_decryption_keypairs(provider, &self.own_leaf_nodes)?;
616
617                // Only load virtual-clients material when the commit was sent
618                // by *our own* leaf — i.e. a sibling emulator of the same
619                // virtual client. A non-emulator member processing a VC
620                // commit never has the per-epoch PPRF / AEAD key in storage
621                // and must fall through to the normal HPKE path.
622                #[cfg(feature = "virtual-clients-draft")]
623                let vc_material = if matches!(&sender, Sender::Member(idx) if *idx == self.own_leaf_index())
624                {
625                    self.load_vc_commit_material(provider, commit)?
626                } else {
627                    None
628                };
629                #[cfg(not(feature = "virtual-clients-draft"))]
630                let _ = commit;
631
632                let staged_commit = self.stage_commit_with_app_data_updates(
633                    &content,
634                    old_epoch_keypairs,
635                    leaf_node_keypairs,
636                    app_data_dict_updates,
637                    provider,
638                    #[cfg(feature = "virtual-clients-draft")]
639                    vc_material,
640                )?;
641
642                ProcessedMessageContent::StagedCommitMessage(Box::new(staged_commit))
643            }
644        };
645
646        Ok(ProcessedMessage::new(
647            self.group_id().clone(),
648            epoch,
649            sender,
650            authenticated_data,
651            content,
652            credential,
653        ))
654    }

656    fn process_internal_authenticated_content<Provider: OpenMlsProvider>(
657        &self,
658        provider: &Provider,
659        content: AuthenticatedContent,
660        credential: Credential,
661    ) -> Result<ProcessedMessage, ProcessMessageError<Provider::StorageError>> {
662        let sender = content.sender().clone();
663        let authenticated_data = content.authenticated_data().to_owned();
664        let epoch = content.epoch();
665
666        let content = match content.content() {
667            FramedContentBody::Application(application_message) => {
668                ProcessedMessageContent::ApplicationMessage(ApplicationMessage::new(
669                    application_message.as_slice().to_owned(),
670                ))
671            }
672            FramedContentBody::Proposal(_) => {
673                let proposal = Box::new(QueuedProposal::from_authenticated_content_by_ref(
674                    self.ciphersuite(),
675                    provider.crypto(),
676                    content,
677                )?);
678
679                if matches!(sender, Sender::NewMemberProposal) {
680                    ProcessedMessageContent::ExternalJoinProposalMessage(proposal)
681                } else {
682                    ProcessedMessageContent::ProposalMessage(proposal)
683                }
684            }
685            #[cfg_attr(not(feature = "virtual-clients-draft"), allow(unused_variables))]
686            FramedContentBody::Commit(commit) => {
687                // Since this is a commit, we need to load the private key material we need for decryption.
688                let (old_epoch_keypairs, leaf_node_keypairs) =
689                    self.read_decryption_keypairs(provider, &self.own_leaf_nodes)?;
690
691                // See `process_internal_authenticated_content_with_app_data_updates`
692                // for the rationale: VC material lookup is gated on the
693                // commit being from our own leaf.
694                #[cfg(feature = "virtual-clients-draft")]
695                let vc_material = if matches!(&sender, Sender::Member(idx) if *idx == self.own_leaf_index())
696                {
697                    self.load_vc_commit_material(provider, commit)?
698                } else {
699                    None
700                };
701
702                let staged_commit = self.stage_commit(
703                    &content,
704                    old_epoch_keypairs,
705                    leaf_node_keypairs,
706                    provider,
707                    #[cfg(feature = "virtual-clients-draft")]
708                    vc_material,
709                )?;
710
711                ProcessedMessageContent::StagedCommitMessage(Box::new(staged_commit))
712            }
713        };
714
715        Ok(ProcessedMessage::new(
716            self.group_id().clone(),
717            epoch,
718            sender,
719            authenticated_data,
720            content,
721            credential,
722        ))
723    }

725    ///  - ValSem240
726    ///  - ValSem241
727    ///  - ValSem242
728    ///  - ValSem244
729    ///  - ValSem246 (as part of ValSem010)
730    fn process_external_authenticated_content<Provider: OpenMlsProvider>(
731        &self,
732        provider: &Provider,
733        content: AuthenticatedContent,
734        credential: Credential,
735    ) -> Result<ProcessedMessage, ProcessMessageError<Provider::StorageError>> {
736        let sender = content.sender().clone();
737        let data = content.authenticated_data().to_owned();
738
739        debug_assert!(matches!(sender, Sender::External(_)));
740
741        // https://validation.openmls.tech/#valn1501
742        match content.content() {
743            FramedContentBody::Application(_) => {
744                Err(ProcessMessageError::UnauthorizedExternalApplicationMessage)
745            }
746            // TODO: https://validation.openmls.tech/#valn1502
747            FramedContentBody::Proposal(Proposal::GroupContextExtensions(_)) => {
748                let content = ProcessedMessageContent::ProposalMessage(Box::new(
749                    QueuedProposal::from_authenticated_content_by_ref(
750                        self.ciphersuite(),
751                        provider.crypto(),
752                        content,
753                    )?,
754                ));
755                Ok(ProcessedMessage::new(
756                    self.group_id().clone(),
757                    self.context().epoch(),
758                    sender,
759                    data,
760                    content,
761                    credential,
762                ))
763            }
764
765            FramedContentBody::Proposal(Proposal::Remove(_)) => {
766                let content = ProcessedMessageContent::ProposalMessage(Box::new(
767                    QueuedProposal::from_authenticated_content_by_ref(
768                        self.ciphersuite(),
769                        provider.crypto(),
770                        content,
771                    )?,
772                ));
773                Ok(ProcessedMessage::new(
774                    self.group_id().clone(),
775                    self.context().epoch(),
776                    sender,
777                    data,
778                    content,
779                    credential,
780                ))
781            }
782            FramedContentBody::Proposal(Proposal::Add(_)) => {
783                let content = ProcessedMessageContent::ProposalMessage(Box::new(
784                    QueuedProposal::from_authenticated_content_by_ref(
785                        self.ciphersuite(),
786                        provider.crypto(),
787                        content,
788                    )?,
789                ));
790                Ok(ProcessedMessage::new(
791                    self.group_id().clone(),
792                    self.context().epoch(),
793                    sender,
794                    data,
795                    content,
796                    credential,
797                ))
798            }
799            // TODO #151/#106
800            FramedContentBody::Proposal(_) => Err(ProcessMessageError::UnsupportedProposalType),
801            FramedContentBody::Commit(_) => {
802                Err(ProcessMessageError::UnauthorizedExternalCommitMessage)
803            }
804        }
805    }

807    /// Performs framing validation and, if necessary, decrypts the given message.
808    ///
809    /// Returns the [`DecryptedMessage`] if processing is successful, or a
810    /// [`ValidationError`] if it is not.
811    ///
812    /// Checks the following semantic validation:
813    ///  - ValSem002
814    ///  - ValSem003
815    ///  - ValSem006
816    ///  - ValSem007 MembershipTag presence
817    ///  - https://validation.openmls.tech/#valn1202
818    pub(crate) fn decrypt_message(
819        &mut self,
820        crypto: &impl OpenMlsCrypto,
821        message: ProtocolMessage,
822        sender_ratchet_configuration: &SenderRatchetConfiguration,
823    ) -> Result<DecryptedMessage, ValidationError> {
824        // Checks the following semantic validation:
825        //  - ValSem002
826        //  - ValSem003
827        self.public_group.validate_framing(&message)?;
828
829        let epoch = message.epoch();
830
831        // Checks the following semantic validation:
832        //  - ValSem006
833        //  - ValSem007 MembershipTag presence
834        match message {
835            ProtocolMessage::PublicMessage(public_message) => {
836                // If the message is older than the current epoch, we need to fetch the correct secret tree first.
837                let message_secrets =
838                    self.message_secrets_for_epoch(epoch).map_err(|e| match e {
839                        SecretTreeError::TooDistantInThePast => ValidationError::NoPastEpochData,
840                        _ => LibraryError::custom(
841                            "Unexpected error while retrieving message secrets for epoch.",
842                        )
843                        .into(),
844                    })?;
845                DecryptedMessage::from_inbound_public_message(
846                    *public_message,
847                    message_secrets,
848                    message_secrets.serialized_context().to_vec(),
849                    crypto,
850                    self.ciphersuite(),
851                )
852            }
853            ProtocolMessage::PrivateMessage(ciphertext) => {
854                // If the message is older than the current epoch, we need to fetch the correct secret tree first
855                DecryptedMessage::from_inbound_ciphertext(
856                    ciphertext,
857                    crypto,
858                    self,
859                    sender_ratchet_configuration,
860                )
861            }
862        }
863    }
