117/// A verified Commit covering AppDataUpdate proposals that cannot be staged
118/// yet.
119///
120/// The AppDataUpdate proposals carry diffs in an application-defined format,
121/// so the application has to interpret them and compute the resulting
122/// [`AppDataUpdates`] before the commit can be staged: the updated
123/// [`AppDataDictionary`] becomes part of the new epoch's GroupContext and
124/// feeds into the key schedule.
125///
126/// Returned by [`MlsGroup::process_message()`] as
127/// [`ProcessedMessageContent::UnresolvedAppDataCommit`]. Inspect the proposals
128/// via [`Self::app_data_update_proposals()`], compute the updates with the
129/// help of [`MlsGroup::app_data_dictionary_updater()`] and resume staging via
130/// [`MlsGroup::stage_app_data_commit()`].
131///
132/// The message signature has already been verified at this point. Dropping
133/// this value discards the commit.
134#[cfg(feature = "extensions-draft")]
135pub struct UnresolvedAppDataCommit {
136    content: AuthenticatedContent,
137    /// The AppDataUpdate proposals covered by the commit, with proposals sent
138    /// by reference already resolved from the proposal store, sorted by
139    /// component id.
140    proposals: Vec<AppDataUpdateProposal>,
141    #[cfg(feature = "virtual-clients-draft")]
142    vc_commit_material: Option<crate::components::vc_derivation_info::VcCommitMaterial>,
143}
144
145#[cfg(feature = "extensions-draft")]
146impl UnresolvedAppDataCommit {
147    /// Returns the AppDataUpdate proposals covered by the commit, sorted by
148    /// component id. Proposals that were committed by reference have already
149    /// been resolved from the proposal store.
150    pub fn app_data_update_proposals(&self) -> impl Iterator<Item = &AppDataUpdateProposal> {
151        self.proposals.iter()
152    }
153}

156impl core::fmt::Debug for UnresolvedAppDataCommit {
157    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
158        let mut debug_struct = f.debug_struct("UnresolvedAppDataCommit");
159        debug_struct
160            .field("content", &self.content)
161            .field("proposals", &self.proposals);
162        // vc_commit_material holds secret key material, so only the epoch id

170                .map(|material| &material.epoch_id),
171        );
172        debug_struct.finish_non_exhaustive()
173    }

177    /// Parses incoming messages from the DS. Checks for syntactic errors and
178    /// makes some semantic checks as well. If the input is an encrypted
179    /// message, it will be decrypted. This processing function does syntactic
180    /// and semantic validation of the message. It returns a [ProcessedMessage]
181    /// enum.
182    ///
183    /// A commit covering AppDataUpdate proposals is returned as
184    /// [`ProcessedMessageContent::UnresolvedAppDataCommit`], since the
185    /// application has to interpret the proposals before the commit can be
186    /// staged via [`MlsGroup::stage_app_data_commit()`].
187    ///
188    /// # Errors:
189    /// Returns an [`ProcessMessageError`] when the validation checks fail
190    /// with the exact reason of the failure.
191    pub fn process_message<Provider: OpenMlsProvider>(
192        &mut self,
193        provider: &Provider,
194        message: impl Into<ProtocolMessage>,
195    ) -> Result<ProcessedMessage, ProcessMessageError<Provider::StorageError>> {
196        let unverified_message = self.unprotect_message(provider, message)?;
197        self.process_unverified_message(provider, unverified_message)
198    }

206    /// Parses and deprotects incoming messages from the DS. Checks for syntactic errors, but only
207    /// performs limited semantic checks.
208    pub(crate) fn unprotect_message<Provider: OpenMlsProvider>(
209        &mut self,
210        provider: &Provider,
211        message: impl Into<ProtocolMessage>,
212    ) -> Result<UnverifiedMessage, ProcessMessageError<Provider::StorageError>> {
213        // Make sure we are still a member of the group
214        if !self.is_active() {
215            return Err(ProcessMessageError::GroupStateError(
216                MlsGroupStateError::UseAfterEviction,
217            ));
218        }
219        let message = message.into();
220
221        // Check that handshake messages are compatible with the incoming wire format policy
222        if !message.is_external()
223            && message.is_handshake_message()
224            && !self
225                .configuration()
226                .wire_format_policy()
227                .incoming()
228                .is_compatible_with(message.wire_format())
229        {
230            return Err(ProcessMessageError::IncompatibleWireFormat);
231        }
232
233        // Parse the message
234        let sender_ratchet_configuration = *self.configuration().sender_ratchet_configuration();
235
236        // Check if this message will modify the secret tree when decrypting a
237        // private message
238        let will_modify_secret_tree = matches!(message, ProtocolMessage::PrivateMessage(_));
239
240        // Resolve the emulator reuse-guard context for `PrivateMessage`
241        // before calling `decrypt_message` so storage errors surface as
242        // `ProcessMessageError::StorageError`. `PublicMessage` carries no
243        // `reuse_guard`, so the lookup is skipped for it. The binding is
244        // looked up at the epoch the message was sent in: a delayed message
245        // from a past epoch must be deprotected with the emulation state
246        // that was bound then, not the latest one.
247        #[cfg(feature = "virtual-clients-draft")]
248        let emulation_state = if let ProtocolMessage::PrivateMessage(private_message) = &message {
249            self.vc_emulation_state_at_epoch(provider.storage(), private_message.epoch())
250                .map_err(|e| match e {
251                    super::VcEmulationStateError::Storage(e) => {
252                        ProcessMessageError::StorageError(e)
253                    }
254                    super::VcEmulationStateError::MissingEmulationEpochState => {
255                        ProcessMessageError::ValidationError(
256                            crate::group::ValidationError::UnableToDecrypt(
257                                crate::framing::errors::MessageDecryptionError::VirtualClientsError(
258                                    crate::components::vc_derivation_info::VirtualClientsError::MissingEmulationEpochState,
259                                ),
260                            ),
261                        )
262                    }
263                })?
264        } else {
265            None
266        };
267        #[cfg(feature = "virtual-clients-draft")]
268        let emulator_ctx: Option<crate::framing::EmulatorReuseGuardCtx<'_>> = emulation_state
269            .as_ref()
270            .map(|state| state.reuse_guard_inputs());
271
272        // Checks the following semantic validation:
273        //  - ValSem002
274        //  - ValSem003
275        //  - ValSem006
276        //  - ValSem007 MembershipTag presence
277        let decrypted_message = self.decrypt_message(
278            provider.crypto(),
279            message,
280            &sender_ratchet_configuration,
281            #[cfg(feature = "virtual-clients-draft")]
282            emulator_ctx.as_ref(),
283        )?;
284
285        // Persist the secret tree if it was modified to ensure forward secrecy
286        if will_modify_secret_tree {
287            provider
288                .storage()
289                .write_message_secrets(self.group_id(), &self.message_secrets_store)
290                .map_err(ProcessMessageError::StorageError)?;
291        }
292
293        let unverified_message = self
294            .public_group
295            .parse_message(decrypted_message, &self.message_secrets_store)
296            .map_err(ProcessMessageError::from)?;
297
298        Ok(unverified_message)
299    }

319    /// Creates a Commit message that covers the pending proposals that are
320    /// currently stored in the group's [ProposalStore]. The Commit message is
321    /// created even if there are no valid pending proposals.
322    ///
323    /// Returns an error if there is a pending commit. Otherwise it returns a
324    /// tuple of `Commit, Option<Welcome>, Option<GroupInfo>`, where `Commit`
325    /// and [`Welcome`] are MlsMessages of the type [`MlsMessageOut`].
326    ///
327    /// [`Welcome`]: crate::messages::Welcome
328    // FIXME: #1217
329    #[allow(clippy::type_complexity)]
330    pub fn commit_to_pending_proposals<Provider: OpenMlsProvider>(
331        &mut self,
332        provider: &Provider,
333        signer: &impl Signer,
334    ) -> Result<
335        (MlsMessageOut, Option<MlsMessageOut>, Option<GroupInfo>),
336        CommitToPendingProposalsError<Provider::StorageError>,
337    > {
338        self.is_operational()?;
339
340        // Build and stage the commit using the commit builder
341        // TODO #751
342        let (commit, welcome, group_info) = self
343            .commit_builder()
344            // This forces committing to the proposals in the proposal store:
345            .consume_proposal_store(true)
346            .load_psks(provider.storage())?
347            .build(provider.rand(), provider.crypto(), signer, |_| true)?
348            .stage_commit(provider)?
349            .into_contents();
350
351        Ok((
352            commit,
353            // Turn the [`Welcome`] to an [`MlsMessageOut`], if there is one
354            welcome.map(|welcome| MlsMessageOut::from_welcome(welcome, self.version())),
355            group_info,
356        ))
357    }

359    /// Merge a [StagedCommit] into the group after inspection. As this advances
360    /// the epoch of the group, it also clears any pending commits.
361    pub fn merge_staged_commit<Provider: OpenMlsProvider>(
362        &mut self,
363        provider: &Provider,
364        staged_commit: StagedCommit,
365    ) -> Result<(), MergeCommitError<Provider::StorageError>> {
366        // Check if we were removed from the group
367        if staged_commit.self_removed() {
368            self.group_state = MlsGroupState::Inactive;
369        }
370        provider
371            .storage()
372            .write_group_state(self.group_id(), &self.group_state)
373            .map_err(MergeCommitError::StorageError)?;
374
375        // Update the per-epoch emulation bindings. Self-removal drops them.
376        // Otherwise the epoch the commit moves the group into is bound to
377        // the emulation epoch of the commit's VC leaf, or, if the commit
378        // does not install a new VC leaf, to the binding of the current
379        // epoch, since the VC leaf stays active across commits by other
380        // members.
381        #[cfg(feature = "virtual-clients-draft")]
382        if staged_commit.self_removed() {
383            provider
384                .storage()
385                .delete_vc_emulation_bindings(self.group_id())
386                .map_err(|e| {
387                    log::error!("vc: drop emulation bindings on self-removal failed: {e:?}");
388                    MergeCommitError::StorageError(e)
389                })?;
390        } else {
391            let mut bindings: crate::components::vc_derivation_info::VcEmulationBindings = provider
392                .storage()
393                .vc_emulation_bindings(self.group_id())
394                .map_err(MergeCommitError::StorageError)?
395                .unwrap_or_default();
396            let epoch_id = staged_commit
397                .vc_emulation_epoch_id
398                .clone()
399                .or_else(|| bindings.get(self.epoch()).cloned());
400            if let Some(epoch_id) = epoch_id {
401                // Keep one entry per retained message-secrets epoch plus
402                // the new current one, so bindings age out in lockstep
403                // with the message secrets they are needed for.
404                let max_entries = self.message_secrets_store.max_epochs.saturating_add(1);
405                bindings.insert(staged_commit.epoch(), epoch_id, max_entries);
406                provider
407                    .storage()
408                    .write_vc_emulation_bindings(self.group_id(), &bindings)
409                    .map_err(|e| {
410                        log::error!("vc: persist emulation bindings at merge failed: {e:?}");
411                        MergeCommitError::StorageError(e)
412                    })?;
413            }
414        }
415
416        // Merge staged commit
417        self.merge_commit(provider, staged_commit)?;
418
419        // Extract and store the resumption psk for the current epoch
420        let resumption_psk = self.group_epoch_secrets().resumption_psk();
421        self.resumption_psk_store
422            .add(self.context().epoch(), resumption_psk.clone());
423        provider
424            .storage()
425            .write_resumption_psk_store(self.group_id(), &self.resumption_psk_store)
426            .map_err(MergeCommitError::StorageError)?;
427
428        // Delete own KeyPackageBundles
429        self.own_leaf_nodes.clear();
430        provider
431            .storage()
432            .delete_own_leaf_nodes(self.group_id())
433            .map_err(MergeCommitError::StorageError)?;
434
435        // Delete a potential pending commit
436        self.clear_pending_commit(provider.storage())
437            .map_err(MergeCommitError::StorageError)?;
438
439        Ok(())
440    }

442    /// Merges the pending [`StagedCommit`] if there is one, and
443    /// clears the field by setting it to `None`.
444    pub fn merge_pending_commit<Provider: OpenMlsProvider>(
445        &mut self,
446        provider: &Provider,
447    ) -> Result<(), MergePendingCommitError<Provider::StorageError>> {
448        match &self.group_state {
449            MlsGroupState::PendingCommit(_) => {
450                let old_state = mem::replace(&mut self.group_state, MlsGroupState::Operational);
451                if let MlsGroupState::PendingCommit(pending_commit_state) = old_state {
452                    self.merge_staged_commit(provider, (*pending_commit_state).into())?;
453                }
454                Ok(())
455            }
456            MlsGroupState::Inactive => Err(MlsGroupStateError::UseAfterEviction)?,
457            MlsGroupState::Operational => Ok(()),
458        }
459    }

461    /// Resolve a commit's virtual-clients derivation info to the per-commit
462    /// `OperationSecret` the receiver needs in order to recreate the path of a
463    /// commit sent by a sibling emulator client, plus the `EpochId` the
464    /// commit binds the group to on merge.
465    ///
466    /// See [`is_sibling_vc_commit`] for the precondition the caller must
467    /// check before invoking this helper. Sibling commits come in two
468    /// shapes:
469    ///
470    ///   * own-leaf VC commits, where a sibling emulator committed through our
471    ///     shared higher-level leaf
472    ///   * sibling-resync external commits, where a sibling emulator joined this
473    ///     higher-level group externally onto a leaf of their own,
474    ///     inline-removing our previous leaf.
475    ///
476    /// Returns `Ok(None)` when the commit carries no virtual-clients
477    /// derivation-info entry on its update-path leaf (path-less commits, or
478    /// commits without an `app_data_dictionary`). Otherwise:
479    ///   - looks up the per-epoch `EmulationEpochState` and operation secret
480    ///     tree the application registered via `register_vc_emulation_epoch`,
481    ///   - decrypts the wrapped `DerivationInfoTbe` with the AEAD key/nonce
482    ///     derived from the epoch encryption key and the path leaf's
483    ///     serialized encryption key,
484    ///   - derives the operation secret positionally from the tree at the
485    ///     sender's emulation-leaf coordinates and persists the advanced
486    ///     tree,
487    ///   - returns the resulting `OperationSecret` and `EpochId`.
488    ///
489    /// A generation the tree reports as already consumed fails with
490    /// `OperationGenerationConsumed`. Operation secrets are consume-once,
491    /// matching the semantics of regular PrivateMessage decryption.
492    #[cfg(feature = "virtual-clients-draft")]
493    pub(super) fn load_vc_commit_material<Provider: OpenMlsProvider>(
494        &self,
495        provider: &Provider,
496        commit: &Commit,
497    ) -> Result<Option<crate::components::vc_derivation_info::VcCommitMaterial>, StageCommitError>
498    {
499        use tls_codec::{DeserializeBytes, Serialize as _};
500
501        use crate::{
502            components::vc_derivation_info::{
503                DerivationInfo, EmulationEpochState, VirtualClientOperationType,
504                VirtualClientsError, VC_COMPONENT_ID,
505            },
506            components::vc_operation_tree::OperationSecretTree,
507            treesync::node::leaf_node::LeafNodeSource,
508        };
509
510        let Some(path) = commit.path.as_ref() else {
511            return Ok(None);
512        };
513        let Some(app_data_dict) = path.leaf_node().extensions().app_data_dictionary() else {
514            return Ok(None);
515        };
516        let Some(derivation_info_bytes) = app_data_dict.dictionary().get(&VC_COMPONENT_ID) else {
517            return Ok(None);
518        };
519        let derivation_info = DerivationInfo::tls_deserialize_exact_bytes(derivation_info_bytes)
520            .map_err(|e| {
521                log::error!("vc: derivation info deserialize failed: {e:?}");
522                VirtualClientsError::DerivationInfoMalformed
523            })?;
524
525        let epoch_id = derivation_info.epoch_id();
526        let storage = provider.storage();
527        let state: EmulationEpochState = storage
528            .vc_emulation_epoch_state(epoch_id)
529            .map_err(|e| {
530                log::error!("vc: load emulation epoch state failed: {e:?}");
531                VirtualClientsError::StorageError
532            })?
533            .ok_or(VirtualClientsError::MissingEmulationEpochState)?;
534        let mut operation_tree: OperationSecretTree = storage
535            .vc_operation_tree(epoch_id)
536            .map_err(|e| {
537                log::error!("vc: load operation tree failed: {e:?}");
538                VirtualClientsError::StorageError
539            })?
540            .ok_or(VirtualClientsError::MissingOperationTree)?;
541        // The receiver uses the emulation epoch's AEAD key and ciphersuite
542        // for `DerivationInfoTbe`. The sender's emulation leaf index travels
543        // on the wire, so it doesn't have to come from storage on this side.
544        let (_state_leaf_index, epoch_encryption_key, emulation_ciphersuite) = state.into_parts();
545
546        let crypto = provider.crypto();
547        let leaf_encryption_key = path
548            .leaf_node()
549            .encryption_key()
550            .tls_serialize_detached()
551            .map_err(VirtualClientsError::from)?;
552        // The operation type is not on the wire. It is inferred from the
553        // carrying leaf's source: key-package leaves map to `KeyPackage`,
554        // update and commit leaves map to `LeafNode`. Only `LeafNode` is
555        // wired up today, and an update-path leaf always has a commit
556        // source. It selects the tagless `DerivationInfoTbe` variant the
557        // plaintext decodes into.
558        let operation_type = match path.leaf_node().leaf_node_source() {
559            LeafNodeSource::KeyPackage(_) => {
560                log::error!("vc: key-package leaf on an update path");
561                return Err(VirtualClientsError::DerivationInfoMalformed.into());
562            }
563            LeafNodeSource::Update | LeafNodeSource::Commit(_) => {
564                VirtualClientOperationType::LeafNode
565            }
566        };
567        let tbe = derivation_info.decrypt(
568            crypto,
569            emulation_ciphersuite,
570            &epoch_encryption_key,
571            &leaf_encryption_key,
572            operation_type,
573        )?;
574        // Carried by an external commit's leaf only; `None` for own-leaf
575        // (regular) VC commits. A sibling uses it as the new epoch's external
576        // init secret instead of decapsulating from the previous epoch's
577        // `external_secret`.
578        let external_init_secret = tbe.external_init_secret().cloned();
579        // The operation context for `LeafNode` operations is the
580        // higher-level group's id.
581        let operation_context = self.group_id().as_slice().to_vec();
582
583        // An already-consumed generation propagates as a hard error here:
584        // operation secrets are consume-once, like the per-generation keys
585        // of regular PrivateMessage decryption.
586        let operation_secret = operation_tree.derive_operation_secret(
587            crypto,
588            emulation_ciphersuite,
589            epoch_id,
590            tbe.leaf_index(),
591            operation_type,
592            tbe.generation(),
593            &operation_context,
594        )?;
595        // Persist the advanced tree immediately, before any key material is
596        // derived from the secret.
597        storage
598            .write_vc_operation_tree(epoch_id, &operation_tree)
599            .map_err(|e| {
600                log::error!("vc: persist advanced operation tree failed: {e:?}");
601                VirtualClientsError::StorageError
602            })?;
603
604        Ok(Some(
605            crate::components::vc_derivation_info::VcCommitMaterial {
606                epoch_id: epoch_id.clone(),
607                operation_secret,
608                external_init_secret,
609            },
610        ))
611    }

613    /// Helper function to read decryption keypairs.
614    pub(super) fn read_decryption_keypairs(
615        &self,
616        provider: &impl OpenMlsProvider,
617        own_leaf_nodes: &[LeafNode],
618    ) -> Result<(Vec<EncryptionKeyPair>, Vec<EncryptionKeyPair>), StageCommitError> {
619        // All keys from the previous epoch are potential decryption keypairs.
620        let old_epoch_keypairs = self.read_epoch_keypairs(provider.storage()).map_err(|e| {
621            log::error!("Error reading epoch keypairs: {e:?}");
622            StageCommitError::MissingDecryptionKey
623        })?;
624
625        // If we are processing an update proposal that originally came from
626        // us, the keypair corresponding to the leaf in the update is also a
627        // potential decryption keypair.
628        let leaf_node_keypairs = own_leaf_nodes
629            .iter()
630            .map(|leaf_node| {
631                EncryptionKeyPair::read(provider, leaf_node.encryption_key())
632                    .ok_or(StageCommitError::MissingDecryptionKey)
633            })
634            .collect::<Result<Vec<EncryptionKeyPair>, StageCommitError>>()?;
635
636        Ok((old_epoch_keypairs, leaf_node_keypairs))
637    }

670    /// This processing function does most of the semantic verifications.
671    /// It returns a [ProcessedMessage] enum.
672    ///
673    /// Checks the following semantic validation:
674    ///  - ValSem008
675    ///  - ValSem010
676    ///  - ValSem101
677    ///  - ValSem102
678    ///  - ValSem104
679    ///  - ValSem106
680    ///  - ValSem107
681    ///  - ValSem108
682    ///  - ValSem110
683    ///  - ValSem111
684    ///  - ValSem112
685    ///  - ValSem113: All Proposals: The proposal type must be supported by all
686    ///    members of the group
687    ///  - ValSem200
688    ///  - ValSem201
689    ///  - ValSem202: Path must be the right length
690    ///  - ValSem203: Path secrets must decrypt correctly
691    ///  - ValSem204: Public keys from Path must be verified and match the
692    ///    private keys from the direct path
693    ///  - ValSem205
694    pub(crate) fn process_unverified_message<Provider: OpenMlsProvider>(
695        &self,
696        provider: &Provider,
697        unverified_message: UnverifiedMessage,
698    ) -> Result<ProcessedMessage, ProcessMessageError<Provider::StorageError>> {
699        // Checks the following semantic validation:
700        //  - ValSem010
701        //  - ValSem246 (as part of ValSem010)
702        //  - https://validation.openmls.tech/#valn1302
703        //  - https://validation.openmls.tech/#valn1304
704        let verified =
705            unverified_message.verify(self.ciphersuite(), provider.crypto(), self.version())?;
706
707        #[cfg_attr(not(feature = "extensions-draft"), allow(unused_mut))]
708        let mut processed = match verified.content.sender() {
709            Sender::Member(_) | Sender::NewMemberProposal | Sender::NewMemberCommit => self
710                .process_internal_authenticated_content(
711                    provider,
712                    verified.content,
713                    verified.credential,
714                    #[cfg(feature = "virtual-clients-draft")]
715                    verified.emulator_sender_leaf_index,
716                )?,
717            Sender::External(_) => self.process_external_authenticated_content(
718                provider,
719                verified.content,
720                verified.credential,
721            )?,
722        };
723        #[cfg(feature = "extensions-draft")]
724        if self.context().safe_aad_required() {
725            processed
726                .try_attach_safe_aad()
727                .map_err(|_| ProcessMessageError::MalformedSafeAad)?;
728        }
729        Ok(processed)
730    }

732    fn process_internal_authenticated_content<Provider: OpenMlsProvider>(
733        &self,
734        provider: &Provider,
735        content: AuthenticatedContent,
736        credential: Credential,
737        #[cfg(feature = "virtual-clients-draft")] emulator_sender_leaf_index: Option<LeafNodeIndex>,
738    ) -> Result<ProcessedMessage, ProcessMessageError<Provider::StorageError>> {
739        let sender = content.sender().clone();
740        let authenticated_data = content.authenticated_data().to_owned();
741        let epoch = content.epoch();
742
743        let content = match content.content() {
744            FramedContentBody::Application(application_message) => {
745                ProcessedMessageContent::ApplicationMessage(ApplicationMessage::new(
746                    application_message.as_slice().to_owned(),
747                ))
748            }
749            FramedContentBody::Proposal(_) => {
750                let proposal = Box::new(QueuedProposal::from_authenticated_content_by_ref(
751                    self.ciphersuite(),
752                    provider.crypto(),
753                    content,
754                )?);
755
756                if matches!(sender, Sender::NewMemberProposal) {
757                    ProcessedMessageContent::ExternalJoinProposalMessage(proposal)
758                } else {
759                    ProcessedMessageContent::ProposalMessage(proposal)
760                }
761            }
762            FramedContentBody::Commit(commit) => {
763                // Load virtual-client derivation info when this commit was
764                // authored by a sibling emulator through a leaf shared with us.
765                // A Commit with an UpdatePath carrying this material is staged
766                // via the VC path below rather than treated as our own (see the
767                // own-commit handling further down). The receiver only loads it
768                // when the commit shape lets it identify itself as a sibling:
769                //
770                // * `Sender::Member(idx)` with `idx == own_leaf_index`: the
771                //   sender committed through our shared higher-level leaf, so
772                //   we are a sibling.
773                // * `Sender::NewMemberCommit` with an inline `Remove(own_leaf)`:
774                //   the sender is a sibling joining externally and the
775                //   auto-Remove targets our previous leaf, so we are the
776                //   sibling being resynced.
777                #[cfg(feature = "virtual-clients-draft")]
778                let vc_commit_material =
779                    if is_sibling_vc_commit(commit, &sender, self.own_leaf_index()) {
780                        self.load_vc_commit_material(provider, commit)?
781                    } else {
782                        None
783                    };
784
785                let is_own_commit =
786                    matches!(&sender, Sender::Member(member) if member == &self.own_leaf_index());
787                #[cfg(feature = "virtual-clients-draft")]
788                let is_own_commit = is_own_commit && vc_commit_material.is_none();
789
790                if is_own_commit {
791                    let received_tag = content
792                        .confirmation_tag()
793                        .ok_or(StageCommitError::ConfirmationTagMissing)?;
794                    if self.matches_pending_commit(received_tag) {
795                        // The Commit is our pending commit this client got
796                        // fanned out by the delivery service: surface
797                        // `OwnPendingCommit` so the caller merges the pending
798                        // commit instead of staging the fanned-out Commit.
799                        return Ok(ProcessedMessage::new(
800                            self.group_id().clone(),
801                            epoch,
802                            sender,
803                            authenticated_data,
804                            ProcessedMessageContent::OwnPendingCommit,
805                            credential,
806                            #[cfg(feature = "virtual-clients-draft")]
807                            emulator_sender_leaf_index,
808                        ));
809                    }
810                    // Not our pending commit. We cannot decrypt a path we
811                    // encrypted to the other members, so a Commit with an
812                    // UpdatePath is unprocessable. A Commit without an
813                    // UpdatePath carries no author-private material and falls
814                    // through to staging (a sibling's Commit without an
815                    // UpdatePath, or our own commit replayed after the pending
816                    // commit was cleared).
817                    if commit.path.is_some() {
818                        return Err(StageCommitError::OwnCommitMismatch.into());
819                    }
820                }
821
822                // A commit covering AppDataUpdate proposals cannot be staged
823                // immediately: the proposals contain diffs in an
824                // application-defined format, so the application has to
825                // interpret them and supply the resulting dictionary entries
826                // first. The verified content is handed back to the caller,
827                // who resumes staging via `MlsGroup::stage_app_data_commit`.
828                #[cfg(feature = "extensions-draft")]
829                {
830                    let app_data_update_proposals =
831                        committed_app_data_update_proposals(commit, self.proposal_store());
832                    if !app_data_update_proposals.is_empty() {
833                        let unresolved_commit = UnresolvedAppDataCommit {
834                            content,
835                            proposals: app_data_update_proposals,
836                            #[cfg(feature = "virtual-clients-draft")]
837                            vc_commit_material,
838                        };
839                        return Ok(ProcessedMessage::new(
840                            self.group_id().clone(),
841                            epoch,
842                            sender,
843                            authenticated_data,
844                            ProcessedMessageContent::UnresolvedAppDataCommit(Box::new(
845                                unresolved_commit,
846                            )),
847                            credential,
848                            #[cfg(feature = "virtual-clients-draft")]
849                            emulator_sender_leaf_index,
850                        ));
851                    }
852                }
853
854                // Since this is a commit, we need to load the private key material we need for decryption.
855                let (old_epoch_keypairs, leaf_node_keypairs) =
856                    self.read_decryption_keypairs(provider, &self.own_leaf_nodes)?;
857
858                let staged_commit = self.stage_commit(
859                    &content,
860                    old_epoch_keypairs,
861                    leaf_node_keypairs,
862                    provider,
863                    #[cfg(feature = "virtual-clients-draft")]
864                    vc_commit_material,
865                )?;
866
867                ProcessedMessageContent::StagedCommitMessage(Box::new(staged_commit))
868            }
869        };
870
871        Ok(ProcessedMessage::new(
872            self.group_id().clone(),
873            epoch,
874            sender,
875            authenticated_data,
876            content,
877            credential,
878            #[cfg(feature = "virtual-clients-draft")]
879            emulator_sender_leaf_index,
880        ))
881    }

883    ///  - ValSem240
884    ///  - ValSem241
885    ///  - ValSem242
886    ///  - ValSem244
887    ///  - ValSem246 (as part of ValSem010)
888    fn process_external_authenticated_content<Provider: OpenMlsProvider>(
889        &self,
890        provider: &Provider,
891        content: AuthenticatedContent,
892        credential: Credential,
893    ) -> Result<ProcessedMessage, ProcessMessageError<Provider::StorageError>> {
894        #[cfg(feature = "virtual-clients-draft")]
895        let emulator_sender_leaf_index: Option<crate::binary_tree::LeafNodeIndex> = None;
896        let sender = content.sender().clone();
897        let data = content.authenticated_data().to_owned();
898
899        debug_assert!(matches!(sender, Sender::External(_)));
900
901        // https://validation.openmls.tech/#valn1501
902        match content.content() {
903            FramedContentBody::Application(_) => {
904                Err(ProcessMessageError::UnauthorizedExternalApplicationMessage)
905            }
906            // TODO: https://validation.openmls.tech/#valn1502
907            FramedContentBody::Proposal(Proposal::GroupContextExtensions(_)) => {
908                let content = ProcessedMessageContent::ProposalMessage(Box::new(
909                    QueuedProposal::from_authenticated_content_by_ref(
910                        self.ciphersuite(),
911                        provider.crypto(),
912                        content,
913                    )?,
914                ));
915                Ok(ProcessedMessage::new(
916                    self.group_id().clone(),
917                    self.context().epoch(),
918                    sender,
919                    data,
920                    content,
921                    credential,
922                    #[cfg(feature = "virtual-clients-draft")]
923                    emulator_sender_leaf_index,
924                ))
925            }
926
927            FramedContentBody::Proposal(Proposal::Remove(_)) => {
928                let content = ProcessedMessageContent::ProposalMessage(Box::new(
929                    QueuedProposal::from_authenticated_content_by_ref(
930                        self.ciphersuite(),
931                        provider.crypto(),
932                        content,
933                    )?,
934                ));
935                Ok(ProcessedMessage::new(
936                    self.group_id().clone(),
937                    self.context().epoch(),
938                    sender,
939                    data,
940                    content,
941                    credential,
942                    #[cfg(feature = "virtual-clients-draft")]
943                    emulator_sender_leaf_index,
944                ))
945            }
946            FramedContentBody::Proposal(Proposal::Add(_)) => {
947                let content = ProcessedMessageContent::ProposalMessage(Box::new(
948                    QueuedProposal::from_authenticated_content_by_ref(
949                        self.ciphersuite(),
950                        provider.crypto(),
951                        content,
952                    )?,
953                ));
954                Ok(ProcessedMessage::new(
955                    self.group_id().clone(),
956                    self.context().epoch(),
957                    sender,
958                    data,
959                    content,
960                    credential,
961                    #[cfg(feature = "virtual-clients-draft")]
962                    emulator_sender_leaf_index,
963                ))
964            }
965            // TODO #151/#106
966            FramedContentBody::Proposal(_) => Err(ProcessMessageError::UnsupportedProposalType),
967            FramedContentBody::Commit(_) => {
968                Err(ProcessMessageError::UnauthorizedExternalCommitMessage)
969            }
970        }
971    }

973    /// Performs framing validation and, if necessary, decrypts the given message.
974    ///
975    /// Returns the [`DecryptedMessage`] if processing is successful, or a
976    /// [`ValidationError`] if it is not.
977    ///
978    /// Checks the following semantic validation:
979    ///  - ValSem002
980    ///  - ValSem003
981    ///  - ValSem006
982    ///  - ValSem007 MembershipTag presence
983    ///  - https://validation.openmls.tech/#valn1202
984    pub(crate) fn decrypt_message(
985        &mut self,
986        crypto: &impl OpenMlsCrypto,
987        message: ProtocolMessage,
988        sender_ratchet_configuration: &SenderRatchetConfiguration,
989        #[cfg(feature = "virtual-clients-draft")] emulator_ctx: Option<
990            &crate::framing::EmulatorReuseGuardCtx<'_>,
991        >,
992    ) -> Result<DecryptedMessage, ValidationError> {
993        // Checks the following semantic validation:
994        //  - ValSem002
995        //  - ValSem003
996        self.public_group.validate_framing(&message)?;
997
998        let epoch = message.epoch();
999
1000        // Checks the following semantic validation:
1001        //  - ValSem006
1002        //  - ValSem007 MembershipTag presence
1003        match message {
1004            ProtocolMessage::PublicMessage(public_message) => {
1005                // If the message is older than the current epoch, we need to fetch the correct secret tree first.
1006                let message_secrets =
1007                    self.message_secrets_for_epoch(epoch).map_err(|e| match e {
1008                        SecretTreeError::TooDistantInThePast => ValidationError::NoPastEpochData,
1009                        _ => LibraryError::custom(
1010                            "Unexpected error while retrieving message secrets for epoch.",
1011                        )
1012                        .into(),
1013                    })?;
1014                DecryptedMessage::from_inbound_public_message(
1015                    *public_message,
1016                    message_secrets,
1017                    message_secrets.serialized_context().to_vec(),
1018                    crypto,
1019                    self.ciphersuite(),
1020                )
1021            }
1022            ProtocolMessage::PrivateMessage(ciphertext) => {
1023                // If the message is older than the current epoch, we need to fetch the correct secret tree first
1024                DecryptedMessage::from_inbound_ciphertext(
1025                    ciphertext,
1026                    crypto,
1027                    self,
1028                    sender_ratchet_configuration,
1029                    #[cfg(feature = "virtual-clients-draft")]
1030                    emulator_ctx,
1031                )
1032            }
1033        }
1034    }

1037/// Collects the AppDataUpdate proposals covered by a commit, sorted by
1038/// component id.
1039///
1040/// Proposals sent by reference are resolved from the proposal store. A
1041/// reference that cannot be resolved is skipped here: staging fails on it
1042/// later with the regular missing-proposal error, so it does not need to be
1043/// surfaced at detection time.
1044#[cfg(feature = "extensions-draft")]
1045fn committed_app_data_update_proposals(
1046    commit: &Commit,
1047    proposal_store: &ProposalStore,
1048) -> Vec<AppDataUpdateProposal> {
1049    use crate::messages::proposals::ProposalOrRef;
1050
1051    let mut proposals: Vec<AppDataUpdateProposal> = commit
1052        .proposals
1053        .iter()
1054        .filter_map(|proposal_or_ref| match proposal_or_ref {
1055            ProposalOrRef::Proposal(proposal) => match proposal.as_ref() {
1056                Proposal::AppDataUpdate(proposal) => Some(proposal.as_ref().clone()),
1057                _ => None,
1058            },
1059            ProposalOrRef::Reference(reference) => proposal_store
1060                .proposals()
1061                .find(|queued_proposal| {
1062                    queued_proposal.proposal_reference_ref() == reference.as_ref()
1063                })
1064                .and_then(|queued_proposal| match queued_proposal.proposal() {
1065                    Proposal::AppDataUpdate(proposal) => Some(proposal.as_ref().clone()),
1066                    _ => None,
1067                }),
1068        })
1069        .collect();
1070
1071    proposals.sort_by_key(|proposal| proposal.component_id());
1072    proposals
1073}

1075/// Determines from the commit's shape whether the receiver is a sibling virtual
1076/// client of the sender of a virtual-clients commit.
1077///
1078/// Returns `true` for:
1079///   * own-leaf commits (`Sender::Member(idx)` with `idx == own_leaf_index`),
1080///     where receiver and sender share the higher-level leaf
1081///   * sibling-resync external commits (`Sender::NewMemberCommit` whose
1082///     proposal list inlines a `Remove` of `own_leaf_index`.
1083///
1084/// `false` for everything else.
1085#[cfg(feature = "virtual-clients-draft")]
1086fn is_sibling_vc_commit(
1087    commit: &Commit,
1088    sender: &super::Sender,
1089    own_leaf_index: crate::binary_tree::LeafNodeIndex,
1090) -> bool {
1091    use crate::messages::proposals::{Proposal, ProposalOrRef};
1092
1093    match sender {
1094        super::Sender::Member(idx) => *idx == own_leaf_index,
1095        super::Sender::NewMemberCommit => commit.proposals.iter().any(|p| {
1096            matches!(
1097                p,
1098                ProposalOrRef::Proposal(boxed)
1099                    if matches!(boxed.as_ref(), Proposal::Remove(r) if r.removed() == own_leaf_index)
1100            )
1101        }),
1102        _ => false,
1103    }
