46    fn derive_epoch_secrets(
47        &self,
48        provider: &impl OpenMlsProvider,
49        apply_proposals_values: ApplyProposalsValues,
50        epoch_secrets: &GroupEpochSecrets,
51        commit_secret: CommitSecret,
52        serialized_provisional_group_context: &[u8],
53    ) -> Result<EpochSecretsResult, StageCommitError> {
54        // Check if we need to include the init secret from an external commit
55        // we applied earlier or if we use the one from the previous epoch.
56        let joiner_secret = if let Some(ref external_init_proposal) =
57            apply_proposals_values.external_init_proposal_option
58        {
59            // Decrypt the content and derive the external init secret.
60            let external_priv = epoch_secrets
61                .external_secret()
62                .derive_external_keypair(provider.crypto(), self.ciphersuite())
63                .map_err(LibraryError::unexpected_crypto_error)?
64                .private;
65            let init_secret = InitSecret::from_kem_output(
66                provider.crypto(),
67                self.ciphersuite(),
68                self.version(),
69                &external_priv,
70                external_init_proposal.kem_output(),
71            )?;
72            JoinerSecret::new(
73                provider.crypto(),
74                self.ciphersuite(),
75                commit_secret,
76                &init_secret,
77                serialized_provisional_group_context,
78            )
79            .map_err(LibraryError::unexpected_crypto_error)?
80        } else {
81            JoinerSecret::new(
82                provider.crypto(),
83                self.ciphersuite(),
84                commit_secret,
85                epoch_secrets.init_secret(),
86                serialized_provisional_group_context,
87            )
88            .map_err(LibraryError::unexpected_crypto_error)?
89        };
90
91        // Prepare the PskSecret
92        // Fails if PSKs are missing ([valn1205](https://validation.openmls.tech/#valn1205))
93        let psk_secret = {
94            let psks: Vec<(&PreSharedKeyId, Secret)> = load_psks(
95                provider.storage(),
96                &self.resumption_psk_store,
97                &apply_proposals_values.presharedkeys,
98            )?;
99
100            PskSecret::new(provider.crypto(), self.ciphersuite(), psks)?
101        };
102
103        // Create key schedule
104        let mut key_schedule = KeySchedule::init(
105            self.ciphersuite(),
106            provider.crypto(),
107            &joiner_secret,
108            psk_secret,
109        )?;
110
111        key_schedule
112            .add_context(provider.crypto(), serialized_provisional_group_context)
113            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
114        Ok(key_schedule
115            .epoch_secrets(provider.crypto(), self.ciphersuite())
116            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?)
117    }

119    /// Stages a commit message that was sent by another group member. This
120    /// function does the following:
121    ///  - Applies the proposals covered by the commit to the tree
122    ///  - Applies the (optional) update path to the tree
123    ///  - Decrypts and calculates the path secrets
124    ///  - Initializes the key schedule for epoch rollover
125    ///  - Verifies the confirmation tag
126    ///
127    /// Returns a [StagedCommit] that can be inspected and later merged into the
128    /// group state with [MlsGroup::merge_commit()]. If the member was not
129    /// removed from the group, the function also returns an
130    /// [ApplicationExportSecret].
131    ///
132    /// This function does the following checks:
133    ///  - ValSem101
134    ///  - ValSem102
135    ///  - ValSem104
136    ///  - ValSem105
137    ///  - ValSem106
138    ///  - ValSem107
139    ///  - ValSem108
140    ///  - ValSem110
141    ///  - ValSem111
142    ///  - ValSem112
143    ///  - ValSem113: All Proposals: The proposal type must be supported by all
144    ///    members of the group
145    ///  - ValSem200
146    ///  - ValSem201
147    ///  - ValSem202: Path must be the right length
148    ///  - ValSem203: Path secrets must decrypt correctly
149    ///  - ValSem204: Public keys from Path must be verified and match the
150    ///    private keys from the direct path
151    ///  - ValSem205
152    ///  - ValSem240
153    ///  - ValSem241
154    ///  - ValSem242
155    ///  - ValSem244 Returns an error if the given commit was sent by the owner
156    ///    of this group.
157    pub(crate) fn stage_commit(
158        &self,
159        mls_content: &AuthenticatedContent,
160        old_epoch_keypairs: Vec<EncryptionKeyPair>,
161        leaf_node_keypairs: Vec<EncryptionKeyPair>,
162        provider: &impl OpenMlsProvider,
163        #[cfg(feature = "virtual-clients-draft")] vc_material: Option<
164            crate::components::vc_derivation_info::OperationSecret,
165        >,
166        #[cfg(feature = "virtual-clients-draft")] vc_emulation_epoch_id: Option<
167            crate::components::vc_derivation_info::EpochId,
168        >,
169    ) -> Result<StagedCommit, StageCommitError> {
170        // Check that the sender is another member of the group
171        #[cfg(not(feature = "virtual-clients-draft"))]
172        if let Sender::Member(member) = mls_content.sender() {
173            if member == &self.own_leaf_index() {
174                return Err(StageCommitError::OwnCommit);
175            }
176        }
177
178        let (commit, proposal_queue, sender_index) = self
179            .public_group
180            .validate_commit(mls_content, provider.crypto())?;
181
182        // Create the provisional public group state (including the tree and
183        // group context) and apply proposals.
184        let mut diff = self.public_group.empty_diff();
185
186        #[cfg(not(feature = "extensions-draft-08"))]
187        let apply_proposals_values =
188            diff.apply_proposals(&proposal_queue, self.own_leaf_index())?;
189
190        #[cfg(feature = "extensions-draft-08")]
191        let apply_proposals_values = diff.apply_proposals_with_app_data_updates(
192            &proposal_queue,
193            self.own_leaf_index(),
194            None,
195        )?;
196        self.stage_applied_proposal_values(
197            apply_proposals_values,
198            diff,
199            commit,
200            proposal_queue,
201            sender_index,
202            mls_content,
203            old_epoch_keypairs,
204            leaf_node_keypairs,
205            provider,
206            #[cfg(feature = "virtual-clients-draft")]
207            vc_material,
208            #[cfg(feature = "virtual-clients-draft")]
209            vc_emulation_epoch_id,
210        )
211    }

268    #[allow(clippy::too_many_arguments)]
269    fn stage_applied_proposal_values(
270        &self,
271        apply_proposals_values: ApplyProposalsValues,
272        mut diff: PublicGroupDiff,
273        commit: &Commit,
274        proposal_queue: ProposalQueue,
275        sender_index: LeafNodeIndex,
276        mls_content: &AuthenticatedContent,
277        old_epoch_keypairs: Vec<EncryptionKeyPair>,
278        leaf_node_keypairs: Vec<EncryptionKeyPair>,
279        provider: &impl OpenMlsProvider,
280        #[cfg(feature = "virtual-clients-draft")] vc_material: Option<
281            crate::components::vc_derivation_info::OperationSecret,
282        >,
283        #[cfg(feature = "virtual-clients-draft")] vc_emulation_epoch_id: Option<
284            crate::components::vc_derivation_info::EpochId,
285        >,
286    ) -> Result<StagedCommit, StageCommitError> {
287        let ciphersuite = self.ciphersuite();
288
289        // A sibling-resync external commit is a VC external commit sent by a
290        // sibling emulator client to onboard itself into this higher-level
291        // group, inline-removing our existing leaf. The receiver-side gate in
292        // `process_internal_authenticated_content[_with_app_data_updates]` (see
293        // `is_sibling_vc_commit`) sets `vc_material = Some(_)` only for
294        // own-leaf VC commits and sibling-resync external commits, so here the
295        // two-condition check fully identifies the resync case:
296        //
297        //   - `vc_material` is `Some`, which after the upstream gate means the
298        //     commit carries a VC derivation-info entry and we hold per-epoch
299        //     state for the referenced `epoch_id`.
300        //   - the sender is `NewMemberCommit`, since own-leaf VC commits arrive as
301        //     `Sender::Member`, so this disambiguates the two sibling shapes.
302        //
303        // When this holds, we (a) skip the `self_removed` short-circuit so we
304        // don't transition to `Inactive`, (b) derive the path from the
305        // per-commit `OperationSecret` (we have no HPKE recipient on the path),
306        // and (c) record the new leaf index on the staged commit so
307        // `merge_commit` can update `own_leaf_index` before filtering owned
308        // encryption keypairs.
309        #[cfg(feature = "virtual-clients-draft")]
310        let is_sibling_resync =
311            vc_material.is_some() && matches!(mls_content.sender(), Sender::NewMemberCommit);
312        #[cfg(not(feature = "virtual-clients-draft"))]
313        let is_sibling_resync = false;
314
315        // Determine if Commit has a path
316        let (commit_secret, new_keypairs, new_leaf_keypair_option, update_path_leaf_node) =
317            if let Some(path) = commit.path.clone() {
318                // Update the public group
319                // ValSem202: Path must be the right length
320                diff.apply_received_update_path(
321                    provider.crypto(),
322                    ciphersuite,
323                    sender_index,
324                    &path,
325                )?;
326
327                // Update group context
328                diff.update_group_context(
329                    provider.crypto(),
330                    apply_proposals_values.extensions.clone(),
331                )?;
332
333                // Check if we were removed from the group. The sibling-resync
334                // discriminator carves out the case where the `Remove` of our
335                // leaf is the auto-Remove paired with a sibling emulator's
336                // external commit. In that case our state survives on the
337                // joiner's new leaf, so we must continue processing.
338                if apply_proposals_values.self_removed && !is_sibling_resync {
339                    // If so, we return here, because we can't decrypt the path
340                    let staged_diff = diff.into_staged_diff(provider.crypto(), ciphersuite)?;
341                    let staged_state = PublicStagedCommitState::new(
342                        staged_diff,
343                        commit.path.as_ref().map(|path| path.leaf_node().clone()),
344                    );
345                    let staged_commit = StagedCommit::new(
346                        proposal_queue,
347                        StagedCommitState::PublicState(Box::new(staged_state)),
348                        #[cfg(feature = "virtual-clients-draft")]
349                        None,
350                    );
351                    return Ok(staged_commit);
352                }
353
354                // When processing a commit sent by a sibling virtual client
355                // (either our own leaf, or a sibling-resync external commit
356                // onto a new leaf), we have no HPKE recipient on the path.
357                // Re-derive the path from the per-commit `OperationSecret`
358                // the receiver computed by evaluating the per-epoch PPRF, and
359                // verify the resulting public keys against the commit. The
360                // non-VC `decrypt_path` is the fallback for everyone else.
361                #[cfg(feature = "virtual-clients-draft")]
362                let vc_path: Option<(Vec<EncryptionKeyPair>, CommitSecret)> =
363                    if sender_index == self.own_leaf_index() || is_sibling_resync {
364                        let operation_secret = vc_material.ok_or(
365                            crate::components::vc_derivation_info::VirtualClientsError::MissingPprf,
366                        )?;
367                        Some(self.recreate_path_for_own_commit(
368                            &diff,
369                            &path,
370                            ciphersuite,
371                            provider.crypto(),
372                            sender_index,
373                            operation_secret,
374                        )?)
375                    } else {
376                        None
377                    };
378                #[cfg(not(feature = "virtual-clients-draft"))]
379                let vc_path: Option<(Vec<EncryptionKeyPair>, CommitSecret)> = None;
380
381                // ValSem203: Path secrets must decrypt correctly
382                // ValSem204: Public keys from Path must be verified and match the private keys from the direct path
383                let (new_keypairs, commit_secret) = if let Some(pair) = vc_path {
384                    pair
385                } else {
386                    let decryption_keypairs: Vec<&EncryptionKeyPair> = old_epoch_keypairs
387                        .iter()
388                        .chain(leaf_node_keypairs.iter())
389                        .collect();
390                    diff.decrypt_path(
391                        provider.crypto(),
392                        &decryption_keypairs,
393                        self.own_leaf_index(),
394                        sender_index,
395                        path.nodes(),
396                        &apply_proposals_values.exclusion_list(),
397                    )?
398                };
399
400                // Check if one of our update proposals was applied. If so, we
401                // need to store that keypair separately, because after merging
402                // it needs to be removed from the key store separately and in
403                // addition to the removal of the keypairs of the previous
404                // epoch.
405                let new_leaf_keypair_option = if is_sibling_resync {
406                    None
407                } else if let Some(leaf) = diff.leaf(self.own_leaf_index()) {
408                    leaf_node_keypairs.into_iter().find_map(|keypair| {
409                        if leaf.encryption_key() == keypair.public_key() {
410                            Some(keypair)
411                        } else {
412                            None
413                        }
414                    })
415                } else {
416                    // We should have an own leaf at this point.
417                    debug_assert!(false);
418                    None
419                };
420
421                // Return the leaf node in the update path so the credential can be validated.
422                // Since the diff has already been updated, this should be the same as the leaf
423                // at the sender index.
424                let update_path_leaf_node = Some(path.leaf_node().clone());
425                debug_assert_eq!(diff.leaf(sender_index), path.leaf_node().into());
426
427                (
428                    commit_secret,
429                    new_keypairs,
430                    new_leaf_keypair_option,
431                    update_path_leaf_node,
432                )
433            } else {
434                if apply_proposals_values.path_required {
435                    // ValSem201
436                    return Err(StageCommitError::RequiredPathNotFound);
437                }
438
439                // Even if there is no path, we have to update the group context.
440                diff.update_group_context(
441                    provider.crypto(),
442                    apply_proposals_values.extensions.clone(),
443                )?;
444
445                (CommitSecret::zero_secret(ciphersuite), vec![], None, None)
446            };
447
448        // Update the confirmed transcript hash before we compute the confirmation tag.
449        diff.update_confirmed_transcript_hash(provider.crypto(), mls_content)?;
450
451        let received_confirmation_tag = mls_content
452            .confirmation_tag()
453            .ok_or(StageCommitError::ConfirmationTagMissing)?;
454
455        let serialized_provisional_group_context = diff
456            .group_context()
457            .tls_serialize_detached()
458            .map_err(LibraryError::missing_bound_check)?;
459
460        #[cfg(feature = "virtual-clients-draft")]
461        let provisional_own_leaf_index = if is_sibling_resync {
462            sender_index
463        } else {
464            self.own_leaf_index()
465        };
466        #[cfg(not(feature = "virtual-clients-draft"))]
467        let provisional_own_leaf_index = self.own_leaf_index();
468
469        let EpochSecretsResult {
470            epoch_secrets,
471            #[cfg(feature = "extensions-draft-08")]
472            application_exporter,
473        } = self.derive_epoch_secrets(
474            provider,
475            apply_proposals_values,
476            self.group_epoch_secrets(),
477            commit_secret,
478            &serialized_provisional_group_context,
479        )?;
480        let (provisional_group_secrets, provisional_message_secrets) = epoch_secrets.split_secrets(
481            serialized_provisional_group_context,
482            diff.tree_size(),
483            provisional_own_leaf_index,
484        );
485
486        // Verify confirmation tag
487        // ValSem205
488        let own_confirmation_tag = provisional_message_secrets
489            .confirmation_key()
490            .tag(
491                provider.crypto(),
492                self.ciphersuite(),
493                diff.group_context().confirmed_transcript_hash(),
494            )
495            .map_err(LibraryError::unexpected_crypto_error)?;
496        if &own_confirmation_tag != received_confirmation_tag {
497            log::error!("Confirmation tag mismatch");
498            log_crypto!(trace, "  Got:      {:x?}", received_confirmation_tag);
499            log_crypto!(trace, "  Expected: {:x?}", own_confirmation_tag);
500            // TODO: We have tests expecting this error.
501            //       They need to be rewritten.
502            // debug_assert!(false, "Confirmation tag mismatch");
503
504            // in some tests we need to be able to proceed despite the tag being wrong,
505            // e.g. to test whether a later validation check is performed correctly.
506            if !crate::skip_validation::is_disabled::confirmation_tag() {
507                return Err(StageCommitError::ConfirmationTagMismatch);
508            }
509        }
510
511        diff.update_interim_transcript_hash(ciphersuite, provider.crypto(), own_confirmation_tag)?;
512
513        let staged_diff = diff.into_staged_diff(provider.crypto(), ciphersuite)?;
514        #[cfg(feature = "extensions-draft-08")]
515        let application_export_tree = ApplicationExportTree::new(application_exporter);
516        #[cfg(feature = "virtual-clients-draft")]
517        let new_own_leaf_index = is_sibling_resync.then_some(provisional_own_leaf_index);
518        let staged_commit_state =
519            StagedCommitState::GroupMember(Box::new(MemberStagedCommitState::new(
520                provisional_group_secrets,
521                provisional_message_secrets,
522                staged_diff,
523                new_keypairs,
524                new_leaf_keypair_option,
525                update_path_leaf_node,
526                #[cfg(feature = "extensions-draft-08")]
527                application_export_tree,
528                #[cfg(feature = "virtual-clients-draft")]
529                new_own_leaf_index,
530            )));
531        let staged_commit = StagedCommit::new(
532            proposal_queue,
533            staged_commit_state,
534            #[cfg(feature = "virtual-clients-draft")]
535            vc_emulation_epoch_id,
536        );
537
538        Ok(staged_commit)
539    }

541    /// Re-derive the path of a commit sent by a sibling virtual client
542    /// (either through our own higher-level leaf, or onto a new leaf via a
543    /// sibling-resync external commit) from the per-commit `OperationSecret`
544    /// resolved by the caller (in `process_message`, via PPRF evaluation), and
545    /// verify that the public keys derived from the path secret match the
546    /// leaf node in the path. Returns the derived parent-node keypairs and
547    /// commit secret, ready to be slotted in where `decrypt_path` would
548    /// normally produce them.
549    ///
550    /// `sender_index` is the leaf the path originates from. For an own-leaf
551    /// VC commit this equals `self.own_leaf_index()`. For a sibling-resync
552    /// external commit it is the joiner's new leaf (the `leftmost_free_index`
553    /// the external-commit builder chose), which is where the path actually
554    /// starts.
555    ///
556    /// Signature key changes are not verified here: not every signature
557    /// scheme has an interoperable seed-to-keypair construction, so the
558    /// application is responsible for supplying any rotated signature key
559    /// pair to the storage provider out-of-band. The new public key on the
560    /// path leaf is already authenticated by the commit's standard
561    /// path-validation against the previous signature key.
562    #[cfg(feature = "virtual-clients-draft")]
563    fn recreate_path_for_own_commit(
564        &self,
565        diff: &PublicGroupDiff,
566        path: &crate::treesync::treekem::UpdatePath,
567        ciphersuite: openmls_traits::types::Ciphersuite,
568        crypto: &impl OpenMlsCrypto,
569        sender_index: LeafNodeIndex,
570        operation_secret: crate::components::vc_derivation_info::OperationSecret,
571    ) -> Result<(Vec<EncryptionKeyPair>, CommitSecret), StageCommitError> {
572        use crate::components::vc_derivation_info::VirtualClientsError;
573
574        let path_secret = operation_secret
575            .derive_path_generation_secret(crypto, ciphersuite)?
576            .into();
577        let (encryption_key_pairs, commit_secret) =
578            diff.recreate_path_from_path_secret(crypto, path_secret, sender_index, path.nodes())?;
579
580        // Verify that the leaf encryption key in the path matches the one
581        // derived from the operation secret.
582        let leaf_keypair = operation_secret
583            .derive_encryption_key_secret(crypto, ciphersuite)?
584            .generate_encryption_key_pair(crypto, ciphersuite)?;
585        if leaf_keypair.public_key() != path.leaf_node().encryption_key() {
586            return Err(VirtualClientsError::EncryptionKeyMismatch.into());
587        }
588
589        // Mirror the sender's `apply_own_update_path`: the leaf keypair is
590        // prepended to the parent keypairs so the merge step has private
591        // keys for all of the new epoch's owned encryption keys.
592        let mut keypairs = Vec::with_capacity(1 + encryption_key_pairs.len());
593        keypairs.push(leaf_keypair);
594        keypairs.extend(encryption_key_pairs);
595        Ok((keypairs, commit_secret))
596    }

598    /// Merges a [StagedCommit] into the group state and optionally return a [`SecretTree`]
599    /// from the previous epoch. The secret tree is returned if the Commit does not contain a self removal.
600    ///
601    /// This function should not fail and only returns a [`Result`], because it
602    /// might throw a `LibraryError`.
603    pub(crate) fn merge_commit<Provider: OpenMlsProvider>(
604        &mut self,
605        provider: &Provider,
606        staged_commit: StagedCommit,
607    ) -> Result<(), MergeCommitError<Provider::StorageError>> {
608        // Get all keypairs from the old epoch, so we can later store the ones
609        // that are still relevant in the new epoch.
610        let old_epoch_keypairs = self
611            .read_epoch_keypairs(provider.storage())
612            .map_err(MergeCommitError::StorageError)?;
613        match staged_commit.state {
614            StagedCommitState::PublicState(staged_state) => {
615                self.public_group
616                    .merge_diff(staged_state.into_staged_diff());
617                self.store(provider.storage())
618                    .map_err(MergeCommitError::StorageError)?;
619                Ok(())
620            }
621            StagedCommitState::GroupMember(state) => {
622                // Save the past epoch
623                let past_epoch = self.context().epoch();
624                // Get all the full leaves
625                let leaves = self.public_group().members().collect();
626                // Merge the staged commit into the group state and store the secret tree from the
627                // previous epoch in the message secrets store.
628                self.group_epoch_secrets = state.group_epoch_secrets;
629
630                // Replace the previous message secrets with the new ones and return the previous message secrets
631                let old_message_secrets = self
632                    .message_secrets_store
633                    .replace_current_message_secrets(state.message_secrets);
634                self.message_secrets_store.add_past_epoch_tree(
635                    past_epoch,
636                    old_message_secrets,
637                    leaves,
638                );
639
640                // Replace the previous exporter tree with the new one.
641                #[cfg(feature = "extensions-draft-08")]
642                {
643                    // The application exporter is only None if the group was
644                    // stored using an older version of OpenMLS that did not
645                    // support the application exporter.
646                    if let Some(application_export_tree) = state.application_export_tree {
647                        // Overwrite the existing exporter tree in the storage.
648
649                        use openmls_traits::storage::StorageProvider as _;
650                        provider
651                            .storage()
652                            .write_application_export_tree(
653                                self.group_id(),
654                                &application_export_tree,
655                            )
656                            .map_err(MergeCommitError::StorageError)?;
657
658                        self.application_export_tree = Some(application_export_tree);
659                    }
660                }
661
662                self.public_group.merge_diff(state.staged_diff);
663
664                #[cfg(feature = "virtual-clients-draft")]
665                let previous_own_leaf_index = self.own_leaf_index;
666
667                // Sibling-resync external commit: install the joiner's new
668                // leaf as our own before filtering keypairs. The call to
669                // `owned_encryption_keys(self.own_leaf_index())` below relies
670                // on this value.
671                #[cfg(feature = "virtual-clients-draft")]
672                if let Some(new_idx) = state.new_own_leaf_index {
673                    self.own_leaf_index = new_idx;
674                }
675
676                let leaf_keypair = if let Some(keypair) = &state.new_leaf_keypair_option {
677                    vec![keypair.clone()]
678                } else {
679                    vec![]
680                };
681
682                // Figure out which keys we need in the new epoch.
683                let new_owned_encryption_keys = self
684                    .public_group()
685                    .owned_encryption_keys(self.own_leaf_index());
686                // From the old and new keys, keep the ones that are still relevant in the new epoch.
687                let epoch_keypairs: Vec<EncryptionKeyPair> = old_epoch_keypairs
688                    .into_iter()
689                    .chain(state.new_keypairs)
690                    .chain(leaf_keypair)
691                    .filter(|keypair| new_owned_encryption_keys.contains(keypair.public_key()))
692                    .collect();
693
694                // We should have private keys for all owned encryption keys.
695                debug_assert_eq!(new_owned_encryption_keys.len(), epoch_keypairs.len());
696                if new_owned_encryption_keys.len() != epoch_keypairs.len() {
697                    return Err(LibraryError::custom(
698                        "We should have all the private key material we need.",
699                    )
700                    .into());
701                }
702
703                // Store the updated group state
704                let storage = provider.storage();
705                let group_id = self.group_id();
706
707                self.public_group
708                    .store(storage)
709                    .map_err(MergeCommitError::StorageError)?;
710                storage
711                    .write_own_leaf_index(group_id, &self.own_leaf_index)
712                    .map_err(MergeCommitError::StorageError)?;
713                storage
714                    .write_group_epoch_secrets(group_id, &self.group_epoch_secrets)
715                    .map_err(MergeCommitError::StorageError)?;
716                storage
717                    .write_message_secrets(group_id, &self.message_secrets_store)
718                    .map_err(MergeCommitError::StorageError)?;
719
720                // Store the relevant keys under the new epoch
721                self.store_epoch_keypairs(storage, epoch_keypairs.as_slice())
722                    .map_err(MergeCommitError::StorageError)?;
723
724                // Delete the old keys.
725                self.delete_previous_epoch_keypairs(
726                    storage,
727                    #[cfg(feature = "virtual-clients-draft")]
728                    previous_own_leaf_index,
729                )
730                .map_err(MergeCommitError::StorageError)?;
731                if let Some(keypair) = state.new_leaf_keypair_option {
732                    keypair
733                        .delete(storage)
734                        .map_err(MergeCommitError::StorageError)?;
735                }
736
737                // Empty the proposal store
738                storage
739                    .clear_proposal_queue::<GroupId, ProposalRef>(group_id)
740                    .map_err(MergeCommitError::StorageError)?;
741                self.proposal_store_mut().empty();
742
743                Ok(())
744            }
745        }
746    }
747}

868    /// Returns the credentials that the caller needs to verify are valid.
869    pub fn credentials_to_verify(&self) -> impl Iterator<Item = &Credential> {
870        let update_path_leaf_node_cred = if let Some(node) = self.update_path_leaf_node() {
871            vec![node.credential()]
872        } else {
873            vec![]
874        };
875
876        update_path_leaf_node_cred
877            .into_iter()
878            .chain(
879                self.queued_proposals()
880                    .flat_map(|proposal: &QueuedProposal| match proposal.proposal() {
881                        Proposal::Update(update_proposal) => {
882                            vec![update_proposal.leaf_node().credential()].into_iter()
883                        }
884                        Proposal::Add(add_proposal) => {
885                            vec![add_proposal.key_package().leaf_node().credential()].into_iter()
886                        }
887                        Proposal::GroupContextExtensions(gce_proposal) => gce_proposal
888                            .extensions()
889                            .iter()
890                            .flat_map(|extension| {
891                                match extension {
892                                    Extension::ExternalSenders(external_senders) => {
893                                        external_senders
894                                            .iter()
895                                            .map(|external_sender| external_sender.credential())
896                                            .collect()
897                                    }
898                                    _ => vec![],
899                                }
900                                .into_iter()
901                            })
902                            // TODO: ideally we wouldn't collect in between here, but the match arms
903                            //       have to all return the same type. We solve this by having them all
904                            //       be vec::IntoIter, but it would be nice if we just didn't have to
905                            //       do this.
906                            //       It might be possible to solve this by letting all match arms
907                            //       evaluate to a dyn Iterator.
908                            .collect::<Vec<_>>()
909                            .into_iter(),
910                        _ => vec![].into_iter(),
911                    }),
912            )
913    }

962    pub(crate) fn safe_export_secret(
963        &mut self,
964        crypto: &impl OpenMlsCrypto,
965        component_id: ComponentId,
966    ) -> Result<Vec<u8>, StagedSafeExportSecretError> {
967        let ciphersuite = self.group_context().ciphersuite();
968        let StagedCommitState::GroupMember(ref mut staged_commit) = self.state else {
969            return Err(StagedSafeExportSecretError::NotGroupMember);

972            return Err(StagedSafeExportSecretError::Unsupported);

980    /// Returns [`ExportSecretError::KeyLengthTooLong`] if the requested
981    /// key length is too long.
982    /// Returns [`ExportSecretError::GroupStateError(MlsGroupStateError::UseAfterEviction)`]
983    /// if the commit removed us from the group.
984    ///
985    /// [`ExportSecretError::GroupStateError(MlsGroupStateError::UseAfterEviction)`]: MlsGroupStateError::UseAfterEviction
986    pub fn export_secret<CryptoProvider: OpenMlsCrypto>(
987        &self,
988        crypto: &CryptoProvider,
989        label: &str,
990        context: &[u8],
991        key_length: usize,
992    ) -> Result<Vec<u8>, ExportSecretError> {
993        if key_length > u16::MAX as usize {
994            log::error!("Got a key that is larger than u16::MAX");
995            return Err(ExportSecretError::KeyLengthTooLong);
996        }
997
998        match &self.state {
999            StagedCommitState::PublicState(_public_staged_commit_state) => Err(
1000                ExportSecretError::GroupStateError(MlsGroupStateError::UseAfterEviction),
1001            ),
1002            StagedCommitState::GroupMember(member_staged_commit_state) => {
1003                Ok(member_staged_commit_state
1004                    .group_epoch_secrets
1005                    .exporter_secret()
1006                    .derive_exported_secret(
1007                        self.group_context().ciphersuite(),
1008                        crypto,
1009                        label,
1010                        context,
1011                        key_length,
1012                    )
1013                    .map_err(LibraryError::unexpected_crypto_error)?)
1014            }
1015        }
1016    }
