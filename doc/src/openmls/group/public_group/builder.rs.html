31impl TempBuilderPG1 {
32    pub(crate) fn with_lifetime(mut self, lifetime: Lifetime) -> Self {
33        self.lifetime = Some(lifetime);
34        self
35    }
36
37    pub(crate) fn with_capabilities(mut self, capabilities: Capabilities) -> Self {
38        self.capabilities = Some(capabilities);
39        self
40    }
41
42    pub(crate) fn with_group_context_extensions(
43        mut self,
44        extensions: Extensions,
45    ) -> Result<Self, InvalidExtensionError> {
46        let is_valid_in_group_context = extensions.application_id().is_none()
47            && extensions.ratchet_tree().is_none()
48            && extensions.external_pub().is_none();
49        if !is_valid_in_group_context {
50            return Err(InvalidExtensionError::IllegalInGroupContext);
51        }
52        self.group_context_extensions = extensions;
53        Ok(self)
54    }
55
56    pub(crate) fn with_leaf_node_extensions(
57        mut self,
58        extensions: Extensions,
59    ) -> Result<Self, InvalidExtensionError> {
60        // Ensure that these extensions are not invalid for leaf nodes.
61        // https://validation.openmls.tech/#valn1601
62        extensions.validate_extension_types_for_leaf_node()?;
63
64        self.leaf_node_extensions = extensions;
65        Ok(self)
66    }
67
68    pub(crate) fn get_secrets(
69        self,
70        provider: &impl OpenMlsProvider,
71        signer: &impl Signer,
72    ) -> Result<(TempBuilderPG2, CommitSecret, EncryptionKeyPair), PublicGroupBuildError> {
73        // If there are no capabilities, we want to provide a default version
74        // plus anything in the required capabilities.
75        let (required_extensions, required_proposals, required_credentials) =
76            if let Some(required_capabilities) =
77                self.group_context_extensions.required_capabilities()
78            {
79                (
80                    Some(required_capabilities.extension_types()),
81                    Some(required_capabilities.proposal_types()),
82                    Some(required_capabilities.credential_types()),
83                )
84            } else {
85                (None, None, None)
86            };
87        let capabilities = self.capabilities.unwrap_or(Capabilities::new(
88            Some(&[ProtocolVersion::default()]),
89            Some(&[self.ciphersuite]),
90            required_extensions,
91            required_proposals,
92            required_credentials,
93        ));
94        let (treesync, commit_secret, leaf_keypair) = TreeSync::new(
95            provider,
96            signer,
97            self.ciphersuite,
98            self.credential_with_key,
99            self.lifetime.unwrap_or_default(),
100            capabilities,
101            self.leaf_node_extensions,
102        )?;
103
104        let group_context = GroupContext::create_initial_group_context(
105            self.ciphersuite,
106            self.group_id,
107            treesync.tree_hash().to_vec(),
108            self.group_context_extensions,
109        );
110        let next_builder = TempBuilderPG2 {
111            treesync,
112            group_context,
113        };
114        Ok((next_builder, commit_secret, leaf_keypair))
115    }
116}
