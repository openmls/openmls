21/// This struct contain the return values of the `apply_proposals()` function
22#[derive(Debug)]
23pub(crate) struct ApplyProposalsValues {
24    pub(crate) path_required: bool,
25    pub(crate) self_removed: bool,
26    pub(crate) invitation_list: Vec<(LeafNodeIndex, AddProposal)>,
27    pub(crate) presharedkeys: Vec<PreSharedKeyId>,
28    pub(crate) external_init_proposal_option: Option<ExternalInitProposal>,
29    pub(crate) extensions: Option<Extensions<GroupContext>>,
30}
31
32impl ApplyProposalsValues {
33    /// This function creates a `HashSet` of node indexes of the new nodes that
34    /// were added to the tree. The `HashSet` will be querried by the
35    /// `resolve()` function to filter out those nodes from the resolution.
36    pub(crate) fn exclusion_list(&self) -> HashSet<&LeafNodeIndex> {
37        // Collect the new leaves' indexes so we can filter them out in the resolution
38        // later
39        let new_leaves_indexes: HashSet<&LeafNodeIndex> = self
40            .invitation_list
41            .iter()
42            .map(|(index, _)| index)
43            .collect();
44        new_leaves_indexes
45    }
46}

48/// Applies a list of proposals from a Commit to the tree.
49/// `proposal_queue` is the queue of proposals received or sent in the
50/// current epoch `updates_key_package_bundles` is the list of own
51/// [`LeafNode`]s corresponding to updates or commits sent in the
52/// current epoch.
53///
54/// If an `own_leaf_index` is provided, `self_removed` in the returned
55/// `ApplyProposalValues` is set to `true` if that leaf index is targeted by a
56/// Remove proposal.
57///
58/// Returns an error if the proposals have not been validated before.
59impl PublicGroupDiff<'_> {
60    pub(crate) fn apply_proposals(
61        &mut self,
62        proposal_queue: &ProposalQueue,
63        own_leaf_index: impl Into<Option<LeafNodeIndex>>,
64    ) -> Result<ApplyProposalsValues, LibraryError> {
65        log::debug!("Applying proposal");
66        let mut self_removed = false;
67
68        // Process external init proposals. We do this before the removes, so we
69        // know that removing "ourselves" (i.e. removing the group member in the
70        // same leaf as we are in) is valid in this case. We only care about the
71        // first proposal and ignore all others.
72        let external_init_proposal_option = proposal_queue
73            .filtered_by_type(ProposalType::ExternalInit)
74            .next()
75            .and_then(|queued_proposal| {
76                if let Proposal::ExternalInit(external_init_proposal) = queued_proposal.proposal() {
77                    Some(*(*external_init_proposal).clone())
78                } else {
79                    None
80                }
81            });
82
83        // Process updates first
84        for queued_proposal in proposal_queue.filtered_by_type(ProposalType::Update) {
85            if let Proposal::Update(update_proposal) = queued_proposal.proposal() {
86                // Check if this is our own update.
87                let sender = queued_proposal.sender();
88                // Only members can send update proposals
89                // ValSem112
90                let sender_index = match sender {
91                    Sender::Member(sender_index) => *sender_index,
92                    // This should not happen with validated proposals
93                    _ => return Err(LibraryError::custom("Update proposal from non-member")),
94                };
95                let leaf_node = update_proposal.leaf_node().clone();
96                self.diff.update_leaf(leaf_node, sender_index);
97            }
98        }
99
100        // Process removes
101        let own_leaf_index = own_leaf_index.into();
102        for queued_proposal in proposal_queue.filtered_by_type(ProposalType::Remove) {
103            if let Proposal::Remove(remove_proposal) = queued_proposal.proposal() {
104                // Check if we got removed from the group
105                match own_leaf_index {
106                    Some(leaf_index) if remove_proposal.removed() == leaf_index => {
107                        self_removed = true
108                    }
109                    _ => (),
110                };
111                // Blank the direct path of the removed member
112                self.diff.blank_leaf(remove_proposal.removed());
113            }
114        }
115
116        // Process self removes
117        for queued_proposal in proposal_queue.filtered_by_type(ProposalType::SelfRemove) {
118            if let Proposal::SelfRemove = queued_proposal.proposal() {
119                // Check if we got removed from the group
120                let Sender::Member(removed) = queued_proposal.sender() else {
121                    // This should not happen with validated proposals
122                    return Err(LibraryError::custom("SelfRemove proposal from non-member"));
123                };
124                if Some(*removed) == own_leaf_index {
125                    self_removed = true;
126                }
127                // Blank the direct path of the removed member
128                self.diff.blank_leaf(*removed);
129            }
130        }
131
132        // Process adds
133        let add_proposals = proposal_queue
134            .filtered_by_type(ProposalType::Add)
135            .filter_map(|queued_proposal| {
136                if let Proposal::Add(add_proposal) = queued_proposal.proposal() {
137                    Some(add_proposal)
138                } else {
139                    None
140                }
141            });
142
143        // Extract KeyPackages from proposals
144        let mut invitation_list = Vec::new();
145        for add_proposal in add_proposals {
146            // XXX: There are too many clones here.
147            let leaf_node = add_proposal.key_package.leaf_node();
148            let leaf_index = self
149                .diff
150                .add_leaf(leaf_node.clone())
151                .map_err(|_| LibraryError::custom("Tree full: cannot add more members"))?;
152            invitation_list.push((leaf_index, *(*add_proposal).clone()))
153        }
154
155        self.diff.trim_tree();
156
157        // Process PSK proposals
158        let presharedkeys: Vec<PreSharedKeyId> = proposal_queue
159            .filtered_by_type(ProposalType::PreSharedKey)
160            .filter_map(|queued_proposal| {
161                if let Proposal::PreSharedKey(psk_proposal) = queued_proposal.proposal() {
162                    Some(psk_proposal.clone().into_psk_id())
163                } else {
164                    None
165                }
166            })
167            .collect();
168
169        // new extensions from group context extension proposal
170        // can be followed by AppDataUpdate proposals that modify
171        // the AppDataDictionary in the extensions
172        let updated_group_context_extensions = proposal_queue
173            .filtered_by_type(ProposalType::GroupContextExtensions)
174            .find_map(|queued_proposal| match queued_proposal.proposal() {
175                Proposal::GroupContextExtensions(extensions) => {
176                    Some(extensions.extensions().clone())
177                }
178                _ => None,
179            });
180
181        let proposals_require_path = proposal_queue
182            .queued_proposals()
183            .any(|p| p.proposal().is_path_required());
184
185        // This flag determines if the commit requires a path. A path is required if:
186        // * at least one proposal requires a path
187        // * (or) the commit is empty which implicitly means it's a self-update
188        // * (or) it is an external commit
189        // External commits always have an ExternalInit proposal, which requires a path. Therefore the last check is redundant.
190        let path_required = proposals_require_path || proposal_queue.is_empty();
191
192        Ok(ApplyProposalsValues {
193            path_required,
194            self_removed,
195            invitation_list,
196            presharedkeys,
197            external_init_proposal_option,
198            extensions: updated_group_context_extensions,
199        })
200    }

202    #[cfg(feature = "extensions-draft-08")]
203    pub(crate) fn apply_proposals_with_app_data_updates(
204        &mut self,
205        proposal_queue: &ProposalQueue,
206        own_leaf_index: impl Into<Option<LeafNodeIndex>>,
207        app_data_dict_updates: Option<AppDataUpdates>,
208    ) -> Result<ApplyProposalsValues, ApplyAppDataUpdateError> {
209        let ApplyProposalsValues {
210            path_required,
211            self_removed,
212            invitation_list,
213            presharedkeys,
214            external_init_proposal_option,
215            extensions,
216        } = self.apply_proposals(proposal_queue, own_leaf_index)?;
217
218        // apply AppDataUpdate proposals to the already-updated GroupContext extensions,
219        // if available, or return the new GroupContext extensions if modified.
220        #[cfg(feature = "extensions-draft-08")]
221        let mut extensions = extensions;
222        #[cfg(feature = "extensions-draft-08")]
223        self.apply_app_data_update_proposals(
224            &mut extensions,
225            proposal_queue,
226            app_data_dict_updates,
227        )?;
228
229        Ok(ApplyProposalsValues {
230            path_required,
231            self_removed,
232            invitation_list,
233            presharedkeys,
234            external_init_proposal_option,
235            extensions,
236        })
237    }

239    /// Return the updated GroupContext extensions after applying AppDataUpdates.
240    #[cfg(feature = "extensions-draft-08")]
241    pub(crate) fn apply_app_data_update_proposals(
242        &self,
243        // group_context_extensions if updated by a GroupContextExtensions proposal
244        updated_group_context_extensions: &mut Option<Extensions<GroupContext>>,
245        proposal_queue: &ProposalQueue,
246        app_data_dict_updates: Option<AppDataUpdates>,
247    ) -> Result<(), ApplyAppDataUpdateError> {
248        let has_app_data_update_proposals = proposal_queue
249            .queued_proposals()
250            .any(|proposal| proposal.proposal().is_type(ProposalType::AppDataUpdate));
251
252        let updates = match (has_app_data_update_proposals, app_data_dict_updates) {
253            // there are AppDataUpdate proposals and we were provided the AooDataUpdates
254            (true, Some(updates)) => updates,
255            // there are no AppDataUpdate proposals in the queue, and we weren't provided any updates
256            (false, None) => return Ok(()),
257
258            // If there are app data update proposals, there must be a list
259            (true, None) => {
260                return Err(ApplyAppDataUpdateError::MissingAppDataUpdates);
261            }
262            // updates were provided, but no proposals could have triggered the update
263            (false, Some(_)) => {
264                return Err(ApplyAppDataUpdateError::SuperfluousAppDataUpdates);
265            }
266        };
267
268        // retrieve the AppDataDictionary to update
269        // (make a copy of the AppDataDictionary, or create a new dictionary).
270        // NOTE: the case where a GroupContextExtensions updates the RequiredCapabilities
271        // to no longer include AppDataUpdate should be already filtered out at this stage.
272        let mut dictionary = self
273            .group_context
274            .app_data_dict()
275            .cloned()
276            .unwrap_or_default();
277
278        // apply updates
279        for (id, data) in updates.into_iter() {
280            if let Some(data) = data {
281                let _ = dictionary.insert(id, data);
282            } else {
283                let _ = dictionary.remove(&id);
284            }
285        }
286
287        // update the new dictionary in the updated extensions if updated by a
288        // GroupContextExtensions proposal, or make a copy of the existing extensions.
289        let extensions_to_update = updated_group_context_extensions
290            .get_or_insert_with(|| self.group_context.extensions().clone());
291
292        // update the dictionary extension
293        extensions_to_update.add_or_replace(Extension::AppDataDictionary(
294            AppDataDictionaryExtension::new(dictionary),
295        ));
296
297        Ok(())
298    }
