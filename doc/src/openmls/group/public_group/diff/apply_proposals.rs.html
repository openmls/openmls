14/// This struct contain the return values of the `apply_proposals()` function
15#[derive(Debug)]
16pub(crate) struct ApplyProposalsValues {
17    pub(crate) path_required: bool,
18    pub(crate) self_removed: bool,
19    pub(crate) invitation_list: Vec<(LeafNodeIndex, AddProposal)>,
20    pub(crate) presharedkeys: Vec<PreSharedKeyId>,
21    pub(crate) external_init_proposal_option: Option<ExternalInitProposal>,
22    pub(crate) extensions: Option<Extensions>,
23}
24
25impl ApplyProposalsValues {
26    /// This function creates a `HashSet` of node indexes of the new nodes that
27    /// were added to the tree. The `HashSet` will be querried by the
28    /// `resolve()` function to filter out those nodes from the resolution.
29    pub(crate) fn exclusion_list(&self) -> HashSet<&LeafNodeIndex> {
30        // Collect the new leaves' indexes so we can filter them out in the resolution
31        // later
32        let new_leaves_indexes: HashSet<&LeafNodeIndex> = self
33            .invitation_list
34            .iter()
35            .map(|(index, _)| index)
36            .collect();
37        new_leaves_indexes
38    }
39}

41/// Applies a list of proposals from a Commit to the tree.
42/// `proposal_queue` is the queue of proposals received or sent in the
43/// current epoch `updates_key_package_bundles` is the list of own
44/// [`LeafNode`]s corresponding to updates or commits sent in the
45/// current epoch.
46///
47/// If an `own_leaf_index` is provided, `self_removed` in the returned
48/// `ApplyProposalValues` is set to `true` if that leaf index is targeted by a
49/// Remove proposal.
50///
51/// Returns an error if the proposals have not been validated before.
52impl PublicGroupDiff<'_> {
53    pub(crate) fn apply_proposals(
54        &mut self,
55        proposal_queue: &ProposalQueue,
56        own_leaf_index: impl Into<Option<LeafNodeIndex>>,
57    ) -> Result<ApplyProposalsValues, LibraryError> {
58        log::debug!("Applying proposal");
59        let mut self_removed = false;
60
61        // Process external init proposals. We do this before the removes, so we
62        // know that removing "ourselves" (i.e. removing the group member in the
63        // same leaf as we are in) is valid in this case. We only care about the
64        // first proposal and ignore all others.
65        let external_init_proposal_option = proposal_queue
66            .filtered_by_type(ProposalType::ExternalInit)
67            .next()
68            .and_then(|queued_proposal| {
69                if let Proposal::ExternalInit(external_init_proposal) = queued_proposal.proposal() {
70                    Some(*(*external_init_proposal).clone())
71                } else {
72                    None
73                }
74            });
75
76        // Process updates first
77        for queued_proposal in proposal_queue.filtered_by_type(ProposalType::Update) {
78            if let Proposal::Update(update_proposal) = queued_proposal.proposal() {
79                // Check if this is our own update.
80                let sender = queued_proposal.sender();
81                // Only members can send update proposals
82                // ValSem112
83                let sender_index = match sender {
84                    Sender::Member(sender_index) => *sender_index,
85                    // This should not happen with validated proposals
86                    _ => return Err(LibraryError::custom("Update proposal from non-member")),
87                };
88                let leaf_node = update_proposal.leaf_node().clone();
89                self.diff.update_leaf(leaf_node, sender_index);
90            }
91        }
92
93        // Process removes
94        let own_leaf_index = own_leaf_index.into();
95        for queued_proposal in proposal_queue.filtered_by_type(ProposalType::Remove) {
96            if let Proposal::Remove(remove_proposal) = queued_proposal.proposal() {
97                // Check if we got removed from the group
98                match own_leaf_index {
99                    Some(leaf_index) if remove_proposal.removed() == leaf_index => {
100                        self_removed = true
101                    }
102                    _ => (),
103                };
104                // Blank the direct path of the removed member
105                self.diff.blank_leaf(remove_proposal.removed());
106            }
107        }
108
109        // Process self removes
110        for queued_proposal in proposal_queue.filtered_by_type(ProposalType::SelfRemove) {
111            if let Proposal::SelfRemove = queued_proposal.proposal() {
112                // Check if we got removed from the group
113                let Sender::Member(removed) = queued_proposal.sender() else {
114                    // This should not happen with validated proposals
115                    return Err(LibraryError::custom("SelfRemove proposal from non-member"));
116                };
117                if Some(*removed) == own_leaf_index {
118                    self_removed = true;
119                }
120                // Blank the direct path of the removed member
121                self.diff.blank_leaf(*removed);
122            }
123        }
124
125        // Process adds
126        let add_proposals = proposal_queue
127            .filtered_by_type(ProposalType::Add)
128            .filter_map(|queued_proposal| {
129                if let Proposal::Add(add_proposal) = queued_proposal.proposal() {
130                    Some(add_proposal)
131                } else {
132                    None
133                }
134            });
135
136        // Extract KeyPackages from proposals
137        let mut invitation_list = Vec::new();
138        for add_proposal in add_proposals {
139            // XXX: There are too many clones here.
140            let leaf_node = add_proposal.key_package.leaf_node();
141            let leaf_index = self
142                .diff
143                .add_leaf(leaf_node.clone())
144                .map_err(|_| LibraryError::custom("Tree full: cannot add more members"))?;
145            invitation_list.push((leaf_index, *(*add_proposal).clone()))
146        }
147
148        self.diff.trim_tree();
149
150        // Process PSK proposals
151        let presharedkeys: Vec<PreSharedKeyId> = proposal_queue
152            .filtered_by_type(ProposalType::PreSharedKey)
153            .filter_map(|queued_proposal| {
154                if let Proposal::PreSharedKey(psk_proposal) = queued_proposal.proposal() {
155                    Some(psk_proposal.clone().into_psk_id())
156                } else {
157                    None
158                }
159            })
160            .collect();
161
162        // apply group context extension proposal
163        let extensions = proposal_queue
164            .filtered_by_type(ProposalType::GroupContextExtensions)
165            .find_map(|queued_proposal| match queued_proposal.proposal() {
166                Proposal::GroupContextExtensions(extensions) => {
167                    Some(extensions.extensions().clone())
168                }
169                _ => None,
170            });
171
172        let proposals_require_path = proposal_queue
173            .queued_proposals()
174            .any(|p| p.proposal().is_path_required());
175
176        // This flag determines if the commit requires a path. A path is required if:
177        // * at least one proposal requires a path
178        // * (or) the commit is empty which implicitly means it's a self-update
179        // * (or) it is an external commit
180        // External commits always have an ExternalInit proposal, which requires a path. Therefore the last check is redundant.
181        let path_required = proposals_require_path || proposal_queue.is_empty();
182
183        Ok(ApplyProposalsValues {
184            path_required,
185            self_removed,
186            invitation_list,
187            presharedkeys,
188            external_init_proposal_option,
189            extensions,
190        })
191    }
