42    /// Checks the following semantic validation:
43    ///  - ValSem002
44    ///  - ValSem003
45    ///  - [valn1307](https://validation.openmls.tech/#valn1307)
46    pub(crate) fn validate_framing(
47        &self,
48        message: &ProtocolMessage,
49    ) -> Result<(), ValidationError> {
50        // ValSem002
51        if message.group_id() != self.group_id() {
52            return Err(ValidationError::WrongGroupId);
53        }
54
55        // ValSem003: Check boundaries for the epoch
56        // We differentiate depending on the content type
57        match message.content_type() {
58            // For application messages we allow messages for older epochs as well
59            ContentType::Application => {
60                if message.epoch() > self.group_context().epoch() {
61                    log::error!(
62                        "Wrong Epoch: message.epoch() {} > {} self.group_context().epoch()",
63                        message.epoch(),
64                        self.group_context().epoch()
65                    );
66                    return Err(ValidationError::WrongEpoch);
67                }
68            }
69            // For all other messages we only only accept the current epoch
70            _ => {
71                // https://validation.openmls.tech/#valn1307
72                if message.epoch() != self.group_context().epoch() {
73                    log::error!(
74                        "Wrong Epoch: message.epoch() {} != {} self.group_context().epoch()",
75                        message.epoch(),
76                        self.group_context().epoch()
77                    );
78                    return Err(ValidationError::WrongEpoch);
79                }
80            }
81        }
82
83        Ok(())
84    }

86    /// Checks the following semantic validation:
87    ///  - ValSem004
88    ///  - ValSem005
89    ///  - ValSem009
90    pub(super) fn validate_verifiable_content(
91        &self,
92        verifiable_content: &VerifiableAuthenticatedContentIn,
93        message_secrets_store_option: Option<&MessageSecretsStore>,
94    ) -> Result<(), ValidationError> {
95        // ValSem004
96        let sender = verifiable_content.sender();
97        if let Sender::Member(leaf_index) = sender {
98            // If the sender is a member, it has to be in the tree, except if
99            // it's an application message. Then it might be okay if it's in an
100            // old secret tree instance.
101            let is_in_secrets_store = if let Some(mss) = message_secrets_store_option {
102                mss.epoch_has_leaf(verifiable_content.epoch(), *leaf_index)
103            } else {
104                false
105            };
106            if !self.treesync().is_leaf_in_tree(*leaf_index) && !is_in_secrets_store {
107                return Err(ValidationError::UnknownMember);
108            }
109        }
110
111        // ValSem005
112        // Application messages must always be encrypted
113        if verifiable_content.content_type() == ContentType::Application {
114            if verifiable_content.wire_format() != WireFormat::PrivateMessage {
115                return Err(ValidationError::UnencryptedApplicationMessage);
116            } else if !verifiable_content.sender().is_member() {
117                return Err(ValidationError::NonMemberApplicationMessage);
118            }
119        }
120
121        // ValSem009
122        if verifiable_content.content_type() == ContentType::Commit
123            && verifiable_content.confirmation_tag().is_none()
124        {
125            return Err(ValidationError::MissingConfirmationTag);
126        }
127
128        Ok(())
129    }

133    /// Validate that all group members support the types of all proposals.
134    /// Implements check [valn0311](https://validation.openmls.tech/#valn0311)
135    pub(crate) fn validate_proposal_type_support(
136        &self,
137        proposal_queue: &ProposalQueue,
138    ) -> Result<(), ProposalValidationError> {
139        // Collect signature keys of removed members s.t. we can skip them
140        // when checking capabilities.
141        let signature_keys: HashSet<_> = proposal_queue
142            .remove_proposals()
143            .filter_map(|p| {
144                let removed_index = p.remove_proposal().removed();
145                self.treesync()
146                    .leaf(removed_index)
147                    .map(|leaf_node| leaf_node.signature_key())
148            })
149            .collect();
150
151        // Iterate over all leaf nodes except the removed ones
152        let mut leaves = self
153            .treesync()
154            .full_leaves()
155            .filter(|(_, leaf_node)| !signature_keys.contains(leaf_node.signature_key()));
156        let Some((_, first_leaf)) = leaves.next() else {
157            return Ok(());
158        };
159
160        // Initialize the capabilities intersection with the capabilities of the
161        // first leaf node.
162        let mut capabilities_intersection = first_leaf
163            .capabilities()
164            .proposals()
165            .iter()
166            .collect::<HashSet<_>>();
167        // Iterate over the remaining leaf nodes and intersect their capabilities
168        for (_, leaf_node) in leaves {
169            let leaf_capabilities_set = leaf_node.capabilities().proposals().iter().collect();
170            capabilities_intersection = capabilities_intersection
171                .intersection(&leaf_capabilities_set)
172                .cloned()
173                .collect();
174        }
175
176        // Check that the types of all non-default proposals are supported by all members
177        for proposal in proposal_queue.queued_proposals() {
178            let proposal_type = proposal.proposal().proposal_type();
179            if !proposal_type.is_default() && !capabilities_intersection.contains(&proposal_type) {
180                return Err(ProposalValidationError::UnsupportedProposalType);
181            }
182        }
183        Ok(())
184    }

186    /// Validate key uniqueness. This function implements the following checks:
187    ///  - ValSem101: Add Proposal: Signature public key in proposals must be unique among proposals & members
188    ///  - ValSem102: Add Proposal: Init key in proposals must be unique among proposals
189    ///  - ValSem103: Add Proposal: Encryption key in proposals must be unique among proposals & members
190    ///  - ValSem104: Add Proposal: Init key and encryption key must be different
191    ///  - ValSem110: Update Proposal: Encryption key must be unique among proposals & members
192    ///  - ValSem206: Commit: Path leaf node encryption key must be unique among proposals & members
193    ///  - ValSem207: Commit: Path encryption keys must be unique among proposals & members
194    ///  - [valn0111]: Verify that the following fields are unique among the members of the group: `signature_key`
195    ///  - [valn0112]: Verify that the following fields are unique among the members of the group: `encryption_key`
196    ///
197    /// [valn0111]: https://validation.openmls.tech/#valn0111
198    /// [valn0112]: https://validation.openmls.tech/#valn0112
199    /// [valn1208]: https://validation.openmls.tech/#valn1208
200    pub(crate) fn validate_key_uniqueness(
201        &self,
202        proposal_queue: &ProposalQueue,
203        commit: Option<&Commit>,
204    ) -> Result<(), ProposalValidationError> {
205        let mut signature_key_set = HashSet::new();
206        let mut init_key_set = HashSet::new();
207        let mut encryption_key_set = HashSet::new();
208
209        // Handle the exceptions needed for https://validation.openmls.tech/#valn0306
210        let remove_proposals = HashSet::<LeafNodeIndex>::from_iter(
211            proposal_queue
212                .remove_proposals()
213                .map(|remove_proposal| remove_proposal.remove_proposal().removed),
214        );
215
216        // Initialize the sets with the current members, filtered by the
217        // remove proposals.
218        for Member {
219            index,
220            encryption_key,
221            signature_key,
222            ..
223        } in self.treesync().full_leaf_members()
224        {
225            if !remove_proposals.contains(&index) {
226                signature_key_set.insert(signature_key);
227                encryption_key_set.insert(encryption_key);
228            }
229        }
230
231        // Collect signature keys from add proposals
232        let signature_keys = proposal_queue.add_proposals().map(|add_proposal| {
233            add_proposal
234                .add_proposal()
235                .key_package()
236                .leaf_node()
237                .signature_key()
238                .as_slice()
239                .to_vec()
240        });
241
242        // Collect encryption keys from add proposals, update proposals, the
243        // commit leaf node and path keys
244        let encryption_keys = proposal_queue
245            .add_proposals()
246            .map(|add_proposal| {
247                add_proposal
248                    .add_proposal()
249                    .key_package()
250                    .leaf_node()
251                    .encryption_key()
252                    .key()
253                    .as_slice()
254                    .to_vec()
255            })
256            .chain(proposal_queue.update_proposals().map(|update_proposal| {
257                update_proposal
258                    .update_proposal()
259                    .leaf_node()
260                    .encryption_key()
261                    .key()
262                    .as_slice()
263                    .to_vec()
264            }))
265            .chain(commit.and_then(|commit| {
266                commit
267                    .path
268                    .as_ref()
269                    .map(|path| path.leaf_node().encryption_key().as_slice().to_vec())
270            }))
271            .chain(
272                commit
273                    .iter()
274                    .filter_map(|commit| {
275                        commit.path.as_ref().map(|path| {
276                            path.nodes()
277                                .iter()
278                                .map(|node| node.encryption_key().as_slice().to_vec())
279                        })
280                    })
281                    .flatten(),
282            );
283
284        // Collect init keys from add proposals
285        let init_keys = proposal_queue.add_proposals().map(|add_proposal| {
286            add_proposal
287                .add_proposal()
288                .key_package()
289                .hpke_init_key()
290                .as_slice()
291                .to_vec()
292        });
293
294        // Validate uniqueness of signature keys
295        //  - ValSem101
296        //  - https://validation.openmls.tech/#valn0111
297        //  - https://validation.openmls.tech/#valn0305
298        //  - https://validation.openmls.tech/#valn0306
299        for signature_key in signature_keys {
300            if !signature_key_set.insert(signature_key) {
301                return Err(ProposalValidationError::DuplicateSignatureKey);
302            }
303        }
304
305        // Validate uniqueness of encryption keys
306        //  - ValSem103
307        //  - ValSem104
308        //  - ValSem110
309        //  - ValSem206
310        //  - ValSem207
311        //  - https://validation.openmls.tech/#valn0112
312        //  - https://validation.openmls.tech/#valn1209
313        for encryption_key in encryption_keys {
314            if init_key_set.contains(&encryption_key) {
315                return Err(ProposalValidationError::InitEncryptionKeyCollision);
316            }
317            if !encryption_key_set.insert(encryption_key) {
318                return Err(ProposalValidationError::DuplicateEncryptionKey);
319            }
320        }
321
322        // Validate uniqueness of init keys
323        //  - ValSem102
324        //  - ValSem104
325        for init_key in init_keys {
326            if encryption_key_set.contains(&init_key) {
327                return Err(ProposalValidationError::InitEncryptionKeyCollision);
328            }
329            if !init_key_set.insert(init_key) {
330                return Err(ProposalValidationError::DuplicateInitKey);
331            }
332        }
333
334        Ok(())
335    }

337    /// Validate capabilities. This function implements the following checks:
338    /// - ValSem106: Add Proposal: required capabilities
339    /// - ValSem109: Update Proposal: required capabilities
340    /// - [valn0113](https://validation.openmls.tech/#valn0113).
341    pub(crate) fn validate_capabilities(
342        &self,
343        proposal_queue: &ProposalQueue,
344    ) -> Result<(), ProposalValidationError> {
345        // ValSem106/ValSem109: Check the required capabilities of the add & update
346        // proposals This includes the following checks:
347        // - Are ciphersuite & version listed in the `Capabilities` Extension?
348        // - If a `RequiredCapabilitiesExtension` is present in the group: Is
349        //   this supported by the node?
350        // - Check that all extensions are contained in the capabilities.
351        // - Check that the capabilities contain the leaf node's credential
352        //   type (https://validation.openmls.tech/#valn0113).
353        // - Check that the credential type is supported by all members of the
354        //   group.
355        // - Check that the capabilities field of this LeafNode indicates
356        //   support for all the credential types currently in use by other
357        //   members.
358
359        // Extract the leaf nodes from the add & update proposals and validate them
360        proposal_queue
361            .queued_proposals()
362            .filter_map(|p| match p.proposal() {
363                Proposal::Add(add_proposal) => Some(add_proposal.key_package().leaf_node()),
364                Proposal::Update(update_proposal) => Some(update_proposal.leaf_node()),
365                _ => None,
366            })
367            .try_for_each(|leaf_node| {
368                self.validate_leaf_node_capabilities(leaf_node)
369                    .map_err(|_| ProposalValidationError::InsufficientCapabilities)
370            })
371    }

373    /// Validate Add proposals. This function implements the following checks:
374    ///  - ValSem105: Add Proposal: Ciphersuite & protocol version must match the group
375    pub(crate) fn validate_add_proposals(
376        &self,
377        proposal_queue: &ProposalQueue,
378    ) -> Result<(), ProposalValidationError> {
379        let add_proposals = proposal_queue.add_proposals();
380
381        // We do the key package validation checks here inline
382        // https://validation.openmls.tech/#valn0501
383        for add_proposal in add_proposals {
384            // ValSem105: Check if ciphersuite and version of the group are correct:
385            // https://validation.openmls.tech/#valn0201
386            if add_proposal.add_proposal().key_package().ciphersuite() != self.ciphersuite()
387                || add_proposal.add_proposal().key_package().protocol_version() != self.version()
388            {
389                return Err(ProposalValidationError::InvalidAddProposalCiphersuiteOrVersion);
390            }
391
392            // Check that the leaf node of the added key package supports all extensions in the group
393            // context.
394            // https://validation.openmls.tech/#valn0502
395            let added_leaf_supports_all_group_context_extensions =
396                self.group_context().extensions().iter().all(|extension| {
397                    add_proposal
398                        .add_proposal()
399                        .key_package
400                        .leaf_node()
401                        .supports_extension(&extension.extension_type())
402                });
403            if !added_leaf_supports_all_group_context_extensions {
404                return Err(ProposalValidationError::InsufficientCapabilities);
405            }
406
407            // https://validation.openmls.tech/#valn0202
408            self.validate_leaf_node(add_proposal.add_proposal().key_package().leaf_node())?;
409        }
410        Ok(())
411    }

413    /// Validate Remove proposals. This function implements the following checks:
414    ///  - ValSem107: Remove Proposal: Removed member must be unique among proposals
415    ///  - ValSem108: Remove Proposal: Removed member must be an existing group member
416    pub(crate) fn validate_remove_proposals(
417        &self,
418        proposal_queue: &ProposalQueue,
419    ) -> Result<(), ProposalValidationError> {
420        let updates_set: HashSet<_> = proposal_queue
421            .update_proposals()
422            .map(|proposal| {
423                if let Sender::Member(index) = proposal.sender() {
424                    Ok(*index)
425                } else {
426                    Err(ProposalValidationError::UpdateFromNonMember)
427                }
428            })
429            .collect::<Result<_, _>>()?;
430
431        let remove_proposals = proposal_queue.remove_proposals();
432
433        let mut removes_set = HashSet::new();
434
435        // https://validation.openmls.tech/#valn0701
436        for remove_proposal in remove_proposals {
437            let removed = remove_proposal.remove_proposal().removed();
438            // The node has to be a leaf in the tree
439            // ValSem108
440            if !self.treesync().is_leaf_in_tree(removed) {
441                return Err(ProposalValidationError::UnknownMemberRemoval);
442            }
443
444            // ValSem107
445            // https://validation.openmls.tech/#valn0304
446            if !removes_set.insert(removed) {
447                return Err(ProposalValidationError::DuplicateMemberRemoval);
448            }
449            if updates_set.contains(&removed) {
450                return Err(ProposalValidationError::DuplicateMemberRemoval);
451            }
452
453            // removed node can not be blank
454            if self.treesync().leaf(removed).is_none() {
455                return Err(ProposalValidationError::UnknownMemberRemoval);
456            }
457        }
458
459        Ok(())
460    }

462    /// Validate Update proposals. This function implements the following checks:
463    ///  - ValSem111: Update Proposal: The sender of a full Commit must not include own update proposals
464    ///  - ValSem112: Update Proposal: The sender of a standalone update proposal must be of type member
465    ///
466    /// TODO: #133 This validation must be updated according to Sec. 13.2
467    pub(crate) fn validate_update_proposals(
468        &self,
469        proposal_queue: &ProposalQueue,
470        committer: LeafNodeIndex,
471    ) -> Result<(), ProposalValidationError> {
472        // Check the update proposals from the proposal queue first
473        let update_proposals = proposal_queue.update_proposals();
474
475        for update_proposal in update_proposals {
476            // ValSem112
477            // The sender of a standalone update proposal must be of type member
478            if let Sender::Member(sender_index) = update_proposal.sender() {
479                // ValSem111
480                // https://validation.openmls.tech/#valn0302
481                // The sender of a full Commit must not include own update proposals
482                if committer == *sender_index {
483                    return Err(ProposalValidationError::CommitterIncludedOwnUpdate);
484                }
485            } else {
486                return Err(ProposalValidationError::UpdateFromNonMember);
487            }
488
489            // https://validation.openmls.tech/#valn0601
490            self.validate_leaf_node(update_proposal.update_proposal().leaf_node())?;
491
492            // Check that the leaf node in the update proposal supports all group context extensions
493            // https://validation.openmls.tech/#valn0602
494            let leaf_node_supports_group_context_extensions =
495                self.group_context().extensions().iter().all(|extension| {
496                    update_proposal
497                        .update_proposal()
498                        .leaf_node()
499                        .supports_extension(&extension.extension_type())
500                });
501
502            if !leaf_node_supports_group_context_extensions {
503                return Err(ProposalValidationError::LeafNodeValidation(
504                    LeafNodeValidationError::UnsupportedExtensions,
505                ));
506            }
507        }
508        Ok(())
509    }

533    /// Validate constraints on an external commit. This function implements the following checks:
534    ///  - ValSem240: External Commit, inline Proposals: There MUST be at least one ExternalInit proposal.
535    ///  - ValSem241: External Commit, inline Proposals: There MUST be at most one ExternalInit proposal.
536    ///  - ValSem242: External Commit must only cover inline proposal in allowlist (ExternalInit, Remove, PreSharedKey)
537    pub(crate) fn validate_external_commit(
538        &self,
539        proposal_queue: &ProposalQueue,
540    ) -> Result<(), ExternalCommitValidationError> {
541        // [valn0401](https://validation.openmls.tech/#valn0401)
542        let count_external_init_proposals = proposal_queue
543            .filtered_by_type(ProposalType::ExternalInit)
544            .count();
545        if count_external_init_proposals == 0 {
546            // ValSem240: External Commit, inline Proposals: There MUST be at least one ExternalInit proposal.
547            return Err(ExternalCommitValidationError::NoExternalInitProposals);
548        } else if count_external_init_proposals > 1 {
549            // ValSem241: External Commit, inline Proposals: There MUST be at most one ExternalInit proposal.
550            return Err(ExternalCommitValidationError::MultipleExternalInitProposals);
551        }
552
553        // ValSem242: External Commit must only cover inline proposal in allowlist (ExternalInit, Remove, PreSharedKey)
554        // [valn0404](https://validation.openmls.tech/#valn0404)
555        let contains_denied_proposal = proposal_queue.queued_proposals().any(|p| {
556            let is_inline = p.proposal_or_ref_type() == ProposalOrRefType::Proposal;
557            let is_allowed_type = matches!(
558                p.proposal(),
559                Proposal::ExternalInit(_)
560                    | Proposal::Remove(_)
561                    | Proposal::PreSharedKey(_)
562                    | Proposal::Custom(_)
563            );
564            is_inline && !is_allowed_type
565        });
566        if contains_denied_proposal {
567            return Err(ExternalCommitValidationError::InvalidInlineProposals);
568        }
569
570        // If a Remove proposal is present,
571        // the credential in the LeafNode MUST present a set of
572        // identifiers that is acceptable to the application for
573        // the removed participant.
574        // This MUST be checked by the application.
575
576        Ok(())
577    }

579    /// Returns a [`LeafNodeValidationError`] if an [`ExtensionType`]
580    /// in `extensions` is not supported by a leaf in this tree.
581    /// Implements check [valn1001](https://validation.openmls.tech/#valn1001).
582    pub(crate) fn validate_group_context_extensions_proposal(
583        &self,
584        proposal_queue: &ProposalQueue,
585    ) -> Result<(), GroupContextExtensionsProposalValidationError> {
586        let iter = proposal_queue.filtered_by_type(ProposalType::GroupContextExtensions);
587
588        for (i, queued_proposal) in iter.enumerate() {
589            // There must at most be one group context extionsion proposal. Return an error if there are more
590            if i > 0 {
591                return Err(GroupContextExtensionsProposalValidationError::TooManyGCEProposals);
592            }
593
594            match queued_proposal.proposal() {
595                Proposal::GroupContextExtensions(extensions) => {
596                    let required_capabilities_in_proposal =
597                        extensions.extensions().required_capabilities();
598
599                    // Prepare the empty required capabilities in case there is no
600                    // RequiredCapabilitiesExtension in the proposal
601                    let default_required_capabilities =
602                        RequiredCapabilitiesExtension::new(&[], &[], &[]);
603
604                    // If there is a RequiredCapabilitiesExtension in the proposal, validate it and
605                    // use that. Otherwise, use the empty default one.
606                    let required_capabilities = match required_capabilities_in_proposal {
607                        Some(required_capabilities_new) => {
608                            // If a group context extensions proposal updates the required capabilities, we
609                            // need to check that these are satisfied for all existing members of the group.
610                            self.check_extension_support(required_capabilities_new.extension_types()).map_err(|_| GroupContextExtensionsProposalValidationError::RequiredExtensionNotSupportedByAllMembers)?;
611                            required_capabilities_new
612                        }
613                        None => &default_required_capabilities,
614                    };
615
616                    // Make sure that all other extensions are known to be supported, by checking
617                    // that they are default extensions or included in the required capabilities.
618                    let all_extensions_are_in_required_capabilities: bool = extensions
619                        .extensions()
620                        .iter()
621                        .map(|ext| ext.extension_type())
622                        .all(|ext_type| {
623                            ext_type.is_default()
624                                || required_capabilities.requires_extension_type_support(ext_type)
625                        });
626
627                    if !all_extensions_are_in_required_capabilities {
628                        return Err(GroupContextExtensionsProposalValidationError::ExtensionNotInRequiredCapabilities);
629                    }
630                }
631                _ => {
632                    return Err(GroupContextExtensionsProposalValidationError::LibraryError(
633                        LibraryError::custom(
634                            "found non-gce proposal when filtered for gce proposals",
635                        ),
636                    ))
637                }
638            }
639        }
640
641        Ok(())
642    }

644    /// Returns an [`AppDataUpdateValidationError`] if:
645    ///   - An [`AppDataUpdateProposal`] appears before a [`GroupContextExtensionProposal`]
646    ///   - The [`GroupContextExtensionProposal`] updates the [`AppDataDictionary`] when the
647    ///     required capabilities include AppDataUpdate proposal type
648    ///   - For any [`ComponentId`], the list of [`AppDataUpdateProposal`]s includes both Updates
649    ///     and Removes
650    ///   - For any [`ComponentId`], the list of [`AppDataUpdateProposal`]s includes more than one
651    ///     Remove
652    #[cfg(feature = "extensions-draft-08")]
653    pub(crate) fn validate_app_data_update_proposals_and_group_context(
654        &self,
655        proposal_queue: &ProposalQueue,
656    ) -> Result<(), AppDataUpdateValidationError> {
657        let no_app_data_updates = proposal_queue.app_data_update_proposals().next().is_none();
658        if no_app_data_updates {
659            return Ok(());
660        }
661
662        // retrieve the GroupContextExtensions proposal, if available
663        let group_context_extension_proposal = proposal_queue
664            .filtered_by_type(ProposalType::GroupContextExtensions)
665            .filter_map(|queued_proposal| match queued_proposal.proposal() {
666                Proposal::GroupContextExtensions(p) => Some(p),
667                _ => None,
668            })
669            .next();
670
671        // check ordering
672        // return an error if an AppDataUpdate appears before a GroupContextExtensions proposal
673        //
674        // From the draft:
675        //
676        // A commit can contain a GroupContextExtensions proposal which modifies
677        // GroupContext extensions other than app_data_dictionary, and can be
678        // followed by zero or more AppDataUpdate proposals.
679        //
680        // https://datatracker.ietf.org/doc/html/draft-ietf-mls-extensions#section-4.7-7
681        //
682        // Note that it is a bit unclear if this is really needed; we already require that default
683        // proposals are processed first.
684        if proposal_queue
685            .queued_proposals()
686            .map(|proposal| proposal.proposal().proposal_type())
687            .skip_while(|proposal_type| *proposal_type != ProposalType::AppDataUpdate)
688            .any(|proposal_type| proposal_type == ProposalType::GroupContextExtensions)
689        {
690            return Err(AppDataUpdateValidationError::IncorrectOrder);
691        }
692
693        if let Some(group_context_extension) = group_context_extension_proposal {
694            let required_capabilities_contain_app_data_update_proposal = group_context_extension
695                .extensions()
696                .required_capabilities()
697                .map(|required_capabilities| {
698                    required_capabilities
699                        .proposal_types()
700                        .contains(&ProposalType::AppDataUpdate)
701                })
702                .unwrap_or(false);
703
704            // From https://datatracker.ietf.org/doc/html/draft-ietf-mls-extensions#section-4.7-6:
705            // When an MLS group contains the AppDataUpdate proposal type in the proposal_types list in
706            // the group's required_capabilities extension, a GroupContextExtensions proposal MUST NOT
707            // add, remove, or modify the app_data_dictionary GroupContext extension. In other words,
708            // when every member of the group supports the AppDataUpdate proposal, a
709            // GroupContextExtensions proposal could be sent to update some other extension(s), but the
710            // app_data_dictionary GroupContext extension, if it exists, is left as it was.
711            if required_capabilities_contain_app_data_update_proposal
712                && group_context_extension.extensions().app_data_dictionary()
713                    != self.group_context().extensions().app_data_dictionary()
714            {
715                return Err(AppDataUpdateValidationError::CannotUpdateDictionaryDirectly);
716            }
717        }
718
719        // From the draft:
720        //
721        // A proposal list is invalid if it includes multiple AppDataUpdate proposals that remove state for the same component_id, or proposals that both update and remove state for the same component_id.
722        //
723        // https://datatracker.ietf.org/doc/html/draft-ietf-mls-extensions#section-4.7-4
724        // NOTE: We depend on the proposals being sorted by component id first and by
725        // type second - for a given ID, removes come first. The sorting is stable and
726        // won't mess with the ordering of updates within a component.
727        // This is ensured in ProposalQueue::app_data_update_proposals.
728        let mut latest = None;
729        for proposal in proposal_queue.app_data_update_proposals() {
730            let proposal = &proposal.app_data_update_proposal;
731            let component_id = proposal.component_id();
732            let operation_type = proposal.operation().operation_type();
733
734            if latest == Some(component_id) {
735                let app_data_update_validation_error = match operation_type {
736                    AppDataUpdateOperationType::Update => {
737                        AppDataUpdateValidationError::CombinedRemoveAndUpdateOperations
738                    }
739                    AppDataUpdateOperationType::Remove => {
740                        AppDataUpdateValidationError::CombinedRemoveAndUpdateOperations
741                    }
742                };
743
744                return Err(app_data_update_validation_error);
745            }
746
747            if proposal.operation().operation_type() == AppDataUpdateOperationType::Remove {
748                // From the draft:
749                //
750                // An AppDataUpdate proposal is invalid if [...] it specifies the removal of
751                // state for a component_id that has no state present.
752                //
753                // https://datatracker.ietf.org/doc/html/draft-ietf-mls-extensions#section-4.7-4
754                let Some(gce) = group_context_extension_proposal else {
755                    // extension gets implicitly created in the group context, so absence is not an
756                    // error condition
757                    return Ok(());
758                };
759                let Some(app_data_dict) = gce.extensions().app_data_dictionary() else {
760                    // extension gets implicitly created in the group context, so absence is not an
761                    // error condition
762                    return Ok(());
763                };
764
765                if app_data_dict.dictionary().get(&component_id).is_none() {
766                    return Err(AppDataUpdateValidationError::CannotRemoveNonexistentComponent);
767                }
768
769                latest = Some(component_id)
770            }
771        }
772
773        Ok(())
774    }

776    fn validate_leaf_node_capabilities(
777        &self,
778        leaf_node: &LeafNode,
779    ) -> Result<(), LeafNodeValidationError> {
780        // Check that the data in the leaf node is self-consistent
781        // Check that the capabilities contain the leaf node's credential
782        // type (https://validation.openmls.tech/#valn0113)
783        // Check that all extension types are valid in leaf node
784        // (https://validation.openmls.tech/#valn1601)
785        leaf_node.validate_locally()?;
786
787        // Check if the ciphersuite and the version of the group are
788        // supported.
789        let capabilities = leaf_node.capabilities();
790        if !capabilities.contains_ciphersuite(VerifiableCiphersuite::from(self.ciphersuite()))
791            || !capabilities.contains_version(self.version())
792        {
793            return Err(LeafNodeValidationError::CiphersuiteNotInCapabilities);
794        }
795
796        // If there is a required capabilities extension, check if that one
797        // is supported (https://validation.openmls.tech/#valn0103).
798        if let Some(required_capabilities) =
799            self.group_context().extensions().required_capabilities()
800        {
801            // Check if all required capabilities are supported.
802            capabilities.supports_required_capabilities(required_capabilities)?;
803        }
804
805        // Check that the credential type is supported by all members of the group (https://validation.openmls.tech/#valn0104).
806        if !self.treesync().full_leaves().all(|(_, node)| {
807            node.capabilities()
808                .contains_credential(leaf_node.credential().credential_type())
809        }) {
810            return Err(LeafNodeValidationError::UnsupportedCredentials);
811        }
812
813        // Check that the capabilities field of this LeafNode indicates
814        // support for all the credential types currently in use by other
815        // members (https://validation.openmls.tech/#valn0104).
816        if !self
817            .treesync()
818            .full_leaves()
819            .all(|(_, node)| capabilities.contains_credential(node.credential().credential_type()))
820        {
821            return Err(LeafNodeValidationError::UnsupportedCredentials);
822        }
823
824        Ok(())
825    }

827    /// Validate a leaf node.
828    ///
829    /// This always validates the lifetime.
830    pub(crate) fn validate_leaf_node(
831        &self,
832        leaf_node: &crate::treesync::LeafNode,
833    ) -> Result<(), LeafNodeValidationError> {
834        // Call the validation function and validate the lifetime
835        self.validate_leaf_node_inner(leaf_node, LeafNodeLifetimePolicy::Verify)
836    }

838    /// Validate a leaf node.
839    ///
840    /// This may skip checking the lifetime when validating a ratchet tree.
841    pub(crate) fn validate_leaf_node_inner(
842        &self,
843        leaf_node: &crate::treesync::LeafNode,
844        validate_lifetimes: LeafNodeLifetimePolicy,
845    ) -> Result<(), LeafNodeValidationError> {
846        // https://validation.openmls.tech/#valn0103
847        // https://validation.openmls.tech/#valn0104
848        // https://validation.openmls.tech/#valn0107
849        self.validate_leaf_node_capabilities(leaf_node)?;
850
851        // https://validation.openmls.tech/#valn0105 is done when sending
852
853        // https://validation.openmls.tech/#valn0106
854        //
855        // Only leaf nodes in key packages contain lifetimes, so this will return None for other
856        // cases. Therefore we only check the lifetimes for leaf nodes in key packages.
857        //
858        // We may want to check these in ratchet trees as well.
859        // However, this may lead to errors when leaf nodes don't get updated
860        // after being added to the tree. RFC 9420 recommends checking the lifetime
861        // but acknowledges already that this may cause issues.
862        // https://www.rfc-editor.org/rfc/rfc9420.html#section-7.3-4.5.1
863        // See #1810 for more background.
864        // We therefore check the lifetime by default, but skip it if ...
865        //
866        // Some KATs use key packages that are expired by now. In order to run these tests, we
867        // provide a way to turn off this check.
868        if matches!(validate_lifetimes, LeafNodeLifetimePolicy::Verify)
869            && !crate::skip_validation::is_disabled::leaf_node_lifetime()
870        {
871            if let Some(lifetime) = leaf_node.life_time() {
872                if !lifetime.is_valid() {
873                    log::warn!(
874                        "offending lifetime: {lifetime:?} for leaf node with {credential:?}",
875                        credential = leaf_node.credential()
876                    );
877                    return Err(LeafNodeValidationError::Lifetime(LifetimeError::NotCurrent));
878                }
879            }
880        }
881
882        // These are done at the caller and we can't do them here:
883        //
884        // https://validation.openmls.tech/#valn0108
885        // https://validation.openmls.tech/#valn0109
886        // https://validation.openmls.tech/#valn0110
887
888        // These are done in validate_key_uniqueness, which is called in the context of changing
889        // this group:
890        //
891        // https://validation.openmls.tech/#valn0111
892        // https://validation.openmls.tech/#valn0112
893
894        Ok(())
895    }

897    /// Returns a [`LeafNodeValidationError`] if an [`ExtensionType`]
898    /// in `extensions` is not supported by a leaf in this tree.
899    pub(crate) fn check_extension_support(
900        &self,
901        extensions: &[crate::extensions::ExtensionType],
902    ) -> Result<(), LeafNodeValidationError> {
903        for (_, leaf) in self.treesync().full_leaves() {
904            leaf.check_extension_support(extensions)?;
905        }
906        Ok(())
907    }
