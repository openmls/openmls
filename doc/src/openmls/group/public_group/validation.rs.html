42    /// Checks the following semantic validation:
43    ///  - ValSem002
44    ///  - ValSem003
45    ///  - [valn1307](https://validation.openmls.tech/#valn1307)
46    pub(crate) fn validate_framing(
47        &self,
48        message: &ProtocolMessage,
49    ) -> Result<(), ValidationError> {
50        // ValSem002
51        if message.group_id() != self.group_id() {
52            return Err(ValidationError::WrongGroupId);
53        }
54
55        // ValSem003: Check boundaries for the epoch
56        // We differentiate depending on the content type
57        match message.content_type() {
58            // For application messages we allow messages for older epochs as well
59            ContentType::Application => {
60                if message.epoch() > self.group_context().epoch() {
61                    log::error!(
62                        "Wrong Epoch: message.epoch() {} > {} self.group_context().epoch()",
63                        message.epoch(),
64                        self.group_context().epoch()
65                    );
66                    return Err(ValidationError::WrongEpoch);
67                }
68            }
69            // For all other messages we only only accept the current epoch
70            _ => {
71                // https://validation.openmls.tech/#valn1307
72                if message.epoch() != self.group_context().epoch() {
73                    log::error!(
74                        "Wrong Epoch: message.epoch() {} != {} self.group_context().epoch()",
75                        message.epoch(),
76                        self.group_context().epoch()
77                    );
78                    return Err(ValidationError::WrongEpoch);
79                }
80            }
81        }
82
83        Ok(())
84    }

86    /// Checks the following semantic validation:
87    ///  - ValSem004
88    ///  - ValSem005
89    ///  - ValSem009
90    pub(super) fn validate_verifiable_content(
91        &self,
92        verifiable_content: &VerifiableAuthenticatedContentIn,
93        message_secrets_store_option: Option<&MessageSecretsStore>,
94    ) -> Result<(), ValidationError> {
95        // ValSem004
96        let sender = verifiable_content.sender();
97        if let Sender::Member(leaf_index) = sender {
98            // If the sender is a member, it has to be in the tree, except if
99            // it's an application message. Then it might be okay if it's in an
100            // old secret tree instance.
101            let is_in_secrets_store = if let Some(mss) = message_secrets_store_option {
102                mss.epoch_has_leaf(verifiable_content.epoch(), *leaf_index)
103            } else {
104                false
105            };
106            if !self.treesync().is_leaf_in_tree(*leaf_index) && !is_in_secrets_store {
107                return Err(ValidationError::UnknownMember);
108            }
109        }
110
111        // ValSem005
112        // Application messages must always be encrypted
113        if verifiable_content.content_type() == ContentType::Application {
114            if verifiable_content.wire_format() != WireFormat::PrivateMessage {
115                return Err(ValidationError::UnencryptedApplicationMessage);
116            } else if !verifiable_content.sender().is_member() {
117                return Err(ValidationError::NonMemberApplicationMessage);
118            }
119        }
120
121        // ValSem009
122        if verifiable_content.content_type() == ContentType::Commit
123            && verifiable_content.confirmation_tag().is_none()
124        {
125            return Err(ValidationError::MissingConfirmationTag);
126        }
127
128        Ok(())
129    }

133    /// Validate that all group members support the types of all proposals.
134    /// Implements check [valn0311](https://validation.openmls.tech/#valn0311)
135    pub(crate) fn validate_proposal_type_support(
136        &self,
137        proposal_queue: &ProposalQueue,
138    ) -> Result<(), ProposalValidationError> {
139        let mut leaves = self.treesync().full_leaves();
140        let Some(first_leaf) = leaves.next() else {
141            return Ok(());
142        };
143        // Initialize the capabilities intersection with the capabilities of the
144        // first leaf node.
145        let mut capabilities_intersection = first_leaf
146            .capabilities()
147            .proposals()
148            .iter()
149            .collect::<HashSet<_>>();
150        // Iterate over the remaining leaf nodes and intersect their capabilities
151        for leaf_node in leaves {
152            let leaf_capabilities_set = leaf_node.capabilities().proposals().iter().collect();
153            capabilities_intersection = capabilities_intersection
154                .intersection(&leaf_capabilities_set)
155                .cloned()
156                .collect();
157        }
158
159        // Check that the types of all non-default proposals are supported by all members
160        for proposal in proposal_queue.queued_proposals() {
161            let proposal_type = proposal.proposal().proposal_type();
162            if !proposal_type.is_default() && !capabilities_intersection.contains(&proposal_type) {
163                return Err(ProposalValidationError::UnsupportedProposalType);
164            }
165        }
166        Ok(())
167    }

169    /// Validate key uniqueness. This function implements the following checks:
170    ///  - ValSem101: Add Proposal: Signature public key in proposals must be unique among proposals & members
171    ///  - ValSem102: Add Proposal: Init key in proposals must be unique among proposals
172    ///  - ValSem103: Add Proposal: Encryption key in proposals must be unique among proposals & members
173    ///  - ValSem104: Add Proposal: Init key and encryption key must be different
174    ///  - ValSem110: Update Proposal: Encryption key must be unique among proposals & members
175    ///  - ValSem206: Commit: Path leaf node encryption key must be unique among proposals & members
176    ///  - ValSem207: Commit: Path encryption keys must be unique among proposals & members
177    ///  - [valn0111]: Verify that the following fields are unique among the members of the group: `signature_key`
178    ///  - [valn0112]: Verify that the following fields are unique among the members of the group: `encryption_key`
179    ///
180    /// [valn0111]: https://validation.openmls.tech/#valn0111
181    /// [valn0112]: https://validation.openmls.tech/#valn0112
182    /// [valn1208]: https://validation.openmls.tech/#valn1208
183    pub(crate) fn validate_key_uniqueness(
184        &self,
185        proposal_queue: &ProposalQueue,
186        commit: Option<&Commit>,
187    ) -> Result<(), ProposalValidationError> {
188        let mut signature_key_set = HashSet::new();
189        let mut init_key_set = HashSet::new();
190        let mut encryption_key_set = HashSet::new();
191
192        // Handle the exceptions needed for https://validation.openmls.tech/#valn0306
193        let remove_proposals = HashSet::<LeafNodeIndex>::from_iter(
194            proposal_queue
195                .remove_proposals()
196                .map(|remove_proposal| remove_proposal.remove_proposal().removed),
197        );
198
199        // Initialize the sets with the current members, filtered by the
200        // remove proposals.
201        for Member {
202            index,
203            encryption_key,
204            signature_key,
205            ..
206        } in self.treesync().full_leave_members()
207        {
208            if !remove_proposals.contains(&index) {
209                signature_key_set.insert(signature_key);
210                encryption_key_set.insert(encryption_key);
211            }
212        }
213
214        // Collect signature keys from add proposals
215        let signature_keys = proposal_queue.add_proposals().map(|add_proposal| {
216            add_proposal
217                .add_proposal()
218                .key_package()
219                .leaf_node()
220                .signature_key()
221                .as_slice()
222                .to_vec()
223        });
224
225        // Collect encryption keys from add proposals, update proposals, the
226        // commit leaf node and path keys
227        let encryption_keys = proposal_queue
228            .add_proposals()
229            .map(|add_proposal| {
230                add_proposal
231                    .add_proposal()
232                    .key_package()
233                    .leaf_node()
234                    .encryption_key()
235                    .key()
236                    .as_slice()
237                    .to_vec()
238            })
239            .chain(proposal_queue.update_proposals().map(|update_proposal| {
240                update_proposal
241                    .update_proposal()
242                    .leaf_node()
243                    .encryption_key()
244                    .key()
245                    .as_slice()
246                    .to_vec()
247            }))
248            .chain(commit.and_then(|commit| {
249                commit
250                    .path
251                    .as_ref()
252                    .map(|path| path.leaf_node().encryption_key().as_slice().to_vec())
253            }))
254            .chain(
255                commit
256                    .iter()
257                    .filter_map(|commit| {
258                        commit.path.as_ref().map(|path| {
259                            path.nodes()
260                                .iter()
261                                .map(|node| node.encryption_key().as_slice().to_vec())
262                        })
263                    })
264                    .flatten(),
265            );
266
267        // Collect init keys from add proposals
268        let init_keys = proposal_queue.add_proposals().map(|add_proposal| {
269            add_proposal
270                .add_proposal()
271                .key_package()
272                .hpke_init_key()
273                .as_slice()
274                .to_vec()
275        });
276
277        // Validate uniqueness of signature keys
278        //  - ValSem101
279        //  - https://validation.openmls.tech/#valn0111
280        //  - https://validation.openmls.tech/#valn0305
281        //  - https://validation.openmls.tech/#valn0306
282        for signature_key in signature_keys {
283            if !signature_key_set.insert(signature_key) {
284                return Err(ProposalValidationError::DuplicateSignatureKey);
285            }
286        }
287
288        // Validate uniqueness of encryption keys
289        //  - ValSem103
290        //  - ValSem104
291        //  - ValSem110
292        //  - ValSem206
293        //  - ValSem207
294        //  - https://validation.openmls.tech/#valn0112
295        for encryption_key in encryption_keys {
296            if init_key_set.contains(&encryption_key) {
297                return Err(ProposalValidationError::InitEncryptionKeyCollision);
298            }
299            if !encryption_key_set.insert(encryption_key) {
300                return Err(ProposalValidationError::DuplicateEncryptionKey);
301            }
302        }
303
304        // Validate uniqueness of init keys
305        //  - ValSem102
306        //  - ValSem104
307        for init_key in init_keys {
308            if encryption_key_set.contains(&init_key) {
309                return Err(ProposalValidationError::InitEncryptionKeyCollision);
310            }
311            if !init_key_set.insert(init_key) {
312                return Err(ProposalValidationError::DuplicateInitKey);
313            }
314        }
315
316        Ok(())
317    }

319    /// Validate capabilities. This function implements the following checks:
320    /// - ValSem106: Add Proposal: required capabilities
321    /// - ValSem109: Update Proposal: required capabilities
322    /// - [valn0113](https://validation.openmls.tech/#valn0113).
323    pub(crate) fn validate_capabilities(
324        &self,
325        proposal_queue: &ProposalQueue,
326    ) -> Result<(), ProposalValidationError> {
327        // ValSem106/ValSem109: Check the required capabilities of the add & update
328        // proposals This includes the following checks:
329        // - Are ciphersuite & version listed in the `Capabilities` Extension?
330        // - If a `RequiredCapabilitiesExtension` is present in the group: Is
331        //   this supported by the node?
332        // - Check that all extensions are contained in the capabilities.
333        // - Check that the capabilities contain the leaf node's credential
334        //   type (https://validation.openmls.tech/#valn0113).
335        // - Check that the credential type is supported by all members of the
336        //   group.
337        // - Check that the capabilities field of this LeafNode indicates
338        //   support for all the credential types currently in use by other
339        //   members.
340
341        // Extract the leaf nodes from the add & update proposals and validate them
342        proposal_queue
343            .queued_proposals()
344            .filter_map(|p| match p.proposal() {
345                Proposal::Add(add_proposal) => Some(add_proposal.key_package().leaf_node()),
346                Proposal::Update(update_proposal) => Some(update_proposal.leaf_node()),
347                _ => None,
348            })
349            .try_for_each(|leaf_node| {
350                self.validate_leaf_node_capabilities(leaf_node)
351                    .map_err(|_| ProposalValidationError::InsufficientCapabilities)
352            })
353    }

355    /// Validate Add proposals. This function implements the following checks:
356    ///  - ValSem105: Add Proposal: Ciphersuite & protocol version must match the group
357    pub(crate) fn validate_add_proposals(
358        &self,
359        proposal_queue: &ProposalQueue,
360    ) -> Result<(), ProposalValidationError> {
361        let add_proposals = proposal_queue.add_proposals();
362
363        // We do the key package validation checks here inline
364        // https://validation.openmls.tech/#valn0501
365        for add_proposal in add_proposals {
366            // ValSem105: Check if ciphersuite and version of the group are correct:
367            // https://validation.openmls.tech/#valn0201
368            if add_proposal.add_proposal().key_package().ciphersuite() != self.ciphersuite()
369                || add_proposal.add_proposal().key_package().protocol_version() != self.version()
370            {
371                return Err(ProposalValidationError::InvalidAddProposalCiphersuiteOrVersion);
372            }
373
374            // https://validation.openmls.tech/#valn0202
375            self.validate_leaf_node(add_proposal.add_proposal().key_package().leaf_node())?;
376        }
377        Ok(())
378    }

380    /// Validate Remove proposals. This function implements the following checks:
381    ///  - ValSem107: Remove Proposal: Removed member must be unique among proposals
382    ///  - ValSem108: Remove Proposal: Removed member must be an existing group member
383    pub(crate) fn validate_remove_proposals(
384        &self,
385        proposal_queue: &ProposalQueue,
386    ) -> Result<(), ProposalValidationError> {
387        let updates_set: HashSet<_> = proposal_queue
388            .update_proposals()
389            .map(|proposal| {
390                if let Sender::Member(index) = proposal.sender() {
391                    Ok(*index)
392                } else {
393                    Err(ProposalValidationError::UpdateFromNonMember)
394                }
395            })
396            .collect::<Result<_, _>>()?;
397
398        let remove_proposals = proposal_queue.remove_proposals();
399
400        let mut removes_set = HashSet::new();
401
402        // https://validation.openmls.tech/#valn0701
403        for remove_proposal in remove_proposals {
404            let removed = remove_proposal.remove_proposal().removed();
405            // The node has to be a leaf in the tree
406            // ValSem108
407            if !self.treesync().is_leaf_in_tree(removed) {
408                return Err(ProposalValidationError::UnknownMemberRemoval);
409            }
410
411            // ValSem107
412            // https://validation.openmls.tech/#valn0304
413            if !removes_set.insert(removed) {
414                return Err(ProposalValidationError::DuplicateMemberRemoval);
415            }
416            if updates_set.contains(&removed) {
417                return Err(ProposalValidationError::DuplicateMemberRemoval);
418            }
419
420            // removed node can not be blank
421            if self.treesync().leaf(removed).is_none() {
422                return Err(ProposalValidationError::UnknownMemberRemoval);
423            }
424        }
425
426        Ok(())
427    }

429    /// Validate Update proposals. This function implements the following checks:
430    ///  - ValSem111: Update Proposal: The sender of a full Commit must not include own update proposals
431    ///  - ValSem112: Update Proposal: The sender of a standalone update proposal must be of type member
432    ///
433    /// TODO: #133 This validation must be updated according to Sec. 13.2
434    pub(crate) fn validate_update_proposals(
435        &self,
436        proposal_queue: &ProposalQueue,
437        committer: LeafNodeIndex,
438    ) -> Result<(), ProposalValidationError> {
439        // Check the update proposals from the proposal queue first
440        let update_proposals = proposal_queue.update_proposals();
441
442        for update_proposal in update_proposals {
443            // ValSem112
444            // The sender of a standalone update proposal must be of type member
445            if let Sender::Member(sender_index) = update_proposal.sender() {
446                // ValSem111
447                // The sender of a full Commit must not include own update proposals
448                if committer == *sender_index {
449                    return Err(ProposalValidationError::CommitterIncludedOwnUpdate);
450                }
451            } else {
452                return Err(ProposalValidationError::UpdateFromNonMember);
453            }
454
455            // https://validation.openmls.tech/#valn0601
456            self.validate_leaf_node(update_proposal.update_proposal().leaf_node())?;
457        }
458        Ok(())
459    }

483    /// Validate constraints on an external commit. This function implements the following checks:
484    ///  - ValSem240: External Commit, inline Proposals: There MUST be at least one ExternalInit proposal.
485    ///  - ValSem241: External Commit, inline Proposals: There MUST be at most one ExternalInit proposal.
486    ///  - ValSem242: External Commit must only cover inline proposal in allowlist (ExternalInit, Remove, PreSharedKey)
487    pub(crate) fn validate_external_commit(
488        &self,
489        proposal_queue: &ProposalQueue,
490    ) -> Result<(), ExternalCommitValidationError> {
491        // [valn0401](https://validation.openmls.tech/#valn0401)
492        let count_external_init_proposals = proposal_queue
493            .filtered_by_type(ProposalType::ExternalInit)
494            .count();
495        if count_external_init_proposals == 0 {
496            // ValSem240: External Commit, inline Proposals: There MUST be at least one ExternalInit proposal.
497            return Err(ExternalCommitValidationError::NoExternalInitProposals);
498        } else if count_external_init_proposals > 1 {
499            // ValSem241: External Commit, inline Proposals: There MUST be at most one ExternalInit proposal.
500            return Err(ExternalCommitValidationError::MultipleExternalInitProposals);
501        }
502
503        // ValSem242: External Commit must only cover inline proposal in allowlist (ExternalInit, Remove, PreSharedKey)
504        // [valn0404](https://validation.openmls.tech/#valn0404)
505        let contains_denied_proposal = proposal_queue.queued_proposals().any(|p| {
506            let is_inline = p.proposal_or_ref_type() == ProposalOrRefType::Proposal;
507            let is_allowed_type = matches!(
508                p.proposal(),
509                Proposal::ExternalInit(_)
510                    | Proposal::Remove(_)
511                    | Proposal::PreSharedKey(_)
512                    | Proposal::Custom(_)
513            );
514            is_inline && !is_allowed_type
515        });
516        if contains_denied_proposal {
517            return Err(ExternalCommitValidationError::InvalidInlineProposals);
518        }
519
520        // If a Remove proposal is present,
521        // the credential in the LeafNode MUST present a set of
522        // identifiers that is acceptable to the application for
523        // the removed participant.
524        // This MUST be checked by the application.
525
526        Ok(())
527    }

529    /// Returns a [`LeafNodeValidationError`] if an [`ExtensionType`]
530    /// in `extensions` is not supported by a leaf in this tree.
531    /// Implements check [valn1001](https://validation.openmls.tech/#valn1001).
532    pub(crate) fn validate_group_context_extensions_proposal(
533        &self,
534        proposal_queue: &ProposalQueue,
535    ) -> Result<(), GroupContextExtensionsProposalValidationError> {
536        let iter = proposal_queue.filtered_by_type(ProposalType::GroupContextExtensions);
537
538        for (i, queued_proposal) in iter.enumerate() {
539            // There must at most be one group context extionsion proposal. Return an error if there are more
540            if i > 0 {
541                return Err(GroupContextExtensionsProposalValidationError::TooManyGCEProposals);
542            }
543
544            match queued_proposal.proposal() {
545                Proposal::GroupContextExtensions(extensions) => {
546                    let required_capabilities_in_proposal =
547                        extensions.extensions().required_capabilities();
548
549                    // Prepare the empty required capabilities in case there is no
550                    // RequiredCapabilitiesExtension in the proposal
551                    let default_required_capabilities =
552                        RequiredCapabilitiesExtension::new(&[], &[], &[]);
553
554                    // If there is a RequiredCapabilitiesExtension in the proposal, validate it and
555                    // use that. Otherwise, use the empty default one.
556                    let required_capabilities = match required_capabilities_in_proposal {
557                        Some(required_capabilities_new) => {
558                            // If a group context extensions proposal updates the required capabilities, we
559                            // need to check that these are satisfied for all existing members of the group.
560                            self.check_extension_support(required_capabilities_new.extension_types()).map_err(|_| GroupContextExtensionsProposalValidationError::RequiredExtensionNotSupportedByAllMembers)?;
561                            required_capabilities_new
562                        }
563                        None => &default_required_capabilities,
564                    };
565
566                    // Make sure that all other extensions are known to be supported, by checking
567                    // that they are default extensions or included in the required capabilities.
568                    let all_extensions_are_in_required_capabilities: bool = extensions
569                        .extensions()
570                        .iter()
571                        .map(|ext| ext.extension_type())
572                        .all(|ext_type| {
573                            ext_type.is_default()
574                                || required_capabilities.requires_extension_type_support(ext_type)
575                        });
576
577                    if !all_extensions_are_in_required_capabilities {
578                        return Err(GroupContextExtensionsProposalValidationError::ExtensionNotInRequiredCapabilities);
579                    }
580                }
581                _ => {
582                    return Err(GroupContextExtensionsProposalValidationError::LibraryError(
583                        LibraryError::custom(
584                            "found non-gce proposal when filtered for gce proposals",
585                        ),
586                    ))
587                }
588            }
589        }
590
591        Ok(())
592    }

594    /// Returns an [`AppDataUpdateValidationError`] if:
595    ///   - An [`AppDataUpdateProposal`] appears before a [`GroupContextExtensionProposal`]
596    ///   - The [`GroupContextExtensionProposal`] updates the [`AppDataDictionary`] when the
597    ///     required capabilities include AppDataUpdate proposal type
598    ///   - For any [`ComponentId`], the list of [`AppDataUpdateProposal`]s includes both Updates
599    ///     and Removes
600    ///   - For any [`ComponentId`], the list of [`AppDataUpdateProposal`]s includes more than one
601    ///     Remove
602    #[cfg(feature = "extensions-draft-08")]
603    pub(crate) fn validate_app_data_update_proposals_and_group_context(
604        &self,
605        proposal_queue: &ProposalQueue,
606    ) -> Result<(), AppDataUpdateValidationError> {
607        let no_app_data_updates = proposal_queue.app_data_update_proposals().next().is_none();
608        if no_app_data_updates {
609            return Ok(());
610        }
611
612        // retrieve the GroupContextExtensions proposal, if available
613        let group_context_extension_proposal = proposal_queue
614            .filtered_by_type(ProposalType::GroupContextExtensions)
615            .filter_map(|queued_proposal| match queued_proposal.proposal() {
616                Proposal::GroupContextExtensions(p) => Some(p),
617                _ => None,
618            })
619            .next();
620
621        // check ordering
622        // return an error if an AppDataUpdate appears before a GroupContextExtensions proposal
623        //
624        // From the draft:
625        //
626        // A commit can contain a GroupContextExtensions proposal which modifies
627        // GroupContext extensions other than app_data_dictionary, and can be
628        // followed by zero or more AppDataUpdate proposals.
629        //
630        // https://datatracker.ietf.org/doc/html/draft-ietf-mls-extensions#section-4.7-7
631        //
632        // Note that it is a bit unclear if this is really needed; we already require that default
633        // proposals are processed first.
634        if proposal_queue
635            .queued_proposals()
636            .map(|proposal| proposal.proposal().proposal_type())
637            .skip_while(|proposal_type| *proposal_type != ProposalType::AppDataUpdate)
638            .any(|proposal_type| proposal_type == ProposalType::GroupContextExtensions)
639        {
640            return Err(AppDataUpdateValidationError::IncorrectOrder);
641        }
642
643        if let Some(group_context_extension) = group_context_extension_proposal {
644            let required_capabilities_contain_app_data_update_proposal = group_context_extension
645                .extensions()
646                .required_capabilities()
647                .map(|required_capabilities| {
648                    required_capabilities
649                        .proposal_types()
650                        .contains(&ProposalType::AppDataUpdate)
651                })
652                .unwrap_or(false);
653
654            // From https://datatracker.ietf.org/doc/html/draft-ietf-mls-extensions#section-4.7-6:
655            // When an MLS group contains the AppDataUpdate proposal type in the proposal_types list in
656            // the group's required_capabilities extension, a GroupContextExtensions proposal MUST NOT
657            // add, remove, or modify the app_data_dictionary GroupContext extension. In other words,
658            // when every member of the group supports the AppDataUpdate proposal, a
659            // GroupContextExtensions proposal could be sent to update some other extension(s), but the
660            // app_data_dictionary GroupContext extension, if it exists, is left as it was.
661            if required_capabilities_contain_app_data_update_proposal
662                && group_context_extension.extensions().app_data_dictionary()
663                    != self.group_context().extensions().app_data_dictionary()
664            {
665                return Err(AppDataUpdateValidationError::CannotUpdateDictionaryDirectly);
666            }
667        }
668
669        // From the draft:
670        //
671        // A proposal list is invalid if it includes multiple AppDataUpdate proposals that remove state for the same component_id, or proposals that both update and remove state for the same component_id.
672        //
673        // https://datatracker.ietf.org/doc/html/draft-ietf-mls-extensions#section-4.7-4
674        // NOTE: We depend on the proposals being sorted by component id first and by
675        // type second - for a given ID, removes come first. The sorting is stable and
676        // won't mess with the ordering of updates within a component.
677        // This is ensured in ProposalQueue::app_data_update_proposals.
678        let mut latest = None;
679        for proposal in proposal_queue.app_data_update_proposals() {
680            let proposal = &proposal.app_data_update_proposal;
681            let component_id = proposal.component_id();
682            let operation_type = proposal.operation().operation_type();
683
684            if latest == Some(component_id) {
685                let app_data_update_validation_error = match operation_type {
686                    AppDataUpdateOperationType::Update => {
687                        AppDataUpdateValidationError::CombinedRemoveAndUpdateOperations
688                    }
689                    AppDataUpdateOperationType::Remove => {
690                        AppDataUpdateValidationError::CombinedRemoveAndUpdateOperations
691                    }
692                };
693
694                return Err(app_data_update_validation_error);
695            }
696
697            if proposal.operation().operation_type() == AppDataUpdateOperationType::Remove {
698                // From the draft:
699                //
700                // An AppDataUpdate proposal is invalid if [...] it specifies the removal of
701                // state for a component_id that has no state present.
702                //
703                // https://datatracker.ietf.org/doc/html/draft-ietf-mls-extensions#section-4.7-4
704                let Some(gce) = group_context_extension_proposal else {
705                    // extension gets implicitly created in the group context, so absence is not an
706                    // error condition
707                    return Ok(());
708                };
709                let Some(app_data_dict) = gce.extensions().app_data_dictionary() else {
710                    // extension gets implicitly created in the group context, so absence is not an
711                    // error condition
712                    return Ok(());
713                };
714
715                if app_data_dict.dictionary().get(&component_id).is_none() {
716                    return Err(AppDataUpdateValidationError::CannotRemoveNonexistentComponent);
717                }
718
719                latest = Some(component_id)
720            }
721        }
722
723        Ok(())
724    }

726    fn validate_leaf_node_capabilities(
727        &self,
728        leaf_node: &LeafNode,
729    ) -> Result<(), LeafNodeValidationError> {
730        // Check that the data in the leaf node is self-consistent
731        // Check that the capabilities contain the leaf node's credential
732        // type (https://validation.openmls.tech/#valn0113)
733        // Check that all extension types are valid in leaf node
734        // (https://validation.openmls.tech/#valn1601)
735        leaf_node.validate_locally()?;
736
737        // Check if the ciphersuite and the version of the group are
738        // supported.
739        let capabilities = leaf_node.capabilities();
740        if !capabilities.contains_ciphersuite(VerifiableCiphersuite::from(self.ciphersuite()))
741            || !capabilities.contains_version(self.version())
742        {
743            return Err(LeafNodeValidationError::CiphersuiteNotInCapabilities);
744        }
745
746        // If there is a required capabilities extension, check if that one
747        // is supported (https://validation.openmls.tech/#valn0103).
748        if let Some(required_capabilities) =
749            self.group_context().extensions().required_capabilities()
750        {
751            // Check if all required capabilities are supported.
752            capabilities.supports_required_capabilities(required_capabilities)?;
753        }
754
755        // Check that the credential type is supported by all members of the group (https://validation.openmls.tech/#valn0104).
756        if !self.treesync().full_leaves().all(|node| {
757            node.capabilities()
758                .contains_credential(leaf_node.credential().credential_type())
759        }) {
760            return Err(LeafNodeValidationError::UnsupportedCredentials);
761        }
762
763        // Check that the capabilities field of this LeafNode indicates
764        // support for all the credential types currently in use by other
765        // members (https://validation.openmls.tech/#valn0104).
766        if !self
767            .treesync()
768            .full_leaves()
769            .all(|node| capabilities.contains_credential(node.credential().credential_type()))
770        {
771            return Err(LeafNodeValidationError::UnsupportedCredentials);
772        }
773
774        Ok(())
775    }

777    /// Validate a leaf node.
778    ///
779    /// This always validates the lifetime.
780    pub(crate) fn validate_leaf_node(
781        &self,
782        leaf_node: &crate::treesync::LeafNode,
783    ) -> Result<(), LeafNodeValidationError> {
784        // Call the validation function and validate the lifetime
785        self.validate_leaf_node_inner(leaf_node, LeafNodeLifetimePolicy::Verify)
786    }
787
788    /// Validate a leaf node.
789    ///
790    /// This may skip checking the lifetime when validating a ratchet tree.
791    pub(crate) fn validate_leaf_node_inner(
792        &self,
793        leaf_node: &crate::treesync::LeafNode,
794        validate_lifetimes: LeafNodeLifetimePolicy,
795    ) -> Result<(), LeafNodeValidationError> {
796        // https://validation.openmls.tech/#valn0103
797        // https://validation.openmls.tech/#valn0104
798        // https://validation.openmls.tech/#valn0107
799        self.validate_leaf_node_capabilities(leaf_node)?;
800
801        // https://validation.openmls.tech/#valn0105 is done when sending
802
803        // https://validation.openmls.tech/#valn0106
804        //
805        // Only leaf nodes in key packages contain lifetimes, so this will return None for other
806        // cases. Therefore we only check the lifetimes for leaf nodes in key packages.
807        //
808        // We may want to check these in ratchet trees as well.
809        // However, this may lead to errors when leaf nodes don't get updated
810        // after being added to the tree. RFC 9420 recommends checking the lifetime
811        // but acknowledges already that this may cause issues.
812        // https://www.rfc-editor.org/rfc/rfc9420.html#section-7.3-4.5.1
813        // See #1810 for more background.
814        // We therefore check the lifetime by default, but skip it if ...
815        //
816        // Some KATs use key packages that are expired by now. In order to run these tests, we
817        // provide a way to turn off this check.
818        if matches!(validate_lifetimes, LeafNodeLifetimePolicy::Verify)
819            && !crate::skip_validation::is_disabled::leaf_node_lifetime()
820        {
821            if let Some(lifetime) = leaf_node.life_time() {
822                if !lifetime.is_valid() {
823                    log::warn!(
824                        "offending lifetime: {lifetime:?} for leaf node with {credential:?}",
825                        credential = leaf_node.credential()
826                    );
827                    return Err(LeafNodeValidationError::Lifetime(LifetimeError::NotCurrent));
828                }
829            }
830        }
831
832        // These are done at the caller and we can't do them here:
833        //
834        // https://validation.openmls.tech/#valn0108
835        // https://validation.openmls.tech/#valn0109
836        // https://validation.openmls.tech/#valn0110
837
838        // These are done in validate_key_uniqueness, which is called in the context of changing
839        // this group:
840        //
841        // https://validation.openmls.tech/#valn0111
842        // https://validation.openmls.tech/#valn0112
843
844        Ok(())
845    }

847    /// Returns a [`LeafNodeValidationError`] if an [`ExtensionType`]
848    /// in `extensions` is not supported by a leaf in this tree.
849    pub(crate) fn check_extension_support(
850        &self,
851        extensions: &[crate::extensions::ExtensionType],
852    ) -> Result<(), LeafNodeValidationError> {
853        for leaf in self.treesync().full_leaves() {
854            leaf.check_extension_support(extensions)?;
855        }
856        Ok(())
857    }
