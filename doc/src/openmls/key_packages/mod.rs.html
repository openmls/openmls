1//! # Key Packages
2//!
3//! Key packages are pre-published public keys that carry information about a
4//! user, allowing for the asynchronous addition of clients to an MLS group.
5//!
6//! A key package object specifies:
7//!
8//! - A **protocol version** and ciphersuite that the client supports
9//! - A **public key** that others can use for key agreement
10//! - A **credential** authenticating the client's application-layer identity
11//! - A list of **extensions** for the key package (see
12//!   [Extensions](`mod@crate::extensions`) for details)
13//!
14//! Key packages are meant to be used only once and SHOULD NOT be reused,
15//! except as a last resort—i.e., when no other key package is available.
16//! Clients MAY generate and publish multiple key packages to support multiple
17//! ciphersuites.
18//!
19//! The HPKE init key MUST be a public key for the asymmetric encryption scheme
20//! defined by the ciphersuite. It MUST be unique among key packages created by
21//! the client. The entire structure is signed using the client's signature key.
22//! A key package object with an invalid signature field is considered malformed.
23//!
24//! ## Creating key package bundles
25//!
26//! Key package bundles are key packages that include their private key. A key
27//! package bundle can be created as follows:
28//!
29//! ```
30//! use openmls::{prelude::{*, tls_codec::*}};
31//! use openmls_rust_crypto::OpenMlsRustCrypto;
32//! use openmls_basic_credential::SignatureKeyPair;
33//!
34//! let ciphersuite = Ciphersuite::MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519;
35//! let provider = OpenMlsRustCrypto::default();
36//!
37//! let credential = BasicCredential::new("identity".into());
38//! let signer =
39//!     SignatureKeyPair::new(ciphersuite.signature_algorithm())
40//!         .expect("Error generating a signature key pair.");
41//! let credential_with_key = CredentialWithKey {
42//!     credential: credential.into(),
43//!     signature_key: signer.public().into(),
44//! };
45//! let key_package = KeyPackage::builder()
46//!     .build(
47//!         ciphersuite,
48//!         &provider,
49//!         &signer,
50//!         credential_with_key,
51//!     )
52//!     .unwrap();
53//! ```
54//!
55//! See [`KeyPackage`] for more details and other ways to create key packages.
56//!
57//! ## Loading key packages
58//!
59//! When getting key packages from another user the serialized bytes are parsed
60//! as follows;
61//!
62//! ```
63//! use openmls::prelude::{*, tls_codec::*};
64//! use openmls::test_utils::hex_to_bytes;
65//! use openmls_rust_crypto::OpenMlsRustCrypto;
66//!
67//! let provider = OpenMlsRustCrypto::default();
68//! let ciphersuite = Ciphersuite::MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519;
69//!
70//! let key_package_bytes = hex_to_bytes(
71//!         "0001000120D4F26FCA6EF6B1CA2FDD8DCAA501730FB003323AD8C781490B94782771\
72//!         B22216208E9BF17CE632EC753A9BFC624F275AA745ACD7316A5CF18B8E39CE71A80EE\
73//!         137205639176E415B378BA54B9E7C678FFAA860676CEFEDFA0DD3FF692F20AC7E2632\
74//!         0001086964656E74697479020001060001000200030200010C0001000200030004000\
75//!         50007020001010000000064A1986E00000000776DA97E004040A71F1B7A5F78D15C3F\
76//!         B215D811BADB0BBBD78B582D42E5C3672085699DCA5F90DAA57BB74A3A973789E7006\
77//!         887FCE85F0E64C19C1F26C28B5752B3C3312FF3040040407B5A96167512061E78414F\
78//!         E3F29B89FF2A954CB8E0A6E976EA039E0A1A0AB91B80664BDDC62BC8CBE64BC9242C4\
79//!         CDC33F56A10E425A384AED029C23E1D467C0E");
80//!
81//! let key_package_in = KeyPackageIn::tls_deserialize(&mut key_package_bytes.as_slice())
82//!     .expect("Could not deserialize KeyPackage");
83//!
84//! let key_package = key_package_in
85//!     .validate(provider.crypto(), ProtocolVersion::Mls10)
86//!     .expect("Invalid KeyPackage");
87//! ```
88//!
89//! See [`KeyPackage`] for more details on how to use key packages.

135/// The unsigned payload of a key package.
136/// Any modification must happen on this unsigned struct. Use `sign` to get a
137/// signed key package.
138///
139/// ```text
140/// struct {
141///     ProtocolVersion version;
142///     CipherSuite cipher_suite;
143///     HPKEPublicKey init_key;
144///     LeafNode leaf_node;
145///     Extension extensions<V>;
146/// } KeyPackageTBS;
147/// ```
148#[derive(Debug, Clone, PartialEq, TlsSize, TlsSerialize, Serialize, Deserialize)]
149struct KeyPackageTbs {
150    protocol_version: ProtocolVersion,
151    ciphersuite: Ciphersuite,
152    init_key: InitKey,
153    leaf_node: LeafNode,
154    extensions: Extensions<KeyPackage>,
155}
156
157impl Signable for KeyPackageTbs {
158    type SignedOutput = KeyPackage;
159
160    fn unsigned_payload(&self) -> Result<Vec<u8>, tls_codec::Error> {
161        self.tls_serialize_detached()
162    }
163
164    fn label(&self) -> &str {
165        SIGNATURE_KEY_PACKAGE_LABEL
166    }
167}

169impl From<KeyPackage> for KeyPackageTbs {
170    fn from(kp: KeyPackage) -> Self {
171        kp.payload
172    }
173}

175/// The key package struct.
176#[derive(Debug, Clone, Serialize, Deserialize, TlsSize)]
177pub struct KeyPackage {
178    payload: KeyPackageTbs,
179    signature: Signature,
180    #[serde(skip)]
181    #[tls_codec(skip)]
182    serialized_payload: Option<Vec<u8>>,
183}
184
185impl TlsSerializeTrait for KeyPackage {
186    fn tls_serialize<W: std::io::Write>(&self, writer: &mut W) -> Result<usize, tls_codec::Error> {
187        let mut written = 0;
188        if let Some(ref bytes) = self.serialized_payload {
189            written += writer.write(bytes)?;
190        } else {
191            written += self.payload.tls_serialize(writer)?;
192        }
193        written += self.signature.tls_serialize(writer)?;
194        Ok(written)
195    }
196}

198impl PartialEq for KeyPackage {
199    fn eq(&self, other: &Self) -> bool {
200        // We ignore the signature in the comparison. The same key package
201        // may have different, valid signatures.
202        self.payload == other.payload
203    }
204}

287    /// Create a new key package for the given `ciphersuite` and `identity`.
288    pub(crate) fn create(
289        ciphersuite: Ciphersuite,
290        provider: &impl OpenMlsProvider,
291        signer: &impl Signer,
292        credential_with_key: CredentialWithKey,
293        extensions: Extensions<KeyPackage>,
294        leaf_node_params: KeyPackageLeafNodeParams,
295    ) -> Result<KeyPackageCreationResult, KeyPackageNewError> {
296        if ciphersuite.signature_algorithm() != signer.signature_scheme() {
297            return Err(KeyPackageNewError::CiphersuiteSignatureSchemeMismatch);
298        }
299
300        // Create a new HPKE key pair
301        let ikm = Secret::random(ciphersuite, provider.rand())
302            .map_err(LibraryError::unexpected_crypto_error)?;
303        let init_key = provider
304            .crypto()
305            .derive_hpke_keypair(ciphersuite.hpke_config(), ikm.as_slice())
306            .map_err(|e| {
307                KeyPackageNewError::LibraryError(LibraryError::unexpected_crypto_error(e))
308            })?;
309        let (key_package, encryption_keypair) = Self::new_from_keys(
310            ciphersuite,
311            provider,
312            signer,
313            credential_with_key,
314            extensions,
315            leaf_node_params,
316            init_key.public.into(),
317        )?;
318
319        Ok(KeyPackageCreationResult {
320            key_package,
321            encryption_keypair,
322            init_private_key: init_key.private,
323        })
324    }

326    /// Create a new key package for the given `ciphersuite` and `identity`.
327    ///
328    /// The HPKE init key must have been generated before and the private part
329    /// has to be stored in the key store.
330    ///
331    /// This function returns the new [`KeyPackage`] as well as the
332    /// encryption key ([`HpkeKeyPair`]) of the leaf node.
333    ///
334    /// The caller is responsible for storing the new values.
335    fn new_from_keys(
336        ciphersuite: Ciphersuite,
337        provider: &impl OpenMlsProvider,
338        signer: &impl Signer,
339        credential_with_key: CredentialWithKey,
340        extensions: Extensions<KeyPackage>,
341        leaf_node_params: KeyPackageLeafNodeParams,
342        init_key: InitKey,
343    ) -> Result<(Self, EncryptionKeyPair), KeyPackageNewError> {
344        // We don't need the private key here. It's stored in the key store for
345        // use later when creating a group with this key package.
346
347        let KeyPackageLeafNodeParams {
348            lifetime,
349            capabilities,
350            extensions: leaf_node_extensions,
351        } = leaf_node_params;
352
353        let new_leaf_node_params = NewLeafNodeParams {
354            ciphersuite,
355            credential_with_key,
356            leaf_node_source: LeafNodeSource::KeyPackage(lifetime),
357            capabilities,
358            extensions: leaf_node_extensions,
359            tree_info_tbs: TreeInfoTbs::KeyPackage,
360        };
361
362        let (leaf_node, encryption_key_pair) =
363            LeafNode::new(provider, signer, new_leaf_node_params)?;
364
365        let key_package_tbs = KeyPackageTbs {
366            protocol_version: ProtocolVersion::default(),
367            ciphersuite,
368            init_key,
369            leaf_node,
370            extensions,
371        };
372
373        let key_package = key_package_tbs.sign(signer)?;
374
375        Ok((key_package, encryption_key_pair))
376    }

378    /// Create a new KeyPackage from caller-provided init and encryption keys.
379    ///
380    /// Mirrors [`KeyPackage::new_from_keys`] but takes a caller-provided
381    /// encryption key pair for the leaf instead of generating a fresh one.
382    /// Used by the virtual-clients KeyPackage build path, where both keys are
383    /// derived from a per-operation secret so a sibling can reproduce them.
384    #[cfg(feature = "virtual-clients-draft")]
385    fn new_from_vc_keys(
386        ciphersuite: Ciphersuite,
387        signer: &impl Signer,
388        credential_with_key: CredentialWithKey,
389        extensions: Extensions<KeyPackage>,
390        leaf_node_params: KeyPackageLeafNodeParams,
391        init_key: InitKey,
392        encryption_key_pair: EncryptionKeyPair,
393    ) -> Result<(Self, EncryptionKeyPair), KeyPackageNewError> {
394        let KeyPackageLeafNodeParams {
395            lifetime,
396            capabilities,
397            extensions: leaf_node_extensions,
398        } = leaf_node_params;
399
400        let new_leaf_node_params = NewLeafNodeParams {
401            ciphersuite,
402            credential_with_key,
403            leaf_node_source: LeafNodeSource::KeyPackage(lifetime),
404            capabilities,
405            extensions: leaf_node_extensions,
406            tree_info_tbs: TreeInfoTbs::KeyPackage,
407        };
408
409        let (leaf_node, encryption_key_pair) = LeafNode::new_with_encryption_key_pair(
410            signer,
411            new_leaf_node_params,
412            encryption_key_pair,
413        )?;
414
415        let key_package_tbs = KeyPackageTbs {
416            protocol_version: ProtocolVersion::default(),
417            ciphersuite,
418            init_key,
419            leaf_node,
420            extensions,
421        };
422
423        let key_package = key_package_tbs.sign(signer)?;
424
425        Ok((key_package, encryption_key_pair))
426    }

553    /// Ensure that a last-resort extension is present in the key package if the
554    /// `last_resort` flag is set.
555    fn ensure_last_resort(&mut self) {
556        if self.last_resort {
557            let last_resort_extension = Extension::LastResort(LastResortExtension::default());
558            if let Some(extensions) = self.key_package_extensions.as_mut() {
559                extensions
560                    .add_or_replace(last_resort_extension)
561                    .expect("LastResort extensions are allowed in key packages");
562            } else {
563                self.key_package_extensions = Some(
564                    Extensions::single(last_resort_extension)
565                        .expect("LastResort extensions are allowed in key packages"),
566                );
567            }
568        }
569    }

595    /// Finalize and build the key package.
596    pub fn build(
597        mut self,
598        ciphersuite: Ciphersuite,
599        provider: &impl OpenMlsProvider,
600        signer: &impl Signer,
601        credential_with_key: CredentialWithKey,
602    ) -> Result<KeyPackageBundle, KeyPackageNewError> {
603        self.ensure_last_resort();
604
605        let leaf_node_params = KeyPackageLeafNodeParams {
606            lifetime: self.key_package_lifetime.unwrap_or_default(),
607            capabilities: self.leaf_node_capabilities.unwrap_or_default(),
608            extensions: self.leaf_node_extensions.unwrap_or_default(),
609        };
610        let KeyPackageCreationResult {
611            key_package,
612            encryption_keypair,
613            init_private_key,
614        } = KeyPackage::create(
615            ciphersuite,
616            provider,
617            signer,
618            credential_with_key,
619            self.key_package_extensions.unwrap_or_default(),
620            leaf_node_params,
621        )?;
622
623        // Store the key package in the key store with the hash reference as id
624        // for retrieval when parsing welcome messages.
625        let full_kp = KeyPackageBundle {
626            key_package,
627            private_init_key: init_private_key,
628            private_encryption_key: encryption_keypair.private_key().clone(),
629        };
630        provider
631            .storage()
632            .write_key_package(&full_kp.key_package.hash_ref(provider.crypto())?, &full_kp)
633            .map_err(|_| KeyPackageNewError::StorageError)?;
634
635        Ok(full_kp)
636    }

638    /// Build a batch of virtual-client KeyPackages a sibling can reproduce.
639    ///
640    /// Allocates a single generation of the `key_package` operation ratchet for
641    /// the emulation epoch identified by `epoch_id`. For each
642    /// `key_package_index` in `0..count` it derives a per-KeyPackage seed
643    /// secret from that one operation secret and derives the KeyPackage's init
644    /// key and leaf encryption key from the seed. Each leaf carries an
645    /// encrypted `DerivationInfo` under
646    /// [`VC_COMPONENT_ID`](crate::components::vc_derivation_info::VC_COMPONENT_ID)
647    /// so a sibling can recover the emulation leaf index, generation, and
648    /// index.
649    ///
650    /// The operation secret and the seeds are derived under the emulation
651    /// epoch's ciphersuite (the operation tree's ciphersuite). The init and
652    /// leaf-encryption keys are derived from each seed under the KeyPackage's
653    /// own `ciphersuite`.
654    ///
655    /// Each leaf must declare `AppDataDictionary` support and list
656    /// [`VC_COMPONENT_ID`](crate::components::vc_derivation_info::VC_COMPONENT_ID)
657    /// in its `AppComponents` entry, otherwise this returns
658    /// [`VirtualClientsError::AppDataDictionaryNotSupported`](crate::components::vc_derivation_info::VirtualClientsError::AppDataDictionaryNotSupported)
659    /// or
660    /// [`VirtualClientsError::VcComponentNotListed`](crate::components::vc_derivation_info::VirtualClientsError::VcComponentNotListed).
661    ///
662    /// Returns a [`VcKeyPackageBatch`] holding the batch's `generation` and one
663    /// `(KeyPackageBundle, KeyPackageInfo)` per KeyPackage. Each bundle is also
664    /// written to storage so the creating client can process its own Welcomes,
665    /// and each
666    /// [`KeyPackageInfo`](crate::components::vc_derivation_info::KeyPackageInfo)
667    /// carries the index the client hands to its sibling.
668    ///
669    /// Returns [`KeyPackageNewError::EmptyBatch`] when `count` is 0, before
670    /// loading any state or consuming a generation.
671    #[cfg(feature = "virtual-clients-draft")]
672    pub fn build_vc_batch(
673        self,
674        ciphersuite: Ciphersuite,
675        provider: &impl OpenMlsProvider,
676        signer: &impl Signer,
677        credential_with_key: CredentialWithKey,
678        epoch_id: crate::components::vc_derivation_info::EpochId,
679        count: u32,
680    ) -> Result<VcKeyPackageBatch, KeyPackageNewError> {
681        use crate::components::vc_derivation_info::{
682            EmulationEpochState, VirtualClientOperationType, VirtualClientsError,
683        };
684        use crate::components::vc_operation_tree::OperationSecretTree;
685
686        if ciphersuite.signature_algorithm() != signer.signature_scheme() {
687            return Err(KeyPackageNewError::CiphersuiteSignatureSchemeMismatch);
688        }
689
690        // Reject an empty batch before loading state or consuming a generation.
691        if count == 0 {
692            return Err(KeyPackageNewError::EmptyBatch);
693        }
694
695        // Resolve and validate the leaf configuration once for the whole batch
696        // (it does not depend on the per-KeyPackage index) before consuming a
697        // generation, so a misconfigured leaf fails without burning one.
698        let resolved_dictionary =
699            crate::components::vc_derivation_info::resolve_vc_leaf_dictionary(
700                self.leaf_node_capabilities.as_ref(),
701                self.leaf_node_extensions.as_ref(),
702                None,
703            )?;
704
705        // Allocate a single generation of the key_package operation ratchet
706        // for the whole batch. The advanced tree is held in memory and only
707        // persisted once every KeyPackage has been built, so a failure during
708        // building consumes no generation.
709        let state: EmulationEpochState = provider
710            .storage()
711            .vc_emulation_epoch_state(&epoch_id)
712            .map_err(|e| {
713                log::error!("vc: load emulation epoch state in build_vc_batch failed: {e:?}");
714                VirtualClientsError::StorageError
715            })?
716            .ok_or(VirtualClientsError::MissingEmulationEpochState)?;
717        let mut operation_tree: OperationSecretTree = provider
718            .storage()
719            .vc_operation_tree(&epoch_id)
720            .map_err(|e| {
721                log::error!("vc: load operation tree in build_vc_batch failed: {e:?}");
722                VirtualClientsError::StorageError
723            })?
724            .ok_or(VirtualClientsError::MissingOperationTree)?;
725        let (emulation_leaf_index, epoch_encryption_key, emulation_ciphersuite) =
726            state.into_parts();
727
728        // The operation context for KeyPackage operations is empty, since a
729        // KeyPackage is not tied to a higher-level group at build time.
730        let (generation, operation_secret) = operation_tree.next_operation_secret(
731            provider.crypto(),
732            emulation_ciphersuite,
733            &epoch_id,
734            emulation_leaf_index,
735            VirtualClientOperationType::KeyPackage,
736            b"",
737        )?;
738
739        let batch_ctx = VcBatchContext {
740            ciphersuite,
741            epoch_id,
742            emulation_ciphersuite,
743            epoch_encryption_key,
744            emulation_leaf_index,
745            generation,
746            operation_secret,
747            resolved_dictionary,
748        };
749
750        let mut key_packages = Vec::with_capacity(count as usize);
751        for key_package_index in 0..count {
752            let entry = self.build_vc_key_package_for_index(
753                provider,
754                signer,
755                credential_with_key.clone(),
756                &batch_ctx,
757                key_package_index,
758            )?;
759            key_packages.push(entry);
760        }
761
762        // Persist the advanced operation tree before the KeyPackages it backs.
763        // If a KeyPackage write fails after this, the burned generation is
764        // harmless, but writing KeyPackages first would let the next batch
765        // reuse the same key material under an unconsumed generation.
766        provider
767            .storage()
768            .write_vc_operation_tree(&batch_ctx.epoch_id, &operation_tree)
769            .map_err(|e| {
770                log::error!("vc: persist advanced operation tree in build_vc_batch failed: {e:?}");
771                VirtualClientsError::StorageError
772            })?;
773        for (full_kp, info) in &key_packages {
774            provider
775                .storage()
776                .write_key_package(&info.key_package_ref, full_kp)
777                .map_err(|_| KeyPackageNewError::StorageError)?;
778        }
779
780        Ok(VcKeyPackageBatch {
781            generation,
782            key_packages,
783        })
784    }
785
786    /// Derive and build a single KeyPackage at `key_package_index` from the
787    /// batch's shared `operation_secret`. Used by [`Self::build_vc_batch`].
788    #[cfg(feature = "virtual-clients-draft")]
789    fn build_vc_key_package_for_index(
790        &self,
791        provider: &impl OpenMlsProvider,
792        signer: &impl Signer,
793        credential_with_key: CredentialWithKey,
794        batch_ctx: &VcBatchContext,
795        key_package_index: u32,
796    ) -> Result<
797        (
798            KeyPackageBundle,
799            crate::components::vc_derivation_info::KeyPackageInfo,
800        ),
801        KeyPackageNewError,
802    > {
803        use crate::components::vc_derivation_info::{
804            DerivationInfo, DerivationInfoTbe, KeyPackageInfo, VirtualClientsError,
805        };
806        use tls_codec::Serialize as _;
807
808        let ciphersuite = batch_ctx.ciphersuite;
809
810        // Derive the per-index seed under the emulation ciphersuite, then the
811        // init and leaf encryption keys from the seed under the KeyPackage's
812        // own ciphersuite.
813        let seed = batch_ctx.operation_secret.derive_key_package_seed_secret(
814            provider.crypto(),
815            batch_ctx.emulation_ciphersuite,
816            key_package_index,
817        )?;
818        let init_key_pair = seed
819            .derive_init_key_secret(provider.crypto(), ciphersuite)?
820            .generate_init_key_pair(provider.crypto(), ciphersuite)?;
821        let encryption_key_pair = seed
822            .derive_encryption_key_secret(provider.crypto(), ciphersuite)?
823            .generate_encryption_key_pair(provider.crypto(), ciphersuite)?;
824
825        // Wrap the TBE bound to the new leaf via its serialized encryption key.
826        // The leaf dictionary was resolved and validated once for the whole
827        // batch, so reuse a clone here.
828        let resolved_dictionary = batch_ctx.resolved_dictionary.clone();
829        let leaf_encryption_key = encryption_key_pair
830            .public_key()
831            .tls_serialize_detached()
832            .map_err(VirtualClientsError::from)?;
833        let tbe = DerivationInfoTbe::KeyPackage {
834            leaf_index: batch_ctx.emulation_leaf_index,
835            generation: batch_ctx.generation,
836            key_package_index,
837        };
838        let derivation_info = DerivationInfo::encrypt(
839            provider.crypto(),
840            batch_ctx.emulation_ciphersuite,
841            &batch_ctx.epoch_encryption_key,
842            batch_ctx.epoch_id.clone(),
843            &leaf_encryption_key,
844            &tbe,
845        )?;
846        let derivation_info_bytes = derivation_info
847            .tls_serialize_detached()
848            .map_err(VirtualClientsError::from)?;
849        let mut builder = self.clone();
850        builder.ensure_last_resort();
851        let leaf_node_extensions = crate::components::vc_derivation_info::merge_vc_derivation_info(
852            builder.leaf_node_extensions.as_ref(),
853            resolved_dictionary,
854            derivation_info_bytes,
855        )
856        .map_err(KeyPackageNewError::LibraryError)?;
857
858        let leaf_node_params = KeyPackageLeafNodeParams {
859            lifetime: builder.key_package_lifetime.unwrap_or_default(),
860            capabilities: builder.leaf_node_capabilities.unwrap_or_default(),
861            extensions: leaf_node_extensions,
862        };
863        let (key_package, encryption_key_pair) = KeyPackage::new_from_vc_keys(
864            ciphersuite,
865            signer,
866            credential_with_key,
867            builder.key_package_extensions.unwrap_or_default(),
868            leaf_node_params,
869            init_key_pair.public.into(),
870            encryption_key_pair,
871        )?;
872
873        let key_package_ref = key_package.hash_ref(provider.crypto())?;
874        let full_kp = KeyPackageBundle {
875            key_package,
876            private_init_key: init_key_pair.private,
877            private_encryption_key: encryption_key_pair.private_key().clone(),
878        };
879
880        Ok((
881            full_kp,
882            KeyPackageInfo {
883                key_package_ref,
884                key_package_index,
885            },
886        ))
887    }
888}

890/// Shared inputs for building each KeyPackage in a `build_vc_batch` call:
891/// the batch's single operation secret plus the epoch material every
892/// KeyPackage in the batch derives against.
893#[cfg(feature = "virtual-clients-draft")]
894struct VcBatchContext {
895    ciphersuite: Ciphersuite,
896    epoch_id: crate::components::vc_derivation_info::EpochId,
897    emulation_ciphersuite: Ciphersuite,
898    epoch_encryption_key: crate::components::vc_derivation_info::EpochEncryptionKey,
899    emulation_leaf_index: crate::binary_tree::LeafNodeIndex,
900    generation: u32,
901    operation_secret: crate::components::vc_derivation_info::OperationSecret,
902    resolved_dictionary: crate::extensions::AppDataDictionary,
903}

905/// The result of [`KeyPackageBuilder::build_vc_batch`]: the single
906/// `key_package` operation generation the batch consumed, and one
907/// `(KeyPackageBundle, KeyPackageInfo)` per KeyPackage built.
908#[cfg(feature = "virtual-clients-draft")]
909#[derive(Debug)]
910pub struct VcKeyPackageBatch {
911    /// The `key_package` operation generation consumed for the whole batch.
912    pub generation: u32,
913    /// One entry per KeyPackage built, in batch-index order. Never empty.
914    pub key_packages: Vec<(
915        KeyPackageBundle,
916        crate::components::vc_derivation_info::KeyPackageInfo,
917    )>,
918}

920/// A [`KeyPackageBundle`] contains a [`KeyPackage`] and the init and encryption
921/// private key.
922///
923/// This is stored to ensure the private key is handled together with the key
924/// package.
925#[derive(Debug, Clone, Serialize, Deserialize)]
926pub struct KeyPackageBundle {
927    pub(crate) key_package: KeyPackage,
928    pub(crate) private_init_key: HpkePrivateKey,
929    pub(crate) private_encryption_key: EncryptionPrivateKey,
930}
931
932// Public `KeyPackageBundle` functions.
933impl KeyPackageBundle {
934    /// Get a reference to the public part of this bundle, i.e. the [`KeyPackage`].
935    pub fn key_package(&self) -> &KeyPackage {
936        &self.key_package
937    }
938
939    /// Extract the key package from the bundle.
940    pub fn into_key_package(self) -> KeyPackage {
941        self.key_package
942    }
943
944    /// Get a reference to the private init key.
945    pub fn init_private_key(&self) -> &HpkePrivateKey {
946        &self.private_init_key
947    }
948
949    /// Get the encryption key pair.
950    pub(crate) fn encryption_key_pair(&self) -> EncryptionKeyPair {
951        EncryptionKeyPair::from((
952            self.key_package.leaf_node().encryption_key().clone(),
953            self.private_encryption_key.clone(),
954        ))
955    }
956}
