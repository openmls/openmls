1//! This module defines the [`MessageSecrets`] struct that can be used for message decryption & verification

3use super::*;

5/// Combined message secrets that need to be stored for later decryption/verification
6#[derive(Serialize, Deserialize)]
7#[cfg_attr(any(test, feature = "test-utils"), derive(Clone))]
8#[cfg_attr(feature = "crypto-debug", derive(Debug))]
9pub(crate) struct MessageSecrets {
10    sender_data_secret: SenderDataSecret,
11    membership_key: MembershipKey,
12    confirmation_key: ConfirmationKey,
13    serialized_context: Vec<u8>,
14    secret_tree: SecretTree,
15}
16
17#[cfg(not(feature = "crypto-debug"))]
18impl core::fmt::Debug for MessageSecrets {
19    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
20        f.debug_struct("MessageSecrets")
21            .field("sender_data_secret", &"***")
22            .field("membership_key", &"***")
23            .field("confirmation_key", &"***")
24            .field("serialized_context", &"***")
25            .field("secret_tree", &"***")
26            .finish()
27    }
28}

30// Public functions
31impl MessageSecrets {
32    ///Create new `MessageSecrets`
33    pub(crate) fn new(
34        sender_data_secret: SenderDataSecret,
35        membership_key: MembershipKey,
36        confirmation_key: ConfirmationKey,
37        serialized_context: Vec<u8>,
38        secret_tree: SecretTree,
39    ) -> Self {
40        Self {
41            sender_data_secret,
42            membership_key,
43            confirmation_key,
44            serialized_context,
45            secret_tree,
46        }
47    }
48
49    /// Get a reference to the message secrets's sender data secret.
50    pub(crate) fn sender_data_secret(&self) -> &SenderDataSecret {
51        &self.sender_data_secret
52    }
53
54    /// Get a reference to the message secrets's membership key.
55    pub(crate) fn membership_key(&self) -> &MembershipKey {
56        &self.membership_key
57    }
58
59    /// Get a reference to the message secrets's confirmation key.
60    pub(crate) fn confirmation_key(&self) -> &ConfirmationKey {
61        &self.confirmation_key
62    }
63
64    /// Get a reference to the message secrets's serialized context.
65    pub(crate) fn serialized_context(&self) -> &[u8] {
66        self.serialized_context.as_ref()
67    }
68
69    /// Get a mutable reference to the message secrets's secret tree.
70    pub(crate) fn secret_tree_mut(&mut self) -> &mut SecretTree {
71        &mut self.secret_tree
72    }
73}

75// Test functions
76impl MessageSecrets {
77    #[cfg(any(feature = "test-utils", test))]
78    pub(crate) fn sender_data_secret_mut(&mut self) -> &mut SenderDataSecret {
79        &mut self.sender_data_secret
80    }
81
82    #[cfg(test)]
83    /// Update the message secrets's serialized context.
84    pub(crate) fn set_serialized_context(&mut self, serialized_context: Vec<u8>) {
85        self.serialized_context = serialized_context;
86    }
87
88    #[cfg(test)]
89    /// Update the membership key.
90    pub(crate) fn set_membership_key(&mut self, membership_key: Secret) {
91        self.membership_key = MembershipKey::from_secret(membership_key);
92    }
93
94    #[cfg(test)]
95    pub(crate) fn random(
96        ciphersuite: Ciphersuite,
97        rng: &impl OpenMlsRand,
98        own_index: LeafNodeIndex,
99    ) -> Self {
100        Self {
101            sender_data_secret: SenderDataSecret::random(ciphersuite, rng),
102            membership_key: MembershipKey::random(ciphersuite, rng),
103            confirmation_key: ConfirmationKey::random(ciphersuite, rng),
104            serialized_context: rng.random_vec(10).expect("Not enough randomness."),
105            secret_tree: SecretTree::new(
106                EncryptionSecret::random(ciphersuite, rng),
107                TreeSize::new(10),
108                own_index,
109            ),
110        }
111    }
112
113    #[cfg(any(feature = "test-utils", test))]
114    pub(crate) fn replace_secret_tree(&mut self, secret_tree: SecretTree) -> SecretTree {
115        std::mem::replace(&mut self.secret_tree, secret_tree)
116    }
117}

119// In tests we allow comparing secrets.
120#[cfg(any(test, feature = "test-utils"))]
121impl PartialEq for MessageSecrets {
122    fn eq(&self, other: &Self) -> bool {
123        self.sender_data_secret == other.sender_data_secret
124            && self.membership_key == other.membership_key
125            && self.confirmation_key == other.confirmation_key
126            && self.secret_tree == other.secret_tree
127    }
