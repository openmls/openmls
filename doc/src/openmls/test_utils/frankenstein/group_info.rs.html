89pub struct FrankenGroupContext {
90    protocol_version: u16,
91    ciphersuite: u16,
92    group_id: VLBytes,
93    epoch: u64,
94    tree_hash: VLBytes,
95    confirmed_transcript_hash: VLBytes,
96    extensions: Vec<FrankenExtension>,

99impl From<GroupContext> for FrankenGroupContext {
100    fn from(value: GroupContext) -> Self {
101        let extensions = value
102            .extensions()
103            .iter()
104            .map(|ext| ext.clone().into())
105            .collect();
106        FrankenGroupContext {
107            protocol_version: match value.protocol_version() {
108                crate::versions::ProtocolVersion::Mls10 => 1,
109                crate::versions::ProtocolVersion::Other(other) => other,
110            },
111            ciphersuite: value.ciphersuite().into(),
112            group_id: value.group_id().as_slice().to_vec().into(),
113            epoch: value.epoch().as_u64(),
114            tree_hash: value.tree_hash().to_vec().into(),
115            confirmed_transcript_hash: value.confirmed_transcript_hash().to_vec().into(),
116            extensions,
117        }
118    }
119}
