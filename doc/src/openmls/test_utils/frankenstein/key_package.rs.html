29impl FrankenKeyPackage {
30    // Re-sign both the KeyPackage and the enclosed LeafNode
31    pub fn resign(&mut self, signer: &impl Signer) {
32        self.payload.leaf_node.resign(None, signer);
33        let new_self = self.payload.clone().sign(signer).unwrap();
34        let _ = std::mem::replace(self, new_self);
35    }
36
37    // Only re-sign the KeyPackage
38    pub fn resign_only_key_package(&mut self, signer: &impl Signer) {
39        let new_self = self.payload.clone().sign(signer).unwrap();
40        let _ = std::mem::replace(self, new_self);
41    }
42}

135#[openmls_test]
136fn test_franken_key_package() {
137    let provider = &Provider::default();
138
139    let config = ciphersuite;
140
141    let (credential, signer) = {
142        let credential = BasicCredential::new(b"test identity".to_vec());
143        let signature_keys = SignatureKeyPair::new(ciphersuite.signature_algorithm()).unwrap();
144        signature_keys.store(provider.storage()).unwrap();
145
146        (credential, signature_keys)
147    };
148    let signature_key = OpenMlsSignaturePublicKey::new(
149        signer.to_public_vec().into(),
150        ciphersuite.signature_algorithm(),
151    )
152    .unwrap();
153
154    let credential_with_key = CredentialWithKey {
155        credential: credential.into(),
156        signature_key: signature_key.into(),
157    };
158
159    let kp = KeyPackage::builder()
160        .build(config, provider, &signer, credential_with_key)
161        .unwrap();
162
163    let ser = kp.key_package().tls_serialize_detached().unwrap();
164    let fkp = FrankenKeyPackage::tls_deserialize(&mut ser.as_slice()).unwrap();
165
166    let ser2 = fkp.tls_serialize_detached().unwrap();
167    assert_eq!(ser, ser2);
168
169    let kp2 = KeyPackage::from(fkp.clone());
170    assert_eq!(kp.key_package(), &kp2);
171}
