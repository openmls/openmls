1//! This module provides the diff functionality for [`TreeSync`].
2//!
3//! # About
4//!
5//! This module provides the [`TreeSyncDiff`] struct, that allows mutable
6//! operations on otherwise immutable [`TreeSync`] instances. It also provides
7//! the [`StagedTreeSyncDiff`] struct, which has to be created from a
8//! [`TreeSyncDiff`] before it can be merged in to the original [`TreeSync`]
9//! instance.
10//!
11//!
12//! # Don't Panic!
13//!
14//! Functions in this module should never panic. However, if there is a bug in
15//! the implementation, a function will return an unrecoverable
16//! [`LibraryError`](TreeSyncDiffError::LibraryError). This means that some
17//! functions that are not expected to fail and throw an error, will still
18//! return a [`Result`] since they may throw a
19//! [`LibraryError`](TreeSyncDiffError::LibraryError).

82/// A [`TreeSyncDiff`] serves as a way to perform changes on an otherwise
83/// immutable [`TreeSync`] instance. Before the changes made to a
84/// [`TreeSyncDiff`] can be merged into the original [`TreeSync`] instance, it
85/// has to be turned into a [`StagedTreeSyncDiff`], upon which a number of
86/// checks are performed to ensure that the changes preseve the [`TreeSync`]
87/// invariants. See [`TreeSync`] for the list of invariants.
88pub(crate) struct TreeSyncDiff<'a> {
89    diff: MlsBinaryTreeDiff<'a, TreeSyncLeafNode, TreeSyncParentNode>,
90}
91
92impl<'a> From<&'a TreeSync> for TreeSyncDiff<'a> {
93    fn from(tree_sync: &'a TreeSync) -> Self {
94        TreeSyncDiff {
95            diff: tree_sync.tree.empty_diff(),
96        }
97    }
98}

101    /// Filtered direct path, skips the nodes whose copath resolution is empty.
102    pub(crate) fn filtered_direct_path(&self, leaf_index: LeafNodeIndex) -> Vec<ParentNodeIndex> {
103        // Full direct path
104        let direct_path = self.diff.direct_path(leaf_index);
105        // Copath resolutions
106        let copath_resolutions = self.copath_resolutions(leaf_index);
107
108        // The two vectors should have the same length
109        debug_assert_eq!(direct_path.len(), copath_resolutions.len());
110
111        direct_path
112            .into_iter()
113            .zip(copath_resolutions)
114            .filter_map(|(index, resolution)| {
115                // Filter out the nodes whose copath resolution is empty
116                if !resolution.is_empty() {
117                    Some(index)
118                } else {
119                    None
120                }
121            })
122            .collect()
123    }

125    /// Filtered copath, skips the nodes whose copath resolution is empty.
126    /// Matches the filtered direct path.
127    pub(crate) fn filtered_copath(&self, leaf_index: LeafNodeIndex) -> Vec<TreeNodeIndex> {
128        // Full copath
129        let copath = self.diff.copath(leaf_index);
130        // Copath resolutions
131        let copath_resolutions = self.copath_resolutions(leaf_index);
132
133        // The two vectors should have the same length
134        debug_assert_eq!(copath.len(), copath_resolutions.len());
135
136        copath
137            .into_iter()
138            .zip(copath_resolutions)
139            .filter_map(|(index, resolution)| {
140                // Filter out the nodes whose copath resolution is empty
141                if !resolution.is_empty() {
142                    Some(index)
143                } else {
144                    None
145                }
146            })
147            .collect()
148    }

150    /// Trims the tree by shrinking it until the last full leaf is in the
151    /// right part of the tree.
152    pub(crate) fn trim_tree(&mut self) {
153        // Nothing to trim if there's only one leaf left.
154        if self.leaf_count() == MIN_TREE_SIZE {
155            return;
156        }
157
158        let rightmost_full_leaf = self.rightmost_full_leaf();
159
160        // We shrink the tree until the last full leaf is the right part of the
161        // tree
162        while self.diff.size().leaf_is_left(rightmost_full_leaf) {
163            let res = self.diff.shrink_tree();
164            // We should never run into an error here, since `leaf_is_left`
165            // returns false when the tree only has one leaf.
166            debug_assert!(res.is_ok());
167        }
168    }
169
170    /// Returns the index of the last full leaf in the tree.
171    fn rightmost_full_leaf(&self) -> LeafNodeIndex {
172        let mut index = LeafNodeIndex::new(0);
173        for (leaf_index, leaf) in self.diff.leaves() {
174            if leaf.node().as_ref().is_some() {
175                index = leaf_index;
176            }
177        }
178        index
179    }

192    /// Updates an existing leaf node and blanks the nodes in the updated leaf's
193    /// direct path.
194    ///
195    /// Returns an error if the target leaf is blank or outside of the tree.
196    pub(crate) fn update_leaf(&mut self, leaf_node: LeafNode, leaf_index: LeafNodeIndex) {
197        self.diff.replace_leaf(leaf_index, leaf_node.into());
198        // This effectively wipes the tree hashes in the direct path.
199        self.diff
200            .set_direct_path_to_node(leaf_index, &TreeSyncParentNode::blank());
201    }

203    /// Find and return the index of either the left-most blank leaf, or, if
204    /// there are no blank leaves, the leaf count.
205    pub(crate) fn free_leaf_index(&self) -> LeafNodeIndex {
206        let mut leaf_count = 0;
207        // Search for blank leaves in existing leaves
208        for (leaf_index, leaf_id) in self.diff.leaves() {
209            if leaf_id.node().is_none() {
210                return leaf_index;
211            }
212            leaf_count += 1;
213        }
214
215        // Return the next free virtual blank leaf
216        LeafNodeIndex::new(leaf_count)
217    }

219    /// Adds a new leaf to the tree either by filling a blank leaf or by
220    /// extending the tree to the right to create a new leaf, inserting
221    /// intermediate blanks as necessary. This also adds the leaf_index of the
222    /// new leaf to the `unmerged_leaves` of the parent nodes in its direct
223    /// path.
224    ///
225    /// Returns the LeafNodeIndex of the new leaf.
226    pub(crate) fn add_leaf(
227        &mut self,
228        leaf_node: LeafNode,
229    ) -> Result<LeafNodeIndex, TreeSyncAddLeaf> {
230        // Find a free leaf and fill it with the new key package.
231        let leaf_index = self.free_leaf_index();
232        // If the free leaf index is within the tree, put the new leaf there,
233        // otherwise extend the tree first.
234        while leaf_index.u32() >= self.diff.size().leaf_count() {
235            self.diff
236                .grow_tree()
237                .map_err(|_| TreeSyncAddLeaf::TreeFull)?;
238        }
239        self.diff.replace_leaf(leaf_index, leaf_node.into());
240
241        // Add new unmerged leaves entry to all nodes in direct path. Also, wipe
242        // the cached tree hash.
243        for parent_index in self.diff.direct_path(leaf_index) {
244            // We know that the nodes from the direct path are in the tree
245            let tsn = self.diff.parent_mut(parent_index);
246            if let Some(ref mut parent_node) = tsn.node_mut() {
247                parent_node.add_unmerged_leaf(leaf_index);
248            }
249        }
250        Ok(leaf_index)
251    }

253    /// Remove a group member by blanking the target leaf and its direct path.
254    /// After blanking the leaf and its direct path, the diff is trimmed, i.e.
255    /// leaves are removed until the right-most leaf in the tree, as well as its
256    /// parent are non-blank.
257    ///
258    /// Returns an error if the target leaf is outside of the tree.
259    pub(crate) fn blank_leaf(&mut self, leaf_index: LeafNodeIndex) {
260        self.diff
261            .replace_leaf(leaf_index, TreeSyncLeafNode::blank());
262        // This also erases any cached tree hash in the direct path.
263        self.diff
264            .set_direct_path_to_node(leaf_index, &TreeSyncParentNode::blank());
265    }

267    /// Derive a new direct path for the leaf with the given index.
268    ///
269    /// Returns an error if the leaf is not in the tree
270    fn derive_path(
271        &self,
272        rand: &impl OpenMlsRand,
273        crypto: &impl OpenMlsCrypto,
274        ciphersuite: Ciphersuite,
275        leaf_index: LeafNodeIndex,
276    ) -> Result<PathDerivationResult, LibraryError> {
277        let path_secret = PathSecret::from(
278            Secret::random(ciphersuite, rand).map_err(LibraryError::unexpected_crypto_error)?,
279        );
280
281        let path_indices = self.filtered_direct_path(leaf_index);
282
283        ParentNode::derive_path(crypto, ciphersuite, path_secret, path_indices)
284    }

286    /// Given a new [`LeafNode`], use it to create a new path starting from
287    /// `leaf_index` and apply it to this diff. The given [`Signer`] reference
288    /// is used to sign the target [`LeafNode`] after updating its parent hash.
289    ///
290    /// Returns the [`CommitSecret`] and the path resulting from the path
291    /// derivation, as well as the newly derived [`EncryptionKeyPair`]s.
292    ///
293    /// Returns an error if the target leaf is not in the tree.
294    #[allow(clippy::too_many_arguments)]
295    pub(crate) fn apply_own_update_path(
296        &mut self,
297        rand: &impl OpenMlsRand,
298        crypto: &impl OpenMlsCrypto,
299        signer: &impl Signer,
300        ciphersuite: Ciphersuite,
301        commit_type: &CommitType,
302        group_id: GroupId,
303        leaf_index: LeafNodeIndex,
304        leaf_node_params: UpdateLeafNodeParams,
305    ) -> Result<UpdatePathResult, TreeSyncAddLeaf> {
306        // For External Commits, we temporarily add a placeholder leaf node to the tree, because it
307        // might be required to make the tree grow to the right size. If we
308        // don't do that, calculating the direct path might fail. It's important
309        // to not do anything with the value of that leaf until it has been
310        // replaced.
311        if let CommitType::External(_) = commit_type {
312            let leaf_node = LeafNode::new_placeholder();
313            self.add_leaf(leaf_node)?;
314        }
315
316        // We calculate the parent hash so that we can use it for a fresh leaf
317        let (path, update_path_nodes, parent_keypairs, commit_secret) =
318            self.derive_path(rand, crypto, ciphersuite, leaf_index)?;
319        let parent_hash = self.process_update_path(crypto, ciphersuite, leaf_index, path)?;
320
321        // We generate the new leaf with all parameters
322        let (leaf_node, node_keypair) = LeafNode::new_with_parent_hash(
323            rand,
324            crypto,
325            ciphersuite,
326            &parent_hash,
327            leaf_node_params,
328            group_id,
329            leaf_index,
330            signer,
331        )?;
332
333        // We insert the fresh leaf into the tree.
334        self.diff.replace_leaf(leaf_index, leaf_node.into());
335
336        // Prepend parent keypairs with node keypair
337        let mut keypairs = vec![node_keypair];
338        keypairs.extend(parent_keypairs);
339
340        Ok((update_path_nodes, keypairs, commit_secret))
341    }

343    /// Set the given path as the direct path of the `sender_leaf_index` and
344    /// replace the [`LeafNode`] in the corresponding leaf with the given one.
345    ///
346    /// Returns an error if the `sender_leaf_index` is outside of the tree.
347    /// ValSem202: Path must be the right length
348    /// TODO #804
349    pub(crate) fn apply_received_update_path(
350        &mut self,
351        crypto: &impl OpenMlsCrypto,
352        ciphersuite: Ciphersuite,
353        sender_leaf_index: LeafNodeIndex,
354        update_path: &UpdatePath,
355    ) -> Result<(), ApplyUpdatePathError> {
356        let path = update_path.nodes();
357
358        // If the committer is a `NewMemberCommit`, we have to add the
359        // leaf to the tree before we can apply an update path. This is
360        // s.t. the tree can be grown if necessary.
361        if self.diff.leaf(sender_leaf_index).node().is_none()
362            || sender_leaf_index.u32() >= self.leaf_count()
363        {
364            let new_leaf_index =
365                self.add_leaf(update_path.leaf_node().clone())
366                    .map_err(|e| match e {
367                        TreeSyncAddLeaf::LibraryError(e) => ApplyUpdatePathError::LibraryError(e),
368                        TreeSyncAddLeaf::TreeFull => ApplyUpdatePathError::TreeFull,
369                    })?;
370            // The new member should have the same index as the claimed sender index.
371            if sender_leaf_index != new_leaf_index {
372                return Err(ApplyUpdatePathError::InconsistentSenderIndex);
373            }
374        }
375
376        // ValSem202: Path must be the right length
377        // https://validation.openmls.tech/#valn1101
378        let filtered_direct_path = self.filtered_direct_path(sender_leaf_index);
379        if filtered_direct_path.len() != path.len() {
380            return Err(ApplyUpdatePathError::PathLengthMismatch);
381        };
382
383        let path = filtered_direct_path
384            .into_iter()
385            .zip(
386                path.iter()
387                    .map(|update_path_node| update_path_node.public_key.clone().into()),
388            )
389            .collect();
390
391        // Verify the parent hash.
392        let parent_hash = self.process_update_path(crypto, ciphersuite, sender_leaf_index, path)?;
393        let leaf_node_parent_hash = update_path
394            .leaf_node()
395            .parent_hash()
396            .ok_or(ApplyUpdatePathError::MissingParentHash)?;
397        if leaf_node_parent_hash != parent_hash {
398            return Err(ApplyUpdatePathError::ParentHashMismatch);
399        };
400
401        // Update the `encryption_key` in the leaf.
402        let leaf: LeafNode = update_path.leaf_node().clone();
403        self.diff.replace_leaf(sender_leaf_index, leaf.into());
404        Ok(())
405    }

407    /// Process a given update path, consisting of a vector of `ParentNode`.
408    /// This function replaces the nodes in the direct path of the given
409    /// `leaf_index` with the the ones in `path`.
410    ///
411    /// Returns the parent hash of the leaf at `leaf_index`. Returns an error if
412    /// the target leaf is outside of the tree.
413    /// TODO #804
414    fn process_update_path(
415        &mut self,
416        crypto: &impl OpenMlsCrypto,
417        ciphersuite: Ciphersuite,
418        leaf_index: LeafNodeIndex,
419        mut path: Vec<(ParentNodeIndex, ParentNode)>,
420    ) -> Result<Vec<u8>, LibraryError> {
421        // Compute the parent hash.
422        let parent_hash = self.set_parent_hashes(crypto, ciphersuite, &mut path, leaf_index)?;
423
424        // While probably not necessary, the spec mandates we blank the direct path nodes
425        let direct_path_nodes = self.diff.direct_path(leaf_index);
426        for node in direct_path_nodes {
427            *self.diff.parent_mut(node) = TreeSyncParentNode::blank();
428        }
429
430        // Set the node of the filtered direct path.
431        // Note, that the nodes here don't have a tree hash set.
432        // TODO #804
433        for (index, node) in path.into_iter() {
434            *self.diff.parent_mut(index) = node.into();
435        }
436
437        Ok(parent_hash)
438    }

440    /// Set the parent hash of the given nodes assuming that they are the new
441    /// filtered direct path of the leaf with the given index.
442    ///
443    /// Returns the parent hash of the leaf node at `leaf_index` and a
444    /// `LibraryError` if an unexpected error occurs.
445    fn set_parent_hashes(
446        &mut self,
447        crypto: &impl OpenMlsCrypto,
448        ciphersuite: Ciphersuite,
449        path: &mut [(ParentNodeIndex, ParentNode)],
450        leaf_index: LeafNodeIndex,
451    ) -> Result<Vec<u8>, LibraryError> {
452        // If the path is empty, return a zero-length string. This is the case
453        // when the tree has only one leaf.
454        if path.is_empty() {
455            return Ok(Vec::new());
456        }
457
458        // Compute the filtered copath corresponding to the given filtered
459        // direct path.
460        let filtered_copath = self.filtered_copath(leaf_index);
461        debug_assert_eq!(filtered_copath.len(), path.len());
462
463        // For every copath node, compute the tree hash.
464        let copath_tree_hashes = filtered_copath.into_iter().map(|node_index| {
465            self.compute_tree_hash(crypto, ciphersuite, node_index, &HashSet::new())
466        });
467
468        // We go through the nodes in the direct path in reverse order and get
469        // the corresponding tree hash for each copath node.
470        let mut previous_parent_hash = vec![];
471        for ((_, path_node), original_sibling_tree_hash) in
472            path.iter_mut().rev().zip(copath_tree_hashes.rev())
473        {
474            path_node.set_parent_hash(previous_parent_hash);
475            let parent_hash =
476                path_node.compute_parent_hash(crypto, ciphersuite, &original_sibling_tree_hash?)?;
477            previous_parent_hash = parent_hash
478        }
479        // The final hash is the one of the leaf's parent.
480        Ok(previous_parent_hash)
481    }

483    /// Helper function computing the resolution of a node with the given index.
484    /// If an exclusion list is given, do not add the leaves given in the list.
485    pub(super) fn resolution(
486        &self,
487        node_index: TreeNodeIndex,
488        excluded_indices: &HashSet<&LeafNodeIndex>,
489    ) -> Vec<(TreeNodeIndex, NodeReference<'_>)> {
490        match node_index {
491            TreeNodeIndex::Leaf(leaf_index) => {
492                // If the node is a leaf, check if it is in the exclusion list.
493                if excluded_indices.contains(&leaf_index) {
494                    vec![]
495                } else {
496                    // If it's not, return it as its resolution.
497                    if let Some(leaf) = self.diff.leaf(leaf_index).node() {
498                        vec![(TreeNodeIndex::Leaf(leaf_index), NodeReference::Leaf(leaf))]
499                    } else {
500                        // If it's a blank, return an empty vector.
501                        vec![]
502                    }
503                }
504            }
505            TreeNodeIndex::Parent(parent_index) => {
506                match self.diff.parent(parent_index).node() {
507                    Some(parent) => {
508                        // If it's a non-blank parent node, get the unmerged
509                        // leaves, exclude them as necessary and add the node to
510                        // the resulting resolution.
511                        let mut resolution = vec![(
512                            TreeNodeIndex::Parent(parent_index),
513                            NodeReference::Parent(parent),
514                        )];
515                        for leaf_index in parent.unmerged_leaves() {
516                            if !excluded_indices.contains(&leaf_index) {
517                                let leaf = self.diff.leaf(*leaf_index);
518                                if let Some(leaf_node) = leaf.node() {
519                                    resolution.push((
520                                        TreeNodeIndex::Leaf(*leaf_index),
521                                        NodeReference::Leaf(leaf_node),
522                                    ))
523                                } else {
524                                    debug_assert!(false, "Unmerged leaves should not be blank.");
525                                }
526                            }
527                        }
528                        resolution
529                    }
530                    None => {
531                        // If it is a blank parent node, continue resolving
532                        // down the tree.
533                        let mut resolution = Vec::new();
534                        let left_child = self.diff.left_child(parent_index);
535                        let right_child = self.diff.right_child(parent_index);
536                        resolution.append(&mut self.resolution(left_child, excluded_indices));
537                        resolution.append(&mut self.resolution(right_child, excluded_indices));
538                        resolution
539                    }
540                }
541            }
542        }
543    }

545    /// Compute the resolution of the copath of the leaf node corresponding to
546    /// the given leaf index. This includes the neighbour of the given leaf. If
547    /// an exclusion list is given, do not add the public keys of the leaves
548    /// given in the list.
549    ///
550    /// Returns a vector containing the copath resolutions of the given
551    /// `leaf_index` beginning with the neighbour of the leaf. Returns an error
552    /// if the target leaf is outside of the tree.
553    pub(crate) fn copath_resolutions(
554        &self,
555        leaf_index: LeafNodeIndex,
556    ) -> Vec<Vec<(TreeNodeIndex, NodeReference<'_>)>> {
557        // If we're the only node in the tree, there's no copath.
558        if self.diff.leaf_count() == MIN_TREE_SIZE {
559            return vec![];
560        }
561
562        // Get the copath of the given leaf index and compute the resolution of
563        // each node.
564        self.diff
565            .copath(leaf_index)
566            .into_iter()
567            .map(|node_index| self.resolution(node_index, &HashSet::new()))
568            .collect()
569    }

571    /// Compute the copath resolutions, but leave out empty resolutions.
572    /// Additionally, resolutions are filtered by the given exclusion list.
573    pub(super) fn filtered_copath_resolutions(
574        &self,
575        leaf_index: LeafNodeIndex,
576        exclusion_list: &HashSet<&LeafNodeIndex>,
577    ) -> Vec<Vec<(TreeNodeIndex, NodeReference<'_>)>> {
578        // If we're the only node in the tree, there's no copath.
579        if self.diff.leaf_count() == 1 {
580            return vec![];
581        }
582
583        let mut copath_resolutions = Vec::new();
584        for node_index in self.diff.copath(leaf_index) {
585            let resolution = self.resolution(node_index, &HashSet::new());
586            if !resolution.is_empty() {
587                let filtered_resolution = resolution
588                    .into_iter()
589                    .filter_map(|(index, node)| {
590                        if let TreeNodeIndex::Leaf(leaf_index) = index {
591                            if exclusion_list.contains(&leaf_index) {
592                                None
593                            } else {
594                                Some((TreeNodeIndex::Leaf(leaf_index), node))
595                            }
596                        } else {
597                            Some((index, node))
598                        }
599                    })
600                    .collect();
601                copath_resolutions.push(filtered_resolution);
602            }
603        }
604        copath_resolutions
605    }

607    /// Verify the parent hashes of all parent nodes in the tree.
608    ///
609    /// Returns an error if one of the parent nodes in the tree has an invalid
610    /// parent hash.
611    ///
612    /// Checks [valn1406](https://validation.openmls.tech/#valn1406).
613    pub(super) fn verify_parent_hashes(
614        &self,
615        crypto: &impl OpenMlsCrypto,
616        ciphersuite: Ciphersuite,
617    ) -> Result<(), TreeSyncParentHashError> {
618        // We traverse the tree by iterating over all parent nodes. For each
619        // parent node, we compute the parent hash and compare it to the parent
620        // hash in its descendants.
621        //
622        // We need to consider all nodes in the the descendants list (the
623        // resolution), since we cannot determine which descedant was the
624        // previous one in the update path when the original commit occurred.
625        //
626        // We look at both righ & left subtree of a give node separately and
627        // compute the original sibling tree hash for the sibling of the subtree
628        // root.
629        //
630        // The parent hash for a given node is valid, when exactly one descendant
631        // carries the parent hash it its parent hash field.
632        for (parent_index, tree_sync_parent_node) in self.diff.parents() {
633            if let Some(parent_node) = tree_sync_parent_node.node() {
634                // We consider both children of the parent node. One of them
635                // takes the role of the descendant, whose resolution carries
636                // the parent hash. The other one is the descendants sibling,
637                // whose original tree hash is used to compute the parent hash.
638                let left_child = self.diff.left_child(parent_index);
639                let right_child = self.diff.right_child(parent_index);
640
641                // We exclude the unmerged leaves from the parent node for the
642                // following computations. Those leaves were obviously added
643                // after the parent node was populated during a commit and must
644                // therefore be removed to recreate the tree state at the time
645                // of the commit.
646                let exclusion_list = HashSet::from_iter(parent_node.unmerged_leaves().iter());
647
648                // Compute the original tree hash (oth) for the left and right child.
649                let oth_left =
650                    self.compute_tree_hash(crypto, ciphersuite, left_child, &exclusion_list)?;
651
652                let oth_right =
653                    self.compute_tree_hash(crypto, ciphersuite, right_child, &exclusion_list)?;
654
655                // Compute the parent hash for both child roles.
656                let parent_hash_left =
657                    parent_node.compute_parent_hash(crypto, ciphersuite, &oth_right)?;
658
659                let parent_hash_right =
660                    parent_node.compute_parent_hash(crypto, ciphersuite, &oth_left)?;
661
662                // Compute the resolution for both children.
663                let left_resolution = self.resolution(left_child, &exclusion_list);
664
665                let right_resolution = self.resolution(right_child, &exclusion_list);
666
667                // Find parent hash in the left resolution.
668                let left_descendant = left_resolution.iter().find(|(_, node)| match node {
669                    NodeReference::Leaf(leaf) => leaf
670                        .parent_hash()
671                        .map(|parent_hash| parent_hash == parent_hash_left)
672                        .unwrap_or(false),
673                    NodeReference::Parent(parent) => parent.parent_hash() == parent_hash_left,
674                });
675
676                // Find parent hash in the right resolution.
677                let right_descendant = right_resolution.iter().find(|(_, node)| match node {
678                    NodeReference::Leaf(leaf) => leaf
679                        .parent_hash()
680                        .map(|parent_hash| parent_hash == parent_hash_right)
681                        .unwrap_or(false),
682                    NodeReference::Parent(parent) => parent.parent_hash() == parent_hash_right,
683                });
684
685                // If one of the parent hashes is in the resolution of the
686                // other child, the parent hash is valid.
687                if left_descendant.is_none() ^ right_descendant.is_some() {
688                    return Err(TreeSyncParentHashError::InvalidParentHash);
689                }
690            }
691        }
692        Ok(())
693    }

696    /// computes and sets the new tree hash.
697    pub(crate) fn into_staged_diff(
698        self,
699        crypto: &impl OpenMlsCrypto,
700        ciphersuite: Ciphersuite,
701    ) -> Result<StagedTreeSyncDiff, LibraryError> {
702        let new_tree_hash = self.compute_tree_hashes(crypto, ciphersuite)?;
703        debug_assert!(self.verify_parent_hashes(crypto, ciphersuite).is_ok());

710    /// Helper function to compute and set the tree hash of the given node and
711    /// all nodes below it in the tree. The leaf nodes in `exclusion_list` are
712    /// not included in the tree hash.
713    pub(crate) fn compute_tree_hash(
714        &self,
715        crypto: &impl OpenMlsCrypto,
716        ciphersuite: Ciphersuite,
717        node_index: TreeNodeIndex,
718        exclusion_list: &HashSet<&LeafNodeIndex>,
719    ) -> Result<Vec<u8>, LibraryError> {
720        match node_index {
721            TreeNodeIndex::Leaf(leaf_index) => {
722                let leaf = self.diff.leaf(leaf_index);
723
724                if exclusion_list.contains(&leaf_index) {
725                    TreeSyncLeafNode::blank().compute_tree_hash(crypto, ciphersuite, leaf_index)
726                } else {
727                    leaf.compute_tree_hash(crypto, ciphersuite, leaf_index)
728                }
729            }
730            TreeNodeIndex::Parent(parent_index) => {
731                // Compute left hash.
732                let left_child = self.diff.left_child(parent_index);
733                let left_hash =
734                    self.compute_tree_hash(crypto, ciphersuite, left_child, exclusion_list)?;
735                // Compute right hash.
736                let right_child = self.diff.right_child(parent_index);
737                let right_hash =
738                    self.compute_tree_hash(crypto, ciphersuite, right_child, exclusion_list)?;
739
740                let node = self.diff.parent(parent_index);
741
742                node.compute_tree_hash(crypto, ciphersuite, left_hash, right_hash, exclusion_list)
743            }
744        }
745    }

761    /// Returns the position of the subtree root shared by both given indices in
762    /// the direct path of `leaf_index_1`.
763    ///
764    /// Returns a [LibraryError] if there's an error in the tree math computation.
765    pub(super) fn subtree_root_position(
766        &self,
767        leaf_index_1: LeafNodeIndex,
768        leaf_index_2: LeafNodeIndex,
769    ) -> Result<usize, TreeSyncDiffError> {
770        let subtree_root_node_index = self.diff.lowest_common_ancestor(leaf_index_1, leaf_index_2);
771        let leaf_index_1_direct_path = self.filtered_direct_path(leaf_index_1);
772
773        leaf_index_1_direct_path
774            .iter()
775            .position(|&direct_path_node_index| direct_path_node_index == subtree_root_node_index)
776            // The shared subtree root has to be in the direct path of both nodes.
777            .ok_or_else(|| LibraryError::custom("index should be in the direct path").into())
778    }

780    /// Compute the position of the highest node in the tree in the filtered
781    /// copath resolution of the given `sender_leaf_index` where a corresponding
782    /// [`EncryptionKeyPair`] can be found.
783    ///
784    /// Returns the resulting position, as well as the private key of that node.
785    /// Returns an error if the given `sender_leaf_index` is outside of the
786    /// tree.
787    pub(crate) fn decryption_key<'private_key>(
788        &self,
789        sender_leaf_index: LeafNodeIndex,
790        excluded_indices: &HashSet<&LeafNodeIndex>,
791        owned_keys: &'private_key [&EncryptionKeyPair],
792        leaf_index: LeafNodeIndex,
793    ) -> Result<(&'private_key EncryptionPrivateKey, usize), TreeSyncDiffError> {
794        // Get the copath node of the sender that is in our direct path, as well
795        // as its position in our direct path.
796        let subtree_root_copath_node_id = self
797            .diff
798            .subtree_root_copath_node(sender_leaf_index, leaf_index);
799
800        let sender_copath_resolution: Vec<EncryptionKey> = self
801            .resolution(subtree_root_copath_node_id, excluded_indices)
802            .into_iter()
803            .map(|(_, node_ref)| match node_ref {
804                NodeReference::Leaf(leaf) => leaf.encryption_key().clone(),
805                NodeReference::Parent(parent) => parent.encryption_key().clone(),
806            })
807            .collect();
808
809        if let Some((keypair, resolution_position)) = sender_copath_resolution
810            .iter()
811            .enumerate()
812            .find_map(|(position, pk)| {
813                owned_keys
814                    .iter()
815                    .find(|&owned_keypair| owned_keypair.public_key() == pk)
816                    .map(|keypair| (keypair, position))
817            })
818        {
819            debug!("Found fitting keypair in the filtered resolution:");
820            debug!("* private key: {:x?}", keypair.private_key());
821            debug!("* public key: {:x?}", keypair.public_key());
822
823            return Ok((keypair.private_key(), resolution_position));
824        };
825        Err(TreeSyncDiffError::NoPrivateKeyFound)
826    }

828    /// Returns a vector of all nodes in the tree resulting from merging this
829    /// diff.
830    pub(crate) fn export_ratchet_tree(&self) -> RatchetTree {
831        let mut nodes = Vec::new();
832
833        // Determine the index of the rightmost full leaf.
834        let max_length = self.rightmost_full_leaf();
835
836        // We take all the leaves including the rightmost full leaf, blank
837        // leaves beyond that are trimmed.
838        let mut leaves = self
839            .diff
840            .leaves()
841            .map(|(_, leaf)| leaf)
842            .take(max_length.usize() + 1);
843
844        // Get the first leaf.
845        if let Some(leaf) = leaves.next() {
846            nodes.push(leaf.node().clone().map(Node::leaf_node));
847        } else {
848            // The tree was empty.
849            return RatchetTree::trimmed(vec![]);
850        }
851
852        // Blank parent node used for padding
853        let default_parent = TreeSyncParentNode::default();
854
855        // Get the parents.
856        let parents = self
857            .diff
858            .parents()
859            // Drop the index
860            .map(|(_, parent)| parent)
861            // Take the parents up to the max length
862            .take(max_length.usize())
863            // Pad the parents with blank nodes if needed
864            .chain(
865                (self.diff.parents().count()..self.diff.leaves().count() - 1)
866                    .map(|_| &default_parent),
867            );
868
869        // Interleave the leaves and parents.
870        for (leaf, parent) in leaves.zip(parents) {
871            nodes.push(parent.node().clone().map(Node::parent_node));
872            nodes.push(leaf.node().clone().map(Node::leaf_node));
873        }
874
875        RatchetTree::trimmed(nodes)
876    }

878    /// Returns the filtered common path two leaf nodes share. If the leaves are
879    /// identical, the common path is the leaf's direct path.
880    pub(super) fn filtered_common_direct_path(
881        &self,
882        leaf_index_1: LeafNodeIndex,
883        leaf_index_2: LeafNodeIndex,
884    ) -> Vec<ParentNodeIndex> {
885        let mut x_path = self.filtered_direct_path(leaf_index_1);
886        let mut y_path = self.filtered_direct_path(leaf_index_2);
887        x_path.reverse();
888        y_path.reverse();
889
890        let mut common_path = vec![];
891
892        for (x, y) in x_path.iter().zip(y_path.iter()) {
893            if x == y {
894                common_path.push(*x);
895            } else {
896                break;
897            }
898        }
899
900        common_path.reverse();
901        common_path
902    }

904    /// Return an iterator over references to all [`EncryptionKey`]s for which
905    /// the owner of the given `leaf_index` should have private key material.
906    pub(crate) fn encryption_keys(
907        &self,
908        leaf_index: LeafNodeIndex,
909    ) -> impl Iterator<Item = &EncryptionKey> {
910        let encryption_keys = if let Some(leaf) = self.diff.leaf(leaf_index).node() {
911            vec![leaf.encryption_key()]
912        } else {
913            // If our own leaf is empty, we don't expect to own any other keys.
914            vec![]
915        };
916        encryption_keys.into_iter().chain(
917            self.filtered_direct_path(leaf_index)
918                .into_iter()
919                .filter_map(move |parent_index| {
920                    // Filter out all blanks.
921                    if let Some(node) = self.diff.parent(parent_index).node().as_ref() {
922                        // Filter all nodes where our leaf is an unmerged leaf.
923                        if !node.unmerged_leaves().contains(&leaf_index) {
924                            Some(node.encryption_key())
925                        } else {
926                            None
927                        }
928                    } else {
929                        None
930                    }
931                }),
932        )
933    }
