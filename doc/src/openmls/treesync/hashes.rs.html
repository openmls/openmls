13/// Helper struct that can be serialized in the course of parent hash
14/// computation.
15///
16/// ```c
17/// // draft-ietf-mls-protocol-17
18/// struct {
19/// HPKEPublicKey encryption_key;
20///     opaque parent_hash<V>;
21///     opaque original_sibling_tree_hash<V>;
22/// } ParentHashInput;
23/// ```
24#[derive(TlsSerialize, TlsSize)]
25pub(super) struct ParentHashInput<'a> {
26    public_key: &'a HpkePublicKey,
27    parent_hash: VLByteSlice<'a>,
28    original_sibling_tree_hash: VLByteSlice<'a>,
29}
30
31impl<'a> ParentHashInput<'a> {
32    /// Create a new [`ParentHashInput`] instance.
33    pub(super) fn new(
34        public_key: &'a HpkePublicKey,
35        parent_hash: &'a [u8],
36        original_sibling_tree_hash: &'a [u8],
37    ) -> Self {
38        Self {
39            public_key,
40            parent_hash: VLByteSlice(parent_hash),
41            original_sibling_tree_hash: VLByteSlice(original_sibling_tree_hash),
42        }
43    }
44
45    /// Serialize and hash this instance of [`ParentHashInput`].
46    pub(super) fn hash(
47        &self,
48        crypto: &impl OpenMlsCrypto,
49        ciphersuite: Ciphersuite,
50    ) -> Result<Vec<u8>, LibraryError> {
51        let payload = self
52            .tls_serialize_detached()
53            .map_err(LibraryError::missing_bound_check)?;
54        crypto
55            .hash(ciphersuite.hash_algorithm(), &payload)
56            .map_err(LibraryError::unexpected_crypto_error)
57    }
58}

60/// Helper struct that can be serialized in the course of tree hash computation.
61///
62/// ```c
63/// // draft-ietf-mls-protocol-16
64/// enum {
65///     reserved(0),
66///     leaf(1),
67///     parent(2),
68///     (255)
69/// } NodeType;
70/// ```
71#[derive(TlsSerialize, TlsSize)]
72#[repr(u8)]
73enum NodeType<'a> {
74    #[tls_codec(discriminant = 1)]
75    Leaf(LeafNodeHashInput<'a>),
76    #[tls_codec(discriminant = 2)]
77    Parent(ParentNodeHashInput<'a>),
78}
79
80/// Helper struct that can be serialized in the course of tree hash computation.
81///
82/// ```c
83/// // draft-ietf-mls-protocol-16
84/// struct {
85///   NodeType node_type;
86/// select (TreeHashInput.node_type) {
87///     case leaf:   LeafNodeHashInput leaf_node;
88///     case parent: ParentNodeHashInput parent_node;
89///   }
90/// } TreeHashInput;
91/// ```
92#[derive(TlsSerialize, TlsSize)]
93pub(super) struct TreeHashInput<'a> {
94    node_type: NodeType<'a>,
95}
96
97impl<'a> TreeHashInput<'a> {
98    /// Create a new [`TreeHashInput`] instance from a leaf node.
99    pub(super) fn new_leaf(leaf_index: &'a LeafNodeIndex, leaf_node: Option<&'a LeafNode>) -> Self {
100        Self {
101            node_type: NodeType::Leaf(LeafNodeHashInput {
102                leaf_index,
103                leaf_node,
104            }),
105        }
106    }
107
108    /// Create a new [`TreeHashInput`] instance from a parent node.
109    pub(super) fn new_parent(
110        parent_node: Option<&'a ParentNode>,
111        left_hash: VLByteSlice<'a>,
112        right_hash: VLByteSlice<'a>,
113    ) -> Self {
114        Self {
115            node_type: NodeType::Parent(ParentNodeHashInput {
116                parent_node,
117                left_hash,
118                right_hash,
119            }),
120        }
121    }
122
123    /// Serialize and hash this instance of [`TreeHashInput`].
124    pub(super) fn hash(
125        &self,
126        crypto: &impl OpenMlsCrypto,
127        ciphersuite: Ciphersuite,
128    ) -> Result<Vec<u8>, LibraryError> {
129        let payload = self
130            .tls_serialize_detached()
131            .map_err(LibraryError::missing_bound_check)?;
132        crypto
133            .hash(ciphersuite.hash_algorithm(), &payload)
134            .map_err(LibraryError::unexpected_crypto_error)
135    }
136}
