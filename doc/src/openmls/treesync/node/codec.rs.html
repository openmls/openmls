20impl Deserialize for UnmergedLeaves {
21    fn tls_deserialize<R: std::io::Read>(bytes: &mut R) -> Result<Self, Error>
22    where
23        Self: Sized,
24    {
25        let list = Vec::tls_deserialize(bytes)?;
26        Self::try_from(list).map_err(|e| match e {
27            UnmergedLeavesError::NotSorted => {
28                Error::DecodingError("Unmerged leaves not sorted".into())
29            }
30        })
31    }
32}
33
34impl DeserializeBytes for UnmergedLeaves {
35    fn tls_deserialize_bytes(bytes: &[u8]) -> Result<(Self, &[u8]), Error>
36    where
37        Self: Sized,
38    {
39        let mut bytes_ref = bytes;
40        let unmerged_leaves = UnmergedLeaves::tls_deserialize(&mut bytes_ref)?;
41        let remainder = &bytes[unmerged_leaves.tls_serialized_len()..];
42        Ok((unmerged_leaves, remainder))
43    }
44}
