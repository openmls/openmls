140/// This struct implements the MLS leaf node.
141///
142/// ```c
143/// // draft-ietf-mls-protocol-17
144/// struct {
145///     HPKEPublicKey encryption_key;
146///     SignaturePublicKey signature_key;
147///     Credential credential;
148///     Capabilities capabilities;
149///
150///     LeafNodeSource leaf_node_source;
151///     select (LeafNode.leaf_node_source) {
152///         case key_package:
153///             Lifetime lifetime;
154///
155///         case update:
156///             struct{};
157///
158///         case commit:
159///             opaque parent_hash<V>;
160///     };
161///
162///     Extension extensions<V>;
163///     /* SignWithLabel(., "LeafNodeTBS", LeafNodeTBS) */
164///     opaque signature<V>;
165/// } LeafNode;
166/// ```
167// TODO(#1242): Do not derive `TlsDeserialize`.
168#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, TlsSerialize, TlsSize)]
169pub struct LeafNode {
170    payload: LeafNodePayload,
171    signature: Signature,
172}

174impl LeafNode {
175    /// Create a new [`LeafNode`].
176    /// This first creates a `LeadNodeTbs` and returns the result of signing
177    /// it.
178    ///
179    /// This function generates a fresh HPKE key pair for the leaf node and
180    /// returns the HPKE key pair along with the new leaf node.
181    /// The caller is responsible for storing the private key.
182    pub(crate) fn new(
183        provider: &impl OpenMlsProvider,
184        signer: &impl Signer,
185        new_leaf_node_params: NewLeafNodeParams,
186    ) -> Result<(Self, EncryptionKeyPair), LibraryError> {
187        let NewLeafNodeParams {
188            ciphersuite,
189            credential_with_key,
190            leaf_node_source,
191            capabilities,
192            extensions,
193            tree_info_tbs,
194        } = new_leaf_node_params;
195
196        // Create a new encryption key pair.
197        let encryption_key_pair =
198            EncryptionKeyPair::random(provider.rand(), provider.crypto(), ciphersuite)?;
199
200        let leaf_node = Self::new_with_key(
201            encryption_key_pair.public_key().clone(),
202            credential_with_key,
203            leaf_node_source,
204            capabilities,
205            extensions,
206            tree_info_tbs,
207            signer,
208        )?;
209
210        Ok((leaf_node, encryption_key_pair))
211    }

259    /// New [`LeafNode`] with a parent hash.

261    pub(in crate::treesync) fn new_with_parent_hash(
262        rand: &impl OpenMlsRand,
263        crypto: &impl OpenMlsCrypto,
264        ciphersuite: Ciphersuite,
265        parent_hash: &[u8],
266        leaf_node_params: UpdateLeafNodeParams,
267        group_id: GroupId,
268        leaf_index: LeafNodeIndex,
269        signer: &impl Signer,
270    ) -> Result<(Self, EncryptionKeyPair), LibraryError> {
271        let encryption_key_pair = EncryptionKeyPair::random(rand, crypto, ciphersuite)?;
272
273        let leaf_node_tbs = LeafNodeTbs::new(
274            encryption_key_pair.public_key().clone(),
275            leaf_node_params.credential_with_key,
276            leaf_node_params.capabilities,
277            LeafNodeSource::Commit(parent_hash.into()),

293    /// Generate a fresh leaf node.
294    ///
295    /// This includes generating a new encryption key pair that is stored in the
296    /// key store.
297    ///
298    /// This function can be used when generating an update. In most other cases
299    /// a leaf node should be generated as part of a new [`KeyPackage`].
300    #[cfg(test)]
301    pub(crate) fn generate_update<Provider: OpenMlsProvider>(
302        ciphersuite: Ciphersuite,
303        credential_with_key: CredentialWithKey,
304        capabilities: Capabilities,
305        extensions: Extensions,
306        tree_info_tbs: TreeInfoTbs,
307        provider: &Provider,
308        signer: &impl Signer,
309    ) -> Result<Self, LeafNodeGenerationError<Provider::StorageError>> {
310        // Note that this function is supposed to be used in the public API only
311        // because it is interacting with the key store.
312
313        let new_leaf_node_params = NewLeafNodeParams {
314            ciphersuite,
315            credential_with_key,
316            leaf_node_source: LeafNodeSource::Update,
317            capabilities,
318            extensions,
319            tree_info_tbs,
320        };
321
322        let (leaf_node, encryption_key_pair) = Self::new(provider, signer, new_leaf_node_params)?;
323
324        // Store the encryption key pair in the key store.
325        encryption_key_pair
326            .write(provider.storage())
327            .map_err(LeafNodeGenerationError::StorageError)?;
328
329        Ok(leaf_node)
330    }
331
332    /// Update a leaf node.
333    ///
334    /// This function generates a new encryption key pair that is stored in the
335    /// key store and also returned.
336    ///
337    /// This function can be used when generating an update. In most other cases
338    /// a leaf node should be generated as part of a new [`KeyPackage`].
339    pub(crate) fn update<Provider: OpenMlsProvider>(
340        &mut self,
341        ciphersuite: Ciphersuite,
342        provider: &Provider,
343        signer: &impl Signer,
344        group_id: GroupId,
345        leaf_index: LeafNodeIndex,
346        leaf_node_parmeters: LeafNodeParameters,
347    ) -> Result<EncryptionKeyPair, LeafNodeUpdateError<Provider::StorageError>> {
348        let tree_info = TreeInfoTbs::Update(TreePosition::new(group_id, leaf_index));
349        let mut leaf_node_tbs = LeafNodeTbs::from(self.clone(), tree_info);
350
351        // Update credential
352        if let Some(credential_with_key) = leaf_node_parmeters.credential_with_key {
353            leaf_node_tbs.payload.credential = credential_with_key.credential;
354            leaf_node_tbs.payload.signature_key = credential_with_key.signature_key;
355        }
356
357        // Update extensions
358        if let Some(extensions) = leaf_node_parmeters.extensions {
359            leaf_node_tbs.payload.extensions = extensions;
360        }
361
362        // Update capabilities
363        if let Some(capabilities) = leaf_node_parmeters.capabilities {
364            leaf_node_tbs.payload.capabilities = capabilities;
365        }
366
367        // Create a new encryption key pair
368        let encryption_key_pair =
369            EncryptionKeyPair::random(provider.rand(), provider.crypto(), ciphersuite)?;
370        leaf_node_tbs.payload.encryption_key = encryption_key_pair.public_key().clone();
371
372        // Store the encryption key pair in the key store.
373        encryption_key_pair
374            .write(provider.storage())
375            .map_err(LeafNodeUpdateError::Storage)?;
376
377        // Set the leaf node source to update
378        leaf_node_tbs.payload.leaf_node_source = LeafNodeSource::Update;
379
380        // Sign the leaf node
381        let leaf_node = leaf_node_tbs.sign(signer)?;
382        self.payload = leaf_node.payload;
383        self.signature = leaf_node.signature;
384
385        Ok(encryption_key_pair)
386    }

451    /// Check whether the this leaf node supports all the required extensions
452    /// in the provided list.
453    pub(crate) fn check_extension_support(
454        &self,
455        extensions: &[ExtensionType],
456    ) -> Result<(), LeafNodeValidationError> {
457        for required in extensions.iter() {
458            if !self.supports_extension(required) {
459                log::error!(
460                    "Leaf node does not support required extension {:?}\n
461                    Supported extensions: {:?}",
462                    required,
463                    self.payload.capabilities.extensions
464                );
465                return Err(LeafNodeValidationError::UnsupportedExtensions);
466            }
467        }
468        Ok(())
469    }
470
471    /// Perform all checks that can be done without further context:
472    /// - the used extensions are not known to be invalid in leaf nodes
473    /// - the types of the used extensions are covered by the capabilities
474    /// - the type of the credential is covered by the capabilities
475    pub(crate) fn validate_locally(&self) -> Result<(), LeafNodeValidationError> {
476        // Check that no extension is invalid when used in leaf nodes.
477        let invalid_extension_types = self
478            .extensions()
479            .iter()
480            .filter(|ext| ext.extension_type().is_valid_in_leaf_node() == Some(false))
481            .collect::<Vec<_>>();
482        if !invalid_extension_types.is_empty() {
483            log::error!(
484                "Invalid extension used in leaf node: {:?}",
485                invalid_extension_types
486            );
487            return Err(LeafNodeValidationError::UnsupportedExtensions);
488        }
489
490        // Check that all extensions are contained in the capabilities.
491        if !self.capabilities().contains_extensions(self.extensions()) {
492            log::error!(
493                "Leaf node does not support all extensions it uses\n
494                Supported extensions: {:?}\n
495                Used extensions: {:?}",
496                self.payload.capabilities.extensions,
497                self.extensions()
498            );
499            return Err(LeafNodeValidationError::UnsupportedExtensions);
500        }
501
502        // Check that the capabilities contain the leaf node's credential type.
503        // (https://validation.openmls.tech/#valn0113)
504        if !self
505            .capabilities()
506            .contains_credential(self.credential().credential_type())
507        {
508            return Err(LeafNodeValidationError::UnsupportedCredentials);
509        }
510
511        Ok(())
512    }

720    pub(crate) fn into_verifiable_leaf_node(self) -> VerifiableLeafNode {
721        match self.payload.leaf_node_source {
722            LeafNodeSource::KeyPackage(_) => {
723                let verifiable = VerifiableKeyPackageLeafNode {
724                    payload: self.payload,
725                    signature: self.signature,
726                };
727                VerifiableLeafNode::KeyPackage(verifiable)
728            }
729            LeafNodeSource::Update => {
730                let verifiable = VerifiableUpdateLeafNode {
731                    payload: self.payload,
732                    signature: self.signature,
733                    tree_position: None,
734                };
735                VerifiableLeafNode::Update(verifiable)
736            }
737            LeafNodeSource::Commit(_) => {
738                let verifiable = VerifiableCommitLeafNode {
739                    payload: self.payload,
740                    signature: self.signature,
741                    tree_position: None,
742                };
743                VerifiableLeafNode::Commit(verifiable)
744            }
745        }
746    }

748    /// Returns the `encryption_key` as byte slice.
