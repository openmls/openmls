37impl TreeSyncDiff<'_> {
38    /// Encrypt the given `path` to the nodes in the copath resolution of the
39    /// owner of this [`TreeSyncDiff`]. The `group_context` is used in the
40    /// encryption of the nodes, while the `exclusion_list` is used to filter
41    /// target leaves from the encryption targets. The given [`LeafNode`] is
42    /// included in the resulting [`UpdatePath`].
43    ///
44    /// Returns the encrypted path (i.e. an [`UpdatePath`] instance).
45    pub(crate) fn encrypt_path(
46        &self,
47        crypto: &impl OpenMlsCrypto,
48        ciphersuite: Ciphersuite,
49        path: &[PlainUpdatePathNode],
50        group_context: &[u8],
51        exclusion_list: &HashSet<&LeafNodeIndex>,
52        own_leaf_index: LeafNodeIndex,
53    ) -> Result<Vec<UpdatePathNode>, LibraryError> {
54        // Copath resolutions with the corresponding public keys.
55        let copath_resolutions = self
56            .filtered_copath_resolutions(own_leaf_index, exclusion_list)
57            .into_iter()
58            .map(|resolution| {
59                resolution
60                    .into_iter()
61                    .map(|(_, node_ref)| match node_ref {
62                        NodeReference::Leaf(leaf) => leaf.encryption_key().clone(),
63                        NodeReference::Parent(parent) => parent.encryption_key().clone(),
64                    })
65                    .collect::<Vec<EncryptionKey>>()
66            })
67            .collect::<Vec<Vec<EncryptionKey>>>();
68
69        // There should be as many copath resolutions.
70        debug_assert_eq!(copath_resolutions.len(), path.len());
71
72        // Encrypt the secrets
73
74        #[cfg(not(target_arch = "wasm32"))]
75        let resolved_path = path.par_iter().zip(copath_resolutions.par_iter());
76        #[cfg(target_arch = "wasm32")]
77        let resolved_path = path.iter().zip(copath_resolutions.iter());
78
79        resolved_path
80            .map(|(node, resolution)| node.encrypt(crypto, ciphersuite, resolution, group_context))
81            .collect::<Result<Vec<UpdatePathNode>, LibraryError>>()
82    }
83
84    /// Decrypt an [`UpdatePath`] originating from the given
85    /// `sender_leaf_index`. The `group_context` is used in the decryption
86    /// process and the `exclusion_list` is used to determine the position of
87    /// the ciphertext in the `UpdatePath` that we can decrypt.
88    ///
89    /// Returns a vector containing the decrypted [`ParentNode`] instances, as
90    /// well as the [`CommitSecret`] resulting from their derivation. Returns an
91    /// error if the `sender_leaf_index` is outside of the tree.
92    ///
93    /// ValSem203: Path secrets must decrypt correctly
94    /// ValSem204: Public keys from Path must be verified and match the private keys from the direct path
95    /// TODO #804
96    pub(crate) fn decrypt_path(
97        &self,
98        crypto: &impl OpenMlsCrypto,
99        ciphersuite: Ciphersuite,
100        params: DecryptPathParams,
101        owned_keys: &[&EncryptionKeyPair],
102        own_leaf_index: LeafNodeIndex,
103    ) -> Result<(Vec<EncryptionKeyPair>, CommitSecret), ApplyUpdatePathError> {
104        let path_position = self
105            .subtree_root_position(params.sender_leaf_index, own_leaf_index)
106            .map_err(|_| LibraryError::custom("Expected own leaf to be in the tree"))?;
107
108        let update_path_node = params
109            .update_path
110            .get(path_position)
111            // We know the update path has the right length through validation, therefore there must be an element at this position
112            // TODO #804
113            .ok_or_else(|| LibraryError::custom("Expected to find ciphertext in update path 1"))?;
114
115        let (decryption_key, resolution_position) = self
116            .decryption_key(
117                params.sender_leaf_index,
118                params.exclusion_list,
119                owned_keys,
120                own_leaf_index,
121            )
122            // TODO #804
123            .map_err(|_| LibraryError::custom("Expected sender to be in the tree"))?;
124
125        let ciphertext = update_path_node
126            .encrypted_path_secrets(resolution_position)
127            // We know the update path has the right length through validation, therefore there must be a ciphertext at this position
128            // TODO #804
129            .ok_or_else(|| LibraryError::custom("Expected to find ciphertext in update path 2"))?;
130
131        // ValSem203: Path secrets must decrypt correctly
132        let path_secret = PathSecret::decrypt(
133            crypto,
134            ciphersuite,
135            ciphertext,
136            decryption_key,
137            params.group_context,
138        )
139        .map_err(|_| ApplyUpdatePathError::UnableToDecrypt)?;
140
141        let common_path =
142            self.filtered_common_direct_path(own_leaf_index, params.sender_leaf_index);
143        let (derived_path, _plain_update_path, keypairs, commit_secret) =
144            ParentNode::derive_path(crypto, ciphersuite, path_secret, common_path)?;
145        // We now check that the public keys in the update path and in the
146        // derived path match up.
147        // ValSem204: Public keys from Path must be verified and match the private keys from the direct path
148        for (update_parent_node, (_, derived_parent_node)) in params
149            .update_path
150            .iter()
151            .skip(path_position)
152            .zip(derived_path.iter())
153        {
154            if update_parent_node.public_key() != derived_parent_node.public_key() {
155                return Err(ApplyUpdatePathError::PathMismatch);
156            }
157        }
158
159        Ok((keypairs, commit_secret))
160    }
161
162    /// Prepare the [`EncryptedGroupSecrets`] for a number of `invited_members`
163    /// based on a [`TreeSyncDiff`]. If a slice of [`PlainUpdatePathNode`] is
164    /// given, they are included in the [`GroupSecrets`] of the path.
165    ///
166    /// Returns an error if
167    ///  - the own node is outside the tree
168    ///  - the invited members are not part of the tree yet
169    ///  - the leaf index of a new member is identical to the own leaf index
170    ///  - the plain path does not contain the correct secrets
171    #[allow(clippy::too_many_arguments)]
172    pub(crate) fn encrypt_group_secrets(
173        &self,
174        joiner_secret: &JoinerSecret,
175        invited_members: Vec<(LeafNodeIndex, AddProposal)>,
176        plain_path_option: Option<&[PlainUpdatePathNode]>,
177        presharedkeys: &[PreSharedKeyId],
178        encrypted_group_info: &[u8],
179        crypto: &impl OpenMlsCrypto,
180        encryptor_leaf_index: LeafNodeIndex,
181    ) -> Result<Vec<EncryptedGroupSecrets>, LibraryError> {
182        let mut encrypted_group_secrets_vec = vec![];
183        for (leaf_index, add_proposal) in invited_members {
184            let key_package = add_proposal.key_package;
185
186            let direct_path_position = self
187                .subtree_root_position(encryptor_leaf_index, leaf_index)
188                // This can only fail if the nodes are outside the tree or identical
189                .map_err(|_| LibraryError::custom("Unexpected error in subtree_root_position"))?;
190
191            // If a plain path was given, there have to be secrets for every new member.
192            let path_secret_option = if let Some(plain_path) = plain_path_option {
193                Some(
194                    plain_path
195                        .get(direct_path_position)
196                        .map(|pupn| pupn.path_secret())
197                        // This only fails if the supplied plain path is invalid
198                        .ok_or_else(|| LibraryError::custom("Invalid plain path"))?,
199                )
200            } else {
201                None
202            };
203
204            // Create the GroupSecrets object for the respective member.
205            let group_secrets_bytes =
206                GroupSecrets::new_encoded(joiner_secret, path_secret_option, presharedkeys)
207                    .map_err(LibraryError::missing_bound_check)?;
208            let ciphertext = hpke::encrypt_with_label(
209                key_package.hpke_init_key().as_slice(),
210                "Welcome",
211                encrypted_group_info,
212                &group_secrets_bytes,
213                key_package.ciphersuite(),
214                crypto,
215            )
216            .map_err(|_| {
217                LibraryError::custom(
218                    "Error while encrypting group secrets. \
219                     This could have really only been a missing bounds check in \
220                     the serialization",
221                )
222            })?;
223            let encrypted_group_secrets =
224                EncryptedGroupSecrets::new(key_package.hash_ref(crypto)?, ciphertext);
225            encrypted_group_secrets_vec.push(encrypted_group_secrets);
226        }
227        Ok(encrypted_group_secrets_vec)
228    }
229}

238/// 8.6. Update Paths
239///
240/// ```text
241/// struct {
242///     HPKEPublicKey public_key;
243///     HPKECiphertext encrypted_path_secret<V>;
244/// } UpdatePathNode;
245/// ```
246#[derive(
247    Debug,
248    Eq,
249    PartialEq,
250    Clone,
251    Serialize,
252    Deserialize,
253    TlsDeserialize,
254    TlsDeserializeBytes,
255    TlsSerialize,
256    TlsSize,
257)]
258pub struct UpdatePathNode {
259    pub(super) public_key: EncryptionKey,
260    pub(super) encrypted_path_secrets: Vec<HpkeCiphertext>,
261}
262
263impl UpdatePathNode {
264    /// Return the `encrypted_path_secrets`.
265    fn encrypted_path_secrets(&self, ciphertext_index: usize) -> Option<&HpkeCiphertext> {
266        self.encrypted_path_secrets.get(ciphertext_index)
267    }
268
269    /// Return the `public_key`.
270    fn public_key(&self) -> &HpkePublicKey {
271        self.public_key.key()
272    }
273
274    pub(crate) fn encryption_key(&self) -> &EncryptionKey {
275        &self.public_key
276    }
277
278    /// Flip the last byte of every `encrypted_path_secret` in this node.
279    #[cfg(test)]
280    fn flip_last_byte(&mut self) {
281        let mut new_eps_vec = Vec::new();
282        for eps in self.encrypted_path_secrets.as_slice() {
283            let mut new_eps = eps.clone();
284            let mut last_bits = new_eps
285                .ciphertext
286                .pop()
287                .expect("An unexpected error occurred.");
288            last_bits ^= 0xff;
289            new_eps.ciphertext.push(last_bits);
290            new_eps_vec.push(new_eps);
291        }
292        self.encrypted_path_secrets = new_eps_vec;
293    }
294
295    /// Flip the last byte of the public key in this node.
296    #[cfg(test)]
297    fn flip_last_pk_byte(&mut self) {
298        use tls_codec::{Deserialize, Serialize};
299
300        let mut new_pk_serialized = self
301            .public_key
302            .tls_serialize_detached()
303            .expect("error serializing public key");
304        let mut last_bits = new_pk_serialized
305            .pop()
306            .expect("An unexpected error occurred.");
307        last_bits ^= 0xff;
308        new_pk_serialized.push(last_bits);
309        self.public_key = EncryptionKey::tls_deserialize(&mut new_pk_serialized.as_slice())
310            .expect("error deserializing pk");
311    }
312}

314/// 8.6. Update Paths
315///
316/// ```text
317/// struct {
318///     LeafNode leaf_node;
319///     UpdatePathNode nodes<V>;
320/// } UpdatePath;
321/// ```
322#[derive(Debug, PartialEq, Eq, Clone, Serialize, Deserialize, TlsSerialize, TlsSize)]
323pub struct UpdatePath {
324    leaf_node: LeafNode,
325    nodes: Vec<UpdatePathNode>,
326}
327
328impl UpdatePath {
329    /// Generate a new update path.
330    pub(crate) fn new(leaf_node: LeafNode, nodes: Vec<UpdatePathNode>) -> Self {
331        Self { leaf_node, nodes }
332    }
333
334    /// Return the `leaf_node` of this [`UpdatePath`].
335    pub(crate) fn leaf_node(&self) -> &LeafNode {
336        &self.leaf_node
337    }
338
339    /// Return the `nodes` of this [`UpdatePath`].
340    pub(crate) fn nodes(&self) -> &[UpdatePathNode] {
341        &self.nodes
342    }
343
344    #[cfg(test)]
345    /// Flip the last bytes of the ciphertexts of all contained nodes.
346    pub fn flip_eps_bytes(&mut self) {
347        let mut new_nodes = Vec::new();
348        for node in self.nodes.as_slice() {
349            let mut new_node = node.clone();
350            new_node.flip_last_byte();
351            new_nodes.push(new_node);
352        }
353        self.nodes = new_nodes;
354    }
355
356    #[cfg(test)]
357    /// Set the path key package.
358    pub fn set_leaf_node(&mut self, leaf_node: LeafNode) {
359        self.leaf_node = leaf_node
360    }
361
362    #[cfg(test)]
363    /// Remove and return the last node in the update path. Returns `None` if
364    /// the path is empty.
365    pub fn pop(&mut self) -> Option<UpdatePathNode> {
366        self.nodes.pop()
367    }
368
369    #[cfg(test)]
370    /// Flip the last bytes of the public key in the last node in the path.
371    pub fn flip_node_bytes(&mut self) {
372        let mut last_node = self.nodes.pop().expect("path empty");
373        last_node.flip_last_pk_byte();
374        self.nodes.push(last_node)
375    }
376}

378#[derive(
379    Debug,
380    PartialEq,
381    Eq,
382    Clone,
383    Serialize,
384    Deserialize,
385    TlsSerialize,
386    TlsDeserialize,
387    TlsDeserializeBytes,
388    TlsSize,
389)]
390pub struct UpdatePathIn {
391    leaf_node: LeafNodeIn,
392    nodes: Vec<UpdatePathNode>,
393}
394
395impl UpdatePathIn {
396    /// Return the `leaf_node` of this [`UpdatePath`].
397    pub(crate) fn leaf_node(&self) -> &LeafNodeIn {
398        &self.leaf_node
399    }
400
401    /// Return a verified [`UpdatePath`].
402    pub(crate) fn into_verified(
403        self,
404        ciphersuite: Ciphersuite,
405        crypto: &impl OpenMlsCrypto,
406        tree_position: TreePosition,
407    ) -> Result<UpdatePath, UpdatePathError> {
408        let leaf_node_in = self.leaf_node().clone();
409        let verifiable_leaf_node = leaf_node_in.into_verifiable_leaf_node();
410
411        // [valn1207](https://validation.openmls.tech/#valn1207)
412        match verifiable_leaf_node {
413            // https://validation.openmls.tech/#valn1208
414            VerifiableLeafNode::Commit(mut commit_leaf_node) => {
415                let pk = &commit_leaf_node
416                    .signature_key()
417                    .clone()
418                    .into_signature_public_key_enriched(ciphersuite.signature_algorithm());
419                commit_leaf_node.add_tree_position(tree_position);
420
421                let leaf_node: LeafNode = commit_leaf_node.verify(crypto, pk)?;
422                Ok(UpdatePath {
423                    leaf_node,
424                    nodes: self.nodes,
425                })
426            }
427            VerifiableLeafNode::Update(_) | VerifiableLeafNode::KeyPackage(_) => {
428                Err(UpdatePathError::InvalidType)
429            }
430        }
431    }
432}
